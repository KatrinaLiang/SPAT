class Person {
	/********* begin *********/
  String name ;
     int age;
     Person(String name,int age)
     {
    	  this.name = name;
    	  this.age = age;
      }
	/********* end *********/
}

class Student extends Person {
	/********* begin *********/
    String school="哈佛大学";
    Student(){
    	super("张三",18);
    }
    public void print(){
    	System.out.println("我是："+name+"，今年："+age+"岁，我在"+school+"上学");
    }
	/********* end *********/
}
