package step1;


public class Person {
    private String name; //姓名
    private String number; //身份证号
    private Date birthday; //出生日期
    
   /************Begin*************/
 	public Person() {
		this.birthday = new Date();
	}
	public Person(String name, String number, Date birthday) {
		this.name = name;
		this.number = number;
		this.birthday = birthday;
	}
   public Person(String name,String number,int year,int month,int day) {
	   this.name = name;
	   this.number = number;
	  this.birthday= new Date(year, month, day);
   }
   	public void setinfo(String name,String number,int year,int month,int day) {
		this.name = name;
	    this.number = number;
	 this.birthday = new Date(year, month, day);
	}
	
	public String toString() {
			return "name="+this.name+" number="+this.number+" birthday:"+this.birthday.getYear()+"."+this.birthday.getMonth()+"."+this.birthday.getDay();
	 }
   
	public String getName() {
		return name;
	}
	public String getNumber() {
		return number;
	}
	public Date getBirthday() {
		return birthday;
	}
	

    
    
    
    
   /************End*************/
}
