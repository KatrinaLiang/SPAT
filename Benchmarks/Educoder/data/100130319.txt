package step1;

public class Task {

	public void sortArray(int[] arr, String sort){
		/********** BEGIN **********/
		int t;
			boolean x = sort.equals("asc");//升序 为 true
			for(int i = 1,j;i < arr.length;i++)
			{
				t = arr[i];
				for(j = i-1;j >= 0 && (arr[j] > t)== x ;j--){
					arr[j+1]=arr[j];
				}
				arr[j+1] = t;
			}
		for(int a:arr)System.out.println(a);
		/********** END **********/
	}
}
