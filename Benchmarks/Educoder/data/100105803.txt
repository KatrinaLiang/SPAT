package step1;

public class Yard {
	int length,width;
    public Yard(int length,int width){
    	this.length=length;
        this.width=width;
    }
    public  double getPerimeter()
    {
        double perimeter=(length+width)*2;
        return perimeter;
    }
    public  double getAcreage()
    {
        
    	double acreage=length*width;
        return acreage;
    }
}
