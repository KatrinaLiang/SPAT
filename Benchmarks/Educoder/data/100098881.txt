package step1;

public class Yard {
    double length;
    double width;
    public Yard(double l, double w)
    {
        length = l;
        width = w;
    }
    public double getPerimeter()
    {
        return 2*(length + width);
    }
    public double getAcreage()
    {
        return length * width;
    }
}
