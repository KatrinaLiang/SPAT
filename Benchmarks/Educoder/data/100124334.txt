package step2;
public class Fibonacci{
	public static void main(String args[]){
		/*********begin********/
		int sum=0;
		int h;
	    int[] arr=new int[20];
	    arr[0]=1;
	    arr[1]=1;
	    for(h=2;h<=19;h++)
	    {      
	       arr[h]=arr[h-1]+arr[h-2];
	    }
	    for(int i=0;i<=19;i++)
	    {
	    	sum=sum+arr[i];
	    }
	    System.out.print("前20个元素的和为"+sum);
		/********end********/
		}
	}
