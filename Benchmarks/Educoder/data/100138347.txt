package step1;

public class ThreadClassOne extends Thread{
	
    public static void main(String[] args) {
        Thread t1 = new ThreadClassOne();
        t1.start();
        ThreadClassTwo t2 = new ThreadClassTwo();
        t2.start();
    }
    public void run(){
    	
        
        for(int i=0;i<10;i++){
        	if(i%2!=0)
                System.out.print(i+" ");
        }
    }
}

class ThreadClassTwo implements Runnable{
	private Thread mythread;
    
    public void run(){
    
        
        for(int i=0;i<=10;i++){
        	if(i%2==0)
                System.out.print(i+" ");
        }
    }
    public void start() {
        if (mythread == null) {
            mythread = new Thread(this);
            mythread.start();
        }
    }
}
