import javax.swing.JOptionPane;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
    public InterRectangle(double width,double length){
        this.width=width;
        this.length=length;
    }
    public double jisuanz(){
		double z=2*width+2*length;
        return z;
    }
    public double jisuans(){
        double s=width*length;
        return s;
    }
	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
        System.out.println("总共有"+total+"块土地");
        double nums=0.0;
        double numz=0.0;
        while(total>0){
		double width=scanner.nextDouble();
        double length=scanner.nextDouble();
 		InterRectangle a=new InterRectangle(width,length);
        numz=numz+a.jisuanz();
           nums=nums+a.jisuans();
            total--;
        }
        System.out.println("土地总面积为:"+nums+"平米");
        System.out.println("土地总周长为:"+numz+"米");
		/*********end*********/
	}

}
