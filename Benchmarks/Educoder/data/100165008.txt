import java.util.Scanner;

public class Box {
    
    double width;
    double height;
    double depth;
    static double v;
    
    public Box(double width,double depth,double height)
    {
    	this.width=width;
    	this.depth=depth;
    	this.height=height;
    }
    
    
    public static void main(String[] args)
{
    	 Scanner scan = new Scanner(System.in);
       
         Box box = new Box(scan.nextDouble(),scan.nextDouble(),scan.nextDouble()); 
         double vol=box.volume(); 
    System.out.println("the volumn is"+vol);
        
        
    }


	public double volume() {
		
		return width*depth*height;
	}
    
}
