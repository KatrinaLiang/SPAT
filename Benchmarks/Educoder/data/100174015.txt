package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int k;
		for(int i=1;i<=9;i++){
			for(int j=1;j<=i;j++){
				k=i*j;
				System.out.print(j+"*"+i+"="+k+"\t");
			}
			System.out.println();
		}
		/*****end*****/
	}
}
