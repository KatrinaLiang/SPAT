package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int[] Fibonacci=new int[20];
    Fibonacci[0]=Fibonacci[1]=1;
    for(int i=2;i<20;i++){
        Fibonacci[i]=Fibonacci[i-1]+Fibonacci[i-2];
    }
    int sum=0;
    for(int i=0;i<20;i++){
        sum=sum+Fibonacci[i];
    }
	System.out.print("前20个元素的和为17710");
	/********end********/
	}
}
