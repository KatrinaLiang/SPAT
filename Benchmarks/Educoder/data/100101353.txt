package step2;

import java.util.Scanner;

public class SystemUpdate {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String name = scanner.next();
        String birth = scanner.next();
        String position = scanner.next();
        double salary = scanner.nextDouble();
        Salary employee = new Salary(name, birth, position, salary);
        employee.introduction();
    }
}

/********** Begin *********/
class Employee {
    private String name;
    private String birth;
    private String position;
    
    public Employee(String name,String birth,String position){
    	this.name=name;
    	this.birth=birth;
    	this.position=position;
    }
    
    public String getName() {
    	return name;
    }
    public String getBirth() {
    	return birth;
    }
    public String getPosition() {
    	return position;
    }
}

class Salary extends Employee {
	
	private double salary;
	
    public Salary(String name, String birth, String position, double salary2) {
    	super(name,birth,position);
		this.salary=salary2;
	}
    
	public void introduction() {
		System.out.print("员工姓名："+super.getName()+" 出生年月："+super.getBirth()+" 职位："+super.getPosition()+" 薪水："+salary);
	}

	
}
/********** End *********/
