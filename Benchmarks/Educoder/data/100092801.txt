package step1;

public class LetterChange {

    // 使用 switch把小写字母转为大写字母
    public char LetterChange(char ch){
      char ch1;
        switch(ch){
           case'a':
           case'b':
           case'c':
           case'd':
           case'e':
                ch1=(char)(ch-32);
                break;
            default:ch1=ch;
                 
                 }
        return ch1;// 请在此提供实现代码
    }
}
