package step1;
import java.text.DecimalFormat;
public class Yard {
	DecimalFormat f=new DecimalFormat("######0.0"); 
	int length;
	int width;
	public Yard(int A,int B) {
		length=A;
		width=B;
	}
	public String getPerimeter() {
        int C;
        C=length+length+width+width;
		return f.format(C);
	}
	public String getAcreage() {
        int S;
        S=length*width;
		return f.format(S);
	}
}
