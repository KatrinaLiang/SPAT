public class Task {
	
	public void sortArray(int[] arr){
for(int i=1;i<arr.length;i++){
    for(int j=i;j>0;j--){
        if(arr[j]<arr[j-1]){
            int t=arr[j];
            arr[j]=arr[j-1];
            arr[j-1]=t;
        }
    }
    int k=0;
    for(k=0;k<arr.length;k++){
        System.out.print(arr[k]+" ");
      
        }
         
            System.out.println();
                
    }
}
		
		
}
