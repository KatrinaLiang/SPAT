package step3;
import java.util.Scanner;

public class Factorial {
	public void factorial(int n) {
        int sum=1;
		for(int i=n;i>0;i--){
        	sum=sum*i;
        }
		System.out.print(sum);		
	}
    public static void main(String args[]){
    	Scanner scan=new Scanner(System.in);
        int n=scan.nextInt();
        Factorial num=new Factorial();
        num.factorial(n);
    }
}
