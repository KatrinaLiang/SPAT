package step2;
public class AlarmGuardDoor extends Door implements Guard,Alarm{

	public void alarm() {		
		System.out.println("刮风下雨收衣服了");
	}

	public void guard() {
		System.out.println("防盗效果好");
	}

	void open() {		
		System.out.println("芝麻开门");
	}
	
	void close() {		
		System.out.println("芝麻关门");
	}

	
	
}
