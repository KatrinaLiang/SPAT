package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
        int remarks=0;
        do
        {
        if(n%2==0)
        {
        	n=n/2;
            if(n==1)
        	{
        		remarks++;
        	}
        }
        else
        {
        	n=n-1;
            n=n/2;
            if(n==1)
        	{
        		remarks++;
        	}
            remarks++;
        }
        }while(n>1);
        
		return remarks;
		
		
		
		
		/********* End *********/
	}
	
}
