public class ThreadHelper {
int num;
public static void calcOnNewThread(int num){
new Thread(new Runnable() {
           public void run(){
    int result = 1;
        for(int i = num ; i > 0; i--)
            result *= i;
   
    System.out.println(result);
}
}).start();
}
}
