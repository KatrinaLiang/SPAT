package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		int i=1,j;
        for(;i<10;i++){
            for(j=1;j<=i;j++){
                System.out.print(j+"*"+i+"="+(j*i)+"\t");
            }
            System.out.print("\n");
        }
		
		
		
		
		
		
		/*****end*****/
	}
}
