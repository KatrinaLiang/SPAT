package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int Sum=0;
        while(n>1){
            if(n%2!=0){
                n=n-1;
                Sum++;
            }
            n=n/2;
        }
		Sum+=1;
		return Sum;
		
		
		/********* End *********/
	}
	
}
