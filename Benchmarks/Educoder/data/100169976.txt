package step1;
import java.util.Scanner;

public class Cal {
	public static void main(String[] args) {
		/*********start*********/
		Scanner sc = new Scanner(System.in);
		System.out.println("请输入第一个整数");
		int i = sc.nextInt();
		System.out.println("请输入第二个整数");
		int k = sc.nextInt();
		int sum1 = i+k;
		int sum2 = i-k;
		int sum3 = i*k;
		int sum4 = i/k;
		int sum5 = i%k;
		System.out.println("两数相加的结果为："+sum1);
		System.out.println("两数相减的结果为："+sum2);
		System.out.println("两数相乘的结果为："+sum3);
		System.out.println("两数相除的结果为："+sum4);
		System.out.println("两数取余数的结果为："+sum5);
		/*********end*********/
	}

}
