package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		double r=0;
		if(n%2==1) {
			for(double i=1;i<=n;i+=2) {
				r+=4/i;
			}
			System.out.println(r);
		}
		else {
			for(double i=2;i<=n;i+=2) {
				r+=3/i;
			}
			System.out.println(r);
		}
	}
}
