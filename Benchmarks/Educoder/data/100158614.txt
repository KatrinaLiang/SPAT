#include<bits/stdc++.h>
#include<cstring>
using namespace std;
main()
{
	char str1[100],str2[100];
	int i,j,x,y,n=0;
	cin >> str1;
	cin >> str2;
	x=strlen(str1);
	y=strlen(str2);
	for(i=0;i<x;i++)
	{
	   
		for(j=0;j<y;j++)
		{
			int temp=0;
			if(str1[i]==str2[j])
			{
				for(int k=0;;k++)
				{
					if(i+k<x&&j+k<y)
					{
						if(str1[i+k]==str2[j+k]) temp++;
						else break;
					}
					else break;
				}
				
			}
			n=max(n,temp);
		}
	}
    cout << n;
    return 0;
}
