package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum = 0;
        if(num1 > num2){
            int n = num1;
            num1 = num2;
            num2 = n;
        }
        for(int j = num1;j <= num2;j++)
        {
            int i = j;
            while(i != 0){
          	  sum = sum + i % 10;
         	   i = i / 10;
      	  }
        }
        return sum;
		/********** END **********/
	}
}
