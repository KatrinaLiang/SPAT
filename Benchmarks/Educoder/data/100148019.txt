public class MyThread extends Thread{
    private int num;

    MyThread(int num){
        this.num = num;
    }

    @Override
    public void run(){
        try {
            Thread.currentThread().sleep(200);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
        int result = 1;
        for (int i = 1; i <= num; i++) {
            result *= i;
        }
        System.out.println(result);
    }
}
