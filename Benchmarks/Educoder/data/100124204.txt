package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int s=0;
		Fibonacci fd=new Fibonacci();
		for(int i=1;i<=20;i++){
			s=s+fd.f(i);
		}
		System.out.print("前20个元素的和为"+s);
	}
	public int f(int n){
        if(n==1|n==2){
            return 1;
        }else{
            return f(n-1)+f(n-2);
        }
    }
	/********end********/
}
