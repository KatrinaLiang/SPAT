package step3;
import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
	int s,temp=0,temp1=0;
       int o;
        Scanner red = new Scanner(System.in);
        System.out.println("请输入一个五位数字：");
        s=red.nextInt();
        o=s;
        for(int i=0;i<5;i++)
        {
        	temp=s%10;
            s=s/10;
            temp1=temp1+temp;
        }
        System.out.println(o+"的各位数字之和是："+temp1);

		/******End******/
	}
}
