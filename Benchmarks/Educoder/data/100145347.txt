package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		
		
		int b=0;
    	while(n!=0)
    	{
    		
    	   if(n%2!=0)
    	   {
    		   n=n-1;
    		   n=n/2;
    		   b++;
    	   }
    	   else
    		    n=n/2;
    	}
    	return b;
		
		
		/********* End *********/
	}
	
}
