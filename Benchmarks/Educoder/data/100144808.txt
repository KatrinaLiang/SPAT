package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		int temp  = 0;
        int m = 0 ;
        int count = 0;
        if(matrix.length%2 ==0)
            count = matrix.length/2;
        else 
            count = (matrix.length -1 )/2;
        for(int i = 0 ;i<count;i++) {
          for(int j = m ;j<matrix.length-1-m;j++){
              temp = matrix[j][matrix.length-1-m];          //2->temp
              matrix[j][matrix.length-1-m] = matrix[i][j];   //1->2
              matrix[i][j] = matrix[matrix.length-1-j][i];       //4->1
              matrix[matrix.length-1-j][i] = matrix[matrix.length-1-m][matrix.length-1-j]; //3->4
              matrix[matrix.length-1-m][matrix.length-1-j] = temp; // temp ->3
             }
            m++;
            }
      
   		
        }
		

		
		
		
		/********* End *********/
	}
