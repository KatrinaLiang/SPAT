public class Task {
    
    public double func1(double x,double y){
     	double tmp = (Math.pow(x,2) + Math.pow(y,2) - 1);
        tmp = Math.pow(tmp,3);
        double tmp2 = Math.pow(x,2) * Math.pow(y,3);
        
        return tmp - tmp2;
    }
    
    public double func2(double x, double y) {
		double tmp = Math.pow(x, 2);
		double tmp2 = 5.0 * y / 4.0 - Math.sqrt(Math.abs(x));
		tmp2 = Math.pow(tmp2, 2);
		return tmp + tmp2 - 1.0;
	}
    
	public void love(int n){
