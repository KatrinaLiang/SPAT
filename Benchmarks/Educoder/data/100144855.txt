package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
	int len=matrix.length;
		int b [][]=new int [len][len];
		for(int i=0;i<len;i++){
			for(int j=0;j<len;j++){
				b[j][len-1-i]=matrix[i][j];
			}
		}
		for(int i=0;i<len;i++)
			for(int j=0;j<len;j++)
				matrix[i][j]=b[i][j];		
		/********* End *********/
	}
}
