package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum=0;
        int number=0;
		for(int i=num1;i<=num2;i++){
            number=i;
            for(int j=1;j<=100;j=j*10){
                if(number/j<10){
                    for(int k=j;k>=1;k=k/10){
                		sum+=number/k;
                        number=number-number/k*k;
                    }
                }
            }
        }
        return sum;
		
		
		/********** END **********/
	}
}
