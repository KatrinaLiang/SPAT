class Person {
	/********* begin *********/
	private String name;
    private int age;
    public void set(String name,int age){
    	this.name = name;
        this.age = age;
    }
    public void get(){
    	System.out.print("姓名："+name+"，年龄：" + age +"，");
    }
	/********* end *********/
}

class Student extends Person {
	/********* begin *********/
	private String school;
    public void setschool(String school){
		this.school = school;    
    }
    public void get(){
    	super.get();
        System.out.println("学校：" + school);
    }
    
	/********* end *********/
}
