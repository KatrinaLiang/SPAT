package step1;
import java.util.Scanner;

public class AbsoluteOfTwo {
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        AbsoluteOfTwo al = new AbsoluteOfTwo();
        int a = sc.nextInt();
        int b = sc.nextInt();
        System.out.println(al.AbsoluteOfTwo(a,b));
    }

    // 求两个整数差的绝对值
    public int AbsoluteOfTwo(int x, int y) {
        if((x-y)<0){
            return (y-x);
        }else if((x-y)>0){
            	return (x-y);
        }else if((x-y)==0){
            return (x-y);
        }
        return 0;// 请在此添加实现代码
    }

}
