package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		 int count = 0;
        while (n >= 1) {
            if (n % 2 == 1) {
                n--;
                count++;
            }
            n = n / 2;
        }
        return count;

		
		
		
		
		/********* End *********/
	}
	
}
