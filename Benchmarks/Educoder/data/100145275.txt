package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		if(n <= 0){
            return 0;
        }
        int sum = 0;
        while (n != 0){
            if((n & 0x01) == 1){
                sum += 1;
                n--;
            }
            n = n >> 1;
        }
        return sum;
		
		
		
		
		/********* End *********/
	}
	
}
