#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n=0,i,j,k;
	int a[1000];
	while (scanf("%d",&a[n])!=EOF)
	{
		n++;
	}
	n--;
	int t,res=0;
	t=a[n];
	scanf("%d",&t);
	for (i=0;i<n;i++)
	{
		for (j=i+1;j<n;j++)
		{
			for (k=j+1;k<n;k++)
			{
				if (a[i]+a[j]+a[k]<t)
				{
					res++;
				}
			}
		}
	}
	cout<<res;
}
