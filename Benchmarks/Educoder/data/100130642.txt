package step3;

import java.util.Arrays;

public class Task {

	public void matrixSum(int[][] arr){
		/********** BEGIN **********/
		int n=arr.length;
        int x=0;
        int y=0;
        int s=0;
        while(x<n){
        s+=arr[x][y];
            x++;
            y++;
        }
		System.out.println(s);



		/********** END **********/
	}
	
	
}
