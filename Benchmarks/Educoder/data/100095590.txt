package step4;

public class DayCount {
	public void dayCount(int year, int month, int day) {
		/********** Begin *********/
        int count;int month2;
        if(year%4==0)
        	month2=29;
        
        else month2=28;
        
		switch(month){
            case 1:
                count=day;
                System.out.print(count);
                break;
            case 2:
                count=31+day;
                System.out.print(count);
                 break;
            case 3:
                count=31+month2+day;
                System.out.print(count);
                 break;
            case 4: 
                count=31*2+month2+day;
                System.out.print(count);
                 break;
            case 5:
                count=31*2+30+month2+day;
                System.out.print(count);
                 break;
            case 6:
                count=31*3+30+month2+day;
                System.out.print(count);
                 break;
            case 7:
                count=31*3+30*2+month2+day;
                System.out.print(count);
                 break;
            case 8:
                count=31*4+30*2+month2+day;
                System.out.print(count);
                 break;
            case 9:
                count=31*5+30*2+month2+day;
                System.out.print(count);
                 break;
            case 10:
                count=31*5+30*3+month2+day;
                System.out.print(count);
                 break;
            case 11:
                count=31*6+30*3+month2+day;
                System.out.print(count);
                 break;
            case 12:
                count=31*6+30*4+month2+day;
                 break;
            default:
            System.out.println("Error");
                 break;
                
         }

		
		/********** End *********/
	}
}
