package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int i,h,a;
        for(i=1;i<=9;i++){
            for(h=1;h<=i;h++){
                a=h*i;
                System.out.print(h+"*"+i+"="+a+"\t");
            }
            System.out.println();
        }
		
		
		
		
		
		/*****end*****/
	}
}
