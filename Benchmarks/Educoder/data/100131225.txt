package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int i=0;
       for(int a=num1;a<=num2;a++)
       {   int b=a/100;
        int c=(a-100*b)/10;
        int d=a%10;
           i=i+b+c+d;
       }
        return i;
		
		
		
		/********** END **********/
	}
}
