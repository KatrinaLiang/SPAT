package step2;

import java.util.Scanner;

public class Task {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int length = sc.nextInt();	//长
		int width = sc.nextInt();	//宽
		/********** BEGIN **********/
		
			int x=length;
			int y=width;
			int z = y;
			while(x%y!=0) {
				z = x%y;
				x=y;
				y=z;
			}
			System.out.println(z);
			int d=(length*width)/(z*z);
			System.out.println(d);
		
		
		
		
		
		
		/********** END **********/
	}
}
