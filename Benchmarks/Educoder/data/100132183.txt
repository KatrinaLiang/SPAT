package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temps){
		/********** BEGIN **********/
		int [] daily = new int [temps.length];
		for(int i=0;i<temps.length-1;i++){
			int day=1;
			for(int j=i+1;j<temps.length;j++){
				int current=temps[i];
				if(current>=temps[j]){
					day++;
					if((j==temps.length-1)&&(current>=temps[j])){
						day=0;
					}
					continue;
				}
				break;
			}
			daily[i]=day;
		}
		return daily;
		/********** END **********/
	}
}
