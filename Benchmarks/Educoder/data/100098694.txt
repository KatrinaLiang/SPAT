package step1;

public class Yard {
        int length;
        int width;
        int Perimeter;
        int Acreage;
        public Yard(int x,int y){
            length=x;
            width=y;
        }
        public float getPerimeter(){
              Perimeter=length*2+width*2;
              return Perimeter;
        }
         public float getAcreage(){
             Acreage=length*width;
             return Acreage;
         }
}
