package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		
		int sum = 0;
		for(int j = num1 ; j <= num2 ; j++) {
		    int i = j;	
		    while(i>=1) {
				sum += i%10;
				
				i = i/10;
			}
			
		}
		return sum;
		
		/********** END **********/
	}
}
