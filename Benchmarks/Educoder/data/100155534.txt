package task4;

public class Box {

    double width;
    double height;
    double depth;

    int boxType;
	public Box(int boxType){
    	this.boxType = boxType;
    }
    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public double volume() {
        //TODO your code
        double v = 0;
        if(boxType == 1) {
        	v = width * height * depth;
        }
        if(boxType == 2) {
        	v = width * width * 3.14 * height;
        }
        return v;
    }

}
