    private double height;
    private double depth;

  /********* Begin *********/        
    public void setWidth(double widthVal)
    {
        width=widthVal;
    }
    public void setHeight(double heightVal)
    {
        height=heightVal;
    }
    public void setDepth(double depthVal)
    {
        depth=depthVal;
    }

    {
        return width;
    }
   public double getHeight()
   { 
       return height;
   } 
    public double getDepth()
    {
        return depth;
    }


  /********* End *********/        

}
