package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum= 0;
		for(int i = num1;i<=num2;i++)
        {
        	int temp = i;
            while(temp != 0)
            {
            	sum += (temp % 10);
                temp /= 10;
            }
        }
        return sum;
		
		
		/********** END **********/
	}
}
