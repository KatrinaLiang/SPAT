package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int result = 0;
        for(int i = num1;i <= num2;i++){
            result += subcribe(i);
        }
		return result;
		
		
		/********** END **********/
	}
    public int subcribe(int num){
     	int temp = 0;
        do{
            temp += num%10;
        }while((num=num/10)>0);
        return temp;
    }
   
}
