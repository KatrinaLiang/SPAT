package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int count = 0;
        while (n>0){
            if (n%2==0){
                n = n/2;
            }else {
                n = (n-1)/2;
                count++;
            }
        }
        return count;
		
		
		
		
		/********* End *********/
	}
	
}
