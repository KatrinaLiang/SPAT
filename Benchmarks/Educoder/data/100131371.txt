package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum=0;
        for(int i=num1;i<=num2;i++)
        {
		int m=i/100;
        int n=i%100;
        int p=n/10;int q=n%10;
        sum=sum+m+p+q;
        }
        return sum;
		/********** END **********/
	}
}
