package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] a=new int[20];
        int sum=0;
        a[0]=1;
        a[1]=1;
        for(int i=2;i<a.length;i++){
        	a[i]=a[i-1]+a[i-2];
        }
        for(int j=0;j<a.length;j++){
        	sum=sum+a[j];
        }
        System.out.print("前20个元素的和为"+sum);

	/********end********/
	}
}
