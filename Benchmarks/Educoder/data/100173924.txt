package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
		
		//在这里打印出正三角形 
		for(int i=1;i<=10;i++){
			for(int b=1;b<=i;b++){
				System.out.print("*");

			}
			System.out.print("\n");
		}
		
		
		
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
		for(int i=1;i<=10;i++){
			for(int b=1;b<=11-i;b++){
				System.out.print("*");

			}
			System.out.print("\n");
		}
		
		
		
		
		/*****end*****/
	}
}
