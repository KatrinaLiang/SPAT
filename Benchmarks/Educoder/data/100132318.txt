package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
				int day[] = new int[temperatures.length];
		for(int i = 0;i<temperatures.length;i++)
		{
			int temp = temperatures[i];
			
			boolean merge = false;
			int k = 1;
			for(int j = i+1;j<temperatures.length;j++,k++)
			{
				if(temperatures[j]>temp) {
					merge = true;
					break;
				}else {
					merge = false;
				}
				
			}
			if(merge) {
				day[i] = k;
			}else {
				day[i] = 0;
			}
			
		}
		return day;
		
		
		/********** END **********/
	}
}
