package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
        int n=matrix.length;
		 int temp;
    for(int i=0;i<n/2;i++){
            for(int j=i;j<n-1-i;j++){
                 temp = matrix[i][j];
                 matrix[i][j]=matrix[n-j-1][i];
                 matrix[n-j-1][i]=matrix[n-i-1][n-j-1];
                 matrix[n-i-1][n-j-1]=matrix[j][n-i-1];
                 matrix[j][n-i-1]=temp;
    }
}
	
		/********* End *********/
	}
}
