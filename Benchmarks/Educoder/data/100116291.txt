package step3;

public class HelloWorld {
	public static void main(String[] args) {
		int count= 0;	//定义变量存储6的倍数出现的次数
		/*****start*****/
		int index = 1;
        while(index<=100){
            if(index++ % 6 == 0){
                count ++;
            }
        }
		

		
		/*****end*****/
		System.out.println("6的倍数出现的次数为：" + count);
	}
}
