package step1;

public class Task {
    
    public static void print(Integer a[]){
        for(int i=0;i<a.length;i++){
            System.out.print(a[i]+" ");
        }
        System.out.println();
    }
	public void sort(Integer[] a){
		/********* Begin *********/
        int len=a.length;
        int flag=0;
        for(int i=0;i<len-1;i++){
            for(int j=0;j<len-i-1;j++){
                if(a[j]>a[j+1]) {
                    flag=1;
                    int temp = a[j];
                    a[j] = a[j + 1];
                    a[j + 1] = temp;
                    if(flag==1){
                         print(a);
                     }
                }
            }
            for(int j=len-i-1;j>0;j--){
                if(a[j]<a[j-1]){
                    flag=1;
                    int temp = a[j];
                    a[j] = a[j - 1];
                    a[j - 1] = temp;
                    if(flag==1){
                        print(a);
                    }
                }
            }
        }
        if(flag==0){
            print(a);
        }
		/********* End *********/
	}
}
