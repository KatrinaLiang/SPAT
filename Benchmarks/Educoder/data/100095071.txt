package step3;

public class Factorial {
	public void factorial(int n) {
		/********** Begin *********/
       // int  a = 7;
    	    int sum=1;
    	    if(n==0)
    	    {
    	    	sum =1;
    	    }else {
    	    	 for(int i=1;i<=n;i++)
    	    	 {
    	    		 sum *= i;
    	    	 }
    	    }
        System.out.print(sum);
		/********** End *********/
	}
}
