class Student{
    private String name;
	private int age;
	private String work;
	private  int qian;
	public Student(String name,int age ,String work,int qian){
		this.name=name;
		this.age =age;
		this.work =work;
		this.qian=qian;
	}
	public void jieshao(){
		System.out.println("姓名："+this.name);
		System.out.println("年龄："+this.age);
		System.out.println("职业："+this.work);
		System.out.println("薪资："+this.qian+"K");
	}
}
