package step3;
import java.util.Scanner;
public class Factorial {
	public static void factorial(int n) {
		/********** Begin *********/
          
        int sum=1;
        for(int i=1;i<=n;i++)
        {
        sum=sum*i;
        }
        System.out.print(sum);
    }
    public static void main(String[] args)
        {
            Scanner scanner=new Scanner(System.in);     
        int x=scanner.nextInt();
        factorial(x);
        }

		/********** End *********/
	}
