package step1;

public class LetterChange {

    // 使用 switch把小写字母转为大写字母
    public char LetterChange(char ch){

  switch(ch){
      case 'a':  ch='A';  break;        
      case 'b':   ch='B';      break;                                   
        case  'c':    ch='C';       break;                 
             case 'd' :     ch='D';   break;
               case 'e':     ch='E';   break;
                default :     
                                    }
        return  ch; 
    }
}
