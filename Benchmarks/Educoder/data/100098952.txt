package step1;

public class Yard {
	private double length,width;
    public Yard()
    {
        length=0;
        width=0;
    }
    public Yard(int x,int y)
    {
        length=x;
        width=y;
    }
    public double getPerimeter()
    {
        return 2*length+2*width;
    }
    public double getAcreage()
    {
        return length*width;
    }
    
}
