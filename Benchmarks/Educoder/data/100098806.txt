package step1;

public class Yard {
	private int length;
    private int width;
    
    public Yard(){}
    public Yard(int length, int width){
    	this.length = length;
        this.width = width;
    }
    public double getPerimeter(){
    	double perimeter;
        perimeter = (double)length+(double)width;
        return perimeter*2;
    }
    public double getAcreage(){
    	double acreage;
        acreage = (double)length*(double)width;
        return acreage;
    }
}
