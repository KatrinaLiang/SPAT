public class ThreadHelper {

	public static void calcOnNewThread(int n) {
    	new Thread (new Runnable () {
            public void run() {
            int result = 1;
        	for (int i = 2; i <= n; i++) {
        	result = result * i;	
        }
        System.out.println(result);
        }}).start();
    
    }
}
