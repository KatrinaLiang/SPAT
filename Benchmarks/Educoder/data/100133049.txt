package step1;

public class Task {

        public static void birthdayCandle(int num){
			int a =0;
            int i ,j;
		for ( i = 18;; i++)
	{
		int sum = 0;
		for ( j = i;; j++)
		{
			sum = (i + j)*(j - i + 1) / 2;
			if (sum == num)
			{
				a = 1;//成功
				break;
			}
			if (sum>num)
			{
				break;//开始年龄数加1
			}
		}
		if (a == 1)break;//成功
	}
		System.out.println("小明今年：" + j + "岁");
		System.out.println("小明从" + i + "岁开始每年举办派对");
	}
}
