package step1;

public class Task {
	public static  int[] dailyTemps(Integer[] temp){
		
		
		
		int[] result = new int[temp.length];
		for(int i = 0; i<temp.length;i++) {
		//System.out.println(i);
		int x = 0;
		 for(int j=i+1;j<temp.length;j++) {
				if(temp[i]<temp[j]) {
					x=j-i;
					
					break;
				}
				
			}
		 result[i]=x;
			
			
		}
		
		
		

		return result;
		
	}
}
