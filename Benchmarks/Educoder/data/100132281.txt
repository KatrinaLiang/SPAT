package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
	    /********** BEGIN **********/
    int n = temperatures.length;
    int[] res = new int[n];
    for (int i = 0; i < n-1; i++) {
      for (int j = i; j < n; j++) {
        //找第一个大于t[i]的
        if (temperatures[j]>temperatures[i]){
          res[i]=j-i;
          break;
        }
      }
    }
    return res;
    /********** END **********/
	}
}
