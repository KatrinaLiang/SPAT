package step3;
import step1.Date;
public class Course {
    private int code; //课程编号
    private String courseName; //课程名称
    private Date testTime;//考试日期
    private int score;//考试成绩
   /******Begin**************/
	public Course(){
		testTime=new Date();
	}
	public Course(int code,String cn,Date d,int s)
	{
		this.code=code;
		this.courseName=cn;
		testTime=d;
		score=s;
	}
	public Course(int code,String cn,int y,int m,int d,int s)
	{
		this.code=code;
		this.courseName=cn;
		testTime=new Date(y,m,d);
		score=s;
	}
	public void setCourse(int code,String cn,int y,int m,int d,int s)
	{
		this.code=code;
		courseName=cn;
		testTime.setDate(y,m,d);
		score=s;
	}
	public int getcode()
	{
		return code;
	}
	public String getCourseName()
	{
		return courseName;
	}
	public Date getTestTime()
	{
		return testTime;
	}
	public int getscore()
	{
		return score;
	}
	public String toString()
	{
		if(courseName==null)
			return code+"  "+testTime.getYear()+"."+testTime.getMonth()+"."+testTime.getDay()+" "+score;
		else
			return code+" "+courseName+" "+testTime.getYear()+"."+testTime.getMonth()+"."+testTime.getDay()+" "+score;
	}



   /******End************/

}
