package step3;

public class SeasonJudge {

    // 根据月份判断季节
	public String SeasonJudge(int month) {
        String a=new String();
        if(month>=3&&month<=5) a="春季";
        else if(month>=6&&month<=8) a="夏季";
        else if(month>=9&&month<=11) a="秋季";
        else if(month==12||month==2||month==1) a="冬季";
        else a="月份错误";
        return a;
        // 请在此提供实现代码
	}
}
