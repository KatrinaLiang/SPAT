package step1;


public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum = 0;
        
        for(int i=num1;i<=num2;i++){
            int x = 10;
            int num = i;
            while((num/x)!=0){
            	int lastnum = num%x;
            	
            	num = num/x;
                sum = lastnum + sum;
                
            }
            sum+=num;
        }
		
		return sum;
		/********** END **********/
	}

}
