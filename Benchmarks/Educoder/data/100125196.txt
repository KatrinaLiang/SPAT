package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;

	/********* 请在此编写构造函数及求周长和面积的方法 *********/
	public InterRectangle(double width, double length) {
		this.width = width;
		this.length = length;
	}

	public double perimeter() {
		double PERIMETER = 0.0;
		PERIMETER = 2 * (width + length);
		return PERIMETER;
	}

	public double area() {
		double AREA;
		AREA = width * length;
		return AREA;
	}

	/********* end **********/
	public static void main(String[] args) {
		Scanner scanner = new Scanner(System.in);
		double total = scanner.nextDouble();// 获取土地数目
		/********* begin *********/
		// 请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
		double TOTAL_PERIMETER = 0.0;
		double TOTAL_AREA = 0.0;
		InterRectangle[] land = new InterRectangle[(int) total];
		for (int i = 0; i < total; i++) {
			land[i] = new InterRectangle(scanner.nextDouble(), scanner.nextDouble());
			TOTAL_PERIMETER += land[i].perimeter();
			TOTAL_AREA += land[i].area();
		}
		System.out.println("总共有" + total + "块土地");
		System.out.println("土地总面积为:" + TOTAL_AREA + "平米");
		System.out.println("土地总周长为:" + TOTAL_PERIMETER + "米");
		/********* end *********/
	}

}
