	public void sort(Integer[] arr){
		/********* Begin *********/
		int count = arr.length;
        int flag=0;
		for (int j = 0; j < arr.length; j++) {

			for (int i = j; i < arr.length - 1; i++) {
				if (arr[i] > arr[i + 1]) {
                    flag=1;
					arr[i] = arr[i] ^ arr[i + 1];
					arr[i + 1] = arr[i] ^ arr[i + 1];
					arr[i] = arr[i] ^ arr[i + 1];
                    for(int x:arr) {
						System.out.print(x+" ");
					}
                    System.out.println();
				}
			}
            
            
			for (int i = arr.length - j - 1; i > 0; i--) {
				if (arr[i] < arr[i - 1]) {
                    flag=1;
					arr[i] = arr[i] ^ arr[i - 1];
					arr[i - 1] = arr[i] ^ arr[i - 1];
					arr[i] = arr[i] ^ arr[i - 1];
                    for(int x:arr) {
						System.out.print(x+" ");
					}
                    System.out.println();
				}
			}
            
            
		}
        
        if(flag==0){
        	for(int x:arr) {
				System.out.print(x+" ");
			}
                 System.out.println();
        }

	}
}
