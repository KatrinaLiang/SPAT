package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] a=new int[20];
        int sum=0;
        int i=0;
        a[0]=1;
        a[1]=1;
        for( i=2;i<20;i++){
        	a[i]=a[i-1]+a[i-2];
        }
        for(i=0;i<20;i++){
        	//System.out.print(a[i]+"\t");
            sum+=a[i];
        }
        System.out.print("前20个元素的和为"+sum);
        

	/********end********/
	}
}
