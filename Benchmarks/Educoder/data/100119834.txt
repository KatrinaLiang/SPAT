package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
		for(int i =0;i<10;i++){
        for(int j=0;j<=i;j++){
        System.out.print("*");}
            System.out.print("\n");
        }
	
		
		
		
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
			for(int i =0;i<10;i++){
        for(int j=10;j>i;j--){
        System.out.print("*");}
            System.out.print("\n");
        }
	
		
		
		
		
		/*****end*****/
	}
}
