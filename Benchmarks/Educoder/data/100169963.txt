package step1;
import java.util.Scanner;

public class Cal {
	public static void main(String[] args) {
		/*********start*********/
		Scanner a = new Scanner(System.in);
		System.out.print("请输入第一个整数\n");
		int m = a.nextInt();
		System.out.print("请输入第二个整数\n");
		int n = a.nextInt();
		int plus = m + n;
		int x = m - n;
		int y = m*n;
		int z = m/n;
		int r = m%n;
		System.out.println("两数相加的结果为："+plus);
		System.out.println("两数相减的结果为："+x);
		System.out.println("两数相乘的结果为："+y);
		System.out.println("两数相除的结果为："+z);
		System.out.println("两数取余数的结果为："+r);
	
		
		
		
		/*********end*********/
	}

}
