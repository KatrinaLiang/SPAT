package step1;

public class Task {
public void sortArray(int[] arr){
		int k;
		int temp;
		for(int i =1;i< arr.length;i++) {
			k = i-1;
			temp = arr[i];
			while(k>=0&&arr[k]>temp) {
				arr[k+1] = arr[k];
                
				k--;
			}
			arr[k+1] = temp;
			for(int j = 0 ;j<arr.length;j++) {
			
			System.out.print(arr[j]+" ");
		}
		System.out.println();
		}
		
		
	}
	
}
