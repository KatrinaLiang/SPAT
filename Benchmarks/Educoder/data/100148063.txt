package task3;
import java.util.Scanner;
public class Task3 {
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        int n=sc.nextInt();
        /********* Begin *********/
        double s = 0;
        double a1 = 2;
        double b1 = 1;
        double c1 = 0;
        for (int i = 0; i < n; i++) {
            s += a1 / b1;
            c1 = a1 + b1;
            b1 = a1;
            a1 = c1;
        }
        System.out.println(s);
        /********* End *********/
    }
}
