package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		
int a[][]=new int[matrix[0].length][matrix.length];
int i=0;
int j=matrix.length-1;
int k=matrix[0].length-1;
for(int p=0;p<=j;p++){
	for(int h=0;h<=k;h++){
		a[p][h]=matrix[p][h];
	}
}
for(int p=0;p<=j;p++){
	for(int h=0;h<=k;h++){
		matrix[h][k-p]=a[p][h];
	}
}

		

		
		
		
		/********* End *********/
	}
}
