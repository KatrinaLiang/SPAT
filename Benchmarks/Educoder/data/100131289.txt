package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum;
        sum=num1>num2?num2:num1;
        num2=num1>num2?num1:num2;
        num1=sum;
        sum=0;
        int f;
        for(int i=num1;i<=num2;i++){
        	f=i;
            while(f>0){
            sum+=(f%10);
            f/=10;}
        }
		return sum;
		
		/********** END **********/
	}
}
