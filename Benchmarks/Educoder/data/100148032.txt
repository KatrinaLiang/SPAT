public class MyRunnable implements Runnable {
	private int b;	
    public MyRunnable() {
    	this(0);
    }   
    public MyRunnable(int b) {    	
    	this.b=b;
    }	
	@Override
	public void run() {
    	int result = 1;  	
		for(int x=1;x<=b;x++){
            result=result*x;}
		try{
            Thread.sleep(50);
        }catch (InterruptedException e){
            e.printStackTrace();
        }
    	System.out.println(result);
	}
}
