package step3;
import java.util.*;
public class Reverse {
    public static StringBuffer start(String data) {
        StringBuffer ans = new StringBuffer();
        /********** Begin *********/
		String[] da = data.split(" ");
		for (int i = 0; i < da.length; i++) {
			for(int j=da[i].length()-1;j>=0;j--)
				{
					ans.append(da[i].charAt(j));
					
				}ans.append(" ");
		}
        /********** End *********/
        return ans;
    }
}
