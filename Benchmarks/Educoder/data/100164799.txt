package task2;
import java.util.Scanner;
public class Box {

    double width=0;
    double height=0;
    double depth=0;
    double volume=0;
/************Begin********/
	public Box(){}
	public Box(double width,double height,double depth){
		//YOUR CODE HERE
           this.width=width;
           this.height=height;
           this.depth=depth;
	}
    public double volume(){
            return width*height*depth;
    }

/************End********/       

}
