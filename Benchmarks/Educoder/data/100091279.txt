package step2;

public class TriangleJudge{

    // 判断三条边能否构成一个三角形
    public boolean TriangleJudge(int a, int b, int c) {
        // 请在此添加实现代码
        if(b>a&&b>c)
        {
        	if(a+c>b)
                return true;
            else
                return false;
         }
        if(c>a&&c>b)
        {
        	if(a+b>c)
                return true;
            else
                return false;
        }   
        if(a>b&&a>c)
        {
        	if(b+c>a)
                return true;
           else
                return false;
        }
        return true;
    }
}
