package step1;

public class Yard {
    private int height;
    private int width;
    Yard(int height,int width){
    	this.height=height;
        this.width=width;
    }
	public float getPerimeter(){
    	return (float)2*(height+width);
    }
    public float getAcreage(){
    	return (float)height*width;
    }
}
