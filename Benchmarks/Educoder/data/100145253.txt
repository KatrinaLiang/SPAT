package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int m = n;
        int count = 0;
        while (m != 1) {
            if (m % 2 != 0) {

                m = m - 1;
                count++;
            }
            m = m / 2;
        }
        
        return count+1;
		
		
		
		
		/********* End *********/
	}
	
}
