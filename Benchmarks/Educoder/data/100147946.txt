public class MyThread extends Thread
{
	private int a,sum=1;
    public MyThread(int a)
    {
    	this.a = a;
    }
	public void run()
    {
    	
        for(int i=1;i<=a;i++)
        {
        	sum *= i;
        }
        try {Thread.sleep(50000);} catch (Exception e) {}
        System.out.println(sum);
    }
}
