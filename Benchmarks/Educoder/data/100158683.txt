#include<bits/stdc++.h>
using namespace std;
int main()
{
	int a[100];int j,i,k,m;char b[100];
	int n;
	int sum=0;
    	for(i=0;;i++)
	{
	scanf("%d%c",&a[i],&b[i]);
	if(b[i]=='\n')
	break;
    }
    scanf("%d",&n);
    for(j=0;j<=(i);j++)
    {
    	for(k=j+1;k<=(i);k++)
    	   {
    	    for(m=k+1;m<=i;m++)
    	       {if(a[j]+a[k]+a[m]<n)
			    sum++;
			   }
           }
    }
    printf("%d",sum);
	return 0; 
}
