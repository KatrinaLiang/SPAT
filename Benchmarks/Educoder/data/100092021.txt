package step1;
import java.util.Scanner;
public class IsOfVowels {
	// 判断输入字符是否是元音字母
	public boolean IsOfVowels(char ch){
		// 请在此添加实现代码
        if(ch=='a'||ch=='e'||ch=='o'||ch=='u'||ch=='i'){
            return true;
        }else if(ch=='A'||ch=='E'||ch=='O'||ch=='U'||ch=='I'){
            return true;
        }else{
            return false;
        }
    }
}
