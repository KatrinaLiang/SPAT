package step2;
class AlarmGuardDoor extends Door implements Guard,Alarm{

	@Override
	public void alarm() {
		System.out.println("刮风下雨收衣服了");
		
	}

	@Override
	public void guard() {
		System.out.println("防盗效果好");
		
	}

	@Override
	public void open() {
		System.out.println("芝麻开门");
		
	}

	@Override
	public void close() {
		System.out.println("芝麻关门");
		
	}
}
