package step1;

public class Task {
	public String inversion(String str){
		/********* Begin *********/
		StringBuilder sb = new StringBuilder();
		for (int i = 0; i < str.length(); i++) {
			sb.append(str.charAt(str.length() - 1 - i));
		}
		return sb.toString();
		
		
		/********* End *********/
	}
}
