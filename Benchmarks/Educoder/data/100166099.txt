package step1;

public class Person {
    private String name; //姓名
    private String number; //身份证号
    private Date birthday; //出生日期
    
   /************Begin*************/
   	private int year,month,day;
	public Person(){ this.birthday=new Date();  }
	public Person(String name,String number,Date birthday){ //构造方法
		this.name=name;
		this.number=number;
		this.birthday=birthday;		
	}
	public Person(String name,String number,int year,int month,int day){
		this.name=name;
		this.number=number;
		this.birthday=new Date(year,month,day);
	}

	public void setinfo(String name,String number,Date birthday){
		this.name=name;
		this.number=number;
		this.birthday=birthday;	
	}
	public void setinfo(String name,String number,int year,int month,int day){
		this.name=name;
		this.number=number;
		this.birthday=new Date(year,month,day);	
	}
	public String getnum(String number) { return number; }
    public String getname(String name) { return name; }
    public Date getbirthday(Date birthday) { return birthday; }
    public String toString() { return "name="+name+" number="+number+" birthday:"+birthday; }

    
    
    
    
   /************End*************/
}
