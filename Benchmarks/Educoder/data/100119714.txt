package step2;

public class ForPractice2 {
	public static void main(String[] args) {
        int a,b;
		for(a=0;a<10;a++){
        for(b=0;b<a;b++){
        System.out.print("*");}
        System.out.println("*");}
		System.out.println("——————————我是华丽的分界线——————————");
		for(a=9;a<10&&a>=0;a--){
        for(b=0;b<a&&b>=0;b++){
        System.out.print("*");}
        System.out.println("*");}
	}
}
