package step2;
import step1.Person;
import step1.Date;
public class Student extends Person{
    private int reg_num; //学号
    private Date reg_day; //注册日期
   // TO DO YOU CODE HERE
	public Student(){
        super();
        
        reg_num=0;
        reg_day=new Date();
    }
    public Student(String name,String number,Date d1,int r,Date d2){
        super(name,number,d1);
        reg_num=r;
        reg_day=d2;
    }
    public Student(String name,String number,int a,int b,int c,int r,int m,int n,int q){
        super(name,number,a,b,c);
        reg_day=new Date(m,n,q);
        reg_num=r;
    }
    public Date getreg_day(){
        return reg_day;
    }
    
    public int getreg_num(){
        return reg_num;
    }
    
	public String toString(){
        String na=super.getname();
        String num=super.getnum();
        Date t=super.getBirthday();
        return "name="+na+" number="+num+" birthday:"+t+"reg_num:"+reg_num+"reg_day:"+reg_day;
    }
    
    




}
