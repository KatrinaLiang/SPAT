package step1;

public class A< T extends Comparable >{
 private T t1,t2; 
/*************BEGIN**************/
public A(T t1,T t2){
	 this.t1 = t1;
	 this.t2 = t2;
 }
 
 public void setT1(T t1) {
	 this.t1 = t1;
 }
 public void setT2(T t1) {
	 this.t2 = t2;
 }
 
 public T getT1() {
	 return t1;
 }
 public T getT2() {
	 return t2;
 }
 
 public T compare() {
	 if(this.t1.compareTo(this.t2)>=0){
		 return this.t1;
	 }
	 else {
		 return this.t2;
	 }
 }




/*************END*****************/

}
