class Person {
	/********* begin *********/
     String name;
    int age;
     void talk(String name,int age){
     System.out.print("我是："+name+"，今年："+age);
    }
	/********* end *********/
}

class Student extends Person {
	/********* begin *********/
    String school;
    public Student(String name,int age,String school){
      super.talk(name,age);
        System.out.println("岁，我在"+school+"上学");
    }
	/********* end *********/
}
