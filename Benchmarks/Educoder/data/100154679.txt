package step1;
import java.util.Scanner;

public class TestJava19_2 {
	static final Scanner sc=new Scanner(System.in);
	public void print(){
		int m,n;
		System.out.println("请输入行数：");
		m=sc.nextInt();
		System.out.println("请输入列数：");
		n=sc.nextInt();
		for(int z=1;z<=m;z++){
			for(int d=1;d<=n;d++){
				System.out.print("*");
			}
			System.out.println();
	}

}
}
