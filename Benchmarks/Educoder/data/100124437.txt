package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
    int[] fib = new int[21];
    fib[0]=0;
    fib[1]=1;
    int sum=1;
	for(int i=2;i<21;i++)
    {
        fib[i]=fib[i-1]+fib[i-2];
        sum+=fib[i];
    }
        System.out.print("前20个元素的和为" + sum);
	/********end********/
	}
}
