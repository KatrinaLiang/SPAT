package step3;
public class NarcissisticNumber {

    //判断一个三位数是否是水仙花数
    public boolean NarcissisticNumber(int n){
        // 请在此添加实现代码
        boolean flag;
    int a,b,c;
    a = n / 100;
    b = n / 10 % 10;
    c = n % 10;
    flag = (a * a * a + b * b * b + c * c * c == n) ? true : false;
    return flag;
    }
}
