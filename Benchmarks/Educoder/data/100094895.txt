package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        long sum=0;
        int temp=a;
		for ( int i=1 ; i<n ; i++ )
		{	
			temp =temp*10+a;
			sum = sum+temp;
		}
        sum = sum +a;
        System.out.print(sum);
		/********** End *********/
	}
}
