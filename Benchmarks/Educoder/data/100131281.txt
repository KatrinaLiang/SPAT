package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
         int sum=0;
		for(int i =0; i<num2-num1+1; i++){
			int index = num1+i;
        	String str1 = String.valueOf(index);
            for(int j=0; j<str1.length(); j++){
            	String strnum = ""+str1.charAt(j);
                sum = sum+Integer.parseInt(strnum);
            }
		}
		return sum;
		
		
		/********** END **********/
	}
}
