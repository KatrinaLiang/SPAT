public class Task {

	public int getNum(int n){
		/********* Begin *********/
		
			 int a=0;
    	while(n>0) {
    		if(n%2==0) {
    			n=n/2;
    		}else {
    			n--;
    			a++;
    			n=n/2;
    		}
    		
    	}
    		 
    	return a; 
		
		
		
		/********* End *********/
	}
	
}
