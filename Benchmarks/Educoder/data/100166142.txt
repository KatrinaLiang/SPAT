package step1;

public class Person {
    private String name; //姓名
    private String number; //身份证号
    private Date birthday; //出生日期
    
   /************Begin*************/
   public String toString(){
       return "name="+this.name+" number="+this.number+" birthday:"+birthday;
   }
    
   public Person(){
       //this.name = null;
       //this.number = null;   
        //System.out.println("name="+this.name+" number="+this.number+" birthday:"+"1999.9.9");
       birthday = new Date();
        
   }
       
   public Person(String name, String num,Date birthday){
       this.name = name;
       number = num;
       this.birthday = birthday;
       // System.out.println("name="+this.name+" number="+this.number+" birthday:"+birthday.toString());
   }
    
    public Person(String name,String num,int year,int month,int day){
        this.name = name;
        number = num;
        birthday = new Date(year,month,day);
        
    }
    
     public String get_name(){ return name;}
    public String get_num(){return number;}
    public Date get_birthday() {   return birthday;  }
    
     public void setinfo(String name,String number,Date birthday){
        this.name=name;
        this.number=number;
        this.birthday=birthday;
    }
    public void setinfo(String name,String number,int year,int month,int day){
        this.name=name;
        this.number=number;
        birthday.setDate(year, month, day);
    }

}
