package step1;

public class Yard {
    int length;
    int width;
    double s,l;
    public Yard(int length,int width){
this.length=length;
        this.width=width;
}
double getPerimeter(){
    l=2*(length+width);
        return l;
    }
double getAcreage(){
    s=length*width;
        return s;
    }
}
