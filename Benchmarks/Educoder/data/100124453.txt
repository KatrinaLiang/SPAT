package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] arr=new int [20];
        arr[0]=1;
        arr[1]=1;
        int b=arr[0]+arr[1];
        for(int a=2;a<arr.length;a++){
            arr[a]=arr[a-1]+arr[a-2];
            b+=arr[a];
        }
        System.out.print("前20个元素的和为"+b);
	/********end********/
	}
}
