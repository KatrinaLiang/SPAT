package task6;

import java.util.*;

public class ScaleBox extends Box{
    private int Type;//1 长方体 2 圆柱体

    public double printVolumn(){
        if (this.Type == 1) {
           return this.volume();
        } else if(this.Type == 2) {
            return 3.14 * this.getWidth() * this.getWidth() * this.getHeight();
        }
        throw new RuntimeException("UNSUPPORTED TYPE");
    }

    public ScaleBox(int boxType) {
        super();
        this.Type = boxType;
    }
    public ScaleBox(double width,double height,double depth,int boxType) {
        super(width, height, depth);
        this.Type = boxType;
    }


    /**********************Begin*******************/
    //TODO write your code here

    public int getType() {
        return Type;
    }

    public void setType(int type) {
        Type = type;
    }


    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        ScaleBox scaleBox = new ScaleBox(scanner.nextInt());
        scaleBox.setWidth(scanner.nextDouble());
        scaleBox.setHeight(scanner.nextDouble());
        scaleBox.setDepth(scanner.nextDouble());
        System.out.println(scaleBox.printVolumn());
    }


    /**********************End*******************/

}
