package step3;

public class PalindromeJudge {
	// 判断一个五位正整数是否是回文数
	public boolean PalindromeJudge(int num){
		// 请在此添加实现代码
    int m,n,m1,n1;
    m=num/10000;
    n=num%10;
    m1=num/1000-m*10;
    n1=(num%100-n)/10;
    if(m==n&&m1==n1)
        return true;
     else 
        return false;
   }
}
