package step1;

public class Task {
	
	public void sort(Integer[] arr){
		/********* Begin *********/
        int len=arr.length;
        int judge=0;
        for (int j = 0; j < len / 2; j++) {
            for (int i = 0; i < len - 1; i++) {
                if (arr[i] > arr[i + 1]) {
                    judge++;
                    int temp;
                    temp = arr[i];
                    arr[i] = arr[i + 1];
                    arr[i + 1] = temp;
                    for (Integer a : arr) {
                        System.out.print(a + " ");
                    }
                    System.out.println();
                }
            }
            for (int i = len - 1; i > 0; i--) {
                if (arr[i] < arr[i - 1]) {
                     judge++;
                    int temp;
                    temp = arr[i];
                    arr[i] = arr[i - 1];
                    arr[i - 1] = temp;
                    for (Integer b : arr) {
                        System.out.print(b + " ");
                    }
                    System.out.println();
                }
            }
        }
        if(judge==0){
             for (Integer b : arr) {
                        System.out.print(b + " ");
                    }
                    System.out.println();
        }
		/********* End *********/
	}
}
