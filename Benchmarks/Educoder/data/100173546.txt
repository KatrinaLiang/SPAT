package step1;

public class ForPractice1 {
	public static void test() {
		/*****start*****/
	 for (int i = 1; i <= 10; i++) {
            System.out.print("*");
            //内层循环 每次输出一个*
            for (int j = 1; j <= 9; j++) {
                System.out.print("*");
            }
            System.out.println();
        }	
		
		
		
		/*****end*****/
	}
}
