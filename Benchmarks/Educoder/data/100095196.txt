package step3;

public class Factorial {
	public void factorial(int n) {
		/********** Begin *********/
        int i,b,sum=1;
        int a=n;
        for(i=0;i<a;i++){
            b=n;
            sum=sum*b;
            n=n-1;
        }
        System.out.print(sum);


		/********** End *********/
	}
}
