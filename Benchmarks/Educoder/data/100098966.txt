package step1;

public class Yard {
    private double length;
    private double width;
    private double perimter;
    private double acreage;
    
    public Yard(double length,double width)
    {
        this.length=length;
        this.width=width;
    }
  
    public double getPerimeter()
    {
      return (length+width)*2;
    }
    public double getAcreage()
    {
      return length*width;
    }
}
