package step3;
import java.util.Scanner;
public class NarcissisticNumber {
    public static void main(String[] args) {
        // 从命令行读入一个三位数
        // 这个整数取自测试集的输入
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        NarcissisticNumber number = new NarcissisticNumber();
        System.out.println(number.NarcissisticNumber(n));
    }
    


    //判断一个三位数是否是水仙花数
    public boolean NarcissisticNumber(int n){
        // 请在此添加实现代码
        int g,s,b;
        g = n%10;
        s = n/10%10;
        b = n/100;
        if( g*g*g+s*s*s+b*b*b != n )
            
            return false ;
        else
            return true;
       
}
}
