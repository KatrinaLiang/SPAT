package step1;
//编写一个Cube类继承Rectangle
public class Cube extends Rectangle{
	/*********begin*********/
  //创建两个构造函数  
  //创建 无参 构造函数  调用父类构造方法默认 宽高 都为10
  public Cube(){
		super(10,10);
	}
	
	public Cube(double d){
		super(d,d);
	}
	
	public double area(){		//求面积		
		return super.area();
	}
	 
	public double perimeter(){	//求周长
		return super.perimeter();
	}
	
   //调用父类中的area方法和perimeter方法
	public double getCube(){	//求tioj
		return super.area()*super.getWidth();
	}

	//编写计算体积的方法

	public static void main(String[] args){
	Cube c=new Cube();//创建两个正方体实例对象，并按相应格式输出正方体的底面周长，底面面积和体积。
    //创建一个 无参 正方体对象
	
	System.out.println("c底面周长："+  c.perimeter()  );
	System.out.println("c底面面积："+ c.area()  );
	System.out.println("c体积："+c.getCube() );

                       
    Cube c1=new Cube(5);//创建一个 有参 正方体对象 参数值为： 5
                       
    System.out.println("c1底面周长："+  c1.perimeter()   );
    System.out.println("c1底面面积："+  c1.area() );
    System.out.println("c1体积："+c1.getCube());                   

	/*********end*********/	

	}
}
