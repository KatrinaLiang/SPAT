package step1;

public class GetScores {

	/********* Begin *********/
    public int getLostScores(int[] nums){
         int l=nums.length;
		 int[] tem = new int[l+4];
		 tem[0]=1;
		 String str="";
		 for(int i=0;i<l;i++){
			 int j1=nums[i];
			 tem[j1]=1;
		 }
		 for(int j=0;j<tem.length;j++){
			 if(tem[j]!=1){
				 str=str+String.valueOf(j);
			 }
		 }
		 Long l2=Long.parseLong(str);
		 Long l3=l2%11;
		 int re=new Long(l3).intValue();
		 return re;
    
	
	}

	/********* End *********/
}
