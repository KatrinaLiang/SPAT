package step1;

public class LetterChange {

    // 使用 switch把小写字母转为大写字母
    public char LetterChange(char ch){
        // 请在此提供实现代码
        
        if((int)ch==97||(int)ch==98||(int)ch==99||(int)ch==100||(int)ch==101)
        {
            int cb = (int)ch-32;
        ch = (char)cb;
        }
        return ch;
    }
}
