package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        int i=2;
        int sum = a;
        while(i<=n){
            int aa=a;
            int t=a;
            for(int j=1;j<i;j++){
                t=t*10;
                aa+=t;
            }
            sum+=aa;
            i++;
        }
        System.out.print(sum);


		/********** End *********/
	}
}
