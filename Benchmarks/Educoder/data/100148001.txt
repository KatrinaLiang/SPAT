public class ThreadHelper{
	static public void calcOnNewThread(int num) {
    	new Thread(new Runnable() {
    	    public void run() {
                try{
            		Thread.sleep(1000);
        		}catch(InterruptedException e){
            		e.printStackTrace();
        		}
    	    	int result = 1;
    			if(num>=1){
					for(int i=1;i<=num;i++)
          	  		result*=i;
        		}
        		else if(num==0)
            		result=1;
    	    	System.out.println(result);
    	    }
    	}).start();
    }
}
