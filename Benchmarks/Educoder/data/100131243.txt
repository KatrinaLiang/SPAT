package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int count =num2-num1;
        int sum=0;
        for(int i=0;i<=count;i++)
        {
            int a=(num1+i)/100;
            int b=(num1+i-a*100)/10;
            int c=(num1+i-a*100-b*10);
            int d=a+b+c;
             sum=sum+d;
        }
       return sum;
            
		
		
		/********** END **********/
	}
}
