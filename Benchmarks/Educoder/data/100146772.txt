package step1;

import java.util.Arrays;

public class GetScores {

	/********* Begin *********/
    public int getLostScores(int[] nums){
       //请在此补全你的代码
        int len = nums.length;
        int[] lose = {0,0,0};
        Arrays.sort(nums);
        int bias = 1;
        int count = 0;
        for(int i = 0;i < len;i++){
            if(nums[i] != i+bias){
                lose[count] = i+bias;
                bias = bias + 1;
                count = count+1;
                if(count >= 3){
                    break;
                }
                i = i-1;
            }
        }
        if(count<3){
            if(count == 0){
                for(int i=0;i<3;i++){
                    lose[i] = len+i+1;
                }
            }else if(count == 1){
                if(nums[len-1] != len+1){
                    lose[1] = len+1;
                    if(nums[len-1] != len+3){
                        lose[2] = len+3;
                    }else {
                        lose[2] = len+2;
                    }
                }else {
                    lose[1] = len+2;
                    lose[2] = len+3;
                }
            }else if (count == 2){
                if(nums[len-1] != len+3){
                    lose[2] = len+3;
                }else if(nums[len-2] != len+2){
                    lose[2] = len+2;
                }else {
                    lose[2] = len+1;
                }
            }
        }
        StringBuffer stringBuffer = new StringBuffer();
        for(int num:lose){
            stringBuffer.append(num);
        }
        long comNum = Long.valueOf(stringBuffer.toString());
        long result = comNum%11;
        return (int)result;
	}

	/********* End *********/
}
