package step3;

public class Task {

	 public int countOfShips(char[][] ferry) {
	  /**************BEGIN************/
         int num = 0;
         for(int i = 0 ; i < ferry.length ; i++){
         	for(int j = 0 ; j < ferry[i].length ; j++){
            	if(ferry[i][j] == '+'){
                	num++;
                    ferry[i][j] = 'o';
                    if(j < ferry[i].length - 1){
                    	for(int row = j+1 ; row < ferry[i].length ; row++){
                        	if(ferry[i][row] == '+'){
                            	ferry[i][row] = 'o';
                            }else{
                            	break;
                            }
                        }
                    }
                    
                    if(i < ferry.length - 1){
                    	for(int line = i+1 ; line < ferry.length ; line++){
                        	if(ferry[line][j] == '+'){
                            	ferry[line][j] = 'o';
                            }else{
                            	break;
                            }
                        }
                    }
                }
            }
         }
         
	 
         return num;
      /***********END ***************/
	 }
}
