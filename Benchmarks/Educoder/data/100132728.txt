package step3;

public class Task {

	 public int countOfShips(char[][] ferry) {
	  /**************BEGIN************/
	 int countOfShips=0;
		for(int i=0;i<ferry.length;i++) {
			for(int j=0;j<ferry[i].length;j++) {
				if(ferry[i][j]!='0') {
					if(ferry[i][j]=='+') {
						countOfShips++;
						if(j+1<ferry[i].length&&ferry[i][j+1]=='+') {
							for(int k=j+1;k<ferry[i].length;k++) {
								if(ferry[i][k]=='o') {
									break;
								}
								else {
									ferry[i][k]='0';
									if(i+1<ferry.length) {
										ferry[i+1][k]='0';
									}
								}
							}
						}
						if(i+1<ferry.length&&ferry[i+1][j]=='+') {
							for(int k=i+1;k<ferry.length;k++) {
								if(ferry[k][j]=='o') {
									break;
								}
								else {
									ferry[k][j]='0';
									if(j+1<ferry[k].length) {
										ferry[k][j+1]='0';
									}
								}
							}
						}
					}
				}
			}
		}
		return countOfShips;
	 
         
      /***********END ***************/
	 }
}
