package step4;

import java.util.Arrays;

public class Task {
    
	/********** BEGIN **********/
    public void sortArray(int[][] arr){
		  int[] sum = new int[100];
        int max;
        int tmp;
        int[] position = new int[100];
        int width = arr.length;
        int n = width;
        for(int i = 0; i < n; i++){
            sum[i] = 0;
            for(int j = 0;j < n;j++){
                sum[i] += arr[i][j];
            }
        }
        for (int i = 0; i < width; ++i)
        {
            position[i] = i;
        }
        for(int i = 0; i < n; i++){
            for (int j = 1; j < n - i; j++)
            { 
                if (sum[j] > sum[j-1]){
                    tmp = sum[j-1];
                    sum[j-1] = sum[j];
                    sum[j] = tmp;
                    tmp = position[j-1];
                    position[j-1] = position[j];
                    position[j] = tmp;
                }
            }
        }
        for(int i = 0; i < n; i++){
            for (int j = 0; j < n ; j++){
                for(int k = 1; k < n;k++){
                    if (arr[i][k] > arr[i][k-1]){
                        tmp = arr[i][k-1];
                        arr[i][k-1] = arr[i][k];
                        arr[i][k] = tmp;
                    }
                }
            }
        }
        for(int i = 0; i < n; i++){
            System.out.print("[");
            int j;
            for (j = 0; j < n-1 ; j++) {
                System.out.print(arr[position[i]][j]);
                System.out.print(", ");
            }
            System.out.print(arr[position[i]][j]);
            System.out.println("]");
        }
    



		
	
    }
    
    /********** END **********/
}
