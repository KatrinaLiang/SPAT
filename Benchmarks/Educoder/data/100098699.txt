package step1;

public class Yard {               
		Yard(double length,double width)
        {   //构造方法 
        		this.length=length;//初始化 边长
            	this.width=width;     //构造方法里面的两个成员
        }  
    public double getPerimeter(){     //成员方法1 返回周长
    			return (length+width)*2;
    }
    public double getAcreage(){       //成员方法2 返回面积
    				return length*width;
    }
    private double length;//为啥private 在后面
    private double width;
}
