package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] arr=new int[20];
        arr[0]=1;
        arr[1]=1;
        int sum=0;
        int i,j;
        for(i=2;i<20;i++){
            
            arr[i]=arr[i-1]+arr[i-2];
        }
		for(j=0;j<20;j++){
            sum=sum+arr[j];
        }
        System.out.print("前20个元素的和为"+sum);


	/********end********/
	}
}
