package step1;
import java.util.Stack;
public class Task {
		public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
		int len=temperatures.length;
		int[] ans=new int[len];
		for(int i=0;i<len;i++) {
			ans[i]=0;
		}
		Stack<Integer> s=new Stack<Integer>();
		for(int i=0;i<len;) {
			if(!s.isEmpty()&&temperatures[i].intValue()>temperatures[s.peek()].intValue()) {
				int id=s.pop();
				ans[id]=i-id;
			}else if(s.isEmpty()||temperatures[i].intValue()<=temperatures[s.peek()].intValue()) {
				s.push(i);
				i++;
			}
		}
		while(!s.isEmpty())s.pop();
		return ans;
		/********** END **********/
	}

}
