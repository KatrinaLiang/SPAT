package step1;

public class Yard {
	int length, width;
	public Yard(int length, int width){
    	this.length = length;
        this.width = width;
    }
	public float getPerimeter(){
    	return (float)(2 * length + 2 * width);
    }
    public float getAcreage(){
    	return (float)(length * width);
    }
}
