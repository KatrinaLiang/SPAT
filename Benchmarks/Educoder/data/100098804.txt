package step1;

public class Yard {
    int length;
    int width;
    int c,s;
    public Yard(int con_length,int con_width){
        length=con_length;
        width=con_width;
    }
       public float getPerimeter() {
		    c=2*(length+width);
           return c;
       }
       public float  getAcreage() {
		   s=length*width;
           return s;
       }
}
