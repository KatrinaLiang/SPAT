public class Task {

	public void sortArray(int[] arr, String sort){
		/********** BEGIN **********/
		int len=arr.length;
		Arrays.sort(arr);
		if(sort.equals("asc")) {
			for(int i=0;i<len;i++) {
				System.out.println(arr[i]);
				}
			}
		else 
			
		{
			for(int i=len-1;i>=0;i--) {
				System.out.println(arr[i]);
				}
			
		}
		


		
		/********** END **********/
	}
}
