package step2;

public class Course {
private int code;
private String courseName;
private Date testTime;
private int score;
public Course(){
    code=0;
    courseName=new String("");
    testTime=new Date();
    score=0;
}
public Course(int code,String courseName,int score,Date testTime){
    this.code=code;
    this.courseName=courseName;
    this.testTime=testTime;
    this.score=score;
}
public void setCourse(int code,String courseName,int score,Date testTime){
    this.code=code;
    this.courseName=courseName;
    this.testTime=testTime;
    this.score=score;
}
public int getCode(){
    return code;
}
public String getCourseName(){
    return courseName;
}
public Date getTestDate(){
    return testTime;
}
public int getScore(){
    return score;
}
public String toString(){
    return code+" "+courseName+" "+" "+testTime+" "+score;
}
}
