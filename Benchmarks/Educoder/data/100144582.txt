public class Task {
	public String inversion(String str){
		/********* Begin *********/		  
            String s = str;
            String temp="";
            for (int i = s.length() - 1; i >= 0; i--) {
                temp = temp+s.charAt(i);
            }
           return temp;
       
		
		
		/********* End *********/
	}
}
