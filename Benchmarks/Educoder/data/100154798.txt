package step3;

public class TestJava19_5 {
	//第一种形式
	public void first() {
        for(int a=1;a<=9;a++){
			for(int b=1;b<=a;b++){
                System.out.print(a*b+"\t");
            }
            System.out.println();
		}System.out.println();
    }

	// 第二种形式
	public void second() {
        for(int a=1;a<=9;a++){
			for(int b=1;b<=a;b++){
                System.out.print(a+"*"+b+"="+(a*b)+"\t");
            }
            System.out.println();
        }
	}

}
