package step2;

import java.util.Scanner;

public class Task {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int length = sc.nextInt();	//长
		int width = sc.nextInt();	//宽
		/********** BEGIN **********/
		int m=length;
        int n=width;
        int t=1;
        while(t!=0)
        {
			t=m%n;
            m=n;
            n=t;
        }
		System.out.println(m);
        System.out.println(length*width/m/m);
		/********** END **********/
	}
}
