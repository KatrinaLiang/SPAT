package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temps){
		/********** BEGIN **********/
		int len=temps.length;
        int[]temps2=new int[len];
        for(int i=0;i<len;i++){
            int k=0;
         for(int j=i+1;j<len;j++){
             if(temps[j]>temps[i]){
             	k=j-i;
                 temps2[i]=k;
                 break;
             }else{
             	temps2[i]=0;
             }
         }
		
        }

		return temps2;
		/********** END **********/
	}
}
