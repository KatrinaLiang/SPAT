package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int i,j;
		int m=1;
		for(i=1;i<=9;i++){
			for(j=1;j<=m;j++){
				int s = i*j;
				System.out.print(j+"*"+i+"="+s+"\t");
			}
			System.out.print("\n");
		    m++;
		}
		
		
		
		
		
		/*****end*****/
	}
}
