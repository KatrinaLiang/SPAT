package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
      	int []Fibonacci=new int[100];
        int num=0;
		Fibonacci[0]=1;
		Fibonacci[1]=1;
		for(int i=2;i<100;i++){
			Fibonacci[i]=Fibonacci[i-1]+Fibonacci[i-2];
			
		}
		for(int i=0;i<20;i++){
			num=num+Fibonacci[i];
		}
        System.out.print("前20个元素的和为"+num);
	/********end********/
	}
}
