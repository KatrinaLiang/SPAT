package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
	public InterRectangle(double length,double width){
        this.width=width;
        this.length=length;
    }
    public double  perimeter(){
		return 2*(length+width);		
	}
	public double area(){
		return length*width;
	}
	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
        InterRectangle rect;
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
         double sumL=0,sumA=0,k=total;
        while(total>0){
        	total--;
        	double length=scanner.nextDouble();
        	double width=scanner.nextDouble();
        	rect=new InterRectangle(length,width);
        	sumL+=rect.perimeter();
        	sumA+=rect.area();        	
        }
        System.out.println("总共有"+k+"块土地");
        System.out.println("土地总面积为:"+sumA+"平米");
        System.out.println("土地总周长为:"+sumL+"米");
        }

		/*********end*********/
	}
