package step2;
public class SquareRoot{

    // 求一个数的算术平方根
    public double SquareRoot(float m){
        // 请在此添加实现代码
         if ( m == 0 ) {
		   return 0;
	   }
		
	   float i = 0;
	   float x1, x2 = 0;
	   while ( ( i * i ) <= m ) {
	          i += 0.01;
	   }
	   x1 = i;
	   for ( int j = 0; j < 10; j++) {
	      x2 = m;
	      x2 /= x1;
	      x2 += x1;
	      x2 /= 2;
	      x1 = x2;
	   }
	   return x2;
    }
}
