package step1;

public class Task {
    private int num1;
    private int num2;
    static private int c=0;
    Task(){
          
    }
    
	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		for(int i=num1;i<=num2;i++){
            int a=i/100;
            int b=(i-a*100)/10;
            int d=i-a*100-b*10;
            c=a+b+c+d;
   
        }
        return c;
		
		
		/********** END **********/
	}
}
