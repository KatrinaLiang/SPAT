package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int num=0;
		for(int i=num1;i<=num2;i++){
			String str=String.valueOf(i);
			for(int j=0;j<str.length();j++)
			{
				int a =str.charAt(j) -'0';
				num+=a;
			}
		}
		return num;
		
		
		/********** END **********/
	}
}
