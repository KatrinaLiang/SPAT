package step1;

public class Yard {
		private int length;
	private int width;
	public Yard(int length, int width) {
		super();
		this.length = length;
		this.width = width;
	}
	
	public double getPerimeter() {
		double c = (length+width)*2;
		return c;
	}
	
	public double getAcreage() {
		double s = length*width;
		return s;
	}
}
