//定义站台类，实现卖票的功能。
public class Station  extends Thread   {	 
		static int a=20;
    public synchronized void change()
    {
         System.out.println("卖出了第"+a+"张票");
        a--;
         if(a==0){
            System.out.println("票卖完了");
    }
    }
    public void run()
    {
        
        while (a>0){
        change();
        }
    }
}
