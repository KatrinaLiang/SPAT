package step3;
import java.io.File;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;
import org.jsoup.Jsoup;
import org.jsoup.nodes.Document;
import org.jsoup.nodes.Element;
import org.jsoup.select.Elements;
public class Task {
    
    //通过filePath文件路径获取Docment对象
	public Document getDoc(String filePath) throws IOException{
		/**********   Begin   **********/
File f=new File(filePath);
Document doc=Jsoup.parse(f,"UTF-8","http://hotels.ctrip.com/");

        
		return doc;
		/**********   End   **********/
	}

	//获取所有链接
	public List<String> getLinks(Document doc){
		/**********   Begin   **********/
		Elements e=doc.select("a[href]");
		List<String> list=new ArrayList<String>();
		for(Element element : e){
			String wb=element.text();
			String lianjie=element.attr("abs:href");
			String bqm=element.tagName();
			String shuchu=bqm+"$"+lianjie+"("+wb+")";
			list.add(shuchu);//
		}

		return list;
		/**********   End   **********/
	}
	
	//获取图片
	public List<String> getMedia(Document doc){
		/**********   Begin   **********/
		List<String> arrays=new ArrayList<String>();
		Elements tagimg=doc.select("img");
		for(Element e :tagimg){
			String bqm=e.tagName();
			String lianjie=e.attr("abs:src");
			String shuchu=bqm+"$"+lianjie;
			arrays.add(shuchu);
			}
		return arrays;
		/**********   End   **********/
	}
	
	//获取link[href]链接
	public List<String> getImports(Document doc){
		/**********   Begin   **********/
		Elements links=doc.select("link");
		List<String> list=new ArrayList<String>();
		for (Element element : links){
			String lianjie=element.attr("rel");
			String bqm=element.tagName();
			String href=element.attr("abs:href");
			String shuchu=bqm+"$"+href+"("+lianjie+")";
			list.add(shuchu);
		}

        return list;
		/**********   End   **********/
	}
	
}
