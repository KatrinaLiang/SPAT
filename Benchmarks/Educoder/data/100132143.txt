package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		int len = temperatures.length;
        int[] number = new int[len];
		for (int i = 0; i < len-1;i++){
        	int j = i +1;
            while(j<len && temperatures[j]-temperatures[i]<=0){
            	j++;
            }
            if(j==len){
            	number[i] = 0;
            }
            else{
            	number[i] = j - i;
            }    
        }
		return number;
	}
}
