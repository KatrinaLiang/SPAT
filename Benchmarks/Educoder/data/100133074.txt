package step1;

public class Task {

	public static void birthdayCandle(int num){
        int x;
        int p;
        int q;
        for(x=19;;x++)
        {
        	int sum=0;
            q=0;
            for(p=x;;p++)
            {
            	sum+=p;
                if(sum>=num) break;
                else q++; 
            }
            if(sum==num) break;
        }
		System.out.println("小明今年：" +  p + "岁");
		System.out.println("小明从" +  x  + "岁开始每年举办派对");
	}
}
