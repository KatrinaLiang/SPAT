package step1;

import java.util.Scanner;

public class Yard {
	private int length;
	private int width;

	//构造函数
	public Yard(int length, int width) {
		this.length = length;
		this.width = width;
	}
	
	//定义周长函数
	public double getPerimeter(){
		double l = (length + width)*2;
		return l;
	}
	
	//定义面积函数
	public double getAcreage(){
		double s = length*width;
		return s;
	}
}
