package step1;

public class Person {
    private String name; //姓名
    private String number; //身份证号
    private Date birthday; //出生日期
    
   /************Begin*************/
   //构造方法
   public Person(){
	   this.birthday = new Date();
   }
   public Person (String name,String num,Date birthday) {
	   this.name = name;
	   this.number = num;
	   this.birthday = birthday;
   }
   public Person (String name, String num,int year,int month,int day) {
	   this.name = name;
	   this.number = num;
	   this.birthday = new Date(year,month,day);
   }

   //set and get
   public void setinfo (String name, String num,int year,int month,int day) {
	   this.name = name;
	   this.number = num;
	   this.birthday = new Date(year,month,day);
   }

   public void setinfo (String name, String num,Date d) {
	   this.name = name;
	   this.number = num;
	   this.birthday = d;
   }

   public String getnum(){
	   return this.number;
   }

   public Date getBirthday(){
	   return this.birthday;
   }

   //toString
   public String toString(){
	   return "name="+this.name+" "+"number="+this.number+" "+"birthday:"+this.birthday;
   }   
    
   /************End*************/
}
