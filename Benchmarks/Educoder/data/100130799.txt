package step5;

import java.util.Scanner;


public class Task {
	
	public int getDepth(TreeNode tree){
		/********** BEGIN **********/

		int i = 0;
		int j = 0;
		if(tree.left!=null) {
			i = getDepth(tree.left);
		}else if(tree.right!=null){
			j = getDepth(tree.right);
		}
		if(i>j) {
			return i+1;
		}else {
			return j+1;
		}
		
		

        
		/********** END **********/
		
	}
	
}
