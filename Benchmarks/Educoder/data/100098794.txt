package step1;

public class Yard {
    public int length;
    public int width;


public void Yard(){
  this.length=5;
  this.width=6;
}

public Yard(int w,int l) {
	 this.length=w;
	 this.width=l;
}
public double getPerimeter(){
    return 2*(this.length+this.width);
}
public double getAcreage(){
    return this.length*this.width;
}
    
}
