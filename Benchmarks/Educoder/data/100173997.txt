public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		 for (int y=1; y<=9; y++) {
            for (int x=1; x<=y; x++) {
                System.out.print(x+"*"+y+"="+x*y+"\t");
            }
            System.out.println();
		 }
		
		
		
		
		
		/*****end*****/
	}
}
