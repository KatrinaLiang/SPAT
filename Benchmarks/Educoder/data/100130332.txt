package step1;

public class Task {

	public void sortArray(int[] arr, String sort){
		/********** BEGIN **********/
		if(sort.equals("asc")){
            for(int i=0;i<arr.length-1;i++){
                int min=i;
                for(int j=i+1;j<arr.length;j++){
                    if(arr[j]<arr[min]){
                        min=j;
                    }
                }
                if(min!=i){
                    int temp=arr[min];
                    arr[min]=arr[i];
                    arr[i]=temp;
                }
            }
        }else if(sort.equals("desc")){
            for(int i=0;i<arr.length-1;i++){
                int max=i;
                for(int j=i+1;j<arr.length;j++){
                    if(arr[j]>arr[max]){
                        max=j;
                    }
                }
                if(max!=i){
                    int temp=arr[max];
                    arr[max]=arr[i];
                    arr[i]=temp;
                }
            }
        }
        for(int i=0;i<arr.length;i++){
            System.out.println(arr[i]);
        }
		/********** END **********/
	}
}
