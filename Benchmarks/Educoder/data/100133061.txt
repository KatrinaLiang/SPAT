package step1;

public class Task {

	public static void birthdayCandle(int num){
        int mark1=0,mark2=0;
		for(int i=18;;i++){
            int flag=0;
            for(int j=18;(i+j)*(j-i+1)<=2*num;j++){
                if((i+j)*(j-i+1)==2*num){
                    mark1=i;
                    mark2=j;
                    flag=1;
                    break;
                }
            }
            if(flag==1) break;
        }
	
		System.out.println("小明今年：" +  mark2  + "岁");
		System.out.println("小明从" + mark1   + "岁开始每年举办派对");
	}
}
