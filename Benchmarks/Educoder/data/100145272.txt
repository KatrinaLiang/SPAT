package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int counter=0;
		while(n/2!=0) {
		if(n%2==0) {
			n=n/2;
		}
		if(n%2!=0) {
			n--;
			counter++;
			n=n/2;
		}
		}
		if(n==1) {
			counter++;
		}
		//System.out.println(counter);
		return counter;
		
		
		
		
		/********* End *********/
	}
	
}
