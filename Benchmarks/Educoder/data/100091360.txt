package step3;
import java.util.Scanner;
public class NarcissisticNumber {
    public static void main(String [] args){
        Scanner sc =new Scanner(System.in);
        int n=sc.nextInt();
        NarcissisticNumber p=new NarcissisticNumber();
         System.out.println(p.NarcissisticNumber(n));
    }

    //判断一个三位数是否是水仙花数
    public boolean NarcissisticNumber(int n){
        int g,s,b;
        g=n%10;
        s=n/10%10;
        b=n/100;
        if(g*g*g+s*s*s+b*b*b==n)
            return true;
        else
            return false;
        // 请在此添加实现代码
    }
}
