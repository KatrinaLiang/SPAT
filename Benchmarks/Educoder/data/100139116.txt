package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
		MyThread my=new MyThread(num);
        my.run();
		
		
	}
}


class MyThread  extends Thread   {

int num;
    MyThread(int num)
    {
        this.num=num;
    }
	public void run()
    {
        int s1=0,s2=1,s=1,i=1;
        
        while(i<num)
        {
	s=s1+s2;
            s1=s2;
            s2=s;
            i++;
        }
        System.out.println("子线程计算结果为："+s);
    }
}
