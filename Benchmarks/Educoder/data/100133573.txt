class Animal {
	/********* begin *********/
	private String name;
    private int age;
    
    public void setName(String name){
    	this.name = name;
    }
    
    public String getNmae(){
    	return name;
    }
    
    public void setAge(int age){
    	this.age = age;
    }
    
    public int getAge(){
    	return age;
    }
    
    public void talk(){
    	System.out.println(name + age+"岁");
    }
	/********* end *********/

}

class Cat extends Animal {
	// 定义Cat类的voice()和eat()方法
	/********* begin *********/
    public Cat(String name , int age){
    	setName(name);
        setAge(age);
    }
	public void voice(){
        System.out.println(getNmae()+"喵喵叫");
    }
    
    public void eat(){
        System.out.println(getNmae()+"吃鱼");
        talk();
    }
	/********* end *********/
}
