package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int i;
        int k;
		for(i=1;i<=9;i++){
		for(k=1;k<=i;k++){
            System.out.print(k+"*"+i+"="+(k*i)+"\t");
        }
            System.out.println();
        }
		/*****end*****/
	}
}
