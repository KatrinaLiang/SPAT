package step3;
import java.util.Scanner;
public class SeasonJudge {
    public static void main(String[] args) {
	Scanner sc= new Scanner(System.in);
	int a= sc.nextInt();
	SeasonJudge sj = new SeasonJudge();
	System.out.println(sj.SeasonJudge(a));
	}	

    // 根据月份判断季节
	public String SeasonJudge(int month) {
        String st = null;
        switch(month) {
		case 3:
		case 4:
		case 5:
			st="春季";break;
		case 6:
		case 7:
		case 8:
			st="夏季";break;
		case 9:
		case 10:
		case 11:
			st="秋季";break;
		case 12:
		case 1:
		case 2:
			st="冬季";break;
            default:
                st="月份错误";break;
		}
        // 请在此提供实现代码
        return st;
	}
}
