package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum = 0;
        for(int i = num1;i<=num2;i++){
            String s = String.valueOf(i);
            for(int j = 0 ;j<s.length(); j++ ){
                sum+=(int)(s.charAt(j)-'0');
            }
        }
        return sum;
		
		
		/********** END **********/
	}
}
