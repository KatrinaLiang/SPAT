package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		if(num1 <= num2){
			int sum = 0;
			for(int i=num1; i<=num2; i++){				
				sum = sum + i/100+(i%100)/10+i%10;
			}
			return sum;
		}else{
			int sum = 0;
			for(int i=num2; i<=num1; i++){				
				sum = sum + i/100+i%100/10+i%10;
			}
			return sum;
		}		
		/********** END **********/
	}
}
