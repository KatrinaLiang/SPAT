
package step4;

import step1.Date;
import step2.Student;
import step3.Course;

public class Report {
    private Student s;//学生基本信息
    private String major; //所学专业  
    private Course[] selCourse;//所选课程
   /************Begin****************/
    public Report(Student s,String major,Course[] selCourse){
        this.s=s;
		this.major=major;
		this.selCourse=selCourse;
    }
    public Student getStudent(){
        return s;
    }
    public String toString(){
        String temp="";
        for(int i=selCourse.length-1;i>=0;i--){
        	temp=selCourse[i]+"\n"+temp;
		}
        return s.toString()+" major:"+major+"\n"+temp;
    }
    /************End****************/

  
}
