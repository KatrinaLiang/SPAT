class Person {
	/********* begin *********/
  String name;
  int age;
public String talk() {
	return "Person [name=" + name + ", age=" + age + "]";
}
  
	/********* end *********/
}

class Student extends Person {
	/********* begin *********/
  String school;

	public Student(String name,int age,String school) {
		this.name 

=name;
		this.age=age;
		this.school = school;
		
	}

	@Override
	public String talk() {
		return "Person [name=" + name + ", age=" + age + "]"+"Student [school=" + school + "]";
	}
  
	/********* end *********/
}
