package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
        Thread t = new MyThread(num);
        t.start();
        int n1=0;
        int n2=1;
        int n=1;
        for(int i=0;i<num-1;i++){
            n=n1+n2;
            n1=n2;
            n2=n;
        }
        System.out.println("子线程计算结果为："+n);
        try {
            t.join();
        } 
        catch (Exception e) {
            e.printStackTrace();
        }

	}
}


class MyThread extends Thread {
    private int num;
    public MyThread(int num){
        this.num=num;
    }
    public void run(){
    	int n1=0;
        int n2=1;
        int n=1;
    	for(int i=0;i<num-1;i++){
            n=n1+n2;
            n1=n2;
            n2=n;
    	}
		try {
                Thread.sleep(30);
        } 
        catch (InterruptedException e) {
                e.printStackTrace();
        }

    }
}
