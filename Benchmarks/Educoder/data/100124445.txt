package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int arg[] =new int[20];
    int sum=0;
        arg[0]=1;
        arg[1]=1;
    for(int i=2;i<20;i++)
		arg[i]=arg[i-1]+arg[i-2];
    for(int j=0;j<20;j++)
        sum+=arg[j];
     System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
