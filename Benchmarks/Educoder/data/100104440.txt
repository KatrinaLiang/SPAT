package step2;

public class ConverseNumber {
	public void Converse (int n) {
		/********** Begin *********/
   		int sum=0;
        while(n>0){//当余数不为0的时候，继续操作
        int d=n%10;  
        sum=sum*10+d;
       	n=n/10;
        }
          System.out.print(sum);
        
        /********** End *********/
	}
}
