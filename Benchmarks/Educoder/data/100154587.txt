package step2;

import java.util.Scanner;

//输出等腰三角形
public class TestJava20_3 {
	
	static final Scanner sc = new Scanner(System.in);

	public void print() {
		int m;
		System.out.println("请输入等腰三角形的行数：");
		m=sc.nextInt();    //m表示行数
		/********* Start ********/
for(int i=1;i<=m;i++){
            int j=1;
        for(;j<2*m;j++){
        if(j>=i&&j<=(2*m-i)){
            System.out.print("*");
        }
       	else if(j>=(2*m-i)){
        	continue;
        }
        
        else{
        	System.out.print(" ");
        }
        
           }
            System.out.println();
            }
        }

		
        
        
		
		/********* End ********/
	

}
