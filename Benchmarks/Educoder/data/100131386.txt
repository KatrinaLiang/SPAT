package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int y = 0;
        for(int x = num1;x<=num2;x++){
        if(x<=99){
        int y1 = x/10+x%10;
            y=y+y1;
        }else{
        int y1 = x/100+x%100/10+x%100%10;
            y=y+y1;
        }
        
        
        }
		
		return y;
		/********** END **********/
	}
}
