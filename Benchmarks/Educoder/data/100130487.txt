package step2;

import java.util.Scanner;

public class Task {
    
	static double [] a = new double[105];
	
	public static void init() {
		a[0] = 0;
		a[1] = 4;
		
		for(int i=2; i<=100; i++) {
			if(i % 2 == 0) {
				a[i] = a[i-2] + 3.0/i;
			} else {
				a[i] = a[i-2] + 4.0/i;
			}
		}
	}
    
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
        
        init();
        
		System.out.println(a[n]);
    }
}
