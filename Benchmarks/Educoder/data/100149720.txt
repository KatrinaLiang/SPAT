package step1;

import java.sql.*;

public class QueryPass {
	
	// 查询数据代码不用上实验报告
	public static void queryDB() {
		
		/********* Begin *********/
        String url = "jdbc:mysql://localhost:3306/tsgc?useUnicode=true&characterEncoding=utf-8";
		PreparedStatement sta = null;
        Connection con = null;
        String sql = null;
        ResultSet rs = null;
		// 第一步：加载驱动
		try{
        	Class.forName("com.mysql.jdbc.Driver");
        }catch(Exception e){
        	e.printStackTrace();
        }
		

		// 第二步：建立连接, "root"和"123123"是针对MySQL设置了用户名(root)和密码(123123)的情况
		// 127.0.0.1:3306是mysql服务器地址及端口   数据库编码格式设置为utf-8
			
		try{
        	con = DriverManager.getConnection(url,"root","123123");
        }catch(Exception e){
        	e.printStackTrace();
        }

		// 第三步：建立statement对象
		try{
            sql = "select * from employee";
            sta = con.prepareStatement(sql);
        }catch(Exception e){
        	e.printStackTrace();
        }
		

		// 第四步：查询数据
		try{
            
        	rs = sta.executeQuery();
            while(rs.next()){
            	System.out.println(
                "no:" + rs.getInt(1) + 
                "\tname:" + rs.getString("name") + 
                "\tpassword:" + rs.getString("password") + 
                    "\tsex:" + rs.getString("sex") + 
                    "\tsalary" + rs.getFloat("salary")
                );
                
            }
            
        }catch(Exception e){
        	e.printStackTrace();
        }
		

		// 第五步：关闭statement对象和连接对象
		
		try{
        	if(con != null) con.close();
            if(sta != null) sta.close();
        }catch(SQLException e){
        	e.printStackTrace();
        }

		/********* End *********/
	}
}
