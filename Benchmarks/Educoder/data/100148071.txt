package task1;

import java.util.Scanner;

public class Task1 {
	/********* Begin *********/
	public static void main(String[] args){
            Scanner input = new Scanner(System.in);
        int n = input.nextInt();
        int sum = 1;
        for (int i = 1; i <= n; i++) {
            sum += Integer.parseInt("2" + i);
        }
        System.out.println(sum);
    }
	/********* End *********/
}
