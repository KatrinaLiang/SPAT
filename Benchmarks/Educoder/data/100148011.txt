public class MyRunnable implements Runnable {
	private int b;
    public MyRunnable() {
    	this(0);
    }
    public MyRunnable(int b) {
        this.b=b;  	
    }
	public void run() {
        try{Thread.sleep(1000);
           }catch(Exception e){}
    	int result = 1;
        int i;
        for(i=1;i<=b;i++){
            result=result*i;
        }
    	System.out.println(result);
	}
}
