package step2;

public class Task {
    
	public int getNum(int n){
		int a=0;
        while(n!=0){
            int c = n%2;
            a = a + c;
            if (c == 1) {
                n = (n -1)/ 2;
            } else {
                n = n/2;
            }
        }
        return a;
	}
	
}
