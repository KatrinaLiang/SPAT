//定义一个数组，数组一共有5个元素，接收用户输入的5个数，找出最大值和最小值，求总和、平均值并输出。

            if(min>s[i]){
            min=s[i];
            }

        System.out.print(s[i]);

        }
        avg=(sum/5);
         System.out.println("中的最大值是"+max+"，最小值是："+min+"，总分是："+sum+"，平均值是："+avg);
