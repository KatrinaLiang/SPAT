package step3;
import step1.Date;
public class Course {
    private int code; //课程编号
    private String courseName; //课程名称
    private Date testTime;//考试日期
    private int score;//考试成绩
   /******Begin**************/
   //构造方法
   public Course(){
	   this.testTime = new Date();
   }
   public Course(int code,String cName,int y,int m,int d,int score){
	   this.code = code;
	   this.courseName = cName;
	   this.testTime = new Date(y,m,d);
	   this.score = score;
   }
   public Course(int code,String cName,Date d,int score){
	   this.code = code;
	   this.courseName = cName;
	   this.testTime = d;
	   this.score = score;
   }

   //set and get
   public void setCourse(int code,String cName,Date d,int score){
	   this.code = code;
	   this.courseName = cName;
	   this.testTime = d;
	   this.score = score;
   }

   public int getcode(){
	   return this.code;
   }

   public String getCourseName(){
	   return this.courseName;
   }

   public Date getTestTime(){
	   return this.testTime;
   }

   public int getscore(){
	   return this.score;
   }

   public String toString(){
	   if(this.courseName == null) {
           return this.code+"  "+this.testTime+" "+this.score;
	   }
	   return this.code+" "+this.courseName+" "+this.testTime+" "+this.score;
   }

   /******End************/

}
