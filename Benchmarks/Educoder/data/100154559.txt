package step1;

import java.util.Scanner;

public class TestJava20_2 {
	static final Scanner sc = new Scanner(System.in);	
	public void print() {
        System.out.println("请输入倒直角三角形的行数：");
		int n=sc.nextInt();   //n表示行数
		/********* Start ********/
		for(int k=n;k>=1;k--,n--){
            for(int i=1;i<=n;i++){
            System.out.print("*"+"\t");
            }
            System.out.println("");
        }
        
        
		/********* End ********/
	}

}
