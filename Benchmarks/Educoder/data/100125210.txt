package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
	public void InterRectangle(){}
    
    public void InterRectangle(double width, double length){
    	this.width = width;
        this.length = length;
    }
    
    public void setWidth(double width){
    	this.width = width;
    }
    
    public double getWidth(){
    	return this.width;
    }
    
    public void setLength(double length){
    	this.length = length;
    }
    
    public double getLength(){
    	return this.length;
    }
    
    public double perimeter(){
    	return 2 * (this.width + this.length);
    }
    
    public double area(){
    	return this.width * this.length;
    }

	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
       	double perimeters = 0;
        double areas = 0;
		for (int i = 0; i < total; i++) {
            double width = scanner.nextDouble();
            double length = scanner.nextDouble();
        	InterRectangle rectangle = new InterRectangle();
            rectangle.setWidth(width);
            rectangle.setLength(length);
            perimeters += rectangle.perimeter();
            areas += rectangle.area();
        }
        System.out.println("总共有" + total + "块土地");
        System.out.println("土地总面积为:" + areas + "平米");
        System.out.println("土地总周长为:" + perimeters + "米");

		/*********end*********/
	}

}
