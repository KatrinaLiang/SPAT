package task4;

public class Box {

    double width;
    double height;
    double depth;

    int boxType;
		public Box(int k){
           boxType=k; 
        }
    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public double volume() {
        if(boxType==1){
         return width*height* depth;  
        }
        else{
            
         return    width*width*3.14*height;
        }
        
    }

}
