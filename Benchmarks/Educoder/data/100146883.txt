package step1;
import java.util.*;
import java.lang.*;


public class GetScores {

	/********* Begin *********/
    public int getLostScores(int[] nums){
       //请在此补全你的代码
    	int[] a={0,0,0} ;
        int k=1;
        
        Arrays.sort(nums);
       
        for(int i=0;i<nums.length;i++){
          if(nums[i] == i+k+1){
            a[k-1]=i+k;
            k++;
          }
           
        }
        
        
        	if((a[2]>0&&a[2]<10)||(a[2]>99&&a[2]<1000)){
                if((a[1]>0&&a[1]<10)||(a[1]>99&&a[1]<1000)){
                    a[1] = -a[1];
                }
                else{
                	a[1] = -a[1];
                    a[2] = -a[2];
                } 
            }
            else{
                if((a[1]>0&&a[1]<10)||(a[1]>99&&a[1]<1000)){
                    a[2] = -a[2];
                }
            }
        
        int  m=(a[0]+a[1]+a[2]);
        return m%11;

	
	}

	/********* End *********/
}
