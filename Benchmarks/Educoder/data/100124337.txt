package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int f[]=new int[20];
        f[0]=1;
        f[1]=1;
        for(int i=2;i<f.length;i++)
        {
            f[i]=f[i-1]+f[i-2];
        }
        int sum=0;
        for(int j=0;j<f.length;j++)
        {
            sum=sum+f[j];
        }
        System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
