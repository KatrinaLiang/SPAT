public class Task {
	public String inversion(String str){
		/********* Begin *********/
		 char[] arr = str.toCharArray();
	//交换
	for (int i = 0; i < arr.length/2 ; i++) {
	char tmp = arr[i];
	arr[i] = arr[arr.length-i-1];
	arr[arr.length-i - 1] = tmp;
	}
	String result = String.valueOf(arr);	
	return result;
		
		
		/********* End *********/
	}
}
