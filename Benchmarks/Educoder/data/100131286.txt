package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum,a,b;
        sum=0;
        a=b=num1;
        while(a<=num2)
        {
            while(b>0)
            {
                sum+=b%10;
           	    b=b/10;
            }
            a++;
            b=a;
        }
		
		return sum;
		/********** END **********/
	}
}
