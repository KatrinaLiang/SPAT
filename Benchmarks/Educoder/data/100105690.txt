package step1;

public class Yard {
    private float length;
    private float width;
    public Yard(float length,float width){
        this.length=length;
        this.width=width;
    }
    public float getPerimeter(){
        float perimeter=(this.length+this.width)*2;
        return perimeter;
    }
    public float getAcreage(){
        float acreage=this.length*this.width;
        return acreage;
    }
}
