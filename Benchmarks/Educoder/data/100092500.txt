package step3;

public class FindMaxNum {
   public int FindMaxNum(int a, int b, int c){
    int max=0;
    if(a>b){
        if(a>c){
            max=a;
        }
        else{
            max=c;
        }
    }
    else{
        if(b>c){
            max=b;
        }
        else{
            max=c;
        }
    }
    return max;
}
}
