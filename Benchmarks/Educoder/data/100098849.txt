package step1;

public class Yard {
     public Yard(double length,double width){
         this.length=length;
         this.width=width;
     }
     public double getPerimeter(){
         double Perimeter = ( length + width ) * 2;
         return Perimeter;
     }
     public double getAcreage(){
         double Acreage = length * width;
         return Acreage;
     }
    
     private double length;
     private double width;
}
