package step2;

public class SwitchToGrade {

    // 请用switch来将学生成绩转换成对应等级
    public String SwitchToGrade(int score) {
        String xx;
         int score1 = score/10;
         switch(score1){
            case 9:
                xx="等级：A";
                 break;
            case 8:
                xx="等级：B";
                 break;
            case 7:
                xx="等级：B";
                 break;
            case 6:
                xx="等级：C";
                 break;
            case 5:
                xx="等级：D";
                 break;
            case 4:
                xx="等级：D";
                 break;
            case 3:
                xx="等级：D";
                 break;
            case 2:
                xx="等级：D";
                 break;
            case 1:
                xx="等级：D";
                 break;
            default:
                if(score1==100){
                    return"等级：A";
                }
                 return"输入成绩非法";
                    }
        return xx;
    }
}
