package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temps){
		/********** BEGIN **********/
		int length = temps.length;
		int [] tempsArr = new int[length];
		
		for(int i = 0; i < length; ++i) {
			for(int j = i + 1; j < length; ++j) {
				int judge = temps[i].intValue() - temps[j].intValue();
				if(judge < 0) {
					tempsArr[i] = j - i;
					break;
				}
			}
		}
		return tempsArr;
		/********** END **********/
	}
}
