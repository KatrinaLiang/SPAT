package step4;

import java.util.Scanner;

public class ForPractice4 {
	public static void main(String[] args) {
		/*****start*****/
		System.out.println("欢迎使用中国人民银行ATM取款机");
		Scanner sc=new Scanner(System.in);
		int qusum=0;
		while(true){
		
			System.out.println("输入取款金额：");
			int qu=sc.nextInt();	
			if(qu<=1000-qusum){
				qusum+=qu;
				System.out.print("剩余金额："+(1000-qusum)+"，是否继续（'1'：结束，'2'：继续）：");
				int b=sc.nextInt();
				System.out.println();
				if(b==2){
					continue;
				}
				else
					System.out.println("取款结束！");break;
			}
			else{
				System.out.println("目前余额："+(1000-qusum)+"无法满足您的取款需求！");
				continue;
			}
				
		}
		
		
		
		
		/*****end*****/
	}
}
