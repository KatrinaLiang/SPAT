package case4;

public class overridingTest {
	public static void main(String[] args) {
		Stu s = new Stu();
		s.talk();
		
	}
}

class Person {
	String name = "张三";
	int age = 18;
	public void talk() {
		System.out.print("我是："+name+"，今年："+age+"岁");
	}
}

class Stu extends Person {
	String school = "哈佛大学";
	public void talk() {
		super.talk();
		System.out.println("，我在"+school+"上学");
	}
}
