package step2;

import java.io.File;

public class Task {
	
    public void getFiles(File file, int index) {
		for(int i=0; i<index; i++)
			System.out.print("  ");
		if(!file.isDirectory()) {
			//file
			System.out.println("--"+file.getName());
		}else{
			System.out.println("+--"+file.getName());
			File [] filelist = file.listFiles();
			for(File nextFile : filelist ) {
				getFiles(nextFile, index+1);
			}
		}
	}
    
	/********** BEGIN **********/
	public void showDirStructure(File file)	{
		getFiles(file, 0);
	}
	


	/********** END **********/
}
