package step3;
import java.util.*;

public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
		Scanner sc=new Scanner(System.in);
		System.out.println("请输入一个五位数字：");
		int f=sc.nextInt();
		int e=f/10000;
		int a=f/1000%10;
		int b=f/100%10;
		int c=f/10%10;
		int d=f%10;
		int sum=(a+b+c+d+e);
		System.out.println(f+"的各位数字之和是："+sum);
		/******End******/
	}
}
