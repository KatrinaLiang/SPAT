
package step3;
import java.util.Scanner;
public class Sum{
	public static void main(String args[]){
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
        int result = recursionSum(num);
        System.out.println(result);
	}
	
    public static int recursionSum(int mum){
        /*********begin*********/
   		int sum=0;
        int i=1;
        for(i=1;i<=mum;i++)
            {
            sum+=i;
        }
        return sum;


        /*********end*********/
    }  
}
