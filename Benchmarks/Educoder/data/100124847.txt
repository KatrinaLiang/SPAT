package step1;

public class Rectangle{
	private double width;
	private double length;
	
	/*********begin*********/
	
    public Rectangle(){}
    
    public Rectangle(double width, double length){
    	this.width = width;
        this.length = length;
    }
    
    public double perimeter(){
    	return 2 * (this.width + this.length);
    }
    
    public double area(){
    	return this.width * this.length;
    }

    public void setLength(double length){
    	this.length = length;
    }
    
    public double getLength(){
    	return this.length;
    }
    
    public void setWidth(double width){
    	this.width = width;
    }
    
    public double getWidth(){
    	return this.width;
    }
    
	/*********end*********/


	public static void main(String[] args) {
		/*********begin*********/
		Rectangle rectangle1 = new Rectangle();
        rectangle1.setLength(6);
        rectangle1.setWidth(5);
        Rectangle rectangle2 = new Rectangle(20, 10);
        System.out.println("r1的周长：" + rectangle1.perimeter());
		System.out.println("r1的面积：" + rectangle1.area());
        System.out.println("r2的周长：" + rectangle2.perimeter());
        System.out.println("r2的面积：" + rectangle2.area());
		/*********end*********/
	}

}
