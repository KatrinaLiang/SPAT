package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
	public InterRectangle(double width,double length)
    {
        this.length=length;
        this.width=width;
    }
    public double perimeter()
    {
        return 2*(width+length);
    }
      
    public double area()
    {
        return (width*length);
    }
        /********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
        double width,length,sum1=0,sum2=0;
        for(int i=0;i<total;i++)
        {
         width=scanner.nextDouble();
         length=scanner.nextDouble();
		InterRectangle a1=new InterRectangle(width,length);
          sum1=sum1+(a1.perimeter());
          sum2=sum2+(a1.area());
        } 
        System.out.println("总共有"+total+"块土地");
        System.out.println("土地总面积为:"+sum2+"平米");
        System.out.println("土地总周长为:"+sum1+"米");
		/*********end*********/
	}

}
