package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum=0;
        for(int i=num1;i<=num2;i++){
            for(int j=1;j<=i;j*=10){
                sum=sum+i/j%10;
            }
        }
        return sum;
		
		
		/********** END **********/
	}
}
