package step3;

import java.util.Scanner;
public class Number_Judge {
		public static void main(String[] args) {
			Scanner scan1=new Scanner(System.in);
			/******Begin******/
           int i, a;
           i = scan1.nextInt();
           if(i % 2 == 0){
               System.out.print("偶数");
           }
           else if(i % 2 == 1 && i > 1){
               for(a = 2; a < i; a++){
                   if(i % a == 0){
                      break;
                   }
               }
               if(a == i){
                 System.out.print("素数");
               }
               else{
                 System.out.print("奇数");
               }
			/******End******/
		}
     }
}
