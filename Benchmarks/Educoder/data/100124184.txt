package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] arr=new int[20];
		int s=0;
		for(int i=0;i<arr.length;i++){
			if(i==0||i==1)
				arr[i]=1;
			if(i>1){
				arr[i]=arr[i-2]+arr[i-1];
			}
			s=s+arr[i];
		}
		System.out.print("前20个元素的和为"+s);	

	/********end********/
	}
}
