package step1;

public class Task {
	
	public void sortArray(int[] arr){
		int i,j,temp;
		for(i=1;i<arr.length;i++) {
            for(j=0;j<i;j++){
			if(arr[i]<arr[j]) {
				temp=arr[i];
				arr[i]=arr[j];
				arr[j]=temp;
			}
            }
			for(j=0;j<arr.length;j++) {
				System.out.print(arr[j]+" ");
			}
            System.out.println("");
		}
		
		
	}
}
