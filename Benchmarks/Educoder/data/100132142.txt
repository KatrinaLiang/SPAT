package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temps){
		/********** BEGIN **********/
		int result[] = new int[temps.length];
		for(int i=0;i<temps.length-1;i++){
			for(int j=i+1;j<temps.length;j++){
				if(temps[j]>temps[i]){
					result[i] = j-i;
					break;
				}
			}
		}
		
		return result;
		/********** END **********/
	}
}
