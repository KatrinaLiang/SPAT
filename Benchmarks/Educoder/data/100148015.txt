public class MyThread extends Thread {
    private int n;
   
    public MyThread() {
		this(0);
	}
	public MyThread (int n) {
    	this.n = n;
    }
    public void run() {
         int result = 1;
    	for (int i = 2; i <= n; i++) {
        	result = result * i;	
        }
        try {
        Thread.sleep(10000);
        }catch (InterruptedException e) {
        	e.printStackTrace();
        }
        System.out.println(result);
    }
}
