package step1;

public class Task {
	public String inversion(String str){
		/********* Begin *********/
		char source[] =str.toCharArray();
        char target[] = new char[source.length];
        for(int i=0;i<source.length;i++){
            target[i]=source[source.length-1-i];
        }
		return String.valueOf(target);
		/********* End *********/
	}
}
