//编写一个方法，统计1990年至2018年之间闰年的个数
public class TestJava23_5 {

	public static void main(String[] args) {
		int a ,b;
	b=TestJava23_5.NumOfLeap(a=1990);
    System.out.println("1990~2018年之间闰年的个数是："+b);
		
	}
	public static int NumOfLeap (int a){
	  int b=0;
	  for(a=1990;a<=2018;a++){
		  if(a%4==0&&100!=0||a%400==0){
			   b++;
		  }
	  }
	  return b;
  }
}
