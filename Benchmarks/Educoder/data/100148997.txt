package step3;

public class Task {
	
	public int concaveValley(Integer[] arr){
		/********* Begin *********/
		 int l=arr.length;
    int d=0;
    int i=0;
    while(i<l-1){
        int count=0;
        while(i<l-1&&(arr[i]<arr[i+1]))
        {
            i++;
        }
        //非降序
        int mh1=i;
        while(i<l-1&&(arr[i]>arr[i+1]))
        {
            i++;
           
        }
        int ml=i;
        while((i<l-1)&&(arr[i]<arr[i+1]))
        {
            i++;
        }
        int mh2=i;
        if((i<l-1)&&(arr[i]==arr[i+1])&&(ml==mh2)){
            count=0;
            i++;
        }
        else{
            count=mh2-mh1+1;
        }
        if(d<count){
            d=count;
        }
        
    }
    return d;
		
		/********* End *********/	
	}
}
