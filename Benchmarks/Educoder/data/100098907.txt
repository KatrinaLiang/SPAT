package step1;

public class Yard {
    public Yard(int length,int width){
        this.length=length;
        this.width=width;
    }
    private int length;
    private int width;
    public double getPerimeter(){
        double c=this.length+this.width;
        return (2*c);
    }
    public double getAcreage(){
        double d=this.length*this.width;
        return d;
    }
        

}
