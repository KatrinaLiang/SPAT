package step1;

public class Yard {
    
    double length;
    double wide;
    
    public Yard(double length, double wide){
         
          this.length= length;
          this.wide= wide;
    }
    public double getPerimeter(){
      return (this.length+this.wide)*2;
    }
    public double getAcreage(){
      return this.length*this.wide;
    }

}
