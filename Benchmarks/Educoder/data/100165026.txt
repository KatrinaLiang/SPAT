package task5;

public class Box {

    double width;
    double height;
    double depth;

    //TODO your code
    public Box(double width, double height, double depth){
    	this.width = width;
        this.height = height;
        this.depth = depth;
    }
    
    public double volume(){
    	double vol;
        vol = width * height * depth;
        return vol;
    }
	public static void main(String[] args){
    	Box box = new Box(10,20,30);
        
        System.out.println("the volume is:" + box.volume());
    }


}
