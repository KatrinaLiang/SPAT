package step2;
import step1.Person;
import step1.Date;
public class Student extends Person{
    private int reg_num; //学号
    private Date reg_day; //注册日期
   // TO DO YOU CODE HERE
	public Student(){
        super();
        reg_num=0;
        reg_day=new Date();
    }
    public Student(String name,String number,Date birthday,int reg_num,Date d2){
        super(name,number,birthday);
        this.reg_num=reg_num;
        reg_day=d2;
    }
    public Student(String name,String number,int year,int month,int day,int reg_num,int year1,int month1,int day1){
        super(name,number,year,month,day);
        this.reg_num=reg_num;
        reg_day=new Date(year1,month1,day1);
    }
    public String toString(){
        String na=super.getName();
        String num=super.getNumber();
        Date t=super.getBirthday();
        
        return "name="+na+" number="+num+" birthday:"+t+"reg_num:"+reg_num+"reg_day:"+reg_day;
    }
   
    public int getreg_num(){
        return reg_num;
    }
    public Date getreg_day(){
        return reg_day;
    }
    public void setDate(int year,int month,int day){
        reg_day=new Date(year,month,day);
    }
        

}
