//定义站台类，实现卖票的功能。
public class Station extends Thread{	 
	public Station(){
       	super();
    }
    static int tick=20;
    static Object ob="aa";
    public void run(){
    	while(tick>0){
        	synchronized(ob){
            	if(tick>0){
                	System.out.println("卖出了第"+tick+"张票");
                    tick--;
                }else{
                	System.out.println("票卖完了");
                }
            }
            try{
            	sleep(1000);
            }catch(InterruptedException e){
            	e.printStackTrace();
            }
        }
    }
	
}
/********* End *********/
