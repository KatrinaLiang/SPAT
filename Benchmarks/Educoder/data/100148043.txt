public class MyRunnable implements Runnable{
    
    int num;
    
    public MyRunnable(int num){
        
        this.num = num;
    }
    
    public MyRunnable(){
        
        this(0);
    }
    
    public void run(){
        
        int result = 1;
            
        for(int i = num ; i > 0 ; i--){

            result *= i;
        }
        
        try {
            Thread.sleep(200l);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
            
        System.out.println(result);
    }
}
