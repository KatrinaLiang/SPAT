package task2;

public class Box {

    double width;
    double height;
    double depth;

    public double volume() {
        return width*height*depth;
}
public static void main(String arg[]){
    Box box = new Box();
    box.width = 10;
    box.height = 20;
    box.depth = 10;
    double volume = box.volume();
    System.out.println("the volumn is:"+box.volume());
}
}
