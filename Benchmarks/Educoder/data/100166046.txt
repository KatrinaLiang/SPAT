package step1;

public class Person{
    private String name; //姓名
    private String number; //身份证号
    private Date birthday; //出生日期
    
   /************Begin*************/
	public Person(){
		birthday=new Date();
	}
	public Person(String name,String num,Date birthday){
		this.name=name;
		this.number=num;
		this.birthday=birthday;
	}
	public Person(String name,String num,int year,int month,int day){
		this.name=name;
		this.number=num;
		birthday=new Date(year,month,day);
	}
	public void setinfo(String name,String number,Date birthday){
		this.name=name;
		this.number=number;
		this.birthday=birthday;
	}
	public String getnum(){
		return this.number;
	}
    public String getname(){
		return this.name;
	}
    public Date getBirthday(){
		return this.birthday;
	}
    public String toString(){
		return "name="+this.getname()+" number="+this.getnum()+" birthday:"+this.getBirthday().getYear()+"."+this.getBirthday().getMonth()+"."+this.getBirthday().getDay();
	}
	
	
    
   /************End*************/
}
