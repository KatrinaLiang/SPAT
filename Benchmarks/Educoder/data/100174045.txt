package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int sum=0;
		for(int a=1;a<=9;a++){
			for(int i=1;i<=a;i++){
				sum=i*a;
				System.out.print(i+"*"+a+"="+sum+"\t");}
			System.out.println();
		}
		
		
		
		
		
		/*****end*****/
	}
}
