package step1;
import java.util.Scanner;
public class Yard {
    int length;
    int width;
    public Yard(int length,int width)
    {
    this.length=length;
    this.width=width;
    }
    public  double getPerimeter()
    {
     return (double)2*length+2*width/1.0;
    }
    public  double getAcreage()
    {
    return (double)length*width/1.0;
    }
}
