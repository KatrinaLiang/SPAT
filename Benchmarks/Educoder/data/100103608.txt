package step3;

import java.io.File;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

public class DeleteFiles {
    public static void start() throws IOException {
        StringBuilder data = new StringBuilder("");
        /********** Begin *********/
        //读取readme.txt文件
		File file1 = new File("src/step3/readme.txt");
        FileReader file1Reader = new FileReader(file1);

        File file2 = new File("src/step3/readmeCopy.txt");
        FileWriter file2Writer = new FileWriter(file2, false);
        
        if(!file2.exists()){
           file2.createNewFile();
        }
        


        //写入readmeCopy.txt文件
		int read = file1Reader.read();
        while(read != -1){
            data.append((char)read);
            read = file1Reader.read();
        }
        file2Writer.write(data.toString());


        //删除readme.txt文件
		file1.delete();
        file1Reader.close();
        file2Writer.close();

        /********** End *********/
    }
}
