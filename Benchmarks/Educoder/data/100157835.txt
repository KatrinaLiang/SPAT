package step;
public class Sort {
	/**
	 *@param arr 待排序数组
	 */
	public void BubbleSort(int[] arr){
		/********* Begin *********/
int n=arr.length;
        int i, j;

    for(i=0; i<n-1; i++){
        for(j=1; j<n-i; j++){
            if(arr[j-1] > arr[j]){
                int temp;
                temp = arr[j-1];
                arr[j-1] = arr[j];
                arr[j]=temp;
            }
        }
        for(int num:arr){
        System.out.print(num+" ");
    }
			System.out.println();
    }
		/********* End *********/
	
}
}
