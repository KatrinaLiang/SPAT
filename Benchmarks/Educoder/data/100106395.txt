package Step3;

public class MultiplyTable {
	public static void main(String[] args)
	{
		for(int i = 1;i<10;i++)
		{
			for (int j = 1;j<=i ;j++ )
			{
				if(j == i){
                    System.out.print(" "+i+"×"+j+"="+j*i);
                    System.out.println();
                }
                else System.out.print(" "+i+"×"+j+"="+j*i);
			}
		}
	}

}
