package step1;

public class Yard {
    double length;
    double width;
    public Yard(double length,double width){
	this.length=length;
    this.width=width;
    }
    public double getPerimeter(){
    	return this.length*2+this.width*2;
    }
    public double getAcreage()
    {
    	return this.length*this.width;
    }

}
