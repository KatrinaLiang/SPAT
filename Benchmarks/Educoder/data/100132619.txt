package step2;

import java.util.Scanner;

public class Task {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int length = sc.nextInt();	//长
		int width = sc.nextInt();	//宽
		/********** BEGIN **********/
		int a=length;
		int b=width;
		while(b!=0){
			int r = a%b;
			a = b;
			b = r;
		}
		int j=a;
		int k=(length/j)*(width/j);
		System.out.println(j);
        System.out.println(k);
		
		
		
		
		
		
		/********** END **********/
	}
}
