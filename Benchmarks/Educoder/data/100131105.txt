package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
			        int sum=0;
				for (int i=num1;i<=num2;i++)
					{	
					int key=i;
					while(key>9)
						{
						sum=sum+key%10;
						key=key/10;
						}
					sum=sum+key;
					}	
				
					//System.out.println(sum);
        return sum;
		/********** END **********/
	}
}
