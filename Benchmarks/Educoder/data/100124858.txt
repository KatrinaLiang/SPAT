package step1;

public class Rectangle{
	private double width;
	private double length;
	
	/*********begin*********/
    String name;
    public Rectangle()
    {
        width = 6.0;
        length = 5.0;
    }
    public Rectangle(double width,double length)
    {
        this.width = width;
        this.length = length;
        
    }
    public void perimeter()
    {
        System.out.println(name+"的周长："+(width*2+length*2));
    }
    public void area()
    {
        System.out.println(name+"的面积："+width*length);
    }
	
	/*********end*********/


	public static void main(String[] args) {
		/*********begin*********/
        Rectangle r1 = new Rectangle();
        r1.name = "r1";
        r1.perimeter();
        r1.area();
        Rectangle r2 = new Rectangle(20.0,10.0);
        r2.name = "r2";
		r2.perimeter();
        r2.area();

		/*********end*********/
	}

}
