package step1;

public class Yard {
	Yard(int length,int width){
		this.length=length;
		this.width=width;
	}
	public double getPerimeter(){
		double perimeter=2*length+2*width;
		return perimeter;
	}
	public double getAcreage(){
		double acreage=width*length;
		return acreage;
	}
	private int length;
	private int width;
}
