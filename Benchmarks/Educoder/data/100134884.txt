class Person {
	/********* begin *********/
    
    public String name;
    public int age;

    public Person(String name,int age)
    {
        this.name = name;
        this.age = age;
    }
    
    public String getName()
    {
        return name;
    }
    
    public int getAge()
    {
        return age;
    }
    
	/********* end *********/
}

class Student extends Person {
	/********* begin *********/
    
    public Student()
    {
        super("张三",18);
    }
    
    public String school;
    
    public void setSchool(String school)
    {
        this.school = school;
    }
    
    public String getScool()
    {
        return school;
    }
    
	/********* end *********/
}
