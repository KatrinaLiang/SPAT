package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int a=0;
        while(n!=0){
        	if(n%2==0){
            	n/=2;
            }
            else{
            	a=a+1;
                n/=2;
            }
            
        }
        return a;
		
		
		
		
		/********* End *********/
	}
	
}
