package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] f=new int [20];
		int i,sum=0;
        f[0]=1;
        f[1]=1;
        for(i=2;i<f.length;i++){
            f[i]=f[i-1]+f[i-2];
        }
        sum=f[0]+f[1];
        for(i=2;i<f.length;i++){
        	sum=sum+f[i];
        }
        System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
