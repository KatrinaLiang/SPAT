package step1;

public class Rectangle{
	private double width;
	private double length;
	
	/*********begin*********/
	public Rectangle(){
        this.width = 5.0;
        this.length = 6.0;
         System.out.println("r1的周长："+ 2*(width+length));
    System.out.println("r1的面积："+ width*length);
    }
    //有参数的构造函数
    public void Rectangle(double width,double length){
        this.width = width;
        this.length = length;
         System.out.println("r2的周长："+ 2*(width+length));
    System.out.println("r2的面积："+ width*length);
    }

	/*********end*********/


	public static void main(String[] args) {
		/*********begin*********/
		Rectangle car = new Rectangle();
        car.Rectangle(10.0,20.0);
        
		/*********end*********/
	}

}
