package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] arr=new int[100];
		arr[0]=1;
		arr[1]=1;
		for(int i=2;i<20;i++)
			arr[i]=arr[i-1]+arr[i-2];
		int ans=0;
		for(int i=0;i<20;i++) ans+=arr[i];
		System.out.print("前20个元素的和为"+ans);

	/********end********/
	}
}
