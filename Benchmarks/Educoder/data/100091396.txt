package step3;
public class NarcissisticNumber {

    //判断一个三位数是否是水仙花数
    public boolean NarcissisticNumber(int n){
        // 请在此添加实现代码
        int p = n;
        int k = 0;
        for(int i = 0;i < 3;i++){
        	int l = n%10;
            k = l*l*l+k;
            n = n/10;
        }
        if(k == p){
        	return true;
        }
        else{
            return false;
        }
    }
}
