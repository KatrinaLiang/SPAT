#include <stdio.h>
#include <string.h>
#include <math.h>
main()
{
	int m[100][3],a,b,d,e=0,f,g,h=1,i,j,z,n;
	char x[100],y[100];
	scanf("%s",x);
	scanf("%s",y);
	a=strlen(x);
	b=strlen(y);
	z=0;
	for(i=0;i<a;i++)
	{
		for(j=0;j<b;j++)
		{
			if(x[i]==y[j])
			{
				if(x[i+1]==y[j+1])
				{
					m[z][1]=i;
					m[z][2]=j;
					z=z+1;
				}
			}
		}
	}
	d=0;
	for(n=0;n<z;n++)
	{
		f=m[n][1];
		g=m[n][2];
		for(i=0;i<100;i++)
		{	
	//	printf("%d %d %d %d %d\n",f,g,i,a,b);
			if(x[f+i]==y[g+i])
			{
				d=d+1;
			//	printf("%c",x[f+i]);
			}
			else
			{
			//	printf("%d",d);
				if(e>d)
				d=0;
				else
				{
					e=d;
					d=0;
				}
			//	printf("\n");
				break;
			}
			if(f+i==a-1||g+i==b-1)
			{
				if(e>d)
				d=0;
				else
				{
					e=d;
					d=0;
				}
				break;
			}
		}
	}
	printf("%d",e);
}
