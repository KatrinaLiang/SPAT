package step3;

import java.util.Arrays;

public class Task {

	public void matrixSum(int[][] arr){
		/********** BEGIN **********/
        int sum=0;
		for(int i=0,j=0;i<arr.length&&j<arr[i].length;j++,i++)
        {
            sum+=arr[i][j];
        }
           
		System.out.println(sum);



		/********** END **********/
	}
	
	
}
