package step3;
public class NarcissisticNumber {

    //判断一个三位数是否是水仙花数
    public boolean NarcissisticNumber(int n){
        // 请在此添加实现代码
        int a = n / 100;//百位
		int b = n / 10 % 10;//十位
		int c = n % 10;//个位
        if((a*a*a) + (b*b*b) + (c*c*c) == n) 
			{
				return true;
			}
        else 
        {
            return false;
        }
    }
}
