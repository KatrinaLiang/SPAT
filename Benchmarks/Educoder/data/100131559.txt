package step2;

import java.io.File;

public class Task {
	
	/********** BEGIN **********/
	public static void showDirStructure(File file) {
		show(file,0);
	}
	public static void show(File file, int leavel){
		for(int i=0; i<leavel; ++i){
			System.out.print(" ");
		}
		if(file.isDirectory()){
			//打印file层目录
			System.out.println("+--" + file.getName());
			File[] files = file.listFiles();
			for(File f : files){
				
	            show(f, leavel+2);
	        }
		}else{
			//打印file层文件
			System.out.println("--" + file.getName());
		}
	}

	/********** END **********/
}
