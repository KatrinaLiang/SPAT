package Pid_203;
import java.util.Scanner;
public class Main {

	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int k=sc.nextInt();
		int m ;
			int t=k/3;
			int T=k%3;
			if(k==3){
				m=2;
			}
			else if(T==1){
				m=(int) (Math.pow(3, t-1))*4;
			}
			else if(T==0) {
				m=(int) (Math.pow(3, t));
			}
			else{
				m=(int) ( Math.pow(3, t)*2);
			}
		
		System.out.print(m);
	}

}
