package step2;

public class Task {
public  int intermediaDrop(int n) {
		if (n % 2 == 0)
			return 0; 
		else
			return 1; 
	}
	public int getNum(int n){
		/********* Begin *********/
		
		
		int total = 0; 
		while (n > 0) {
			int x = intermediaDrop(n); 
			n -= x; 
			n /= 2; 
			total += x; 

	}		
		return total; 
		
		
		
		/********* End *********/
	}
	
}
