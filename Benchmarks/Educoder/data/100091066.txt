package step2;
import java.util.*;
public class SquareRoot{
    public static void main(String[]args){
    Scanner ss= new Scanner(System.in);
        int a = ss.nextInt();
        System.out.println(SquareRoot(a));
    }
    // 求一个数的算术平方根
    public static double SquareRoot(float n){
        return Math.sqrt(n);
    }
}
