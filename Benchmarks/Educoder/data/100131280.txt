package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        
        String t ="";
        int f = 0;
       for(int i=num1;i<=num2;i++){
          t += String.valueOf(i);
          
           
       }
        for(int x=0;x<t.length();x++){
            char c = t.charAt(x);
            f+=Integer.parseInt(String.valueOf(c));
        }
		
		return f;
		
		/********** END **********/
	}
}
