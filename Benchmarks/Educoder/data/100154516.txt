package step1;

import java.util.Scanner;

public class TestJava20_2 {
	static final Scanner sc = new Scanner(System.in);	
	public void print() {
      
		/********* Start ********/
	Scanner sc=new Scanner(System.in);
	System.out.println("请输入倒直角三角形的行数：");
	int n=sc.nextInt();
	for(int i=1;i<=n;n--){
		for(int j=0;j<n;j++){
			System.out.print("*"+"\t");
		}
		System.out.println();
	}
        
        
        
        
		/********* End ********/
	}

}
