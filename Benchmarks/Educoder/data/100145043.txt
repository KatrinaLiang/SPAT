package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
	    int hang = matrix.length;
        int lie = matrix[0].length;
        int [][] temp1=new int [hang][lie];
        int [][] temp2=new int [hang][lie];
        for(int i= 0;i<hang;i++)
            for(int j =0;j<lie;j++){
                temp1[i][j]=matrix[j][i];
            }
        for(int i= 0;i<hang;i++)
            for(int j =0;j<lie;j++){
                temp2[i][j]=temp1[i][lie-1-j];
            }
        for(int i= 0;i<hang;i++)
            for(int j =0;j<lie;j++) {
                matrix[i][j] = temp2[i][j];
              
            }	
		

		
		
		
		/********* End *********/
	}
}
