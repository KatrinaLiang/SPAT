package step1;
import java.util.Arrays;
public class Task {

	public void sortArray(int[] arr, String sort){
		/********** BEGIN **********/
        Arrays.sort(arr);
		 if(sort.compareTo("asc")==0){
             for(int k = 0;k<arr.length;k++){
                 System.out.println(arr[k]);
             }
         }
        else if(sort.compareTo("desc")==0){
            for(int k = arr.length - 1;k>=0;k--){
                 System.out.println(arr[k]);
            }
        }
		/********** END **********/
	}
}
