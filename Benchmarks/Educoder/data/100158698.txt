#include <iostream>
using namespace std;
int cf(int x){
	int i,xcf=1;
	for(i=0;i<x;i++){
		xcf*=3;
	}
	return xcf;
}
int main()
{
    int k,n,max,i;
    cin>>k;
    if(k==1){
    	max=0;
    }
    else if(k==2){
    	max=1;
    }
    else if(k==3){
    	max=2;
    }
    else if(k==4){
    	max=4;
    }
    else{
    	if(k%3==1){
    	    n=k/3-1;
    	    max=cf(n)*4;
        }
        else if(k%3==2){
    	    n=k/3;
    	    max=cf(n)*2;
        }
        else{
        	n=k/3;
    	    max=cf(n);
        }
    }
    cout<<max;
    return 0;
}
