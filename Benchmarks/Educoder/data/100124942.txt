package step1;

public class Rectangle{
	private double width;
	private double length;
	
	/*********begin*********/
    	static int i=1;
		public Rectangle() {
		super();
		this.width=6;
		this.length=5;
	}
	public Rectangle(double width, double length) {
		super();
		this.width = width;
		this.length = length;
	}
	public void perimeter(){
		System.out.println("r"+i+"的周长："+2*(width+length));
	}
	public void area(){
		System.out.println("r"+i+"的面积："+width*length);
        i++;
	}
	/*********end*********/


	public static void main(String[] args) {
		/*********begin*********/
		Rectangle r1=new Rectangle();
		Rectangle r2=new Rectangle(20,10);
		r1.perimeter();
		r1.area();
		r2.perimeter();
		r2.area();

		/*********end*********/
	}

}
