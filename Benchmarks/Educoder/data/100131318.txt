package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum = 0;
        while(num1 <= num2){
            sum = sum + num1%10 + num1/10%10 + num1/100;
            num1 = num1 + 1;
        }
		return sum;
		
		/********** END **********/
	}
}
