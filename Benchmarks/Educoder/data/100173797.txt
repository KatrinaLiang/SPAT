package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
		int i,a;
		//在这里打印出正三角形 
		for(i=1;i<=10;i++){
            for(a=1;a<i;a++){
				System.out.print("*");
			}
			System.out.println("*");
		}
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
		for(i=10;i>=1;i--){
             for(a=1;a<i;a++){
				 System.out.print("*");
			 }
			 System.out.println("*");
		}
	    /*****end*****/
	}
}
