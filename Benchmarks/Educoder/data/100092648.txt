package step3;

public class FindMaxNum {
    // 找出三个整数的最大值
    public int FindMaxNum(int a, int b, int c){
        // 请在此添加实现代码
        int t=0;
        if(a>b && a>c)
        {
        	t=a;
        }
        else if(b>a && b>c)
        {
        	t=b;
        }
        else if(c>a && c>b)
        {
        	t=c;
        }
        return t;
	}
}
