	/*******Start*********/
	//此处定义一个方法，名为SumOfEven，实现计算并返回1~50的偶数和的功能
		public static int SumOfEven(){
		int s=0,k;
		for(k=1;k<=50;k++){
			if(k%2==0){
				s+=k;
			}
		}
		return s;
	}
