class Person {
	/********* begin *********/
    String name ,age;
    Person(String _name,String _age){
        name = _name;
        age = _age;
    }
	/********* end *********/
}

class Student extends Person {
	/********* begin *********/
    String school;
    Student(String _name,String _age,String _school){
        super(_name,_age);
        school = _school;
    }
	/********* end *********/
}
