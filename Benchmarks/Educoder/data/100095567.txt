public class DayCount {
	public void dayCount(int year, int month, int day) {
		/********** Begin *********/
		int[][] a={{31,28,31,30,31,30,31,31,30,31,30,31},{31,29,31,30,31,30,31,31,30,31,30,31}};
        int flag=0,i,s=0;
		if((year%4==0&&year%100!=0)||year%400==0)
            flag=1;
        else
            flag=0;
		for(i=0;i<month-1;i++)
        {
            s=s+a[flag][i];
        }
        s=s+day;
        System.out.print(s);
		/********** End *********/
	}
}
