#include <stdio.h>
#include <string.h>

int main() {
    char a[100], b[100];
    int alength, blength, minlength, max = 0, maxcount = 0, count = 0;
    int i, j =0;
    scanf("%s", a);
    scanf("%s", b);
    alength = strlen(a);
    blength = strlen(b);
  
	for (i = 0; i<alength; i++) {
		for (j = 0; j<blength; j++) {
			int k, f, count = 0;
			k = i;
			f = j;
			while (a[k] == b[f]) {
				if (k < alength && f < blength) {
					k++;
					f++;
					count++;
				} else {
					break;
				}
			} 
			if (count >= maxcount) {
				maxcount = count;
			}
		}
    }
    
    printf("%d", maxcount);
}
