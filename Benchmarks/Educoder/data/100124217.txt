package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int[] array = new int[20];
		int sum = 0;
		for(int i=0;i<20;i++){
			if(i==0||i==1){
				array[i] = 1;
			}
			if(i>=2){
				array[i] = array[i-1] + array[i-2];

			}
			sum = sum +array[i];
		}
		System.out.print("前20个元素的和为"+sum);

	/********end********/
	}
}
