package step1;
public class DynamicBind {
    public static void main(String[] args) {
        /*********begin*********/
        //定义Fruit的一个引用，分别赋予三个子类对象，顺序为Apple，Pear，Banana
        Apple a = new Apple();
        Banana b = new Banana();
        Pear p = new Pear();
        Fruit f = a;
        f.show();
        f = p;
        f.show();
        f = b;
        f.show();       
        
        /**********end*********/
    }
}
