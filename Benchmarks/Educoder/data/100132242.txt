package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
		int k=0;
		int [] a=new int [temperatures.length];
		for(int i=0;i<temperatures.length;i++){
			k=0;
			for(int j=i;j<temperatures.length;j++){
				if((temperatures[i].compareTo(temperatures[j]))==-1 && j!=i){
					k=j-i;
					break;
				}else{
					k=0;
				}
			}
			a[i]=k;
		}

		return a;
		/********** END **********/
	}
}
