package step1;
import java.lang.Math;
//编写一个Cube类继承Rectangle
public class Cube extends Rectangle{
	/*********begin*********/
    private double lengthOfSide;
	//创建两个构造函数  
    //创建 无参 构造函数  调用父类构造方法默认 宽高 都为10
	public Cube(){
		super(10,10);
        lengthOfSide=10;
    }
    
    
    //创建 有参 构造函数 参数为 double d
    public Cube(double d){
    	this.lengthOfSide=d;
    }

	//调用父类中的area方法和perimeter方法
	public double area(){
		return lengthOfSide*lengthOfSide;
	}	
    public double perimenter(){
		return lengthOfSide*4;
	}

	//编写计算体积的方法
	public double cubeVolume(){
		return Math.pow(lengthOfSide, 3);
	}
	public static void main(String[] args){
	//创建两个正方体实例对象，并按相应格式输出正方体的底面周长，底面面积和体积。
    //创建一个 无参 正方体对象
	Cube myCube=new Cube();
	System.out.println("c底面周长："+ myCube.perimenter()  );
	System.out.println("c底面面积："+ myCube.area()  );
	System.out.println("c体积："+ myCube.cubeVolume()     );

                       
    //创建一个 有参 正方体对象 参数值为： 5
    Cube myCube_2=new Cube(5);               
    System.out.println("c1底面周长："+ myCube_2.perimenter()  );
    System.out.println("c1底面面积："+ myCube_2.area()   );
    System.out.println("c1体积："+  myCube_2.cubeVolume()    );                   

	/*********end*********/	

	}
}
