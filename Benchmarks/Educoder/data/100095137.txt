package step3;
import java.util.*;

public class Factorial {
	public void factorial(int n) {
		/********** Begin *********/
        int sum=1;
		for(int i=2;i<=n;i++)
			sum=sum*i;
        System.out.print(sum);
    }
	public static void main(String[] args){
        Scanner reader=new Scanner(System.in);
        int n;
        n=reader.nextInt();
        Factorial a=new Factorial();
        a.factorial(n);
		/********** End *********/
	}
}
