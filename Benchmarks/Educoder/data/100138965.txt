package step2;

import java.util.Scanner;

public class Task{

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
		MyThread t=new MyThread();
		System.out.println("子线程计算结果为："+t.run(num));
		
	}
}

class MyThread{
 int num;
int f1=1;
int f2=1;
int f3;
public int run(int num) {
	if(num==1||num==2) {
		return 1;
	}
    else {
		for(int i=2;i<num;i++)
		{
            f3=f2;
          
            f2=f1+f2;
		  
            f1=f3;
		
		}
        return f2;
	}


}
	

}
