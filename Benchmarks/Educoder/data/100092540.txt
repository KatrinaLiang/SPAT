package step3;

public class FindMaxNum {
    // 找出三个整数的最大值
    public int FindMaxNum(int a, int b, int c){
       int d=0;
        if(a>b){
           if(a>c){
               d=a;
           }            
            else{
                d=c;
            }
        }
        else{
            if(b>c){
                d=b;
            }
            else{
                d=c;
            }
      }
        return d;
    }
}
