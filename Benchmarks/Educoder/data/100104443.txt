package step2;

public class ConverseNumber {
	public void Converse (int n) {
		/********** Begin *********/
        int sum=0,m;        
        do{ 
            m=n%10;
            sum=sum*10+m;
            n=n/10; 
        }while(n>0);
        System.out.print(sum);
        /********** End *********/
	}
}
