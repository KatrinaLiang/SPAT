package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] arr = new int[20];
        int sum = 0;
        for(int i = 0;i<arr.length;i++){
        	arr[i]=f(i+1);
            sum+=arr[i];
        }
		System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
    
    public static int f(int n){
        if(n==1||n==2){
            return 1;
        }
        else{
            return f(n-1)+f(n-2);
        }
    }
    
}
