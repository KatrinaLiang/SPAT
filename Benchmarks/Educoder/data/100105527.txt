package step1;

public class Yard {
    public int length,width; //定义变量
	public Yard(int x,int y){   //构造器设值
		this.length=x;
		this.width=y;

}
    public double getPerimeter(){  //返回周长，调用构造器传过来的数值进行运算，不用自身传参
		double s=length*2+width*2;
		return s;
	}
	public double getAcreage(){  //返回获取面积
		double s=length*width;
		return s;
	}
	
}
