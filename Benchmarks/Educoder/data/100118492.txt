public class HelloWorld {
    public static void main(String[] args) {


        print();// 调用无参的方法

        String name  = "educoder";
        print(name);// 调用带有一个字符串参数的方法
		int kk = 666;
        // 调用带有一个整型参数的方法
        print(kk);
    }
    
    // 无参print方法的定义
	public static void print(){
    	System.out.println("无参的print方法");
    }
    
    // 字符串print方法的定义
    public  static void print(String ss){
    	System.out.println("带有一个字符串参数的print方法，参数值为："+ss);
    }
    
    // 整型print方法的定义
	public static void print(int kk){
    	System.out.println("带有一个整型参数的print方法，参数值为："+kk);
    }    
}
