package step5;

import java.util.Scanner;


public class Task {
	
	public int getDepth(TreeNode tree){
		/********** BEGIN **********/


		  if (tree.left == null && tree.right == null) return 1;
        if (tree.left == null && tree.right !=null) return getDepth(tree.right)+1;
        if (tree.left!=null && tree.right == null) return getDepth(tree.left)+1;
        int rightTreeDepth = getDepth(tree.right);
        int leftTreeDepth = getDepth(tree.left);
        return rightTreeDepth>leftTreeDepth?rightTreeDepth+1:leftTreeDepth+1;
		/********** END **********/
		
	}
	
}
