package step3;

public class SeasonJudge {

    // 根据月份判断季节
	public String SeasonJudge(int month) {
        // 请在此提供实现代码
        switch (month / 3) {
            case 1 :
                return "春季";
            case 2 :
                return "夏季";
            case 3 :
                return "秋季";
            case 0 :
                return "冬季";
            default :
             if (month == 12){
                 return "冬季";
             }
                return "月份错误";
	}
}
}
