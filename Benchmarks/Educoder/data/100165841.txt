
package step4;

import step1.Date;
import step2.Student;
import step3.Course;

public class Report {
    private Student s;//学生基本信息
    private String major; //所学专业  
    private Course[] selCourse;//所选课程
   /************Begin****************/
   //构造方法
    public Report(Student s,String major,Course[] selCourse){
		this.s = s;
		this.major = major;
		this.selCourse = selCourse;
	}

    //set and get
	public Student getStudent(){
		return this.s;
	}

	public String getmajor(){
		return this.major;
	}

	public Course[] getCourse(){
		return this.selCourse;
	}

    //toString
    public String toString(){
		String str = "";
		for(int i=0;i<this.selCourse.length;i++){
			str = str+this.selCourse[i]+"\n";
		}
		return s+" major:"+this.major+"\n"+str;
	}

    /************End****************/
}
