package step1;

public class A< T extends Comparable >{
 private T t1,t2; 
/*************BEGIN**************/
public A(T a,T b){
	this.t1=a;
	this.t2=b;
}
public T compare(){
	int result=t1.compareTo(t2);
	if(result>0){
		return t1;
	}else if(result<0){
		return t2;
	}else{
		return t1;
	}
} 




/*************END*****************/

}
