package step1;

public class Task {
	public String inversion(String str){
		/********* Begin *********/
		String str_new ="";
        for(int i=str.length()-1;i>=0;i--){
            char c = str.charAt(i);
            str_new = str_new+c+"";
        }
        return str_new;
		/********* End *********/
	}
}
