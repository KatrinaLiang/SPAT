package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		int m=matrix.length;
	int n=matrix[0].length;
	int[] mat=new int[m*n];
	int p=0;
	for(int j=0;j<n;j++)
		for(int i=m-1;i>-1;i--)
		{
			mat[p]=matrix[i][j];
	        p++;
		}
	int k=0;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			{
			    matrix[i][j]=mat[k];
			    k++;
			}
		

		
		
		
		/********* End *********/
	}
}
