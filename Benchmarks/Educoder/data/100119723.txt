package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
		
		//在这里打印出正三角形 
		int i,a;
for(i=1;i<=10;i++){		
		for(a=1;a<=i;a++){
		System.out.print("*");}
    System.out.print("\n");}
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
		int j,b;
		for(j=10;j>=1;j--){
for(b=j;b>=1;b--){		
		System.out.print("*");}
            System.out.print("\n");}
		
		/*****end*****/
	}
}
