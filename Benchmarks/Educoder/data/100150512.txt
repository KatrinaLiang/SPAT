package step3;
import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
		System.out.println("请输入一个五位数字：");
		Scanner input1=new Scanner(System.in);
		String input;
		input=input1.next();
		int abcde=Integer.parseInt(input);
		
		int f=abcde%10;
		int g=abcde/10%10;
		int h=abcde/10/10%10;
		int i=abcde/10/10/10%10;
		int j=abcde/10/10/10/10%10;
		int s=f+g+h+i+j;
		System.out.println("85698的各位数字之和是："+s);
		/******End******/
	}
}
