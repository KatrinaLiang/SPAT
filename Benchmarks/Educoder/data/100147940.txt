/***begin your code here***/
public class MyRunnable implements Runnable
{	
	private int a,sum=1;
    public MyRunnable(int a)
    {
    	this.a = a;
    }
	public void run()
    {
    	
        for(int i=1;i<=a;i++)
        {
        	sum *= i;
        }
        try {Thread.sleep(40000);} catch (Exception e) {}
        System.out.println(sum);
    }
}
