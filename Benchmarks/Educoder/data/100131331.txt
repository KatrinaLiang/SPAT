package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int i;
        int a;
        int b;
        int c;
        int sum=0;
        for(i=num1;i<=num2;i++){
        	a=i/100;
            b=(i%100)/10;
            c=i%10;
            sum=sum+a+b+c;
        }
		return sum;
		
		/********** END **********/
	}
}
