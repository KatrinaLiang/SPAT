	public void sortArray(int[] arr, String sort){
		/********** BEGIN **********/
        String s1="asc";
        String s2="desc";
		if(sort.equals(s1)/*升序*/)
        {for(int i=0;i<arr.length;i++)
        	{for(int j=0;j<arr.length;j++)
            	{if(arr[i]<arr[j])
                	{int temp=arr[i];
                     arr[i]=arr[j];
                     arr[j]=temp;}
            }}}
        if(sort.equals(s2)/*降序*/)
        {for(int i=0;i<arr.length;i++)
        	{for(int j=0;j<arr.length;j++)
            	{if(arr[i]>arr[j])
                	{int temp=arr[i];
                     arr[i]=arr[j];
                     arr[j]=temp;}
            }}}
        for(int k=0;k<arr.length;k++)
        {
        	System.out.println(arr[k]);
        }
		


		
		/********** END **********/
	}
