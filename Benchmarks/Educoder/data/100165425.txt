package step2;
import step1.Person;
import step1.Date;
public class Student extends Person{
    private int reg_num; //学号
    private Date reg_day; //注册日期
   // TO DO YOU CODE HERE
   public Student(){
		super();
		reg_day=new Date();
	}
	public Student(String name,String number,Date birthday,int reg_num,Date reg_day)
	{
		super(name,number,birthday);
		this.reg_num=reg_num;
		this.reg_day=reg_day;
	}
	public Student(String name,String number,int y,int m,int d,int reg_num,int reg_y,int reg_m,int reg_d){
		super(name,number,y,m,d);
		this.reg_num=reg_num;
		this.reg_day=new Date(reg_y,reg_m,reg_d);
	}
	public void setinfo(String name,String number,int y,int m,int d,int reg_num,int reg_y,int reg_m,int reg_d){
		super.setinfo(name,number,y,m,d);
		this.reg_num=reg_num;
		this.reg_day.setDate(reg_y,reg_m,reg_d);
	}
	public int getreg_Num()
	{
		return reg_num;
	}
	public Date getreg_day()
	{
		return reg_day;
	}
	public String toString()
	{
		return super.toString()+"reg_num:"+reg_num+"reg_day:"+reg_day.getYear()+"."+reg_day.getMonth()+"."+reg_day.getDay();
	}




}
