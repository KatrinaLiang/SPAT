package step1;

public class GetScores {

	/********* Begin *********/
    public int getLostScores(int[] nums){
       //请在此补全你的代码
    	
        int[] b=new int[10001];

        for(int i=0;i<9997;i++)
        {
            //标记
            
           int x=nums[i];
            b[x]=1;      
        }
        
        int[] s=new int[3];
        for(int i=1,j=0;i<=10000;i++)
        {
           
            if(b[i]!=1)
            {
                s[j]=i;
                j++;
            }
        }

        long x=0;
        x=Long.parseLong(String.valueOf(s[0])+String.valueOf(s[1])+String.valueOf(s[2]));
        x=x%11;
        int y=(int)x;
        return y;
	}

	/********* End *********/
}
