package step3;

public class HelloWorld {
	public static void main(String[] args) {
		int count= 0;	//定义变量存储6的倍数出现的次数
		/*****start*****/
		int a = 6;
        while(a<=100){
        	if(a%6==0){count++;}
            a++;
        }
        
		

		
		/*****end*****/
		System.out.println("6的倍数出现的次数为：" + count);
	}
}
