package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int arr[] = new int[20];
        int s = 0;
        arr[0] = 1;
        arr[1] = 1;
        for(int i = 2;i<arr.length;i++){
           arr[i] = arr[i-2] + arr[i-1];
        }
        for (int j = 0;j<arr.length;j++){
            s += arr[j];
        }
        System.out.print("前20个元素的和为"+s);

	/********end********/
	}
}
