package step3;
import step1.Date;
public class Course {
    private int code; //课程编号
    private String courseName; //课程名称
    private Date testTime;//考试日期
    private int score;//考试成绩
   /******Begin**************/
public Course(int code, String Name, int year, int month, int day, int score) {
		this.code = code;
		this.courseName = Name;
		this.testTime = new Date(year, month, day);
		this.score = score;
	}
	public Course() {
		this.code = 0;
		this.testTime = new Date();
		this.score = 0;
	}
	public Course(int code, String Name, Date date, int score) {
		this.code = code;
		this.setCourseName(Name);
		this.testTime = date;
		this.score = score;
	}
	public int get_code() {
		return code;
	}
	public String gettime() {
		return testTime.toString();
	}
	public int getscore() {
		return score;
	}
	public String toString() {
        if(courseName == null)
            return get_code()+"  "+gettime()+" "+getscore(); 
        else
			return get_code()+" "+courseName+" "+gettime()+" "+getscore();  
	}
	public String getCourseName() {
		return courseName;
	}
	public void setCourseName(String courseName) {
		this.courseName = courseName;
	}




   /******End************/

}
