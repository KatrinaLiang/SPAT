package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int a,b,c;
		for(a=1;a<=9;a++){
			for(b=1;b<=a;b++){
				c=a*b;
				System.out.print(b+"*"+a+"="+c+"\t");
			}
			System.out.println();
		}
		
		
		
		
		
		/*****end*****/
	}
}
