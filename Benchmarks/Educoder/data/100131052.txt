package step1;

public class Task {
	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum=0;
		for(int i=num1;i<=num2;i++){
			if(i/10==0)
				sum+=i;
			else{
				sum+=(i%10);
				sum+=getSum(i/10,i/10);
			}
		}
		return sum;
		
		
		/********** END **********/
	}
}
