package step3;

public class Reverse {
    public static StringBuffer start(String data) {
        StringBuffer ans = new StringBuffer();
        /********** Begin *********/
        String str=data.toString();
        String [] sp=str.split(" ");
        for(int i=0; i<sp.length;i++){
            ans.append(new StringBuffer(sp[i]).reverse());
            ans.append(' ');
        }
        



        /********** End *********/
        return ans;
    }
}
