package case1;

public class TestPersonDemo {
	public static void main(String[] args) {
		/********* begin *********/
		// 声明并实例化一Person对象p
    Person p=new Person();
		// 给p中的属性赋值
 p.talk("张三，",18);
		// 调用Person类中的talk()方法
System.out.println("我是："+p.getName()+"今年："+p.getAge()+"岁");
		/********* end *********/

	}
}

// 在这里定义Person类
class Person {
	/********* begin *********/
private String name;
	private int age;
	public void talk(String name,int age) {
	     this.name = name;
	     this.age=age;
	 }
	public int getAge() {
	     return age;
	 }
	public String getName() {
	     return name;
	 }
	/********* end *********/
}
