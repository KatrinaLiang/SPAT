public class MyThread extends Thread{
    public int n;
    public MyThread(){
        this(0);
    }
    public MyThread(int n){
        this.n=n;
    }
    public void run(){
        try{
            Thread.sleep(1000);
        }catch(Exception e){}
        int result=1;
        for(int i=1;i<=n;i++){
            result=result*i;
        }
        System.out.println(result);
    }
}
