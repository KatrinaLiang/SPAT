package step1;

public class Task {
        public  int[] dailyTemps(Integer[] temperatures){
		int size= temperatures.length;
		
		int i,j;
		int flag=0;
		int[] answer=new int[size];
		for(i=0;i<size;i++){
			flag=0;
			for(j=i+1;j<size&&flag==0;j++){
				if(temperatures[j]>temperatures[i]){
					answer[i]=j-i;
					flag=1;
				}
			}
			if(j==size&&flag==0){
				answer[i]=0;
			}
		}
	
		return answer;
            /********** END **********/
        }
}
