package step3;
import java.util.Scanner;
public class SumOfDigit{
	public static void main(String[] args) {
		/******Begin******/
		Scanner a=new Scanner(System.in);
        int s=a.nextInt();
        int q=s/10000;
        int w=s/1000%10;
        int e=s/100%10;
        int r=s/10%10;
        int t=s%10;
        int he=q+w+e+r+t;
        System.out.println("请输入一个五位数字：");
        System.out.println(s+"的各位数字之和是："+he);
		/******End******/
	}
}
