package step1;

public class DecomposeFactor {
	public void decompose(int score) {
		/********** Begin *********/
        double rank;
        if(score<60){
            rank=0;}
        else if(score<70)
        {rank=2.5;}
        else if(score<80)
        {rank=3.0;}
        else if(score<90)
        {rank=3.5;}
        else
        {rank=4.0;}
        System.out.print(rank);


		/********** End *********/
	}
}
