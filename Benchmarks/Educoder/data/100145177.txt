package step2;
import java.util.Scanner;
public class Task {

	public int getNum(int n){
        int k=0;
		while(n>0){
            if((n+2)%2!=0){
                k++;
            }
            n=n/2;
        }
		return k;
	}
    public static void main(String[] args){
        Scanner x=new Scanner(System.in);
        int y=x.nextInt();
        Task t1=new Task();
        t1.getNum(y);                                               
    }	
}
