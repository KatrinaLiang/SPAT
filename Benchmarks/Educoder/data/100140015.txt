package step2;

import java.util.Scanner;

public class NameSearch {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        /********** Begin *********/
	String data=scanner.nextLine();
	String name=scanner.nextLine();
    int oldposition=-1;
    int firstpostion=data.indexOf(name);
    while(firstpostion>-1){
        oldposition+=firstpostion+1;
        System.out.println(oldposition);
        if(firstpostion+name.length()>data.length())
            break;
        data=data.substring(firstpostion+name.length()-1);
        firstpostion=data.indexOf(name);
    }
    
        /********** End *********/
    }
}
