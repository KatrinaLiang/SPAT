package step4;
import step1.Date;
import step2.Student;
import step3.Course;


public class Report {
    private Student s;//学生基本信息
    private String major; //所学专业  
    private Course[] selCourse;//所选课程
   /************Begin****************/
    public Report(){
    s = new Student();
    selCourse = new Course[1];}
    public Report(Student s,String major,Course selCourse[]){
        this.s = s;
        this.major = major;
        this.selCourse = new Course[selCourse.length];
        for(int i=0;i< selCourse.length;i++){
        	this.selCourse[i]=selCourse[i];
        }
    }
    public Student getStudent(){return s;
    }
    public String getMajor(){return major;}
    
    public String toString(){
        String str="";
        for(int i=0; i<selCourse.length;i++){
        	str = str+selCourse[i].getcode()+" "+selCourse[i].getCourseName()+" "+selCourse[i].getTestTime()+" "+selCourse[i].getscore()+"\n";
        }
        
        return "name="+s.getName()+" number="+s.getNum()+" birthday:"+s.getBirthday()+"reg_num:"+s.getreg_Num()+"reg_day:"+s.getreg_day()+" major:"+getMajor()+"\n"+str;
        
    }
}
