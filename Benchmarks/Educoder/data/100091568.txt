package step2;

public class SegmentFunCal {
	// 求分段函数的值
	public int SegmentFunCal(int x){
		// 请在此添加实现代码
        //新建一个作为返回的变量
        int y;
        //然后根据传进来的x进行计算
        if(x > 0) {
            y = x + 2;
        } else if(x < 0) {
        	y = x - 4;
        } else {
        	y = x - 2;
        }
        //计算出结果了,将结果返回
        return y;
	}
}
