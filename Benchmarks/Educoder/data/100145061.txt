package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		
			int temp[][] = new int[matrix.length][matrix[1].length];
		for (int i = 0; i < matrix.length; i++)
		System.arraycopy(matrix[i], 0, temp[i], 0, matrix[i].length);
		int n = matrix.length;
		int m = matrix[1].length;
		for (int i = n - 1; i >= 0; i--)
			for (int k = 0; k < m; k++) {
				matrix[k][i] = temp[m - i - 1][k];

			}


		
		
		
		/********* End *********/
	}
}
