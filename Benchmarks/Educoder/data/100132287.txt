package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
		int l = temperatures.length;
		int [] res = new int [l];
		int i , j;
		for(i=0;i<l-1;i++) {
			for(j=i+1;j<l;j++) {
				if(temperatures[j]>temperatures[i]) {
					res[i]=j-i;
					break;
				}
			}
		}
		return res;
		/********** END **********/
	}
}
