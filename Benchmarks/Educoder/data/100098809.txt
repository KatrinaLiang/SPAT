package step1;

public class Yard {
	private int length;
	private int width;
	public Yard(int length, int width) {
		this.length=length;
    	this.width=width;
	}
    public float getPerimeter() {
    	float r=Float.valueOf(length*2+width*2);
    	return r;
    }
    public float getAcreage() {
    	
    	return Float.valueOf(length*width);
    }
}
