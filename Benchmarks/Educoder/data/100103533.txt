package step3;

import java.io.File;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

public class DeleteFiles {
    public static void start() throws IOException {
        StringBuilder data = new StringBuilder("");
        /********** Begin *********/
        //读取readme.txt文件
        File file = new File("src/step3/readme.txt");	
		FileReader fileReader = new FileReader(file);
		int read= fileReader.read();
		while (read != -1) {   
            data.append((char)read);
      			
			read= fileReader.read();   
		}
        fileReader.close();
        //写入readmeCopy.txt文
		File file3 = new File("src/step3/readmeCopy.txt");
		FileWriter fileWriter = new FileWriter(file3, false);
        fileWriter.write(data.toString());
		fileWriter.close();
        //删除readme.txt文件
        File file4 = new File("src/step3/readme.txt");   //file对象指向src文件夹下的step1文件夹下的Hello World.txt文件
        if (file4.exists()) {   //如果文件存在
         file4.delete();   //删除该文件，并输出布尔值
		
		}
        /********** End *********/
    }
}
