
package step3;
import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		int x;
		int y;
		int sum=0;
        Scanner scan = new Scanner(System.in);
        System.out.println("请输入一个五位数字：");
        x=scan.nextInt();
        y=x;
        for(int i=0;i<5;i++)
        {
        sum=sum+(x%10);
        x=x/10;
        }
        System.out.println(y+"的各位数字之和是："+sum);
	}

}
