		for(int i = 0 ; i< arr.length ; i++){
			arr[i] = sc.nextInt();
		}
		/********** Begin **********/
        Arrays.sort(arr);
        //System.out.println(Arrays.toString(arr));
        int[] b = new int[arr.length];
         int j=arr.length-1;
		for(int i=0;i<arr.length;i++){
            b[j]=arr[i];
            j--;
            }
        System.out.println(Arrays.toString(b));
		
		
		
		
		
		
		/********** End **********/
	}
}
