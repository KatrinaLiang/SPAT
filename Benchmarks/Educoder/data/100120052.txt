package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		
		
		for(int hang=1;hang<10;hang++){
            
            for(int lie=1;lie<=hang;lie++){
                 System.out.print(lie+"*"+hang+"="+(hang*lie));
               if(lie==hang)
                   System.out.println("\t");
               else System.out.print("\t"); 
               
                   
            }
         
        }
		
		
		
		/*****end*****/
	}
}
