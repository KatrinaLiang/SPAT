package case4;

public class overridingTest {
	public static void main(String[] args) {
		// 实例化子类对象s，调用talk()方法打印信息
		/********* begin *********/
		Student s=new Student("张三","哈佛",18);
        s.talk();
		/********* end *********/
		
	}
}

class Person {
	/********* begin *********/
	String name;
    int age;
    public void talk(){
	System.out.print("我是："+name+"，今年："+age+"岁，");
    }
	/********* end *********/
}

class Student extends Person {
	/********* begin *********/
	String school;
	public Student(String name,String school, int age){
    	super.name=name;
        super.age=age;
        this. school=school;
        
    }
    public void talk(){
        super.talk();
        System.out.println("我在"+school+"大学上学");
        
    }
	/********* end *********/
}
