package step1;

public class Task {

	public static void birthdayCandle(int num){
		int [] t = new int [200];
		int i,j,age=0,st=0;
		for(i=19;i<200;i++) {
			t[i]=t[i-1]+i;
		}
		boolean f =false;
		for(i=18;i<200;i++) {
			for(j=i+1;j<200;j++) {
				if(t[j]-t[i]==num) {
					st=i+1;
					age=j;
					f=true;
					break;
				}
			}
			if(f)
				break;
		}
		System.out.println("小明今年：" +age    + "岁");
		System.out.println("小明从" +  st  + "岁开始每年举办派对");
	}
}
