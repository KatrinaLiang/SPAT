package step1;

public class Task {
	public String inversion(String str){
		/********* Begin *********/
		
		StringBuffer s = new StringBuffer();
        for(int i = str.length() - 1;i >= 0;i --)
        {
			if(i == str.length() - 1 && str.charAt(i) == ' ')
                continue;
            s.append(str.charAt(i));
        }
        return s.toString();
		
		/********* End *********/
	}
}
