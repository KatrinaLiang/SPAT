package step1;

import java.util.Scanner;

public class TestJava20_2 {
	static final Scanner sc = new Scanner(System.in);	
	public void print() {
        System.out.println("请输入倒直角三角形的行数：");
		int n=sc.nextInt();   //n表示行数
		/********* Start ********/
        for(;n>=1;n--){
            for(int b=1;b<=n;b++){
                System.out.print("*"+"\t");
            }
            System.out.println("");
        }

        
        
        
		/********* End ********/
	}

}
