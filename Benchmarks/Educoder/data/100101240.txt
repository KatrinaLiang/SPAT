package step2;

import java.util.Scanner;

public class SystemUpdate {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String name = scanner.next();  //读入姓名
        String birth = scanner.next();  //读入年龄
        String position = scanner.next();//读入职位
        double salary = scanner.nextDouble();//读入薪水
        
        Salary employee = new Salary(name, birth, position, salary);
        employee.introduction();
    }
}

/********** Begin *********/
class Employee {
    public String name;
    public String birth;
    public String position;
    
}

class Salary extends Employee {
    private double salary;
    public Salary(String name,String birth,String position,double salary)
    {   this.name=name;
     	this.birth=birth;
     	this.position=position;
     	this.salary=salary;
    }
    public void introduction()
    { System.out.print("员工姓名："+name+" "+"出生年月："+birth+" "+"职位："+position+" "+"薪水："+salary);
    }
}
/********** End *********/
