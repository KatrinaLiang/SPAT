package step3;

import java.util.Scanner;

/*用 while语句编写一个程序，输出 1-50这 50个数中所有奇数之和以及所有偶数之和。 */
public class TestJavaKJ15_1 {

	public static void main(String[] args) {
		/*********begin**********/
		int i=1,s=0,h=0,m=0;
		
		while(m<=100){
			h=h+m;
			m=m+2;
		}
		System.out.println("1-100之间所有偶数之和是："+h);
        while(i<=100){
			s=s+i;
			i=i+2;
		}
		System.out.println("1-100之间所有奇数之和是："+s);
	
		
	
		
		/*********end**********/
	}
}
