package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] a=new int[20];
		a[0]=1;
		a[1]=1;
		int i,s=0;
		for(i=2;i<a.length;i++){
			a[i]=a[i-1]+a[i-2];
		}
		for(i=0;i<a.length;i++)
            s+=a[i];
        System.out.print("前20个元素的和为"+s);
	/********end********/
	}
}
