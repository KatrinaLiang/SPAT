package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum = 0;
		for(int i = num1;i <=num2 ;i ++)
		{
			String S = String.valueOf(i);
			for(int j = 0;j < S.length();j ++)
			{
				String s = S.substring(j, j+1);
				int a = Integer.parseInt(s);
				sum = sum + a;
			}
		}
	    return sum;
		
		
		/********** END **********/
	}
}
