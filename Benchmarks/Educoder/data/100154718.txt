package step1;
import java.util.Scanner;

public class TestJava19_2 {
	static final Scanner sc=new Scanner(System.in);
	public void print(){
		int m,n;
		System.out.println("请输入行数：");
		m=sc.nextInt();
		System.out.println("请输入列数：");
		n=sc.nextInt();
		/*********Start********/
		for(int i=1;i<=m;i++){
			for(int k=1;k<=n;k++){
				System.out.print("*");
			}
			System.out.println();
		}
		
		
		
		/*********End********/
	}

}
