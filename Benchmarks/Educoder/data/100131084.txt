package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum=0;
		for(;num1<=num2;num1++){
               int  baiwei=num1/100;
			   int gewei=num1/10%10;
			   int shiwei=num1%10;
            sum=sum+baiwei+gewei+shiwei;
          
        }
		
		
		/********** END **********/
        return sum;
	}
}
