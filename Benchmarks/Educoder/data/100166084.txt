package step1;

public class Person {
    private String name; //姓名
    private String number; //身份证号
    private Date birthday; //出生日期
    
   /************Begin*************/
	private int year;
	private int month;
	private int day;
	public Person(){
         birthday=new Date();
    }
	public Person(String name,String number,Date birthday){
		this.name=name; 
		this.number=number; 
		this.birthday=birthday;
	}
	public Person(String name,String number,int year,int month,int day){
		this.name=name; //姓名
		this.number=number;
		birthday=new Date(year,month,day);
	}
	public void setinfo(String name,String number,int year,int month,int day){
	this.name=name; 
	this.number=number;
	this.year=year;
	this.month=month;
	this.day=day;
	}
	public String getname(){
		return name;
		
	}
	public String getnum(){
		return number;
		
	}
	public Date getBirthday(){
		return birthday;
    }
		
	public String toString(){ 
	    return "name="+name+" number="+number+" birthday:"+birthday;
	}


}
