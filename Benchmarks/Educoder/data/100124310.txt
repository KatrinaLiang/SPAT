package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] Fib = new int[20];
          Fib[0] = 1;
          Fib[1] = 1;
        int sum =2;
        for(int i = 2; i < Fib.length; i++){
			Fib[i] = Fib[i-1] + Fib[i-2]; 
            sum = sum+ Fib[i];
           }

        System.out.print("前20个元素的和为" + sum);

	/********end********/
	}
}
