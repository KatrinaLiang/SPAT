package step3;

public class Task {

	 public int countOfShips(char[][] ferry) {
	  /**************BEGIN************/
	      int count = 0;
		  for(int row = 0; row < ferry.length; row++) {
	      	for (int col = 0; col < ferry[0].length; col ++) {
				if (ferry[row][col] == 'o') {
					continue;
				}
				if (ferry[row][col] == '+') {
					count += 1;
				}
				int var = 0;
				while ((col + var) < ferry[0].length && ferry[row][col + var] == '+') {
					ferry[row][col + var] = 'o';
					var += 1;
				}

				var = 1;
				while ((row + var) < ferry.length && ferry[row + var][col] == '+') {
					ferry[row + var][col] = 'o';
					var += 1;
				}     
	        }
	      }
	 
      return count;
         
      /***********END ***************/
	 }
}
