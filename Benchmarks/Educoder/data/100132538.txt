package step2;

import java.util.Scanner;

public class Task {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int length = sc.nextInt();	//长
		int width = sc.nextInt();	//宽
		/********** BEGIN **********/
        //求出最大公约数即可求出 正方形最大边长 
        int temp =1; 
        int r =length * width ;
        while(temp !=0 )
        {
            temp = length %width ;
            length = width;
            width = temp;
        }
        int  j = length;         //得出最大边长
        int k =  r/(j*j);         //得出个数
        
        System.out.println(j);
        System.out.println(k);
        
		

		
		/********** END **********/
	}
}
