package step2;
public class Fibonacci{
	public static void main(String args[]){
		int i;
		int[] f=new int[20];
		f[0]=1;
		f[1]=1;
		for(i=2;i<f.length;i++)
			f[i]=f[i-1]+f[i-2];
        int sum=0;
        for( i=0;i<20;i++){
            sum+=f[i]; 
        }
        System.out.print("前20个元素的和为"+sum);
	}
}
