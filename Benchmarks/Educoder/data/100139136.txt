public class Task {

	public static void main(String[] args) {

		int num = sc.nextInt();
		MyThread a = new MyThread();
        a.start();
        a.jisuan(num);
		
		
	}
}


class MyThread  extends Thread {
    public void jisuan(int c){
    	int i,j;
       int temp1=1;
        int temp2 = 1;

        	j= temp2;
            temp2 = temp2+temp1;
            temp1 =j;
        }
        System.out.println("子线程计算结果为："+temp2);
    }

	
	

}
