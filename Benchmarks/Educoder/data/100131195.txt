package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
        int sum=0;
        int i=0;
		/********** BEGIN **********/
        for( i=num1;i<=num2;i++)
        {
            int d=i;
            int a=d/100;
            d=d%100;
            int b=d/10;
            d=d%10;
            int c=d;
            sum+=a+b+c;
        }
        return sum;
		
		
		/********** END **********/
	}
}
