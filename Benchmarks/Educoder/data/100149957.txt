
package step1;

	public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
		/******Begin******/

		avg=sc.nextInt();
		math=sum*avg;
		System.out.println("长方形的面积是："+math);
		/******End******/
	}

}
