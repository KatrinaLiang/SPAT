package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int counter = 1;
		while ( n != 1) {
			int divided = n / 2;
			int remian = n % 2;
			if (remian != 0) 
                counter++;
			n = divided;
		}
		return counter;
		
		/********* End *********/
	}
	
}
