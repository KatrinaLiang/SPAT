package step1;

public class Task {
	
	public void sort(Integer[] arr){
		/********* Begin *********/
        int count,num;
        num = 0;
        while(true){
        	count = 0;
            int i,j,temp;
            for(i=0; i<arr.length-1; i++){
                if(arr[i] > arr[i+1]){
                    temp = arr[i];
                    arr[i] = arr[i+1];
                    arr[i+1] = temp;
                    for(j=0; j<arr.length; j++){
                        System.out.print(arr[j] + " ");
                    }
                    System.out.println();
                    count++;
                }
            }
            for(i=arr.length-1; i>1; i--){
                if(arr[i] <arr[i-1]){
                    temp = arr[i];
                    arr[i] = arr[i-1];
                    arr[i-1] = temp;
                    for(j=0; j<arr.length; j++){
                        System.out.print(arr[j] + " ");
                    }
                    System.out.println();
                    count++;
                }
            }
            if(count == 0){
            	if(num == 0) {
            		for(i = 0; i<arr.length; i++) {
            			System.out.print(arr[i] + " ");
            		}
            		System.out.println();
            	}
                break;
            }
            num++;
        }
		/********* End *********/
	}
}
