public class ThreadHelper {
	/**
	 * 在子线程中计算参数的阶乘并输出
	 * @param num
	 */
    static public void calcOnNewThread(int num) {
    	//使用Thread匿名对象以及Runnable匿名对象创建并执行子线程
    	new Thread(new Runnable() {
    	    @Override public void run() {
    	    	int result = 1;
    	    	/***begin your code here***/ 
    			int i=num;
                while(i>0){
		          result=result*i;
                  i=i-1;
                }
    			/***end your code***/
    	    	//直接输出
    	    	System.out.println(result);
    	    }		
    	}).start();
    }
}
