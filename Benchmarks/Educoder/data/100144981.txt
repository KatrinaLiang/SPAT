package step3;

public class Task {
	public void rotate(int[][] matrix){
			/********* Begin *********/
			int row = matrix.length;
			int col = row;
			int[][] m = new int[row][col];
			for(int i=0; i<row; i++) {
				for(int j=0; j<col; j++) {
					m[i][j] = matrix[col - j - 1][i];
				}
			}
			for(int i=0; i<row; i++) {
				for(int j=0; j<col; j++) {
					matrix[i][j] = m[i][j];
				}
			}
			/********* End *********/
		}
}
