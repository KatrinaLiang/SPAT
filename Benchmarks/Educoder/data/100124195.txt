package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] a = new int[20];
		a[0] = 1;
		a[1] = 1;
		for (int i = 2; i < a.length; i++) {
			int b = i;
			a[i] = a[--b] + a[--b];
		}
		int sum = 0;
		for (int i = 0; i < a.length; i++) {
			sum += a[i];
		}
        System.out.print("前20个元素的和为"+sum);

	/********end********/
	}
}
