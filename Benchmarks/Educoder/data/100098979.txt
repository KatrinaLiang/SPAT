package step1;

public class Yard {
    
    private int length;
    private int width;
    public Yard(int length,int width){
        this.length=length;
        this.width=width;
    
    }
    
    public float getPerimeter(){
        float x=(length+width)*2;
        return x;
    }
    public float getAcreage(){
        float x=length*width;
        return x;
    }

}
