package step1;

public class Yard {
        int length,width;
     	public Yard(int length,int width){
         	   this.length=length;
            this.width=width;
        }
    
		public double getPerimeter(){
        	double a= (double)(2*length+2*width);
            return a;
        
        }
		public double getAcreage(){
        	double b=(length*width);
            return b;
        
        }
}
