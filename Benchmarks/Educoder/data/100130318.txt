import java.util.Scanner;
public class Task {
void swap(int A[],int i,int j)
{
    int temp=A[i];
    A[i]=A[j];
    A[j]=temp;
}
void Asc(int A[],int n)
{
    for(int j=0;j<n-1;j++)
    {
        for(int i=0;i<n-1-j;i++)
        {
            if(A[i]<A[i+1])
                swap(A,i,i+1);
        }
    }
}
void Desc(int A[],int n)
{
    for(int j=0;j<n-1;j++)
    {
        for(int i=0;i<n-1;i++)
        {
            if(A[i]>A[i+1])
                swap(A,i,i+1);
        }
    }
}
