package Q103;
import java.util.*;
public class Main {

	public static void main(String[] args) {
		// write your code here
		int dp[] = new int[10000];
		Scanner cin = new Scanner(System.in);
		int n = cin.nextInt();
		dp[0] = 1;
		dp[1] = 1;
		for (int i = 2; i <= n; ++ i) dp[i] = dp[i - 1] + dp[i - 2];
		System.out.print(dp[n]);
	}

}
