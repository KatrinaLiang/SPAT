package step3;

public class Reverse {
    public static StringBuffer start(String data) {
        StringBuffer ans = new StringBuffer(data);
        /********** Begin *********/
         ans.reverse();
		data=ans.toString();
        ans.delete(0,ans.length());
        String [] word=data.split(" ");
        for(int i=word.length-1;i>=0;i--)
        {
            ans.append(word[i]+" ");
        }
        
        /********** End *********/
        return ans;
    }
}
