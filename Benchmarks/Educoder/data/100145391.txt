package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int x = 0;
        while(n>=1){
        	int carry = n/2;
            int a = n%2;
            x = x+a;
            n = carry;
        }
        
        return x;
		
		
		
		
		/********* End *********/
	}
	
}
