package step3;

public class Reverse {
    public static StringBuffer start(String data) {
        StringBuffer ans = new StringBuffer();
        /********** Begin *********/
        
        
        String[] names= data.split(" ");
        StringBuffer[] name=new StringBuffer[names.length];
        for(int i=0;i<names.length;i++){
            name[i] = new StringBuffer(names[i]);

        }
        
        for(int i=0;i<names.length;i++){
            name[i]=name[i].reverse();
        }
        for(int i=1;i<names.length;i++){
            
            name[0]=name[0].append(" ");
            name[0]=name[0].append(name[i]);
            if(i==names.length-1)
                name[0]=name[0].append(" ");
            
        }
        ans = name[0];

        /********** End *********/
        return ans;
    }
}
