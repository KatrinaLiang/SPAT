package step1;

public class Task{

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int  sum,a,b,c;
        sum=0;
      
        for(int i=num1;i<=num2; i++){
       a=i%10;
            b=(i-a)%100;
            c=(i-a-b)/100;
            sum=sum+a+(b/10)+c;
            
        
    }
		return sum;
        
 
		
		/********** END **********/
	}
}
