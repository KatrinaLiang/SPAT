public class Task {
    
	public boolean isPowerCut(String date){
		String rest[]= {"7-1","7-3","7-6","7-10","7-15","7-21","7-28","8-5","8-14","8-24","9-4","9-16","9-29"};
		for(int i=0;i<rest.length;i++) {
			if(date.equals(rest[i])) {
				return true;
			}
		}
		return false;
		
	}
}
