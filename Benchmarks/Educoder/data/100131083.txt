package step1;

public class Task {

	
	public int getSum(int m, int n) {
		/********** BEGIN **********/
		int g,s,b;
        int sum=0;
        if(m>n){
        	int temp=m;
            m=n;
            n=temp;
        }
        
        if(m>=1&&n<=999&&m<n){
        	for(;m<=n;m++){
            	g=m%10;
                s=m/10%10;
                b=m/100;
                sum=sum+g+s+b;
            }
            return sum;
        }
        else{
        	return -1;
        }

		/********** END **********/
	}
}
