package step4;

public class DayCount {
	public void dayCount(int year, int month, int day) {
		/********** Begin *********/
		int[] m_days = {0,31,28,31,30,31,30,31,31,30,31,30,31};
        int sum = 0;
        if((year%4 == 0 && year%100!= 0) || year % 400 == 0)
            m_days[2] = 29;
        int i = 1;
		while(i<month){
        	sum += m_days[i];
            i++;
        }
        sum += day;
		System.out.printf("%d", sum);
		/********** End *********/
	}
}
