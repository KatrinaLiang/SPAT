public class MyThread extends Thread{
    
    int num;
    
    public MyThread(int num){
        
        this.num = num;
    }
    
    public MyThread(){
        
        this(0);
    }
    
    public void run(){
        
        int result = 1;
            
        for(int i = num ; i > 0 ; i--){

            result *= i;
        }
        
        try {
            Thread.sleep(300l);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }

        System.out.println(result);
    }
}
