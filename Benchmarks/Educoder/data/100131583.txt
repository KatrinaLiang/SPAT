public class Task {
	
	/********** BEGIN **********/
    int date =0;

	public void showDirStructure(File file)	{
        if(file.isDirectory()){
          System.out.print("+--"+file.getName()+"\n");  
          new Task().printFile(file,2);
        }else{
            //文件
            System.out.print("--"+file.getName()+"\n");
        }
        
        
	}
    public void printFile(File file,int lever){
        String strNspace="";
        for(int j=0;j<lever;j++){
            strNspace=strNspace+" ";
        }
        
        
        File []fileList = file.listFiles();
        
        
          for(int i = 0;i<file.listFiles().length;i++){
            if(fileList[i].isDirectory()){
                //文件夹
                System.out.print(strNspace+"+--"+fileList[i].getName()+"\n");
                printFile(fileList[i], lever+2);
            }else{
                System.out.print(strNspace+"--"+fileList[i].getName()+"\n");
            }
          }
        
        
    }
