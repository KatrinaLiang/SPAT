package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int result = 0;
		for(int i = num1; i <= num2; ++i) {
			int a = i / 10,b = i % 10 ;
			result += b;
			while(a!= 0) {
				b = a % 10;
				a = a / 10;
				result += b;
				
			}
		}
		return result;
		
		
		/********** END **********/
	}
}
