public class MyRunnable implements Runnable {
	private int b;
	
    public MyRunnable() {
    	this(0);
    }
    
    public MyRunnable(int b) {
    	this.b=b;
    }
	
	@Override
	public void run() {
		try{
            Thread.sleep(100);
        }catch(InterruptedException e){
            e.printStackTrace();
        }
        int result =1;
        for(int i=1;i<=b;i++){
            result=result*i;
        }
    	System.out.println(result);
	}

}
