package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		
		 int sum=0;
	 for(int i=num1;i<=num2;i++){
        if(i<100)
           sum=sum+(i%10)+(i/10);
        else 
           sum=sum+i/100+i/10%10+i%100%10;
        }  
        
		return sum;
		/********** END **********/
	}
}
