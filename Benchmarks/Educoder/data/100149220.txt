package step2;

public class Task {
    
	public int virusArea(int n,int m,char[][] area){
		/********* Begin *********/
		if(area.length == 0 || area[0].length ==0){
            return 0;
        }
        n = area.length;
        m = area[0].length;
        int res = 0;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(area[i][j]=='o'){                    
                    res++; 
                    toZero(area,i,j,n,m);
                }
            }
        }
        return res;
        
		
		/********* End *********/	
	}
    private void toZero(char[][] area,int i,int j,int n,int m){
        if(i<0 || i>=n || j<0 || j>=m){
            return ;
        }
        if(area[i][j]!='o'){
            return;
        }
        area[i][j]='x';
        toZero(area,i+1,j,n,m);
        toZero(area,i-1,j,n,m);
        toZero(area,i,j+1,n,m);
        toZero(area,i,j-1,n,m);
    }
}
