package task01;

import java.util.Scanner;

class Student
{
	String name;
	int age;
	void speak() {
		System.out.println("我的姓名:"+name+",年龄:"+age);
	}
}

public class Task01
{
	public static void main(String[] args)
	{
		Scanner scan = new Scanner(System.in);	
				Student p=new Student();
				if(scan.hasNextLine())
				{
				String n=scan.next();
				int a=scan.nextInt();
				p.name=n;
				p.age=a;
				
				p.speak();
				}
		
	}
}
