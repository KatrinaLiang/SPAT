package step1;
public class DynamicBind {
    public static void main(String[] args) {
        /*********begin*********/
        //定义Fruit的一个引用，分别赋予三个子类对象，顺序为Apple，Pear，Banana
        
        Fruit fruit = new Apple("红富士苹果");
		fruit.show();
		fruit=new Pear("香梨");
		fruit.show();
		fruit = new Banana("仙人蕉");
		fruit.show();
        
        
        /**********end*********/
    }
}
