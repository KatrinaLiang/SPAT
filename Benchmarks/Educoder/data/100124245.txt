package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	long fub1 = 1,fub2 = 1,sum = 1;
    for (long i = 1;i<20; i++){
        sum += fub1;
        long tmp = fub1;
        fub1 += fub2;
        fub2 = tmp;
    }
        System.out.print("前20个元素的和为" + sum);

	/********end********/
	}
}
