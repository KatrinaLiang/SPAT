package step3;

import java.util.Arrays;

public class Task {

	public void matrixSum(int[][] arr){
		/********** BEGIN **********/
		int l=arr.length;
        int sum=0;
        for(int i=0;i<l;i++){
            for(int k=0;k<l;k++){
                if(k==i)
        sum=sum+arr[i][k];
            }
        }
		System.out.println(sum);



		/********** END **********/
	}
	
	
}
