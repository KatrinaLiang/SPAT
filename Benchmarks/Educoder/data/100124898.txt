package step1;

public class Rectangle{
	private double width;
	private double length;
	
	/*********begin*********/
	public   Rectangle(){
        this.width = 5.0;
        this.length = 6.0;
    }
    public Rectangle(double width, double length){
        this.width = width;
        this.length = length;
    }
    public double perimeter(){
        return (this.width + this.length) * 2.0;
    }
    public double area(){
        return this.width * this.length;
    }
