package step1;

public class Task {
	public void sort(Integer[] arr){
		/********* Begin *********/
		int num=arr.length,start=0,sum=arr.length;
		boolean sign=true;
		for(start=0;start<num;start++)
		{
			for(int i=start;i<num-1;i++)
			{
				if(arr[i]>arr[i+1])
				{
					sign=false;
					int t=arr[i];
					arr[i]=arr[i+1];
					arr[i+1]=t;
					for(int j=0;j<sum;j++)
					{
						System.out.print(arr[j]);
						if(j<sum)
						{System.out.print(' ');}
					}
					System.out.println();
				}
			}
			num--;
			for(int i=num-1;i>0;i--)
			{
				if(arr[i]<arr[i-1])
				{
					sign=false;
					int t=arr[i];
					arr[i]=arr[i-1];
					arr[i-1]=t;
					for(int j=0;j<sum;j++)
					{
						System.out.print(arr[j]);
						if(j<sum)
						{System.out.print(' ');}
					}
					System.out.println();
				}
			}
			start++;
		}
		if(sign)
		{
			for(int j=0;j<sum;j++)
			{
				System.out.print(arr[j]);
				if(j<sum)
				{System.out.print(' ');}
			}
            System.out.println();
		}
		
		/********* End *********/
	}
}
