package step2;

import java.io.File;
import java.io.IOException;

public class FileUtil {
	public static void start() {
		/********** Begin *********/
File file=new File("src/step2");
File[] fileFile=file.listFiles();
if(fileFile!=null){
    for(File f: fileFile){
        if(f.isDirectory()){
            System.out.println(f.getName());
        }
    }
    for(File f: fileFile){
        if(f.isFile()){
            System.out.println(f.getName());
        }
    }
}
        


		/********** End *********/
	}
}
