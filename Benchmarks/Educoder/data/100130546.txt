package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int a = sc.nextInt();
		/********** BEGIN **********/
		
		double count = 0;
		if(a%2==0){
        	for(int i=2;i<=a;i+=2){
        		count+=3.0/i;
        	}
		}
		else{
			for(int i=1;i<=a;i+=2){
        		count+=4.0/i;
        	}
		}
    			System.out.println(count);
		
		/********** END **********/
	}
}
