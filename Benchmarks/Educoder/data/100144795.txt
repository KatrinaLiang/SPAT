package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		
		int n=matrix.length;
		int m=matrix[0].length;
		
		int[][] temp=new int[m][n];
		int count=n;
		for(int i=0;i<n;i++) {
			for(int j=0;j<m;j++) {
				temp[j][count-1]=matrix[i][j];
//				temp.get(j).set(count-1, matrix.get(i).get(j));
			}
			count--;
		}
		for(int i=0;i<n;i++) {
			for(int j=0;j<m;j++) {
//				matrix.get(i).set(j,temp[i][j]);
				matrix[i][j]=temp[i][j];
			}
		}
		
		

		
		
		

		
		
		
		/********* End *********/
	}
}
