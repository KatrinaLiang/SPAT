package step3;

public class MyThread implements Runnable {   
	    
    public static void main(String[] args) throws Exception {  
        Object e = new Object();
        Object d = new Object();
        Object u = new Object();
        Thread thread1 = new Thread(new MyThread("E", u, d, e));
        Thread thread2 = new Thread(new MyThread("D", e, u, d));
        Thread thread3 = new Thread(new MyThread("U", d, e, u));
        thread1.start();
        Thread.sleep(100);
        thread2.start();
        Thread.sleep(100);
        thread3.start();
        Thread.sleep(100);
    }  
    
    private String name;
    private Object self;
    private Object pre;
    private Object prepre;
    
    public MyThread (String name, Object self, Object pre, Object prepre){
        this.name = name;
        this.self = self;
        this.pre = pre;
        this.prepre = prepre;
    }
    public void run(){
        int count = 5;
        while(count > 0){
            synchronized(pre){
                synchronized(self){
                    System.out.print(name);
                    count--;
                    self.notify();
                }
                try{
                    pre.wait();
                }catch(InterruptedException e){
                    e.printStackTrace();
                }

            } 
        }
        System.exit(0);
    }
}
