public class MyRunnable implements Runnable{
    private int b;
    public MyRunnable(int b){
        this.b=b;
    }
    public void run(){
        int result = 1;
        for(int i=1;i<=b;i++){
            result =result * i;
        }
        try{
        Thread.currentThread().sleep(100);
        }catch(Exception e){}
        System.out.println(result);
    }
}
