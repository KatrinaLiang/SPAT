package Step3;

public class MultiplyTable {
	public static void main(String args[]){
		for ( int n=1;n<=9;n++)
		{
			for(int m=1 ; m<=9 ; m++)
			{
				if(m==n)
				{
					System.out.println( " " + n + "×" + m + "=" + (n*m));
					break;
				}
				else
					System.out.print( " " + n + "×" + m + "=" + (n*m));


			}
		}
}
}
