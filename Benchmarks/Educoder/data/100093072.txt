package step2;

public class SwitchToGrade {

    // 请用switch来将学生成绩转换成对应等级
    public String SwitchToGrade(int score) { 
        // 请在此提供实现代码  
        String ch=null;
        if(score>=0&&score<=100)
        {
            score/=10;
            switch(score)
            {
                case 9:ch="等级：A";break;
                case 8:case 7:ch="等级：B";break;
                case 6:ch="等级：C";break;
                default:ch="等级：D";
            }
            return ch;
        }
        return ch="输入成绩非法";
	}
}
