package step4;

public class HelloWorld {
	public static void main(String[] args) {
		/********** Begin **********/
		int [][] a={{92,85},{91,65},{90,33}};
		for(int i=0;i<a.length;i++){
			for(int j=0;j<a[i].length;j++){
				System.out.println(a[i][j]);
				a[i][j] = j+1;					//超级牛逼！！！！！
			}
		}

		for(int i=0;i<a.length;i++){
			for(int j=0;j<a[i].length;j++){
				System.out.println(a[i][j]);
			}
		}
		
		
		
		
		
		
		
		/********** End **********/
	}
}
