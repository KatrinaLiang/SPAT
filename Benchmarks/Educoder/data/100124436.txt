package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] f=new int[20];
        int i;
        f[0]=1;
        f[1]=1;
        int sum=0;
        for(i=2;i<20;i++){
            f[i]=f[i-1]+f[i-2];
        }
        for(i=0;i<20;i++)
            sum=sum+f[i];
        System.out.print("前20个元素的和为"+sum);

	/********end********/
	}
}
