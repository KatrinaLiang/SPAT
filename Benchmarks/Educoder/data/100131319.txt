package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum = 0;
		for(int i=num1; i<=num2; i++) {
            sum += getDigitSum(i);
        }
		return sum;
		/********** END **********/
	}
    
    public int getDigitSum(int num) {
        int sum = 0;
        while(num != 0) {
            sum += num%10;
            num = num/10;
        }
        return sum;
    }
}
