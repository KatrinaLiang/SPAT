package step3;
import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
		int a,b;
        int s=0;
        Scanner scan=new Scanner(System.in);
        System.out.print("请输入一个五位数字：\n");
        a=scan.nextInt();
        while(a>0)
        {
           b=a%10;
           s=s+b;
           a=a/10;
        }
        System.out.println("85698的各位数字之和是："+s);
		/******End******/
	}
}
