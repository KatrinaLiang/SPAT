package task3;

public class Box {

    double width;
    double height;
    double depth;

    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public double volume(int boxType) {
        //TODO write your code here
        double vol=0;
        switch(boxType)
        {
            case 2:
                vol=3.14*width*width*height;
                break;
            case 1:
                vol=width*height*depth;
                break;
        }
        return vol;
    }

}
