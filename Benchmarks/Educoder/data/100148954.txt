package step3;

public class Task {
	
	public int concaveValley(Integer[] arr){
		int maxLength = 0;
		for (int i=1;i<arr.length-1;i++){
			int length = 1;
			int left = i;
			while(left!=0 && arr[left-1]>arr[left]){
				left--;
				length++;
			}
			int right = i;
			while(right!=arr.length-1 && arr[right+1]>arr[right]){
				right++;
				length++;
			}
			if(length>maxLength){
				maxLength = length;
			}
		}
		if (maxLength == 1)
			return 0;
		return maxLength;
	}
}
