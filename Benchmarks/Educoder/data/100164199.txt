package step3;

import java.util.Scanner;
public class Number_Judge {
		public static void main(String[] args) {
			Scanner scan1=new Scanner(System.in);
			/******Begin******/
			int x=scan1.nextInt();
            Number_Judge a=new Number_Judge();
			if(x%2==0)
            {
            	System.out.print("偶数");
            }
	        else
            {
            	if(x==1)
                System.out.print("奇数");
                else
                {
                	if(a.judge(x)==1)
                    System.out.print("素数");
                    else
                    System.out.print("奇数");
                }
            }
			/******End******/
		}
    	public int judge(int m)
        {
        	for(int i=2;i<=Math.sqrt(m);i++)
            {
            	if(m%i==0)
                return 0;
            }
            return 1;
        }
}
