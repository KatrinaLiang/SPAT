package step4;
import java.util.Scanner;
public class Josephus {

   public static void main(String args[]){
		   /**
			 * 数组解决约瑟夫环问题
			 * @param total   总人数（比如41）
			 * @param number  要报的数（比如3）
			 */
		   int total,number;
		   Scanner scan=new Scanner(System.in);
		   total=scan.nextInt();  //输入参与游戏的总人数
		   number=scan.nextInt(); //输入要报的数字
		   System.out.print("最后一个人编号："+Josecount(total,number));
			  
	  }	
	  public static int Josecount(int personNumber ,int num) { //返回最后那个人的原始编号
	    int[] persons = new int[personNumber]; //建立数组，存放所有人的原始编号，从1开始.....personNumber最后一人
	    for (int i = 0; i < personNumber; i++) {
	        persons[i] = i+1;
	    }
	 
	   /*************begin***************/
	    int index = 0;//从0开始
	    int dunNumbers = 0;//记录已经蹲下的人数，跳过几个人
	    int duns = 0;//记录蹲下人的个数
	 
	    while (duns !=personNumber ) {  //有人没有蹲下，即蹲下的人数不等于总人数	       
	 
	        if ( persons[index % persons.length] == 0) { // 判读当前位置是否已经蹲下，数组值为0表示蹲下
	            dunNumbers++; //已蹲下,跳过它，继续数数
	            index++;
	        }
	        else { //没有蹲下,判读数的数是否是number的数
	            
	            if ((index + 1 -dunNumbers) % num == 0) { //是	               
	                //用0标识蹲下
	            	//System.out.print(" "+persons[index % persons.length]);
	                persons[index%persons.length] = 0;
	                duns++;   //蹲下的人数++	               
	                index++;  //继续数数
	            }else {
	                //否
	                //继续数数
	                index++;
	            }
	 
	        }
	     
	    }
	    return index % persons.length;
	    /*************end***************/
	  }
	
	}
