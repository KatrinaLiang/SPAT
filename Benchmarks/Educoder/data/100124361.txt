package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		
int n=20;
		int n1=1;
		int	n2=1;
		int n3=0;
		int sum=2;
	    for(int i=2;i<n;i++)
	    {
	        n3=n1+n2;
	        n1=n2;
	        n2=n3;
	        sum+=n3;
	    }
        System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
