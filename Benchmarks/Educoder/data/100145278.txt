package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
        int drop=0;
        while(n!=1){
        if(n%2==1){
        drop++;
        n--;
        n=n/2;
        }
        else{
        n=n/2;
        } 
           }
        drop++;
        return drop;		
		/********* End *********/
	}
	
}
