public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        int sum=a;
        int i=2;
        int k=a;
	while(i<=n)
    {
        a=a*10+k;
        sum+=a;
        i++;
    }
	System.out.print(sum);
		/********** End *********/
	}
}
