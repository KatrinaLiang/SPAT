import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
		SubThread mt = new SubThread(num);
		
		
	}
}


class SubThread extends Thread{
    public int n;
    public SubThread(int _n){
        n =_n;
        start();
    }
    public void run(){
        System.out.println("子线程计算结果为："+fab(n));
    }
    private int fab(int n){
        if(n == 1|n ==2)
            return 1;
        else
            return fab(n-1)+fab(n-2);
    }

	
	

}
