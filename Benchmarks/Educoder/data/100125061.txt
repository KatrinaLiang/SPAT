package step1;

public class Rectangle{
	private double width;
	private double length;
	
	/*********begin*********/    
    public Rectangle(){
   	System.out.print("new Rectangle ");
    }
    
    public Rectangle(double width,double length){
   			this.width=width;
        	this.length=length;
    }
    public double perimeter(){
    	 double Perimeter=2*(this.width+this.length);
        return Perimeter;
    }
    public double area(){
         double Area=this.width*this.length;
        return Area;
    }
        
    
   
	
	/*********end*********/


	public static void main(String[] args) {
		/*********begin*********/
		 Rectangle r1=new Rectangle(6,5);
		 Rectangle r2=new Rectangle(20,10);
        System. out.println("r1的周长："+r1.perimeter());
        System. out.println("r1的面积："+r1.area());
        System. out.println("r2的周长："+r2.perimeter());
        System. out.println("r2的面积："+r2.area());
		/*********end*********/
	}

}
