		File file=new File("src/step3/input/input.txt");
        File file1=new File("src/step3/output/output.txt");
        if(!file1.exists()){
            File file2=new File("src/step2/output");
            if(!file2.exists())
            	file2.mkdirs();
            file1.createNewFile();

		FileInputStream fi=new FileInputStream(file);

        //System.out.println(str);
