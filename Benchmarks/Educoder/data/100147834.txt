package step4;
public class MyRunnable implements Runnable {
	private int num;
    public MyRunnable() {
    	this(0);
    }
    public MyRunnable(int num) {
    	this.num = num;
    }
	
	public void run() {
        
    	int result = 1;
		int i;
        for(i=1;i<=num;i++){
            result *= i;
        }
        try {
                Thread.sleep(200);
            } catch (Exception e) {
            }

    	System.out.println(result);
	}
    
}
