package step2;

import java.util.Scanner;

public class SystemUpdate {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String name = scanner.next();
        String birth = scanner.next();
        String position = scanner.next();
        double salary = scanner.nextDouble();
        Salary employee = new Salary(name, birth, position, salary);
        employee.introduction();
    }
}

/********** Begin *********/
class Employee {
	private String name;
	private String brith;
	private String position;
	
	public String getName() {
		return name;
	}
	public void setName(String name) {
		this.name = name;
	}
	public String getBrith() {
		return brith;
	}
	public void setBrith(String brith) {
		this.brith = brith;
	}
	public String getPosition() {
		return position;
	}
	public void setPosition(String position) {
		this.position = position;
	}
	public Employee(String name, String brith, String position) {
		this.name = name;
		this.brith = brith;
		this.position = position;
	}
	
		
}

class Salary extends Employee {

	private double salary;
	public Salary(String name, String brith, String position,double salary) {
		super(name, brith, position);
		this.salary = salary;
	}
	
	public void introduction() {
		System.out.print("员工姓名："+ super.getName()+" 出生年月："+super.getBrith() 
						+" 职位："+super.getPosition()+ " 薪水："+salary);
	}

}
