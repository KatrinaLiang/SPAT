package task3;

public class Box {

    double width;
    double height;
    double depth;

    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public double volume(int boxType) {
        //TODO write your code here
        double v;
        if(boxType==2)
        {
            v=3.14*(height/2)*(height/2)*width*2;
        }
        else
        {
            v=width*height*depth;
        }
        return v;
    }

}
