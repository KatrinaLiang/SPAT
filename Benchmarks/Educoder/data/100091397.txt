package step3;
public class NarcissisticNumber {

    //判断一个三位数是否是水仙花数
    public boolean NarcissisticNumber(int n){
        boolean flag=true;
        int a,b,c;
        a=n/100;
        b=n/10%10;
        c=n%10;
        if (a*a*a+b*b*b+c*c*c!=n){
            flag=false;
        }
        return flag;
        // 请在此添加实现代码
    }
}
