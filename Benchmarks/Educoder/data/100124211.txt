package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
        int []s = new int[20];
        int sum = 0;
        for(int i=0;i<s.length;i++){
            s[0] = 1;
            s[1] = 1;
            if(i>=2){
                s[i] = s[i-1]+s[i-2];
            }
            sum = sum + s[i];
        }
        System.out.print("前20个元素的和为"+sum);
		

	/********end********/
	}
}
