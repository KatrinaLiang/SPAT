package step2;

// 利用 Runnable 接口实现
/********** Begin *********/
/********** End *********/
    public class MyThread implements Runnable{
	private String name;
    public MyThread(String name) {
        /********** Begin *********/
        this.name=name;
		/********** End *********/
      
    }  
    public void run() {
        /********** Begin *********/
       System.out.println("运行:"+name);
		/********** End *********/
        
    }
}
