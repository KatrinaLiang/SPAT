package step2;
import step1.Person;
import step1.Date;
public class Student extends Person{
    private int reg_num; //学号
    private Date reg_day; //注册日期
   // TO DO YOU CODE HERE
    public Student(){
        super();
        reg_day=new Date();}
    public Student(String name,String number,int year,int month,int day,int a,int b,int c,int d){
        super(name,number,year,month,day);
        reg_num=a;
        reg_day=new Date(b,c,d);}
    public Student(String name,String number,Date d1,int a,Date d2){
        super(name,number,d1);
        reg_num=a;
        reg_day=d2;}
    public Date getreg_day(){
        return reg_day;}
    

    public String toString(){
        return super.toString()+"reg_num:"+reg_num+"reg_day:"+reg_day.toString();}





}
