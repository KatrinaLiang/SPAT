package step3;

public class HelloWorld {
	public static void main(String[] args) {
		int count= 0;	//定义变量存储6的倍数出现的次数
        int num = 1;
		/*****start*****/
		do{
            if(num%6==0)
                count++;
            num++;
        }while(num<=100);
		

		
		/*****end*****/
		System.out.println("6的倍数出现的次数为：" + count);
	}
}
