package step2;

public class Student extends Person{
    private int reg_num; //学号
    private Date reg_day; //注册日期
   // TO DO YOU CODE HERE
	public Student(){
		super();
		reg_day = new Date(1999, 9 ,9);	
	}
	public Student(String name, String number, Date birthday, int reg_num, int year, int month, int day){
		super(name, number, birthday);
		this.reg_num = reg_num;
		this.reg_day = new Date(year, month, day);
	}
	public void setinfo(String name, String num, Date day) {
		super.setperson(name, num, day);
	}
	public Student(String name, String number, int year, int month, int day, int reg_num, Date reg_day){
		super(name, number, year, month, day);
		this.reg_num = reg_num;
		this.reg_day = reg_day;
	}
	public Student(String name, String number, int year, int month, int day, int reg_num, int year1, int month1, int day1){
		super(name, number, year, month, day);
		this.reg_num = reg_num;
		this.reg_day = new Date(year1, month1, day1);
	}
	public Student(String name, String number, Date d1, int reg_num, Date d2){
		super(name, number, d1);
		this.reg_num = reg_num;
		this.reg_day = d2;
	}
	public String toString(){
        return super.toString()+"reg_num:"+reg_num+"reg_day:"+reg_day;
    }
	public Date getreg_day(){
		return reg_day;
	}
	public void setDate(int year, int month, int day){
		reg_day = new Date(year, month, day);
	}
