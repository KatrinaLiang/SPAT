public class Task {

	public void sortArray(int[] arr, String sort){
		/********** BEGIN **********/
        int temp = 0;
        for(int i = 0;i<arr.length-1;i++)
            for(int j=0;j<arr.length-1;j++){
            	if(arr[j]>arr[j+1])
                {
                	temp = arr[j];
                    arr[j] = arr[j+1];
                    arr[j+1] = temp;
                }
            }
		 if(sort.equals("asc"))
        {
        	for(int i = 0;i<arr.length;i++)
            {
            	System.out.println(arr[i]);
            }
           
        
        }
       if(sort.equals("desc"))
            
        {
        	for(int i = arr.length-1;i>=0;i--)
                System.out.println(arr[i]);
        }
		


		
		/********** END **********/
	}
}
