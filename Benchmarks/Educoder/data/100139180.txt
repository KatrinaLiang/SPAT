package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
		Thread t = new MyThread(num);
		t.start();
        try{
           t.join();
        }catch(Exception e){
          e.printStackTrace();   
        }
	}
}


class MyThread  extends Thread   {
    private int a;
    int[] t=new int[100];
    MyThread( int num){
    this.a = num;
    }
    public void run(){
       if(a==0)
       {
           t[0]=0;
           System.out.println("子线程计算结果为："+t[0]);
       }
       else if(a==1)
       {   
           t[1]=1;
           System.out.println("子线程计算结果为："+t[1]);
       }
       else if(a>=2)
       {
           t[0]=0;t[1]=1;
           for(int i=2;i<=a;i++)
           {
               t[i]=t[i-1]+t[i-2];
           }
           System.out.println("子线程计算结果为："+t[a]);
       }
    }
	
	

}
