package Step3;

public class MultiplyTable {
	public static void main(String args[]){
		/********** Begin *********/
			for(int k = 1;k<=9;k++){         //外循环用于控制行数        

        for(int j = 1;j<=k;j++){             

           System.out.print(" "+k+"×"+j+"="+(j*k));     //"\t"为制表符 

               }    

                System.out.println();  //换行  

		 /********** End *********/
            }
    }
}
