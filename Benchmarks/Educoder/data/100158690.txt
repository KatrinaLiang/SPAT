#include<stdio.h>
#define N 1001
int main(void){
	int t;
	int a[N];
    char c;
    int i=0;
    while(1)
    {
        scanf("%d",&a[i]);
        c=getchar();
        i++;
        if(c=='\n')break;
    }
    int len=i;
	scanf("%d",&t);
	int count=0;
	for(int i=0;i<len;i++){
		for(int j=i+1;j<len;j++){
			for(int k=j+1;k<len;k++){
				if(a[i]+a[j]+a[k]<t)count++;
			}
		}
	}
	printf("%d",count);
	return 0;
}
