package step2;
import java.util.Scanner;
public class InterRectangle {
		public static double changdu(double width,double length) {
			return (width+length)*2;
		}
		public static double mianji(double width,double length) {
			return width*length;
		}
		public static void main(String[] args) {
			  Scanner sin=new Scanner(System.in);
			  double n=sin.nextDouble();
                         double[]width=new double[100];	
			 double[]length=new double[100];
			 for(int i=0;i<n;i++){
                           width[i]=sin.nextDouble();
			   length[i]=sin.nextDouble();
             }
   		          double a=0;double b=0;
                          for(int i=0;i<n;i++)
{                         a+=mianji(width[i],length[i]);
                          b+=changdu(width[i],length[i]);
                               }
			  System.out.println("总共有"+n+"块土地");
			  System.out.println("土地总面积为:"+a+"平米");
			  System.out.println("土地总周长为:"+b+"米");
		}
	}
