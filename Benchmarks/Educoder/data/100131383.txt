package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int max=num1>num2?num1:num2;
        int min=num1<num2?num1:num2;
        int sum=0;
        for(int i=min;i<=max;i++){
            int cur=i;
            while(cur>0){
                int p=cur%10;
                cur/=10;
                sum+=p;
            }
        }
		return sum;
		/********** END **********/
	}
}
