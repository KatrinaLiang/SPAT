package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
	 InterRectangle(double width2,double length2){
	        this.width = width2;
	        this.length = length2;
	    }
	
	    double perimeter() {
	    	return   (this.width+this.length)*2;
	    }
	    double area() {
	    	return (this.width*this.length);
	    }
	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
		double totalArea=0,totalPerimeter=0;
		double width,length;
		for(double i=0;i<total;i++) {
			width = scanner.nextDouble();
			length = scanner.nextDouble();
			totalArea  +=new InterRectangle(width,length).area();
			totalPerimeter += new InterRectangle(width,length).perimeter();
		}
		System.out.println("总共有"+total+"块土地");
		System.out.println("土地总面积为:"+totalArea+"平米");
		System.out.println("土地总周长为:"+totalPerimeter+"米");
		/*********end*********/
	}

}
