package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
		int h,l;
        for(h=1;h<=10;h++){
            for(l=1;l<=h;l++){
                System.out.print("*");
            }
            System.out.println();
        }
		//在这里打印出正三角形 
		
		
		
		
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
		for(h=1;h<=10;h++){
            for(l=10;l>=h;l--){
                System.out.print("*");
            }
            System.out.println();
        }
		
		
		
		
		/*****end*****/
	}
}
