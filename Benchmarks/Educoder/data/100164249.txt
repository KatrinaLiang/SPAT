package step3;

import java.util.Scanner;
public class Number_Judge {
		public static void main(String[] args) {
			Scanner scan1=new Scanner(System.in);
			/******Begin******/
			int number1=scan1.nextInt();
            if(number1%2==0){
            	System.out.print("偶数");
            }else if(number1%2!=0){
            	for(int i=2 ; i<=number1;i++){    
            		if(number1% i == 0&& i != number1){
             			System.out.print("奇数");
           				break;
           		}else if(i == number1){
           			 System.out.print("素数");
            	}
                	
            	}
            }
			/******End******/
		}
}
