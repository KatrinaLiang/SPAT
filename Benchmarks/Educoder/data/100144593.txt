package step1;
public class Task {
	public String inversion(String str){
		/********* Begin *********/
		int length = str.length();
		if (length <= 1){
		   return str;
		  }
		String left = str.substring(0, length / 2);
		String right = str.substring(length / 2, length);
		  
        return inversion(right) + inversion(left);
		
		
		/********* End *********/
	}
}
