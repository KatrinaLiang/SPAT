public class Task {

	public int getNum(int n){
		/********* Begin *********/
		
		    int m=0;
while(n!=0)
{
    if(n==1){
        m++;break;
    }
    if(n%2==0){
        n=n/2;
    }
    if(n%2==1){
        n=n-1;
        n=n/2;
        m++;
    }
    
}
return m;

	}
	
}
