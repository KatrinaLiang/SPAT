package Q103;
import java.util.*;
public class Main {

	public static void main(String[] args) {
		Scanner in=new Scanner(System.in);
		int n=in.nextInt();
		if(n==1||n==2) {
			System.out.printf("1");
			return;
		}
		
		int k1=1,k2=2,k3=0;//斐波拉契数列
		for(int i=3;i<=n;i++) {
			k3=k1+k2;
			k1=k2;
			k2=k3;//迭代
		}
		System.out.print(k3);

		// write your code here
	}

}
