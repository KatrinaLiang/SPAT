package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        
        int a,b,c,s=0,i;
		for(i=num1;i<=num2;i++)
        {
            a=i/100;
            b=i/10%10;
            c=i%10;
            s=a+b+c+s;
        }
		return(s);
		/********** END **********/
	}
}
