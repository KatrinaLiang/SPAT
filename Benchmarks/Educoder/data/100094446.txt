package step1;

public class DecomposeFactor {
	public void decompose(int score) {
		/********** Begin *********/
			double s = 0;
        if (score <0)
            s = 0.0;
        else if(score>90)
            s = 4.0;
        else if (score > 80)
            s = 3.5;
        else if(score > 70)
            s = 3.0;
        else if(score>60)
            s = 2.5;
        else
            s = 0.0;
        
        System.out.print(s);


		/********** End *********/
	}
}
