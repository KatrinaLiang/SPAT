package task5;

public class Box {

    double width;
    double height;
    double depth;

    //TODO your code
	public Box(double width, double height, double depth){
    	this.width = width;
        this.height = height;
        this.depth = depth;
    }
	public double volume() {
    	double v = width * height * depth;
        return v;
    }

}
