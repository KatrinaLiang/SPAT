package step4;
		//创建类 添加属性和方法
class WuMingFen{
    String theMa;
    int quantity;
    boolean likeSoup;
    public WuMingFen(String ma,int q,boolean soup)
    {
        theMa=ma;
        quantity=q;
        likeSoup=soup;
    }
    public WuMingFen(String ma,int q)
    {
        theMa=ma;
        quantity=q;
        likeSoup=false;
    }
    public WuMingFen()
    {
        theMa="酸辣";
        quantity=2;
        likeSoup=true;
    }
    public void check()
    {
        System.out.println("面码："+theMa+"，粉的份量："+quantity+"两，是否带汤："+likeSoup);
    }
}
