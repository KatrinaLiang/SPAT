package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum = 0;
        int n = num1;
 		
        while( n <= num2 ){
            int m = n;
        	for( int i = 1; i <= 3; i ++ ){
        	    sum =sum + m%10;
        	    m = m/10;
     	   }
            n++;
        }
		return sum;
		/********** END **********/
	}
}
