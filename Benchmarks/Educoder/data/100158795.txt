package Q103;
import java.util.Scanner;
public class Main {

	public static void main(String[] args) {
				int a[]=new int[100];
		Scanner in=new Scanner(System.in);
		int n;
		n=in.nextInt();
		in.close();
		int result;
		 a[0]=1;
		 a[1]=2;
		for(int i=2;i<=n-1;i++){
			a[i]=a[i-1]+a[i-2];
		}
		result=a[n-1];
		System.out.print(result);
	}

}
