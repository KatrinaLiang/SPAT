package Pid_108;
import java.util.Scanner;
public class Main {

	public static void main(String[] args) {
				Scanner in=new Scanner(System.in);
        int n=in.nextInt();
        int a=1;
        for(int i=1;i<=n;i++){
        		n=n-i;
        		if(n>=i+1)
        			a++;
        }
        System.out.print(a);
	}

}
