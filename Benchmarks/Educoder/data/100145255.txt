package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int s = 0;
        while (n / 2 != 0) {
            if (n % 2 != 0) {
                n = n - 1;
                s++;
            } else {
                n = n / 2;
                if(n==1)
                    s++;
            }
        }
        return s;
		
		
		
		
		/********* End *********/
	}
	
}
