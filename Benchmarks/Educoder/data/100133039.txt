package step1;

public class Task {

	public static void birthdayCandle(int num){
		int a=0;
		int b=0;
		game: for(a=18;a<=num;a++) {
				for(b=a;b<=num;b++) {
					if((a+b)*(b-a+1)/2==num) {
						break game;
				}
			}
		}
		System.out.println("小明今年：" +b+ "岁");
		System.out.println("小明从" +a+ "岁开始每年举办派对");
	}
}
