package step5;

import java.util.Scanner;

public class HelloWorld {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		System.out.println("请给定一个自然数N：");
		int n = sc.nextInt();//获取输入的整数N
		int sum=1,j;
      for(j=1;j<=n;j++){
            sum=sum*j;} 
		System.out.println("自然数N的阶乘为" + sum);
	}
}
