package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int i,j;
		for(i=1;i<10;i++)
		{
			for(j=1;j<i+1;j++)
			System.out.print(j+"*"+i+"="+j*i+"\t");
			System.out.println();
		}
		
		
		
		
		
		/*****end*****/
	}
}
