package step2;

import java.util.Scanner;

public class Task {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int length = sc.nextInt();	//长
		int width = sc.nextInt();	//宽
		/********** BEGIN **********/
		int a = length;
        int b = width;
		int c = a % b;
        while(c!=0)
        {
            a = b;
            b = c;
			c = a % b;
        }
		System.out.println(b);
		System.out.println( (length* width) / ( b* b) );
		/********** END **********/
	}
}
