class Person {
	/********* begin *********/
	public String name;
    public int age;
    public void setAge(int age,String name){
        if(age>120||age<0){
            System.out.println("ERROR:error age intput...");
        }else{
            this.age=age;
        }
    
    
        this.name=name;
    }
	/********* end *********/
}

class Student extends Person {
	/********* begin *********/
	public String school;
    public Student(){
        super();
    }
    public void setSchool(String school){
        this.school=school;
        
    }
    public void talk(){
        System.out.println("姓名：" + super.name + "，年龄：" + super.age + "，学校：" + school);
    }
	/********* end *********/
}
