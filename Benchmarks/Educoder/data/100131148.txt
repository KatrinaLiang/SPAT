package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int delta = num2 - num1;
        int sum = 0;
        for(int i = 0; i < delta + 1; i++){
            int num = i + num1;
      	    int subSum = 0;
            String stringNum = String.valueOf(num);
            for(int j = 0; j < stringNum.length(); j++){
                int temp = Integer.parseInt(stringNum.substring(j, j+1));
                subSum += temp;
        	}
            sum += subSum;
    	}
        return sum;
		/********** END **********/
	}
}
