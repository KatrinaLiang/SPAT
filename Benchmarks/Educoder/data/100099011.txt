package step1;

public class Yard {
	double length;
    double width;
    public Yard(){}
    public Yard(double length,double  width)
    {
    	this.length=length;
        this.width=width;
      
    }
    public double getPerimeter()
    {
    	return 2*width +length*2;
    }
    public double getAcreage()
    {
    	return length*width;
    }
    
    
}
