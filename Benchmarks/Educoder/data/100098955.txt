package step1;

public class Yard {
    
    private int length;
    private int width;
    
	Yard(int l, int w)
    {
        length = l;
        width = w;
    }
    
    public double getPerimeter()
    {
        return 2 * (length + width);
    }
    
    public double getAcreage()
    {
        return length * width;
    }
}
