package step1;

import java.sql.*;

public class QueryPass {
	
	// 查询数据代码不用上实验报告
	public static void queryDB() {
		
		/********* Begin *********/
		try{
		// 第一步：加载驱动
		Class.forName("com.mysql.jdbc.Driver");
		

		// 第二步：建立连接, "root"和"123123"是针对MySQL设置了用户名(root)和密码(123123)的情况
		// 127.0.0.1:3306是mysql服务器地址及端口   数据库编码格式设置为utf-8
		Connection conn = DriverManager.getConnection("jdbc:mysql://127.0.0.1:3306/tsgc?useUnicode=true&characterEncoding=UTF-8","root", "123123");
		

		// 第三步：建立statement对象
		Statement state = conn.createStatement();
		

		// 第四步：修改数据
		String sql ="select * from employee;";
        ResultSet rs = state.executeQuery(sql);
            while(rs.next()){
            	System.out.print("no:"+rs.getString(1));
                System.out.print("\tname:"+rs.getString(2));
                System.out.print("\tpassword:"+rs.getString(3));
                System.out.print("\tsex:"+rs.getString(4));
                System.out.print("\tsalary"+rs.getFloat(5));
                System.out.println();
            }
            rs.close();
        	state.close();
        	conn.close();
       
        }catch(Exception e){
        	e.printStackTrace();
        }
		

		// 第五步：关闭statement对象和连接对象
		
		

		/********* End *********/
	}
}
