package step1;

public class Test {
	public static void main(String[] args) {
		Dog mydog = new Dog("五花肉");//创建Dog对象
		mydog.Color("棕色");//设置Dog对象的属性
		mydog.Type("阿拉斯加");
		//输出小狗的属性
		System.out.println("名字：" +mydog.getName()+ "，毛色：" +mydog.getColor()+ "，品种：" +mydog.getType());
		
		//调用方法
        mydog.hab();
        mydog.act();
		
	}
}

//在这里定义Dog类
class  Dog {
    public Dog(String name){
  		this.name = name; 
    }
    public String getName(){
        return name;
    }
    public void Color(String color){    
        this.color = color;
    }
    public String getColor(){
        return color;
    }
    public void Type(String type){
        this.type = type;
    }
    public String getType(){
        return type;
    }
    public void hab(){
        System.out.println("啃骨头");
    }
    public void act(){
        System.out.println("叼着骨头跑");
    }
    private String name;
    private String color;
    private String type;
}
