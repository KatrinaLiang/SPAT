package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
		int sum=0;
        do{
            int m=a;
			for(int i=0;i<n-1;i++){
            m=m*10+a;
            }
		sum=m+sum;
        n--;
        }while(n!=0);
        System.out.print(sum);
        

		/********** End *********/
	}
}
