package step3;

import java.util.Scanner;

public class Test {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		String name = sc.next();
		int age = sc.nextInt();
		String sex = sc.next();
		Person p = new Person(name,age,sex);
		p.display(name,age,sex);
	}
}

class Person{
	String name = "张三";
	int age = 18; 
	String sex = "男";
	
	public Person(String name,int age,String sex){
        //this();
		this(age);
		this.name = name;
		this.sex = sex;
	}
	
	public Person(int age){
        //this();
		this.age = age;
	}
	
	public void display(String name,int age,String sex){
		this.name = "李四";
		this.age = 11;
		this.sex = "男";
		System.out.println("name：" + name);
		System.out.println("age：" + age);
		System.out.println("sex：" + sex);
	}
}
