package step1;

public class Yard {  
    private int length;
	private int width;
	public Yard(){
        this.length = 0;
        this.width = 0;
    }
	public Yard(int length,int width){
        this.length = length;
        this.width = width ;
    }
	public float getPerimeter(){
        return 2 * (length + width);
    }
	public float  getAcreage (){
        return length * width;
    }
}
