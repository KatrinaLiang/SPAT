package step1;

public class Task {
	
	public void sortArray(int[] arr){
        for(int i=1;i<arr.length;i++){
            int n = i;
            while(arr[n]<arr[n-1]){
                int temp = arr[n];
                arr[n] = arr[n-1];
                arr[n-1] = temp;
                n--;
                if(n==0)
                    break;
            }
            for(int a:arr){
                System.out.print(a+" ");
            }
            System.out.println();
        }
        

		
		
	}
}
