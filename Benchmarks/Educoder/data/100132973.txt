package step1;

public class Task {
	
	public void sortArray(int[] arr){
		for (int i = 1; i < arr.length; i++) {
			int x = arr[i];
			int j = i;
			for (; j > 0; j--) {
				if (x < arr[j - 1]) {
					arr[j] = arr[j - 1];
				} else {
					break;
				}
			}
			arr[j] = x;
			for(int a:arr) {
				System.out.print(a+" ");
			}
			System.out.println();
		}
		
		
	}
}
