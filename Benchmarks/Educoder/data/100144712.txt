package step1;

public class Task {
	public String inversion(String str){
		/********* Begin *********/
		StringBuffer c = new StringBuffer();
		char [] d = str.toCharArray();
		for(int i=d.length-1;i>=0;i--) {
			c.append(d[i]);
		}
		return c.toString();
		
		
		/********* End *********/
	}
    
}
