package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		
        
        int count=0;
        for(int i=num1;i<=num2;i++){
            int sum=0;
            int t=i;
            while(t>0){
            sum=sum+t%10;
            t=t/10;
            }
            count=count+sum;
        }
        return count;
		
		/********** END **********/
	}
}
