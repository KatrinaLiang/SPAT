public class HelloWorld {
    public static void main(String[] args) {
HelloWorld cls=new HelloWorld();
        
cls.wucan();
        // 调用无参的方法
        String names=new String("educoder");
cls.zfccs(names);
        // 调用带有一个字符串参数的方法
        int sum=666;
cls.zxcs(sum);
        // 调用带有一个整型参数的方法
        
    }
    public void wucan(){
        System.out.println("无参的print方法");
    }
    // 无参print方法的定义
    public void zfccs(String names){
        System.out.println("带有一个字符串参数的print方法，参数值为："+names);
    }
    // 字符串print方法的定义
    public void zxcs(int sum){
        System.out.println("带有一个整型参数的print方法，参数值为："+sum);
    }
    
    // 整型print方法的定义
    
}
