package step3;

import java.util.Arrays;
import java.util.Scanner;

public class RecordRecovery {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int[] record = new int[10];
        int[] recordRecovery = new int[10];
        /********** Begin *********/
       
        int q=0;
        for(int j=0;j<10;j++)
        {
            int numCandy = scanner.nextInt();
            record[j]=numCandy;
        }
        int j=1;
        while(j<10)
        {
            for(j=0;j<10;j++)
            {
                if(j==0)
                {
                    recordRecovery[q]=record[j];
                    q++;
                }
                if(j>0)
                {
                    int flag=0;
                    for(int y=0;y<q;y++)
                    {
                        if(record[j]==recordRecovery[y])
                        {
                            flag=1;
                            break;
                        }
                    }
                    if(flag==0)
                    {
                       recordRecovery[q]=record[j];
                        q++;
                    }
                }
                
            }
        }
        Arrays.sort(recordRecovery);
        int[] recordRecovery2 = new int[q];
        {
           for(int p=0,k=10-q;k<10;k++,p++)
           {
             recordRecovery2[p]=recordRecovery[k];
           }
        }
        
        /********** End *********/
        for (int i = 0; i < q; i++) {
            System.out.print(recordRecovery2[i]);
            if (i !=q-1) {
                System.out.print(' ');
            }
        }
    }
}
