package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
    int s=1;
    int m=0;
    int sum=0;
    int a=0;
	int[] arr=new int[20];
    for(int i=0;i<19;i++)
    {
        sum=s+m;
        arr[i]=sum;
        m=s;
        s=sum;
        a=a+arr[i];
        
    }
    System.out.print("前20个元素的和为"+(a+1));

	/********end********/
	}
}
