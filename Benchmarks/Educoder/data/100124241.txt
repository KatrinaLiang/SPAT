package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		
			int i,s=0;
			int[] f=new int[20];
			f[0]=1;
			f[1]=1;
			for(i=2;i<f.length;i++)
				f[i]=f[i-1]+f[i-2];

			for(i=0;i<f.length;i++)
                s=f[i]+s;
				System.out.print("前20个元素的和为"+s);

	/********end********/
	}
}
