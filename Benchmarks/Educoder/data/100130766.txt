package step5;

import java.util.Scanner;


public class Task {
	
	public int getDepth(TreeNode tree){
        /********** BEGIN **********/
        int node = 1;
        if(tree!=null){
            node = 1 + getNode(tree);
        }
        int depth = 0;
        while(node!=0){
            node = node/2;
            depth++;
        }


        return depth;
        /********** END **********/

    }

    private static int getNode(TreeNode node) {
        int num=0;
        if(node.left!=null){
            num = num + 1+ getNode(node.left);
        }
        if(node.right!=null){
            num = num + 1+  getNode(node.right);
        }
        return num;
    }
}
