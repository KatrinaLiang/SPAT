package step1;

public class Yard {
    public double length;
    public double width;
    public Yard(double length,double width) {
        this.length = length;
        this.width = width;
    }	
    public double getPerimeter() {
		return (length*2)+(width*2);
    }
    public double getAcreage() {
		return length*width;
    }
}
