public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
		Thread t=new MyThread(num);
        t.start();
		
		
	}
}


class MyThread   extends Thread  {

int i=3;
    int x=1;
    int y=1;
 private  int num;
    public MyThread(int num){
        this.num=num;}
    public void run(){
    if(num==2||num==1) System.out.println("子线程计算结果为：1");
        else{
            for(;i<=num;i++){
                int t;
                t=x+y;x=y;y=t;
                if(i==num) System.out.println("子线程计算结果为："+y);
        
        }
    }
	
    }}
