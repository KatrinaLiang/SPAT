package step2;

public class TestJava18_4 {

	public void print() {
		/********* Start ********/
	for(int a=2000,b=0;a<=3000;a++){
			if(a%4==0&&a%100!=0||a%400==0){//闰年条件
				System.out.print(a+"\t");//满足条件时输出a
				b++;//每输出一次a，b加1
				if(b%10==0){//条件当b是10的倍数时
					System.out.println();//满足条件时换行
				}
			}
		}
		/********* End ********/
	}

}
