package step1;

public class Rectangle{
	private double width;
	private double length;
	
	/*********begin*********/
    Double perimeter(){
        this.width=6.0;
        this.length=5.0;
        return width*2+length*2;
    }
    Double perimeter(double width,double length){
        this.width=width;
        this.length=length;
        return width*2+length*2;
    }
    Double area(){
        this.width=6.0;
        this.length=5.0;
        return width*length;
    }
    Double area(double width,double length){
        this.width=width;
        this.length=length;
        return width*length;
    }
	/*********end*********/


	public static void main(String[] args) {
		/*********begin*********/
        Rectangle r1=new Rectangle();
        Rectangle r2=new Rectangle();
        System.out.println("r1的周长："+r1.perimeter());
        System.out.println("r1的面积："+r1.area());
        System.out.println("r2的周长："+r2.perimeter(20,10));
        System.out.println("r2的面积："+r2.area(20,10));

		/*********end*********/
	}

}
