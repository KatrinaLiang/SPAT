package step1;
import java.util.Scanner;

public class Cal {
	public static void main(String[] args) {
		/*********start*********/
				int sum,sub,mul,div,rem;
		Scanner sc=new Scanner(System.in);
		System.out.print("请输入第一个整数\n");
		int a=sc.nextInt();
		System.out.print("请输入第二个整数\n");
		int b=sc.nextInt();
		sum=a+b;
		sub=a-b;
		mul=a*b;
		div=a/b;
		rem=a%b;
		System.out.print("两数相加的结果为："+ sum+"\n");
		System.out.print("两数相减的结果为："+ sub+"\n");
		System.out.print("两数相乘的结果为："+ mul+"\n");
		System.out.print("两数相除的结果为："+ div+"\n");
		System.out.print("两数取余数的结果为："+ rem+"\n");
		/*********end*********/
	}

}
