				if(isDj==true) {
					thisMax++;
				}else {
					isDj = false;
				}
				isDz = true;
			}else {
				thisMax = 1;
			}
			if(max<thisMax) {
				max = thisMax;
			}
		}
		if(max==1) {
			return 0;
