package Pid_101;
import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		// write your code here
        Scanner in = new Scanner(System.in);
		String strA, strB;
		
		strA = in.nextLine();
		strB = in.nextLine();
		
		int max = 0;
		for(int i=0;i<strA.length();i++){
			for(int j=0;j<strB.length();j++){
				int maxLength=0;
				if(strB.charAt(j) == strA.charAt(i)){
					for(int z=j;z<strB.length();z++)
						if(i+(z-j)>=strA.length()  || strB.charAt(z) != strA.charAt(i+(z-j)))
							break;
						else if(strB.charAt(z) == strA.charAt(i+(z-j)))
							maxLength++;
				}
				if(maxLength > max)
					max = maxLength;
			}
				
		}
		
		
		
		System.out.print(max);
	}

}
