package step2;

import java.util.Scanner;


public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
        MyThread t = new MyThread(num);
        t.start();
        try{
            t.join();
        }catch(Exception e) {
            e.printStackTrace();
        }
        System.out.println("子线程计算结果为：" + t.getValue());
	}
}

class MyThread extends Thread{
    private int value = 0;
    private int num;
    
    public MyThread(int num) {
        this.num = num;
    }
    
    public void run(){
        value = fibonacci(num);
    }
    public int fibonacci(int num){
        if(num == 0) return 0;
        if(num == 1) return 1;
        else return fibonacci(num-1)+fibonacci(num-2); 
    }
    public int getValue(){
        return value;
    }
}
