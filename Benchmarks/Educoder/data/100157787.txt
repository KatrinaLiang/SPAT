
import java.util.*;
public class Main {

	public static void main(String[] args) {
        Scanner in=new Scanner(System.in);
        int x=in.nextInt();
		System.out.print(zhuan(x));
	}
    public static int zhuan(int n){
            if(n==0||n==1)
                return 1;
            else
                return zhuan(n-1)+zhuan(n-2); 
        }

}
