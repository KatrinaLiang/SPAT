package step1;

public class ShuShu {

	// 第一题
	public void first() {
		/*********Start********/
        
        int k=1;
        while(k<=100)
        {
        System.out.print(k+"\t");
            if(k%10==0)
            {
            System.out.println("");
            }
            k++;                                                                                                                                                                                                          
        }
		
		/*********End********/
	}
	
	// 第二题
	public void second() {
		/*********Start********/
        System.out.println("");
        int k=1;
        do{
        System.out.print(k+"\t");
            
            if(k%10==0)
            {
            System.out.print("\n");
            }
            k++;
        }while(k<=100);
            
		
		/*********End********/
	}
	
	// 第三题
	public void third() {
		/*********Start********/
        System.out.println("");
        int n=0,x=1;
        for(;x<=100;x++)
        {
        if(x>100){
        break;
        }System.out.print(x+"\t");
            n++;
            if(n%10==0){
            System.out.print("\n");
            }
        }

		
		/*********End********/
	}
}
