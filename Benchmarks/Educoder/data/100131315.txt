package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int num,sum=0;
        while(num1<=num2){
        	num=num1;
			while(num!=0){
            	sum=sum+num%10;
            	num=num/10;
        	}
            num1++;
        }
		return sum;
		
		/********** END **********/
	}
}
