package step1;

import java.util.Scanner;

public class TestJava20_2 {
	static final Scanner sc = new Scanner(System.in);	
	public void print() {
        System.out.println("请输入倒直角三角形的行数：");
		int a=sc.nextInt();   //n表示行数
		for(int i=1;i<=a;i++){
            for(int n=a;n>=i;n--){
                System.out.print("*"+"\t");
            }System.out.println();
        }
            
	}

}
