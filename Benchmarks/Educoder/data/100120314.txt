package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		for(int cai=1;cai<=9;cai++){
        	for(int yy=1;yy<=cai;yy++){
            	System.out.print(yy+"*"+cai+"="+(yy*cai)+"\t");
            }
             System.out.print("\n"); 
        }
        
		/*****end*****/
	}
}
