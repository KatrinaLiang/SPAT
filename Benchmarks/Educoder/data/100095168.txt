package step3;

public class Factorial {
	public void factorial(int n) {
		/********** Begin *********/
        if(n!=0){
        int sum = 1;
        for(;n>0;n--){
            sum = sum * n;  
        }
            System.out.print(sum);
        }
        else
            System.out.print("0");


		/********** End *********/
	}
}
