package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        int ans=0,res=a;
		for(int i=1;i<=n;i++)
        {
            ans+=res;
            res=res*10+a;
        }
        System.out.print(ans);
		/********** End *********/
	}
}
