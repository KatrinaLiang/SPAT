package step3;

public class CatAndDog {
    public static void main(String[] args) {
        /********** Begin *********/
        Animal cat= new cat;
        Animal dog= new dog;
        /********** End *********/
        cat.eat();
        cat.work();
        dog.eat();
        dog.work();
    }
}

abstract class Animal {
    abstract void eat();
    abstract void work();
}

/********** Begin *********/
class Cat extends Animal {
	void eat(){
	System.out.print("吃鱼");	} 
	void work(){
	System.out.print("抓老鼠");	
	} 
	
	

}

class Dog extends Animal {
	void eat(){
		System.out.print("吃骨头");
	}
void work(){
	System.out.print("看家");
}
}
/********** End *********/
