package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
       int[] result;
        result = new int[temperatures.length];
		int i,j,count = 0;
    for(i = 0;i < temperatures.length;i++){
        count = 0;
        for(j = i + 1;j < temperatures.length;j++){
        count++;
        if(temperatures[j] > temperatures[i]){
             result[i] = count;
             break;
         }
        }
    }
		return result;
		/********** END **********/
	}
}
