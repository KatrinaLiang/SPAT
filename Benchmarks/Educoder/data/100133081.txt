package step2;

import java.util.Arrays;

public class Task {

    public int getTheWay(int num,int bagCapacity,long[] snacks){
        Arrays.sort(snacks);
        int ways = 0;
        long bags = 0;
        boolean[] flag = new boolean[snacks.length];
        int i= 0;
        while(i!=-1){
            if(put(i,bags,bagCapacity,snacks)) {
                flag[i]=true;
                bags+=snacks[i];
                i++;
            }else{
                ways++;
                while(--i!=-1&&!flag[i]);
                if(i==-1) return ways;
                flag[i]=false;
                bags-=snacks[i];
                i++;
            }
        }
        return ways;

    }
    boolean put(int i,long bags,int bagCapacity,long[] snacks){
        if(i==snacks.length||bags+snacks[i]>bagCapacity) return false;
        else return true;
    }
}
