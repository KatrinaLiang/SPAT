
package step3;
import java.util.Scanner;
public class Sum{
	public static void main(String args[]){
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
        int result = recursionSum(num);
        System.out.println(result);
	}
	
    public static int recursionSum(int mum){
        /*********begin*********/
        int i=0,sum=0;
        //int a[]=new int[mum];
   		while(i<mum){
            //a[i]=i+1;
            i++;
            sum=sum+i;
        }
        
        return sum;
        /*********end*********/
    }  
}
