package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
        	/********begin********/
        int []arr=new int[20];
     arr[0]=1;
     arr[1]=1;
        
	for(int a=2;a<arr.length;a++){
	
		arr[a]=arr[a-1]+arr[a-2];
	}
	int s=0;
	for(int i=0;i<arr.length;i++){
	s=s+arr[i];	
	}
      System.out.print("前20个元素的和为"+s);  
        

	/********end********/
	}
}
