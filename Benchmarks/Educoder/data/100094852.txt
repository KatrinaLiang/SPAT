package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        double sum=0;
		for(int i=n;i>0;i--)
        {
            double s=0;
            double x;
        	for(int j=i;j>0;j--)
            {
            	x=Math.pow(10,j-1)*a;
                s+=x;
            }
            sum+=s;
        }
		System.out.print((int)sum);
		/********** End *********/
	}
}
