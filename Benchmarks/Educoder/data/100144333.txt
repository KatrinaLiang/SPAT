package step4;

import java.io.File;
import java.io.FileFilter;

public class Task {
	
	/********** Bgin **********/
	public void showDirStructure(File file)	{
		if (file.isDirectory()) {
			System.out.println("+--" + file.getName());
		}
		int Blank = 2;
		haha(file, Blank);






	}
	public static void haha(File file, int Blank ) {
        FileFilter filter = filek -> {
      if (filek.isFile()) {
        String fileName = filek.getName().toLowerCase();
        if (fileName.endsWith(".jpg") || fileName.endsWith(".bmp") || fileName.endsWith(".png")) {
          return true;
        }
      }
      if(filek.isDirectory())
          return true;
      return false;
    };
		File[] files = file.listFiles(filter);
		for (int i = 0; i < files.length; i++) {
			if (files[i].isDirectory()) {
				for (int k = 0; k < Blank; k++) {
					System.out.print(" ");
				}
				System.out.println("+--" + files[i].getName());
				haha(files[i], Blank+2);//通过迭代不断调用文件,用blank控制前面的空格数
			} else {
				for (int k = 0; k < Blank; k++) {
					System.out.print(" ");
				}
				System.out.println("--" + files[i].getName());
			}
		}
    }

	/********** End **********/
}
