package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
        int N=matrix.length;
		//调换反对角线元素
		for(int i=0;i<N-1;i++) {
			for(int j=0;j<N-1-i;j++) {
				int t=matrix[i][j];
				matrix[i][j]=matrix[N-1-j][N-1-i];
				matrix[N-1-j][N-1-i]=t;
			}
		}
		//调换i行和n-i-1行元素
		for(int i=0;i<N/2;i++) {
			for(int j=0;j<N;j++) {
				int t=matrix[i][j];
				matrix[i][j]=matrix[N-i-1][j];
				matrix[N-i-1][j]=t;
			}
		}
		
		/********* End *********/
	}
}
