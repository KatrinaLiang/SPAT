package step1;

public class Person {
    private String name; //姓名
    private String number; //身份证号
    private Date birthday; //出生日期
    
   /************Begin*************/
    public Person(){this.birthday=new Date();}
	public Person(String name,String num,Date birthday){
    this.name=name;
    this.number=num;
    this.birthday=birthday;
    }
	public Person(String name,String num,int year,int month,int day){
    this.name=name;
    this.number=num;
    this.birthday=new Date(year,month,day);
    }
    public String toString(){
        return "name="+this.name+" number="+this.number+" birthday:"+birthday;
    }
	
            public String getName() {
            return name;
        }

        public void setName(String name) {
            this.name = name;
        }

        public String getNumber() {
            return number;
        }

        public void setNumber(String number) {
            this.number = number;
        }

        public Date getBirthday() {
            return birthday;
        }

        public void setBirthday(Date birthday) {
            this.birthday = birthday;
        }
    
    
    
    
   /************End*************/
}
