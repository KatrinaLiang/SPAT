package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int Fibonacci[]=new int[20];
        Fibonacci[0]=1;
         Fibonacci[1]=1;
        int sum=0;
        for(int i=2;i<20;i++)
             Fibonacci[i]= Fibonacci[i-1]+Fibonacci[i-2];
        for(int k=0;k<=19;k++)
            sum=sum+Fibonacci[k];
        System.out.print("前20个元素的和为"+sum);

	/********end********/
	}
}
