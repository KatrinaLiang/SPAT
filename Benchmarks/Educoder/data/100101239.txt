package step2;

import java.util.Scanner;

public class SystemUpdate {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String name = scanner.next();
        String birth = scanner.next();
        String position = scanner.next();
        double salary = scanner.nextDouble();
        Salary employee = new Salary(name, birth, position, salary);
        employee.introduction();
    }
}

/********** Begin *********/
class Employee {
    protected String name;
    protected String birth;
    protected String position;
    public Employee(String n, String b, String p) {
        name = n;
        birth = b;
        position = p;
    }
}

class Salary extends Employee {
    private double salary;
    public Salary(String n, String b, String p, double s) {
        super(n, b, p);
        salary = s;
    }
    public void introduction() {
        System.out.print("员工姓名：");
        System.out.print(super.name);
        System.out.print(" 出生年月：");
        System.out.print(super.birth);
        System.out.print(" 职位：");
        System.out.print(super.position);
        System.out.print(" 薪水：");
        System.out.print(salary);
    }
}
/********** End *********/
