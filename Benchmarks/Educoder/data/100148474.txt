package step2;

/********* Begin *********/
//定义站台类，实现卖票的功能。
public class Station extends Thread    {	 
	public static int ticket=20;
	public Object ob="1";
	public void run() {
		while(ticket>0) {
			synchronized(ob) {
				if(ticket>0) {
					System.out.println("卖出了第"+ticket+"张票");
					ticket--;
				}else
					System.out.println("票卖完了");
			}
			try {
				sleep(1000);
			}
			catch(InterruptedException e) {
				System.out.println(e);
			}
		}
	}
}
/********* End *********/
