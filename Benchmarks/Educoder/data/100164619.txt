package step2;

public class Equation {
		double a, b, c,d; 
    double  X1,X2;
                 //一元二次方程的三个系数
		public Equation(){}
		public Equation(double a,double b,double c){
			this.a=a;
            this.b=b;
            this.c=c;
		}
		/*************Begin*******************/
	    //YOUR CODE HERE 
        
         
       
        public double getX1()
        {
            d=Math.sqrt(b*b-4*a*c);
            X1=(-b+d)/(2*a);
            return X1;
        }
         public double getX2()
        {
            d=Math.sqrt(b*b-4*a*c);
            X2=(-b-d)/(2*a);
             return X2;
        }
        public double deta()
           {
              return b*b-4*a*c;
           }

       
    
          
    	  /***********End*******************/
		
		
	}
