package step1;

public class Task {

	 public double solve(double x1, double y1, double r1,
              double x2, double y2, double r2) {
		  double juli1=Math.pow(x2-x1, 2)+Math.pow(y2-y1, 2);
        double d=Math.pow(juli1,1/2);	 
         double sum=0.0;
         
         if(d>r2+r1)
         {
             sum=Math.PI*r1*r1+Math.PI*r2*r2;
         }
         else
         {
        double a1=2*Math.acos((r1*r1+d*d-r2*r2)/2/r1/d);
        double a2=2*Math.acos((r2*r2+d*d-r1*r1)/2/r2/d);
        sum=r1*r1*a1/2+r2*r2*a2/2-r1*r1*Math.sin(a1)/2-r2*r2*Math.sin(a2)/2;
         }
         return sum; 
         
	  }
}
