package step1;

public class Yard {
double length;
double width;
     public Yard(double length,double width)
    {
        this.length=length;
        this.width=width;
}
    
    public double getAcreage()
    {
        return length*width;
    }
    public double getPerimeter()
    {
        return 2*length+2*width;
    }
 
}
