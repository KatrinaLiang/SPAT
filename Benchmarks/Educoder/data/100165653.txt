package step3;
import step1.Date;
public class Course {
    private int code; //课程编号
    private String courseName; //课程名称
    private Date testTime;//考试日期
    private int score;//考试成绩
   /******Begin**************/
   	public Course(){
		super();
		this.testTime=new Date(1999,9,9);
		this.courseName="";
		this.score=0;
		this.code=0;
	}

	public Course(int code, String courseName, int year,int month,int day, int score) {
		super();
		this.code = code;
		this.courseName = courseName;
		this.testTime = new Date(year,month,day);
		this.score = score;
	}
   	public Course(int code, String courseName, Date testTime, int score) {
		super();
		this.code = code;
		this.courseName = courseName;
		this.testTime = testTime;
		this.score = score;
	}

	public int getCode() {
		return code;
	}
	public void setCode(int code) {
		this.code = code;
	}
	public String getCourseName() {
		return courseName;
	}
	public void setCourseName(String courseName) {
		this.courseName = courseName;
	}
	public Date getTestTime() {
		return testTime;
	}
	public void setTestTime(Date testTime) {
		this.testTime = testTime;
	}
	public int getScore() {
		return score;
	}
	public void setScore(int score) {
		this.score = score;
	}
    
	public String toString(){
		if (this.code!=0){
			return String.format("%-4d", this.code)+this.courseName+" "+this.testTime+" "+this.score;            }
		else{
 			return this.code+"  "+this.courseName+this.testTime+" "+this.score;            }
		}
	


 

   /******End************/

}
