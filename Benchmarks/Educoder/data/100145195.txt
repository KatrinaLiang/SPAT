package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int m,ttao;
        for(m=0,ttao=n;ttao>1;)
        {
            if(ttao%2==1)
            {
                ttao=(ttao-1)/2;
                      m++;
            }
		else
            ttao/=2;
        }
		if(ttao==1)
            m++;
        
		return m;
    }
		/********* End *********/
	}
