package step2;
import java.util.Scanner;
public class SumOfDigit {
		public static void main(String[] args) {
			int number;
			int sum=0;
			Scanner scan1=new Scanner(System.in);
	        number=scan1.nextInt();
			/******Begin******/
			while(true){
				int add=number%10;
				sum+=add;
				number=number/10;
				if(number==0){
					break;
				}
			}
			System.out.print(sum);
	         
	        
	                    
			/******End******/
		}
}
