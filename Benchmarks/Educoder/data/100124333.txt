package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
        int arr[] = new int[50];
        arr[0] = 0;
        arr[1] = 1;
        int sum = 1; 
        for(int i = 2 ; i <= 20 ; i++ ){
            arr[i] = arr[i-1] + arr[i-2];
            sum += arr[i];
        }
        System.out.print("前20个元素的和为"+sum);

	/********end********/
	}
}
