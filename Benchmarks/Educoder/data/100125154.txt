package step1;

public class Rectangle{
	private double width;
	private double length;	
	/*********begin*********/
    public Rectangle(){
    this.width=5;
    this.length=6;
    }
    public Rectangle(double w,double h){
    this.width=w;
    this.length=h;
    }
    public double perimeter(){
   double perimeter = (this.width+this.length)*2;
        return perimeter;
    }
    public double area(){
    double area = this.width*this.length;
        return area;
    }
	/*********end*********/
	public static void main(String[] args) {
		/*********begin*********/
        Rectangle sc = new Rectangle();
        System.out.println("r1的周长："+sc.perimeter());
         System.out.println("r1的面积："+sc.area());
        Rectangle sc1 =new Rectangle(10,20);
         System.out.println("r2的周长："+sc1.perimeter());
         System.out.println("r2的面积："+sc1.area());
		/*********end*********/
	}

}
