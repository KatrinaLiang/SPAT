package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		 int num = 0;
        for(int i = num1;i<=num2;i++){
        	String s1 = i+"";
            int count = 0;
            for(int j = 0;j<s1.length();j++){
            	   char c = s1.charAt(j);
                	int i1 = c-'0';
                count += i1;
            }
            num += count;
        }
		
		
		/********** END **********/
       return num;
	}
}
