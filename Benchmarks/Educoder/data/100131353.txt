package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum=0;
        int seq=0;
        for(int i=num1;i<=num2;i++){
            int j=i;
        while(j!=0){
            seq=j%10;
            sum=sum+seq;
            j=j/10;
        }
        }
		return sum;
		/********** END **********/
	}
}
