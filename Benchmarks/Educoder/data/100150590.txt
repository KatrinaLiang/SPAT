package step3;
import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
			int a,b,c,d,e;
		Scanner scan=new Scanner(System.in);
        System.out.println("请输入一个五位数字：");
       int x=scan.nextInt();
     a=x/10000;
     b=(x/1000)%10;
     c=(x/100)%10;
     d=((x/10)%100)%10;
     e=(x%100)%10;
        System.out.println(x+"的各位数字之和是："+(a+b+c+d+e));
		/******End******/
	}
}
