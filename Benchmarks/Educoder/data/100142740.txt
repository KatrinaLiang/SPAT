package step3;

import java.io.*;
public class Task 
{
	private InputStream input;
    private OutputStream output;
    private static int length;
	public void task() throws IOException
    {
		/********* Begin *********/
		 input = new FileInputStream("src/step3/input/input.txt");
         output = new FileOutputStream("src/step3/output/output.txt");
		 length = input.available(); 
                // 创建缓存区域
         byte[] buffer = new byte[length];
                // 将文件中的数据写入缓存数组
         input.read(buffer);
                 // 将缓存数组中的数据输出到文件
         output.write(buffer);
		/********* End *********/		
	}
}
