package step2;

public class Task {
    
	public static int virusArea(int n,int m,char[][] area){
       
        /********* Begin *********/
        int[][] v = new int[100][100];
        char[][] newArray = new char[n+10][m+10];
        for(int i = 1; i <= n; i++){
            for(int j = 1; j <= m; j++){
                newArray[i][j] = area[i-1][j-1];
            }
        }
        int ans = 0;
        
        for(int i = 1; i <= n; i++){
            for(int j = 1; j <= m; j++){
                if(newArray[i][j] == 'o' && v[i][j] == 0){
                    dfs(i,j,v,newArray);
                    ans++;
                }
            }
        }

        return ans;


        /********* End *********/
    }

    public static void dfs(int x, int y, int[][] v, char[][] area){
        int[] dx = {1,0,-1,0};
        int[] dy = {0,1,0,-1};
        v[x][y] = 1;
        for(int i = 0; i < 4; i++){
            if(area[x+dx[i]][y+dy[i]]=='o'&&v[x+dx[i]][y+dy[i]]==0){
                v[x+dx[i]][y+dy[i]] = 1;
                dfs(x+dx[i],y+dy[i],v,area);
            }
        }
        return;
    }
}
