package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int i;
        int sum=0;
		int[] f=new int[20];
		
		for(i=0;i<f.length;i++){
            if(i==1||i==0){
            f[i]=1;
           continue;}
		
f[i]=f[i-1]+f[i-2];
            
        }
        for(i=0;i<f.length;i++){
        sum=sum+f[i];}
        
			
                 System.out.print("前20个元素的和为"+sum);
	

	/********end********/
	}
}
