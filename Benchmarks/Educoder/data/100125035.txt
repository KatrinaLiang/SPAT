package step1;

public class Rectangle{
	private static double width;
	private static double length;
	
	/*********begin*********/
    public Rectangle(int length,int width)
    {
        this.width = width;
        this.length = length;
    }
    public static double perimeter()
    {
        return (length+width)*2;
    }
    public static double area()
    {
        return length * width;
    }
    
	
	/*********end*********/


	public static void main(String[] args) {
		/*********begin*********/
		Rectangle rect1 = new Rectangle(6,5);
        System.out.println("r1的周长：" + rect1.perimeter());
        System.out.println("r1的面积：" + rect1.area());
        
        Rectangle rect2 = new Rectangle(20,10);
		System.out.println("r2的周长：" + rect2.perimeter());
        System.out.println("r2的面积：" + rect2.area());

		/*********end*********/
	}

}
