package step3;

public class Task {
	
	public int concaveValley(Integer[] arr){
		/********* Begin *********/
		int max = 0;
		int index = 0;
		boolean judge = false;
		for(int i = 0; i < arr.length - 1; i++) {
			index = 0;
			if(arr[i+1] < arr[i]) {
				int j = i;
				judge = false;
				while(j+1 < arr.length && arr[j+1] < arr[j]) {
					j++;
					index++;
				}
				while(j+1 < arr.length && arr[j+1] > arr[j]) {
					j++;
					index++;
					judge = true;
				}
				if(!judge) index = 0;
				else index++;
				if(index > max) max =  index;
			}
		}
		return max;
		/********* End *********/	
	}
}
