package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
		//assume : sum < Integer.Max_Value
        int sum = 0;
        
        for(int i=1;i<=n;i++){
         // 加法
            
            int a_i = 0; // i个a的值
            for(int j=0;j<i;j++){
                a_i = a_i *10 +a;
            }
            sum+= a_i;
        }
        System.out.print(sum);
		/********** End *********/
	}
}
