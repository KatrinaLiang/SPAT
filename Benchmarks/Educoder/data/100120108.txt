public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/

            for(int i = 1; i <= j; i++){

               System.out.print(i+"*"+j+"="+sum+"\t");}
             System.out.println();
        }
		
		
		
		
		
		/*****end*****/
	}
}
