package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        int sum=0;
        int i=0;
      
	do{
        int s=0;
		for(int j=0;j<=i;j++) {
			s=(int) (s+(a*Math.pow(10, j)));
		}
        sum=sum+s;
        i++;
    }while(i<n);
	System.out.print(sum);
		/********** End *********/
	
}
}
