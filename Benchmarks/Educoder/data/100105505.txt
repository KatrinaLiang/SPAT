package step1;

public class Yard {
    int length;
    int width;
    public Yard(int length,int width){
        this.width=width;
        this.length=length;
    
    
    }
    public float getPerimeter ()
    { 
   return((this.width+this.length)*2);
    
        
    }
    public float getAcreage()
    {
        return ( this.width*this.length);
        
    }
    
        
    
    
    
    }
