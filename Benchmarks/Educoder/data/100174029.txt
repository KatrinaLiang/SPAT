package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
	for(int i=1;i<10;i++){
		for(int j=1;j<=i;j++){
			int a=j*i;
			System.out.print(j + "*" + i + "=" + a +"\t");
		}
		System.out.println();
	}	
		
		
		
		
		
		/*****end*****/
	}
}
