package step1;

public class Yard {
	 static int length;
     static int width;
    public Yard(int length,int width){
    	this.length=length;
        this.width=width;
    }
    public static float getPerimeter(){
   return (length+width)*2;
    }
    public static float getAcreage(){
    return (length*width);
    }
}
