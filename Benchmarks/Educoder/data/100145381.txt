package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		   int su=0;
		while(n>1)
        {
            int i;
            i=n/2;
            if(n>i*2)
            {
                su=su+1;
            }
            n=i;
        }
        su=su+1;
		
		return su;
		
		
		
		
		/********* End *********/
	}
	
}
