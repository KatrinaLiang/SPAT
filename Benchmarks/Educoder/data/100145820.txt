package step4;

import java.util.Calendar;
import java.text.DateFormat;
import java.text.SimpleDateFormat;
import java.util.Date;

public class Task {
	
	public String getDate(String releaseDate,int day){
		/********* Begin *********/
				
                String year1=releaseDate.substring(0,4);
                String month1=releaseDate.substring(5,7);
        		String day1=releaseDate.substring(8,10);
                int year2=Integer.parseInt(year1);
                int month2=Integer.parseInt(month1);
        		int day2=Integer.parseInt(day1);
                Calendar cal=Calendar.getInstance();
                cal.set(Calendar.YEAR,year2);
        		cal.set(Calendar.MONTH,month2-1);
        		cal.set(Calendar.DATE,day2);
                cal.add(Calendar.DATE, day);
                
                DateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd");
                Date date=cal.getTime();
                String dateStr=dateFormat.format(date);
        		return dateStr;










        /********* End *********/
	}
	
	
}
