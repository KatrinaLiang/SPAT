package step3;
import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
        System.out.println("请输入一个五位数字：");
		Scanner scan = new Scanner(System.in);
        int x = scan.nextInt();
        int y=x;
        int sum = 0;
        while(x!=0){
         sum = sum + x % 10;
            x = x / 10;
        }
        System.out.print(y);
        System.out.println("的各位数字之和是："+sum);
		/******End******/
	}
}
