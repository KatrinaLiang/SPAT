public class LeapYear {
	public static void main(String[] args) {
		/*********begin***********/
		int a=2000,k=0;
        while(a<=2100){
            if(a%4==0&a%100!=0||a%400==0){
                k++;
                System.out.print(a+" ");
            }if(k%10==0){
                System.out.println();
            }
            a++;
        }
               
               
            
		
		/*********end*************/

	}

}
