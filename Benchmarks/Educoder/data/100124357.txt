package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] arr = new int[20];
        int i,total = 2;
        arr[0] = 1;arr[1] = 1;
        for(i=2;i<arr.length;i++){
        	arr[i] = arr[i-1] + arr[i-2];
            total += arr[i];
        }
        System.out.print("前"+i+"个元素的和为"+total);

	/********end********/
	}
}
