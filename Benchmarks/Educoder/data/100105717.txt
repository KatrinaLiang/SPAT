package step1;

public class Yard {
		int length,width;
    public  Yard(int length,int width){
    	this.length=length;
       	this.width=width;
    }
      public float getPerimeter(){
    	float z;
        z=length*2+width*2;
        return z;
    }
     public float getAcreage(){
        float z;
        z=length*width;
        return z;
    }
}
