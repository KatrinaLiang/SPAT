package step1;
import java.util.Scanner;
public class Yard 
{
    private int length;
    private int width;
    private double Perimeter;
    private double Acreage;
    public Yard(int length,int width)
    {
        this.length=length;
        this.width=width;
    }
    double getPerimeter()
    {
        Perimeter=(length+width)*2;
        return Perimeter;
    }
    double getAcreage()
    {
        Acreage=length*width;
        return Acreage;
    }
}
