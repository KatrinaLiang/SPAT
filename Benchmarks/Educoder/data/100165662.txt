package step3;
import step1.Date;
public class Course {
    private int code; //课程编号
    private String courseName; //课程名称
    private Date testTime;//考试日期
    private int score;//考试成绩
   /******Begin**************/
    public Course(){
        code=0;
        courseName="";
        testTime=new Date();
        score=0;}
    public Course(int code,String math,int year,int month,int day,int score){
        this.code=code;
        this.courseName=math;
        testTime=new Date(year,month,day);
        this.score=score;}
    public Course(int code,String math,Date d1,int score){
        this.code=code;
        this.courseName=math;
        testTime=d1;
        this.score=score;}
    public String toString(){
        return code+" "+courseName+" "+testTime+" "+score;}





   /******End************/

}
