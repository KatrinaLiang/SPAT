package step3;

public class Reverse {
    public static StringBuffer start(String data) {
        StringBuffer ans = new StringBuffer(data);
         StringBuffer ann = new StringBuffer();
        /********** Begin *********/
         ans.reverse();
		data=ans.toString();
       
        String [] word=data.split(" ");
        for(int i=word.length-1;i>=0;i--)
        {
            ann.append(word[i]+" ");
        }
       ans = ann;
        
        /********** End *********/
        return ans;
    }
}
