package step4;

public class DayCount {
	public void dayCount(int year, int month, int day) {
		/********** Begin *********/
        int a[]={31,28,31,30,31,30,31,31,30,31,30,31}; 
        if((year%4==0&&year%100!=0)||year%400==0)
            a[1]=29;
        int ans=0;
        for(int i=0;i<month-1;i++)
            ans+=a[i];
        ans+=day;
        System.out.print(ans);
		/********** End *********/
	}
}
