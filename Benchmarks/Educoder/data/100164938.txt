package task3;

public class Box {

    double width;
    double height;
    double depth;

    //TODO write your code here
    public double volume(double width,double height,double depth){

		this.width = width;
		this.height = height;
		this.depth = depth;

		double v= this.width*this.height*this.depth;
		return v;

	}

}
