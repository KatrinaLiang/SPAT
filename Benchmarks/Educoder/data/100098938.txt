package step1;

public class Yard {
	public Yard(int length, int width){
    	this.length = length;
        this.width = width;
    }
    public double getPerimeter(){
        return 2.0 * (length + width);
    }
    public double getAcreage(){
    	return (length * width) * 1.0;
    }
    private int length, width;
}
