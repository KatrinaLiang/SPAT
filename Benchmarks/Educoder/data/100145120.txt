package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
	int m = matrix.length;
	int n = matrix[1].length;
    int [][] arr = new int[n][m];
	for(int i=0; i<m; i++){
		for(int j=0; j<n; j++){
			arr[j][m-1-i]=matrix[i][j];	
		}
	}
	for(int i=0; i<m; i++){
		for(int j=0; j<n; j++){			
			matrix[i][j]=arr[i][j];
        }
    }	
		
		/********* End *********/
	}
}
