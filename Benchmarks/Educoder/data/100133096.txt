package step3;

public class Task {
    
	public int getBlockNum(int n,int m,char[][] arr){
        int num = 0;
		for(int i = 0; i < n; i++)
        {
        	for(int j = 0; j < m; j++)
            {
            	if(arr[i][j] == 'x')
                {
                	find(i, j, arr, n, m);
                    num++;
                }
            }
        }
        return num;
	}
    
    public static void find(int i, int j, char[][] arr,int n, int m)
    {
    	arr[i][j] = '*';
        
        
        
        if(i-1>=0 && j-1>=0 && arr[i-1][j-1] == 'x')
        {
        	arr[i-1][j-1] = '*';
            find(i-1,j-1,arr,n,m);
        }
        if(i-1>=0 && arr[i-1][j] == 'x')
        {
        	arr[i-1][j] = '*';
            find(i-1,j,arr,n,m);
        }
        if(i-1>=0 && j+1<m && arr[i-1][j+1] == 'x')
        {
        	arr[i-1][j+1] = '*';
            find(i-1,j+1,arr,n,m);
        }
        
        //*****
        
        if(j-1>=0 && arr[i][j-1] == 'x')
        {
        	arr[i][j-1] = '*';
            find(i,j-1,arr,n,m);
        }
        if(j+1<m && arr[i][j+1] == 'x')
        {
        	arr[i][j+1] = '*';
            find(i,j+1,arr,n,m);
        }
        
        //*****
        
        if(i+1<n && j-1>=0 && arr[i+1][j-1] == 'x')
        {
        	arr[i+1][j-1] = '*';
            find(i+1,j-1,arr,n,m);
        }
        if(i+1<n && arr[i+1][j] == 'x')
        {
        	arr[i+1][j] = '*';
            find(i+1,j,arr,n,m);
        }
        if(i+1<n && j+1<m && arr[i+1][j+1] == 'x')
        {
        	arr[i+1][j+1] = '*';
            find(i+1,j+1,arr,n,m);
        }
        
        
    }
}
