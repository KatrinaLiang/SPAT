package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		/********** BEGIN **********/
		double s = 0;
        if(n%2 == 0){
        	for(double i=2;i<=n;i+=2)
                s += 3/i;
        }
        if(n%2 != 0){
        	for(double i=1;i<=n;i+=2)
                s += 4/i;
        }
        System.out.println(s);



		/********** END **********/
	}
}
