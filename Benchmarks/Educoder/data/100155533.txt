package task4;

public class Box {

    double width;
    double height;
    double depth;

    int boxType;
    
    public Box(int boxType)
    {
        this.boxType=boxType;
    }

    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;
    
    

    public double volume() {
        //TODO your code
        
        if(boxType==2)
        return (width*width*3.14*height);
        if(boxType==1)
        return (width*depth*height);   
        else 
            return 0;
    }

}
