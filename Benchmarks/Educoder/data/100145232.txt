package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int num = 0;
		boolean flag = true;
		if (n == 1)
			return 1;
		if (n == 2)
			return 1;
		while (flag) {
			if (n == 1) {
				num++;
				break;
			}
			if (n == 2) {
				num++;
				break;
			}
			if (n % 2 == 1)
				num++;
			n /= 2;
		}
		return num;
		
		/********* End *********/
	}
	
}
