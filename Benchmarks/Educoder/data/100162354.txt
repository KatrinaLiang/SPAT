/**************** begin *****************/
// 在此定义学生类名
class Student
/**************** end *****************/
{
	/**************** begin *****************/
	 // 将姓名name和年龄age属性定义为私有的

	/**************** end *****************/
private String name;
private int age;

	/**************** begin 
	 * @param age2 *****************/
	/*
	 * 定义三个重载的构造方法，
	 * 无参的构造方法：方法体为空；
	 * 接收一个String类型参数的构造方法：方法体中用参数为姓名赋值；
	 * 接收String类型和int类型两个参数的构造方法：方法体中用参数为姓名和年龄赋值；
	 */
public Student() {
	return ;
}
public Student(String a) {
	name=a;
}
public Student(String a,int b) {
	name=a;
	age=b;
}
 /**************** end *****************/
	
	void speak() {
		System.out.println("我的名字是" + name + ",今年" + age + "岁");
	}
}
public class Task04{
	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);

		/**************** begin *****************/
		 // 使用无参的构造方法创建一个Student对象，并调用speak()方法
      Student p=new Student();
      p.speak();
		/**************** end *****************/
		
		/**************** begin *****************/
		/*
		 * 使用一个参数的构造方法创建一个Student对象，参数值为运行时接收从键盘输入，
		 * 然后调用speak()方法
		 */
      String name=scan.next();
      Student p2=new Student(name);
      p2.speak( );
		/**************** end *****************/


		/**************** begin *****************/
		/*
		 * 使用两个参数的构造方法创建一个Student对象，参数值为运行时接收从键盘输入，
		 * 然后调用speak()方法
		 */
     String name2=scan.next();
     int age=scan.nextInt();
     Student p3=new Student(name2,age);
     p3.speak();
		/**************** end *****************/
	}
}
