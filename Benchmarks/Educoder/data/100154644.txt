package step3;

import java.util.Scanner;

public class TestJava20_6 {
	static final Scanner sc=new Scanner(System.in);
	public void print() {
		int m;
		System.out.println("请输入菱形上半部的行数：");
		m=sc.nextInt();   //m表示行数
		/********* Start ********/
        for(int a=1;a<=2*m-1;a++){
            for(int b=1;b<=Math.abs(m-a);b++){
                System.out.print(" ");
            }
            for(int c=1;c<=2*(m-Math.abs(m-a))-1;c++){
                System.out.print("*");
            }
            System.out.println();
        }

        
        
		
		
		/********* End ********/
	}

}
