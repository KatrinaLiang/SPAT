package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        int sum = 0;
        while(n>0){
            StringBuffer sb = new StringBuffer();
            for(int i=0;i<n;i++){
                sb.append(a);
            }
            sum+=Integer.valueOf(sb.toString());
            sb=null;
            n--;
        }
        System.out.print(sum);


		/********** End *********/
	}
}
