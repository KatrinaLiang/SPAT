package step1;
import java.util.*;
public class LetterChange { 
    public static void main(String [] args){
Scanner in = new Scanner(System.in);
    String s = in.next();
    char ch=s.charAt(0);
    // 使用 switch把小写字母转为大写字母
    LetterChange p=new LetterChange();
    System.out.println(p.LetterChange(ch));
}
    public char LetterChange(char ch){
        switch(ch){
                case 'a':ch-=32;break;
                case 'b':ch-=32;break;
                case 'c':ch-=32;break;
                case 'd':ch-=32;break;
                case 'e':ch-=32;break;
                default:break;
        }
        return ch;
        // 请在此提供实现代码
    }
}
