package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int sum=0;
	int[] arr=new int[20];
	arr[0]=1;
	arr[1]=1;
	arr[2]=2;
	for(int n=3;n<20;n++){
		arr[n]=arr[n-1]+arr[n-2];
	}
	for(int i=0;i<20;i++){
		sum+=arr[i];
	}
	System.out.print("前20个元素的和为"+sum);

	/********end********/
	}
}
