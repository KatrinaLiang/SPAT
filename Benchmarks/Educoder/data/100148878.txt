package step1;

public class Task {
	
    public void output(Integer[] arr){
    	for(int i=0;i<arr.length;i++){
        	System.out.print(arr[i]+" ");
        }
        System.out.println();
    }
    
	public void sort(Integer[] arr){
		/********* Begin *********/
        boolean flag = false;
		for(int i=0;i<arr.length/2;i++){
            for(int j=i;j<arr.length-1;j++){
            	if(arr[j]>arr[j+1]){
                	Integer t = arr[j];
                    arr[j] = arr[j+1];
                    arr[j+1] = t;
                    flag = true;
                    output(arr);
                }
            }
            for(int j = arr.length-i-1;j>i;j--){
            	if(arr[j]<arr[j-1]){
                	Integer t = arr[j];
                    arr[j] = arr[j-1];
                    arr[j-1] = t;
                    flag = true;
                    output(arr);
                }
            }
            if(!flag)
                output(arr);
            
        }
		
		/********* End *********/
	}
}
