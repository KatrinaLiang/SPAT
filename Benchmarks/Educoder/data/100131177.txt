package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum=0;
        for(int i=num1;i<=num2;i++){
        if(i>=10&&i<100){
        int a=i/10;
        int b=i%10;
        sum+=a+b;
        }else if(i>=100){
            int a=i/100;
            int b=i%100;
            int c=b/10;
            int d=b%10;
            sum+=a+c+d;
        }else if(i<10){
        sum+=i;
        }
            
        }
        return sum;
		/********** END **********/
    }
}
