package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		 int[]arr=new int[20];
        int i,s=0;
        arr[0]=1;
        arr[1]=1;
        for(i =2;i<20;i++)
        arr[i]=arr[i-2]+arr[i-1]; 
        for(i=0;i<20;i++)
            s+=arr[i];
        System.out.print("前20个元素的和为"+s);
	/********end********/
	}
}
