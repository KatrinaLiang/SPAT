package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] arr=new int[20];
        
        int i,sum=0;
            
        for(i=0;i<20;i++){
           
        if(i<=1)
               arr[i]=1;
        else
            arr[i]=arr[i-1]+arr[i-2];
        
        sum=sum+arr[i];
        }
        System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
