package step4;

public class DayJudge {

    // 根据输入年、月、日判断这一天是当年的第几天
    public String DayJudge(int year, int month, int day){
        String d;// 请在此提供实现代码
        int t;
        switch(month)
        {
            case 1:t=0;break;
            case 2: t=31;break;
            case 3: t=31+28;break;
            case 4:t= 31+28+31;break;
            case  5: t=31+28+31+30; break;
            case 6: t=31+28+31+30+31;break;
            case 7:  t=31+28+31+30+31+30;break;
            case 8:  t=31+28+31+30+31+30+31;break;
            case 9:  t=31+28+31+30+31+30+31+31;break;
            case 10:  t=31+28+31+30+31+30+31+30+31;break;
            case 11:  t=31+28+31+30+31+30+31+30+31+31;break;
            default :  t=31+28+31+30+31+30+31+30+31+31+30;break;
        }
        t=t+day;
        if(year%4==0&&year%100!=0||year%400==0)
        {
            if(month>2)
                t++;
        }
            d=year+"年的第"+t+"天";
        return d;
	}
}
