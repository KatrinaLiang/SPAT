package step1;

public class Yard {
    private int length;
    private int width;
    public double getPerimeter(){
     	return  Math.round(2*(width+length)*10)/10.0;  
    }
    public double getAcreage(){
     	return Math.round((width*length)*10)/10.0;  
    }
    public Yard(int length,int width){
        this.length=length;
        this.width=width;
    }
		
}
