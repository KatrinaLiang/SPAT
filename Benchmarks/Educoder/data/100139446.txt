package step3;

public class MyThread implements Runnable {
    private String name;
    private Object prev;
    private Object self;
    private MyThread(String name,Object prev,Object self){
        this.name = name;
        this.prev = prev;
        this.self = self;
    }
    public void run(){
        int count = 5;
        while(count > 0){
            synchronized(prev){
                synchronized(self){
                    System.out.print(name);
                    count--;
                    self.notify();
                }
                try{
                    prev.wait();
                }catch(InterruptedException e){
                    e.printStackTrace();
                }
            }
        }
        System.exit(0);                
    }
	    
    public static void main(String[] args) throws Exception {   
        Object a = new Object();
        Object b = new Object();
        Object c = new Object();
        MyThread te = new MyThread("E",a,b);
        MyThread td = new MyThread("D",b,c);
        MyThread tu = new MyThread("U",c,a);
        
        new Thread(te).start();
        Thread.sleep(100);
        new Thread(td).start();
        Thread.sleep(100);
        new Thread(tu).start();
        Thread.sleep(100);
    }   
}
