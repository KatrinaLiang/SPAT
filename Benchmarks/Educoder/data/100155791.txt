package step1;

public class Task {
	
	public int search(int n,int[] nums,int T){
		/********* Begin *********/
		int l = 0;
		int r = n-1;
		while (l<= r) {
			int m = (l+r)/2;
			if(nums[m]<T){
				l=m+1;
			}else if (nums[m]>T) {
				r = m-1;
			}else if (nums[m] == T){
				return m;
			}
		}
		return -1;		
		/********* End *********/
	}
	
}
