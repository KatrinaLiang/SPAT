package step1;

public class Rectangle{
	private double width;
	private double length;
	
	/*********begin*********/
	 public  Rectangle(){
	       this.length=6;
	       this.width=5;
	    }
    Rectangle(double a,double b){
        this.length=a;
        this.width=b;
    
    }
  
	public double area(){
        return this.length*this.width;
    
    }
    public double permeter(){
        return (this.length+this.width)*2;
    
    }
	/*********end*********/
	public static void main(String[] args) {
		/*********begin*********/
		double Rperimeter,Rarea;
		 Rectangle r1=new Rectangle();
        Rperimeter=r1.permeter();
        Rarea=r1.area();
        System.out.println("r1的周长："+Rperimeter);
        System.out.println("r1的面积："+Rarea);
        Rectangle r2=new Rectangle(20,10);
        Rperimeter=r2.permeter();
        Rarea=r2.area();
        System.out.println("r2的周长："+Rperimeter);
        System.out.println("r2的面积："+Rarea);

		/*********end*********/
	}



}
