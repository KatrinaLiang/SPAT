package task04;

import java.util.Scanner;
class Student{
private String name;
private int age;
Student()
{
	
	}
Student(String n)
{
	name=n;
	}
Student(String n,int i)
{
	name=n;
	age=i;
	}
void speak() {
	System.out.println("我的名字是" + name + ",今年" + age + "岁");
}
}

	

public class Task04{
	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
Student stu1=new Student();
		stu1.speak();
		String n;
		n=scan.next();
		Student stu2=new Student(n);
		stu2.speak();
		String m;
		m=scan.next();
		int i;
		i=scan.nextInt();
		Student stu3=new Student(m,i);
		stu3.speak();
		

	}
}
