package step1;

public class Yard {
    private static int length;
   private static int width; 
    public Yard(int length,int width){
        this.length = length;
        this.width = width;
    }
    public static double getPerimeter(){
        return 2*(length+width);
    }
    public static double getAcreage(){
        return length*width;
    }
}
