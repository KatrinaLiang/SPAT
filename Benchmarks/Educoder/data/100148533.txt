//定义站台类，实现卖票的功能。
public  class Station extends Thread {	
      public static  int count=20;
     public static Object prev=new Object();
      
    
     public  void run() { 
          
         
         while(count>1)
         {
             synchronized (prev){
                 if(count>0){
                  System.out.println("卖出了第"+count+"张票");
          count--;}
             }
         }
         
         if(count==0){
         System.out.println("票卖完了");
          count=-1;

}
        
}
}
