package step1;

public class Yard {
	Yard(int length,int width){
		this.width=width;
		this.length=length;
	}
	public double getPerimeter(){
		double perimeter=2*(length+width);
		return perimeter;
	}
	public double getAcreage() {
		double acreage=width*length;
		return acreage;
	}
	private int length;
	private int width;

}
