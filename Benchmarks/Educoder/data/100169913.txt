package step1;
import java.util.Scanner;

public class Cal {
	public static void main(String[] args) {
		/*********start*********/
		Scanner sc = new Scanner(System.in);
		int re1 = 0,re2 = 0,re3 = 0,re4 = 0,re5 = 0;
		System.out.println("请输入第一个整数");
		int a = sc.nextInt();	
		System.out.println("请输入第二个整数");
		int b = sc.nextInt();
		re1 = a + b;
		re2 = a - b;
		re3 = a * b;
		re4 = a / b;
		re5 = a % b;
		System.out.println("两数相加的结果为：" + (a+b));
		System.out.println("两数相减的结果为：" + re2);
		System.out.println("两数相乘的结果为：" + re3);
		System.out.println("两数相除的结果为：" + re4);
		System.out.println("两数取余数的结果为：" + re5);
		sc.close();
		/*********end*********/
	}

}
