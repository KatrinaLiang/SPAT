package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		for(int i=1;i<=9;i++){
            for(int j=1;j<=9;j++){
                for(int k=1;k<=j;k++){
                    System.out.print(k+"*"+j+"="+j*k+"\t");
                }
                System.out.print("\n");
            }
            break;
        }
		
		
		
		
		
		/*****end*****/
	}
}
