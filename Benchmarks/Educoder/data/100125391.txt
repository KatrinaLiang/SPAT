package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
	public InterRectangle(){
        
    }
    public InterRectangle(double length,double width){
        this.length=length;
        this.width=width;
    }
    public double per(){
        return 2*(this.length+this.width);
    }
    public double area(){
        return this.length*this.width;
    }

	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
		System.out.println("总共有"+total+"块土地");
        int i;
		double totalPer=0;
		double totalArea=0;
		InterRectangle r=new InterRectangle();
		for(i=1;i<=total;i++){
			r.width=scanner.nextDouble();
			r.length=scanner.nextDouble();
			totalPer=totalPer+r.per();
			totalArea=totalArea+r.area();
		}
		System.out.println("土地总面积为:"+totalArea+"平米");
		System.out.println("土地总周长为:"+totalPer+"米");

		/*********end*********/
	}

}
