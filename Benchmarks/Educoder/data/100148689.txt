package step1;

public class Task {
	
	public void sort(Integer[] arr){
		/********* Begin *********/
		int temp;
		if(isAsc(arr))
		{
			printA(arr);
		}
		while(!isAsc(arr))
		{
			for(int i=0;i<arr.length-1;i++)
			{
				if(arr[i+1]<arr[i])
				{
					temp=arr[i];
					arr[i]=arr[i+1];
					arr[i+1]=temp;
					printA(arr);
					
				}
				else{
					continue;
				}
			}
			if(isAsc(arr))
			{
				break;
			}
			for(int i=arr.length-1;i>=1;i--)
			{
				if(arr[i-1]>arr[i])
				{
					temp=arr[i-1];
					arr[i-1]=arr[i];
					arr[i]=temp;
					printA(arr);
				}
				else{
					continue;
				}
			}
			if(isAsc(arr))
			{
				break;
			}
		}
		
		
		/********* End *********/
	}
    public boolean isAsc(Integer[] arr)
	{
		for(int i=0;i<arr.length-1;i++)
		{
			if(arr[i+1]<arr[i])
			{
				return false;
			}
			else{
				continue;
			}
		}
		return true;
	}
	public void printA(Integer[] arr)
	{
		for(int i=0;i<arr.length;i++)
		{
			System.out.print(arr[i]+" ");
		}
		System.out.print("\n");
	}
}
