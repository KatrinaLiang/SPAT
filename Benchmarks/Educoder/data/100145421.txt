package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int number=0;
        while(n>0){
        if(n%2==1){
          number++;
            n=n/2;
          }
        if(n%2==0){
            n=n/2;
          }
        }
        return number;
		
		
		
		
		/********* End *********/
	}
	
}
