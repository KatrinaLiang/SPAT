package step2;
import step1.Person;
import step1.Date;
public class Student extends Person{
    private int reg_num; //学号
    private Date reg_day; //注册日期
   // TO DO YOU CODE HERE
    public Student()
    {
        super();
        reg_num=0;
        reg_day=new Date(1999,9,9);
    }
    public Student(String name,String num,Date birthday,int num1,Date b)
    {
        super(name,num,birthday);
        reg_num=num1;
        reg_day=new Date(b.getYear(),b.getMonth(),b.getDay());
    }
	public Student(String name,String num,int year,int month,int day,int b,int y,int m,int d)
    {
         super(name,num,year,month,day);
        reg_num=b;
        reg_day=new Date(y,m,d);
    }
public Date getreg_day()
{
   return reg_day;
}

public String toString()
    {
      return  super.toString()+"reg_num:"+reg_num+"reg_day:"+reg_day.toString();
    }
}
