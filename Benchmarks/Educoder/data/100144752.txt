package step1;

public class Task {
	public String inversion(String str){
		/********* Begin *********/
		StringBuilder sb=new StringBuilder();
		for (int i=str.length()-1;i>=0;i--){
            char c = str.charAt(i);
            sb.append(c);
        }
        String string=sb+"";
        return string;
		
		/********* End *********/
	}
}
