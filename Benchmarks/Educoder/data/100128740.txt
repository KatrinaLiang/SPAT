//在这里定义Dog类
class Dog
{
    String name;
    String color;
    String variety;
    public Dog()
    {
        
    }
    public Dog(String name,String color,String variety)
    {
        this.name=name;
        this.color=color;
        this.variety=variety;
    }
    public void eat()
    {
        System.out.println("啃骨头");
    }
    public void run()
    {
        System.out.println("叼着骨头跑");
    }
}
