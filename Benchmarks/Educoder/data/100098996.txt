package step1;

public class Yard {
    int length;
    int width;
    public Yard(int length,int width)
    {
    	this.length=length;
        this.width=width;
    }
    public double getPerimeter()
    {
        double p;
        p=2*(length+width);
        return p;
    }
    public double getAcreage()
    {
        double a;
        a=length*width;
        return a;
    }
}
