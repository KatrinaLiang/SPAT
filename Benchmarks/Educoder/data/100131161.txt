package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int s=0;
        String string;
		for(int i=num1;i<=num2;i++)
		{
			string=String.valueOf(i);
			char[] b = string.toCharArray();
			for(int j=0;j<b.length;j++)
			{
				
				s=s+Integer.valueOf(b[j])-48;
				
			}
		}
		
		return s;
		/********** END **********/
	}
}
