package step1;

public class Yard {
    private double width;
    private double length;	
    public Yard(){
    }
    public Yard(double width,double length){
        this.width=width;
        this.length=length;
    }
    public double getPerimeter(){
        return 2*(width+length);
    }
   public double getAcreage(){
   return width*length;
   }


}
