package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
	public InterRectangle(){
    }
    
    public InterRectangle(double width , double length){
        
        this.width = width;
        this.length = length;
    }

	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
        double totalper = 0;
        double totalarea = 0;
        for(int i = 1 ; i <= total ; i++){
            double width = (double)scanner.nextInt();
            double length = (double)scanner.nextInt();
            totalper += 2.0 * (width + length);
            totalarea += width * length;
        }
        System.out.println("总共有" + total + "块土地");
        System.out.println("土地总面积为:" + totalarea + "平米");
        System.out.println("土地总周长为:" + totalper + "米");
		/*********end*********/
	}

}
