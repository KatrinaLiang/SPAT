
package step4;
import step1.Date;
import step2.Student;
import step3.Course;

public class Report {
    private Student s;//学生基本信息
    private String major; //所学专业  
    private Course[] selCourse = new Course[10];//所选课程4
    private int l;
   /************Begin****************/
    
    public Report(){}
    public Report(Student s, String major, Course[] selCourse) {
    	this.s = s;
        this.major = major;
        this.l = selCourse.length;
        for(int i = 0; i < l; i++) {
         this.selCourse[i] = selCourse[i];
        }
    }
    
    public void setReport(Student s, String major, Course[] selCourse) {
    	this.s = s;
        this.major = major;
        for(int i = 0; i < selCourse.length; i++) {
         this.selCourse[i] = selCourse[i];
        }
    }
        
    public Course[] getCourse(){return selCourse;}
    public Student getStudent(){return s;}  
    public String getmajor() {return major;}
    
    public String toString() {
        String a = "";
        for(int i = 0; i < l; i++) {
        	a = a + selCourse[i].toString() + "\n";
        }
    	return s.toString() + " major:" +major+ "\n" + a;
    }




    /************End****************/

  
}
