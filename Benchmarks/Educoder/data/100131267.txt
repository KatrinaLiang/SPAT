package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum=0;
        int a=0;
        int b=0;
        for(int i=num1;i<=num2;i++){
            b=i;
           while(b!=0){
            a=b%10;
            sum=sum+a;
            b=b/10;
        } 
        }
		return sum;
		/********** END **********/
	}
}
