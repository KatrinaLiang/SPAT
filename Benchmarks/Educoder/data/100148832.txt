package step1;

public class Task {
	
	public void sort(Integer[] arr){
		/********* Begin *********/
	if(arr.length<=3){
	        for(int i=0;i<arr.length;i++){
					if(i+1<arr.length){
					if(arr[i]>arr[i+1]){
						int tmp = arr[i];
						arr[i] = arr[i+1];
						arr[i+1] = tmp;
					}
					}					
				}
				for(int i=arr.length-1;i>0;i--){
					if(arr[i]<arr[i-1]){
						int tmp = arr[i];
						arr[i] = arr[i-1];
						arr[i-1] = tmp;						
					}
				}
	        for (Integer integer : arr) {
						System.out.print(integer+" ");
					}
					System.out.println();
	    }else{
		so(arr);
		for (int i=0;i<arr.length-1;i++) {
			while (arr[i].compareTo(arr[i+1])>0) {
				sort(arr);
			}
				
			}
        
		
		/********* End *********/
	}
}
    public static void so(Integer[] arr){
		for(int i=0;i<arr.length;i++){
			if(i+1<arr.length){
			if(arr[i]>arr[i+1]){
				int tmp = arr[i];
				arr[i] = arr[i+1];
				arr[i+1] = tmp;
				for (Integer integer : arr) {
					System.out.print(integer+" ");
				}
				System.out.println();
			}
			}			
		}
		for(int i=arr.length-1;i>0;i--){
			if(arr[i]<arr[i-1]){
				int tmp = arr[i];
				arr[i] = arr[i-1];
				arr[i-1] = tmp;
				for (Integer integer : arr) {
					System.out.print(integer+" ");
				}
				System.out.println();
			}
		}
        
	}
}
