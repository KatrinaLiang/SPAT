package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		int a,b;
        for(a=1;a<=10;a++){
            for(b=1;b<=a;b++){
                System.out.print("*");
            }
                System.out.print("\n");
        }
		
		
		
		System.out.println("——————————我是华丽的分界线——————————");
		 for(a=10;a>0;a--){
             for(b=1;b<=a;b++){
              System.out.print("*");
             }
              System.out.print("\n");
         }
		
		
		
		
		
		/*****end*****/
	}
}
