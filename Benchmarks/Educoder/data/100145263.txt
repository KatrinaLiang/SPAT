package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int num = 0;
		if (n == 0) {
			return 0;
		} else if (n == 1) {
			return 1;
		}
		while (n != 1) {
			if (n % 2 == 1) {
				num++;
				n--;
			} 
			if (n % 2 == 0) {
				n = n/2;
			}
		}
        
        
		return num+1;
		
		
		
		
		/********* End *********/
	}
	
}
