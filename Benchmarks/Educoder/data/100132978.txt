package step1;

public class Task {
	
	public void sortArray(int[] arr){

        int a;
        int b;
		for(a=1;a<arr.length;a++){
        	for(b=a-1;b>=0;b--){
            	if(arr[b]>arr[b+1]){
                	int c;
                    c=arr[b+1];
                    arr[b+1]=arr[b];
                    arr[b]=c;
                }
            }
            int i;
            for(i=0;i<arr.length;i++){
            	System.out.printf("%d ",arr[i]);
            }
            System.out.println();
        }
		
	}
}
