package step4;
import java.util.Scanner;
public class NiXu {
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        System.out.println("请输入一个正整数：");
        int a=sc.nextInt();
        do {
            int num=a%10;
            a/=10;
            System.out.print(num);
        }while (a != 0);
System.out.println();
    }

}
