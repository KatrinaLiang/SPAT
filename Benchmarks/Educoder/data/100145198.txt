package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
       int k[]=new int[100];
        int s,i,sum;
        s=n;i=0;sum=0;
        while(s!=0)
        {
            s=s/2;
            k[i]=s;
            i++;
        }
        for(i=0;i<k.length;i++)
        {
            sum=sum+k[i];
        }
        return (n-sum);
		/********* End *********/
	}
	
}
