class Person {
	/********* begin *********/
	private String name;
    private int age;
    public void talk(String name,int age){
         this.name=name;
        this.age=age;
       System.out.print("我是："+name+"，今年："+age+"岁，我在");}
	/********* end *********/
}

class Student extends Person {
	/********* begin *********/
	private String school;
    public Student(String name,int age,String school){
        this.school=school;
        talk(name,age);
        System.out.println(school+"上学");
    }
    public void talk(String name,int age){
        super.talk(name,age);
    }
	/********* end *********/
}
