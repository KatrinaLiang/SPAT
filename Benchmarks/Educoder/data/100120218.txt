package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
        int i,n,a;
		for(i=1;i<=9;i++)
        {
            for(n=1;n<=i;n++)
            {
                a=i*n;
                System.out.print(n+"*"+i+"="+a+"\t");
            }System.out.println();
        }
		
		
		
		
		/*****end*****/
	}
}
