package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int[ ] F = new int[20];
		F[0]=1; F[1]=1;
		int sum=0;
		for(int i=2;i<20;i++)
		{
			F[i]=F[i-1]+F[i-2];
		}
		for(int j=0;j<20;j++)
		{
			sum=sum+F[j];
		}
    System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
