package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
		int length = temperatures.length;
        int[] outputDays = new int[length];
        int i, j, temprature;
        for(i = 0; i < length; i++){
            temprature = temperatures[i];
            for (j = i; j < length; j++){
                if(j == length - 1){
                    break;
                }
                if (temprature < temperatures[j + 1]){
                    break;
                }
            }
            if (j == length - 1){
                outputDays[i] = 0;
            }else {
                outputDays[i] = j - i + 1;
            }
        }

       
        return outputDays;
		/********** END **********/
	}
}
