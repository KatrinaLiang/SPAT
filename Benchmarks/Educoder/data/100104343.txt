package step3;
import java.util.Scanner;
public class GcdAndGcm {
	public static void main(String[] args) {
		/********** Begin *********/
		Scanner sc=new Scanner(System.in);
        
        int m=sc.nextInt(),a;
        int n=sc.nextInt(),b; //键盘输入数据
        int r;
        a=m;b=n;
        do{
           r=m%n;
        	m=n;
            n=r;
        }while(r!=0);
         System.out.println("最大公约数："+m);
         System.out.println("最小公倍数："+(a*b/m));
        /********** End *********/
		}
}
