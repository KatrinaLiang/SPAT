
#include<stdio.h>
#include<string.h>
//#define N 10005


int main(){
    char s1[1005],s2[1005];
	int a,b,p,q,count,max,i,j;
	scanf("%s%s",&s1,&s2);
	a=strlen(s1);
	//scanf("%s",&s2);
	b=strlen(s2);	
	max=0;
	for( i=0;i<a;i++){		
		for( j=0;j<b;){
			count=0;
			if(s1[i]==s2[j]){				
				p=i;
				q=j;
				while(s1[p]==s2[q]){
					count++;
					p++;
					q++;
				}
				if(count>max) {
					max=count;
				}
					
				j=q;
				
			}
			else{
				j++;
			}
		}
	}
	 printf("%d",max);
}
