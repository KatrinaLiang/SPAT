package step4;
import java.util.*;

public class Task {
	
	public String getDate(String releaseDate,int day){
		/********* Begin *********/
		
		String[] sp = releaseDate.split("-");
		
		int in[] = new int[sp.length];
		for(int i = 0; i < sp.length; i++) {
			in[i] = Integer.parseInt(sp[i]);
		}

		
		Date date = new Date(in[0]-1900,in[1]-1,in[2]);

		
		Calendar calendar = Calendar.getInstance();
		
		calendar.setTime(date);
		
		calendar.add(Calendar.DAY_OF_MONTH,day);
		
		int year = calendar.get(Calendar.YEAR);
		int month = calendar.get(Calendar.MONTH)+1;
		int dayOfMonth = calendar.get(Calendar.DAY_OF_MONTH);
		
		if(month < 10) {
			if(dayOfMonth < 10)
				return year + "-0" + month + "-0" + dayOfMonth;
			else
				return year + "-0" + month + "-" + dayOfMonth;
		}else {
			if(dayOfMonth < 10)
				return year + "-" + month + "-0" + dayOfMonth;
			else
				return year + "-" + month + "-" + dayOfMonth;
		}
		

        /********* End *********/
	}
	
	
}
