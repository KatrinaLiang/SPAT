package step1;

import java.sql.*;

public class QueryPass {
    
    static String driver = "com.mysql.jdbc.Driver";
    static String url = "jdbc:mysql://localhost:3306/tsgc";
    static String username = "root";
    static String password = "123123";
    
    static Connection conn = null;
    static PreparedStatement pstmt = null;
	
	// 查询数据代码不用上实验报告
	public static void queryDB() throws Exception{
		
		/********* Begin *********/

		// 第一步：加载驱动
		Class.forName(driver);
		

		// 第二步：建立连接, "root"和"123123"是针对MySQL设置了用户名(root)和密码(123123)的情况
		// 127.0.0.1:3306是mysql服务器地址及端口   数据库编码格式设置为utf-8
		conn = (Connection) DriverManager.getConnection(url, username, password);
		

		// 第三步：建立statement对象
		String sql = "select * from employee";
		pstmt = (PreparedStatement) conn.prepareStatement(sql);
		ResultSet rs = pstmt.executeQuery();
        int col = rs.getMetaData().getColumnCount();
        ResultSetMetaData rsmd = rs.getMetaData();
        while (rs.next()) {
            
            
            for (int i = 1; i <= col; i++) {
                if(i == col){
                    String val = rs.getString(i);
                    val = val.substring(0,val.length()-1);
                	System.out.print(rsmd.getColumnName(i)+val);
                }else{
                	System.out.print(rsmd.getColumnName(i) +":"+rs.getString(i)+"\t");
                }
                
                
                
             }
            System.out.println();
        }
        
        
		

		// 第五步：关闭statement对象和连接对象
		pstmt.close();
        conn.close();
		

		/********* End *********/
	}
}
