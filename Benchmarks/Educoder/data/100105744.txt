package step1;

public class Yard {
    double x,y;
    public Yard(int length,int width){
        x=(double)length;
        y=(double)width;
    }
    double getPerimeter(){
        return (this.x+this.y)*2;
        
    }
    double getAcreage(){
        return this.x*this.y;
        
        
    }

}
