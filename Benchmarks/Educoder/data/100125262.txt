package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
    public InterRectangle(){
		super();
	}
	
	public InterRectangle(double width,double length){
		super();
		this.width = width;
		this.length = length;
	}
	
	
	
	public double getWidth() {
		return width;
	}

	public void setWidth(double width) {
		this.width = width;
	}

	public double getLength() {
		return length;
	}

	public void setLength(double length) {
		this.length = length;
	}

	public double area(){
		return width*length;
	}
	
	public double perimeter(){
		return (width+length)*2;
	}
	

	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
        InterRectangle r = new InterRectangle();
		double sumArea = 0;
		double sumPerimeter = 0;
		for(int i = 0; i < total; i++){
			double length=scanner.nextDouble();
			double width=scanner.nextDouble();
			r.setLength(length);
			r.setWidth(width);
			sumArea += r.area();
			sumPerimeter += r.perimeter();
		}
		System.out.println("总共有" + total + "块土地");
		System.out.println("土地总面积为:" + sumArea + "平米");
		System.out.println("土地总周长为:" + sumPerimeter + "米");
		

		/*********end*********/
	}

}
