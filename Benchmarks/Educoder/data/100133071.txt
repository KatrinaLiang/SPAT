package step1;

public class Task {

	public static void birthdayCandle(int num){
        int i,j,k;
		for (i=18;i<=100;i++)
        {
            for (j=i;j<=100;j++)
            {
                int res=0;
                for (k=i;k<=j;k++)
                {
                    res+=k;
                }
                if (res==num)
                {
                    System.out.println("小明今年：" +  j + "岁");
					System.out.println("小明从" +  i  + "岁开始每年举办派对");
                    return;
                }
            }
        }
		
	}
}
