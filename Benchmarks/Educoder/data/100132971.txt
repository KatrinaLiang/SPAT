package step1;

public class Task {
	
	public void sortArray(int[] arr){
		if(arr.length==1)
		System.out.println(arr[0]);
	else {
		int i,j,t;
		for(i=1;i<arr.length;i++) {
			if(arr[i]<arr[i-1]) {
				j=i;
				while(j-1>=0 && arr[j]<arr[j-1]) {
					t=arr[j-1];
					arr[j-1]=arr[j];
					arr[j]=t;
                    j--;
				}
			}
			for(j=0;j<arr.length;j++) {
				System.out.print(arr[j]);
				System.out.print(' ');
			}
			System.out.println();
		}
	}	
	
		
		
	}
}
