package step2;
import java.util.Scanner;
public class SumOfDigit {
		public static void main(String[] args) {
			int number;
			int sum=0;
			Scanner scan1=new Scanner(System.in);
	        number=scan1.nextInt();
			/******Begin******/
		sum+=number%10;
        sum+=number%100/10;
        sum+=number%1000/100;
        sum+=number%10000/1000;
        sum+=number/10000;
		System.out.print(sum);
	         
	        
	                    
			/******End******/
		}
}
