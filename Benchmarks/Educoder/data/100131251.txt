package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum=0;
        int ten=0;
        int one=0;
        for(int i=num1;i<=num2;i++)
        {
            ten=i;
            while(ten!=0)
            {
                one=ten%10;
                sum+=one;
                ten=ten/10;
            }
        }
		return sum;
		
		/********** END **********/
	}
}
