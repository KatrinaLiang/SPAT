package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] arr=new int[20];
        arr[0]=1;
        arr[1]=1;
        int sum=0;
        int i;
        for(i=2;i<20;i++){
            int n=i-1;
            int m=i-2;
            arr[i]=arr[n]+arr[m];
        }
		for(int j=0;j<20;j++){
            sum=sum+arr[j];
        }
        System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
