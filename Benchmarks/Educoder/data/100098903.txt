package step1;

public class Yard {
double width;
    double length;
    public Yard(){

    }
    public Yard(int width,int length){
        this.width=width;
        this.length=length;
    }
    public double getPerimeter(){
        return 2*(this.width+this.length);
    }
    public double getAcreage(){
        return this.width*this.length;
    }
}
