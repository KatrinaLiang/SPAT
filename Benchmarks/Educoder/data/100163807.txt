package step2;
import java.util.Scanner;
public class SumOfDigit {
		public static void main(String[] args) {
			int number;
			int sum=0;
			Scanner scan1=new Scanner(System.in);
	        number=scan1.nextInt();
			/******Begin******/
			int[] a=new int[5];

			a[0]=number%10;
			a[1]=number/10%10;
			a[2]=number/100%10;
			a[3]=number/1000%10;
			a[4]=number/10000%10;

			for(int i=0;i<5;i++)
			sum+=a[i];

			System.out.print(sum);


	         
	        
	                    
			/******End******/
		}
}
