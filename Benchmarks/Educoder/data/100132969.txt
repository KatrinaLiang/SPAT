package step1;

public class Task {
	
	private void print(int[] arr) {
		for(int i=0; i<arr.length; i++) {
			System.out.print(arr[i] +  " ");
		}
		System.out.println();
	}
	
	public void sortArray(int[] arr){
		for(int i=1; i<arr.length; i++) {
			for(int j=i; j>0; j--) {
				if(arr[j - 1] > arr[j]) {
					int t = arr[j];
					arr[j] = arr[j - 1];
					arr[j - 1] = t;
				}
			}
			print(arr);
		}
	}
}
