package step1;

import java.util.Scanner;

public class Yard {
    private double length;
    private double width;

    public  Yard(double length,double width)
    {
        this.length=length;
        this.width=width;
    }
    
    public double getPerimeter()
    {
        return 2*(length+width);
    }
    public double getAcreage()
    {
         return width*length;
    }
    
}
