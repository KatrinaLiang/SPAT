package step1;

public class Yard {
   public double length;
   public double width; 
	public  Yard(double a,double b)
    {	
        this.length=a;
        this.width=b;
    }
    public double getAcreage()
    {	
        
        return length*width;
    }
    public double getPerimeter()
    {
        return ((length+width)*2);
    }
}
