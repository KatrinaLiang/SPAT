package step3;

import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;
import java.io.File;
public class Task {
	
	public void task() throws IOException{
		/********* Begin *********/
		File filein= new File("src/step3/input/input.txt");
        File fileout= new File("src/step3/output/output.txt");
        File fileoutPrent =fileout.getParentFile();
        if(!fileoutPrent.exists())
        {
            fileoutPrent.mkdirs();
        }
        fileout.createNewFile();
        
        FileReader fileRD=new FileReader(filein);
        FileWriter fileWT=new FileWriter(fileout);
        char buf[] =new char[1024];
        int readlen;
        while((readlen=fileRD.read(buf))!=-1)
        {
            fileWT.write(buf,0,readlen);
        }
        fileWT.close();
        
		
		
		
		
		/********* End *********/		
	}
}
