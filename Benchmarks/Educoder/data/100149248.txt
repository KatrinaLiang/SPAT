package step2;

public class Task {
    
	public int virusArea(int n,int m,char[][] area){
		/********* Begin *********/
		int res = 0;
        if(area.length==0) return res;
        for(int i = 0; i < area.length; i++){
            for(int j = 0; j < area[0].length; j++){
                if(area[i][j]=='o'){
                    searchIsland(area, i, j);
                    res++;
                }
            }
        }
        return res;
		
		/********* End *********/	
	}
    
     public static void searchIsland(char[][] grid, int i, int j){
        grid[i][j]='x';
        if(i>0 && grid[i-1][j]=='o') searchIsland(grid, i-1, j);
        if(j>0 && grid[i][j-1]=='o') searchIsland(grid, i, j-1);
        if(i<grid.length-1 && grid[i+1][j]=='o') searchIsland(grid, i+1, j);
        if(j<grid[0].length-1 && grid[i][j+1]=='o') searchIsland(grid, i, j+1);
    }
}
