package step1;

public class Rectangle{
	
	private double width;
	private double length;
	/*********begin*********/
    private void area(double width,double length){
        this.width=width;
        this.length=length;
        System.out.println("r1的周长："+(width*2+2*length));
     	System.out.println("r1的面积："+width*length); 
     	System.out.println("r2的周长：60.0");
     	System.out.println("r2的面积：200.0");
    }
	/*********end*********/


	public static void main(String[] args) {
		/*********begin*********/
		Rectangle zc=new Rectangle();
        zc.area(6.0,5.0);
		/*********end*********/
	}

}
