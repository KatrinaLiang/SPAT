package step3;

import java.util.Scanner;
public class Number_Judge {
		public static void main(String[] args) {
			Scanner scan1=new Scanner(System.in);
			/******Begin******/
			int n = scan1.nextInt();

			if(n%2 == 0){
				System.out.print("偶数");
			}
			else{
				int flag = 1;//是素数
				if(n>1){
					for(int i=2;i<n;i++){
						if(n%i == 0){
							System.out.print("奇数");
							flag = 0;
							break;
						}
					}
					if(flag == 1){
						System.out.print("素数");
					}	
				}
				
			}
	        
			/******End******/
		}
}
