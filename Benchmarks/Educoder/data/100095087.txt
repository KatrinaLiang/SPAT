package step3;

import java.util.Scanner;

public class Factorial {
	public static void factorial(int n) {
		/********** Begin *********/
		int sum = 1;
		for(int i = n ; i >= 1 ; i--) {
			
			sum *= i;
		}
		
		System.out.print(sum);

		/********** End *********/
	}
	
	public static void main(String args[]) {
		
		Scanner sc = new Scanner(System.in);
		int in = sc.nextInt();
		factorial(in);
	}
}
