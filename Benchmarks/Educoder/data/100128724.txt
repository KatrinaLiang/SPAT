package step1;

public class Test {
	public static void main(String[] args) {
		//创建Dog对象
		Dog dog = new Dog("五花肉","棕色","阿拉斯加");
		//设置dog对象的属性
		
		
		//输出小狗的属性
		System.out.println("名字：" +  dog.name   + "，毛色：" + dog.color   + "，品种：" + dog.variety );
		
		//调用方法
		dog.eat();
		dog.run();
	}
}

//在这里定义Dog类
class Dog{
    	public String name;
    	public String color;
    	public String variety;
    	
    	public Dog(){
    		
    	}
    	public Dog(String name, String color, String variety){
    		this.name = name;
    		this.color = color;
    		this.variety = variety;
    	}
    	
    	public void eat(){
    		System.out.println("啃骨头");
    	}
    	
    	public void run(){
    		System.out.println("叼着骨头跑");
    	}
}
