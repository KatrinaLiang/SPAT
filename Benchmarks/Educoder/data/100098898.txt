package step1;

class Yard {
    public int length=0;
    public int width=0;
public  Yard(int length,int width)
{
       this.length=length;
        this.width=width;
}
    public double getPerimeter()
    {
        double Perimeter=(2*length+2*width);
        return Perimeter;
    }
    public double getAcreage()
    { 
        double Acreage=length*width;
        return Acreage;
    }
}
