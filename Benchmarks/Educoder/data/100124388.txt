package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] Fib=new int[20];
        Fib[0]=1;
        Fib[1]=1;
        int sum=2;
        for(int i=1;i<19;i++){
        	Fib[i+1]=Fib[i]+Fib[i-1];
            sum+=Fib[i+1];
        }
        System.out.print("前20个元素的和为"+sum);

	/********end********/
	}
}
