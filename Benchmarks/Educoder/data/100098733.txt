package step1;

public class Yard {
    int length;
    int width;
    public Yard(int l,int w){
    length=l;
    width=w;}
public  float getPerimeter(){
    float sum;
    sum=(float)(length+width)*2;
    return sum;}
public float getAcreage(){
  float sum;
    sum=(float)(length*width);
    return sum;}
}
