package step2;

public class SwitchToGrade {

    // 请用switch来将学生成绩转换成对应等级
    public String SwitchToGrade(int score) { 
        // 请在此提供实现代码
        String a;
        if(score>=0&&score<=100)
        {
        switch(score=score/10)
               { case 10:
                 case 9:a="等级：A";break;
                 case 8:
                 case 7:a="等级：B";break;
                 case 6:a="等级：C";break;
                 default:a="等级：D";
	}
        }
        else a="输入成绩非法";
        return a;
}
}
