package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		 int sum=0;
      		  while(true){
       		     if(n==1) {
         		       sum++;
         		       break;
        		    }
        		    if(n%2==0)
          		      n=n/2;
        		    else {
          		      n = (n - 1) / 2;
           		     sum++;
         		   }
   		     }
     		   return sum;
		
		
		
		/********* End *********/
	}
	
}
