package step1;

public class Task {
	
	public void sortArray(int[] arr){
        int longs;
       longs= arr.length;
        int temp;
		for(int i=1;i<longs;i++)
        {
              temp = arr[i];
            for(int j=i-1;j>=0;j--)
            {
               if(arr[j]>temp){
                    arr[j+1]=arr[j];
                    arr[j]=temp;
                    temp=arr[j];
                } 
            }
            for(int k=0;k<longs;k++)
            {
                System.out.print(arr[k]+" ");
            }
            System.out.println();
        }
		
	}
}
