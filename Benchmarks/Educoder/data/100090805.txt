package step2;
import java.util.*;
public class ListNode {
	public ListNode next;
	public int data;
	
	public ListNode() {
		next = null;
		data = Integer.MIN_VALUE;
	}
	
	public ListNode(int data) {
		next = null;
		this.data = data;
	}

	    // 获取节点的data值
    public int getData() {
        return data;
    }
    // 设置节点的data值
    public void setData(int data) {
        this.data = data;
    }
    // 返回当前ListNode的next对象
    public ListNode getNext(){
        return next;
    }
    // 设置当前ListNode的next对象
    public void setNext (ListNode next){
        this.next = next;
    
	}

}
