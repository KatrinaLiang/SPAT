package step3;

public class Task {
	public  int xx[] = {0,1,0,-1};
	public  int yy[] = {-1,0,1,0};
	public  int visited[][];
	
	public  void dfs(char[][] ferry,int x,int y,int ind) {
		int l = ferry.length;
		int w = ferry[0].length;
		if(x < 0 || x >= l || y < 0 || y >= w) {
			return ;
		}
		if(visited[x][y] >0 || ferry[x][y] !='+') {
			return ;
		}
		visited[x][y] = ind;
		for(int i=0;i<4;i++) {
			dfs(ferry,x+xx[i],y+yy[i],ind);
		}
	}
	
	public  int countOfShips(char[][] ferry) {
		int l = ferry.length;
        if(l==0){
        	return 0;
        }
		int w = ferry[0].length;
		visited = new int[l][w];
		
		int cnt = 0;
		for(int i=0;i<l;i++) {
			for(int j=0;j<w;j++) {
				if(visited[i][j]==0&&ferry[i][j]=='+') {
					dfs(ferry,i,j,++cnt);
				}
			}
		}
		
		//System.out.println(cnt);
		
		return cnt;
	}

}
