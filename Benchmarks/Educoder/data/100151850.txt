        int a = sc.nextInt();
        System.out.println("请输入第二个整数：");
        int b = sc.nextInt();
        System.out.println("请输入第三个整数：");
        int c = sc.nextInt();
        int max = 0 ;
        if (a>b && a>c)
        {
            max = a ;
        }
        if (b>a && b>c)
        {
            max = b ;
        }
        if (c>a && c>b)
        {
            max = c ;
        }
        System.out.println(a+"，"+b+"，"+c+"这三个整数中，"+max+"是最大值");

	}
