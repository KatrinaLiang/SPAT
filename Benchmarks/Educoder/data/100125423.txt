package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
	public double perimeter(){
    	return 2*(this.width+this.length);
    }
    public double area(){
    	return this.width*this.length;
    }
    public InterRectangle(){
    	this.width=0;
        
        this.length=0;
    }
    public InterRectangle(double l,double w){
    	this.width=w;
        this.length=l;
    }

	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
      	double totalarea=0,totalperimeter=0;
		for(int i=1;i<=total;i++){
            double l=scanner.nextDouble();
            double w=scanner.nextDouble();
        	InterRectangle r=new InterRectangle(l,w);
            totalarea+=r.area();
            totalperimeter+=r.perimeter();
        }
        System.out.println("总共有"+total+"块土地");
        System.out.println("土地总面积为:"+totalarea+"平米");
        System.out.println("土地总周长为:"+totalperimeter+"米");
        } 

		/*********end*********/
	}
