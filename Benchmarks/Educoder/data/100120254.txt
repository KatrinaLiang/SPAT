package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int j,k;
        for(j=1;j<=9;j++){
            for(k=1;k<=j;k++){
                System.out.print(k+"*"+j+"="+j*k+"\t");
            }
            System.out.println();
        }	
		/*****end*****/
	}
}
