//定义站台类，实现卖票的功能。
public class Station extends Thread{	 
	static int p1=20;	
    public void run(){
          f();
        
        }
   public  void f(){
       synchronized(this){ for(int i=1;i<=7;i++){ 
           if(p1!=0)System.out.println("卖出了第"+p1--+"张票");
           if(p1==0){System.out.println("票卖完了");
                    break;
                    }
       }
    }
   }
}
