package step2;

public class Task {

	public int getNum(int number){
		/********* Begin *********/
		int sum=0;
		while(number>1) {
			if(number%2==1) {
				number-=1;
				sum+=1;
			}else {
				number/=2;
			}
		}
		if(number==1) {
			sum++;
		}
		return sum;
		
		
		
		
		/********* End *********/
	}
	
}
