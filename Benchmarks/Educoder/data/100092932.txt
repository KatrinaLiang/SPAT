package step1;

public class LetterChange {

    // 使用 switch把小写字母转为大写字母
    public char LetterChange(char ch){
        // 请在此提供实现代码
        switch(ch)
        {
            case 'a': return 'A';
        	case 'b': return 'B';
        case 'c': return 'C';case 'd': return 'D';case 'e': return 'E';
        default :
        return ch;
        }
    }
}
