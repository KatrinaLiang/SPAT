package step2;

import java.io.File;

public class Task {
	
	/********** BEGIN **********/
	public void showDirStructure(File file)	{
		
	File[] fileArr = file.listFiles();
		String perStr="";
		
		int num = 0;
		for (int b=0 ;b<file.getPath().length();b++)
			if (file.getPath().charAt(b)=='/' )
				num++; 
		int level=num-2;
		for(int i=0; i<level; i++) 
			perStr += "  ";
		System.out.println(perStr+"+--"+file.getName());
		for(File f : fileArr)
			if(f.isDirectory())
				showDirStructure(f);			
			else 
				System.out.println(perStr+"  "+"--"+f.getName());
	}
	/********** END **********/
}
