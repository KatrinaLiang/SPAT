	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
		  int i0=1,i1=1,s=0;
           if(num>2)
           {
             for(int i =3;i<=num;i++)
             {   s=i0+i1;
                 i0=i1;
                 i1=s;
             }
               System.out.println("子线程计算结果为："+s);
           }
           else
               System.out.println("子线程计算结果为：1");
		
	}
}
