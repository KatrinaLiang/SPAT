package step3;

public class HelloWorld {

	public void judge(String fileName,String email){
		if(fileName.endsWith(".java")&&fileName.length()>5)System.out.println("Java文件名正确");
        else System.out.println("Java文件名无效");
        int k=email.lastIndexOf("@");
        int k1=email.lastIndexOf(".");
        if(k>k1||k==-1)System.out.println("邮箱名无效");
        else System.out.println("邮箱名正确");
	}
	
	
}
