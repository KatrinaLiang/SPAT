package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int s=0,i,n,t;
        for(i=num1;i<=num2;i++){
            t=i;
            while(t>0){
                n=t%10;
                s=s+n;
                t=t/10;
        }
        }
       return s;
		
		/********** END **********/
	}
}
