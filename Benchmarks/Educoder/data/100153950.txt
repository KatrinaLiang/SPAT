package step4;
import java.util.Scanner;
public class TestJava17_5 {
	public void print() {
		/*****start*****/
		Scanner input = new Scanner(System.in);
		int score = 0;
		int sum = 0;
		for( int i=0; i<5; ++i )
		{
			System.out.println("请输入第"+(i+1)+"门课的成绩");
			score=input.nextInt();
			sum+=score;
		}
		double avg = sum / 5;
		System.out.println("5门课的平均值是："+avg);
		input.close();
		/*****end*****/
	}

}
