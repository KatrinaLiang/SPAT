package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
		
		int[] a = new int[temperatures.length];
		for(int i = 0;i<temperatures.length;i++) 
		{
			int b = temperatures[i];
			int j;
			for( j = i;j<temperatures.length;j++) 
			{
				if(b<temperatures[j])
				{
					a[i]=j-i;
					break;
				}
			}
			if(j==temperatures.length) 
			{
				a[i]=0;
			}
				
		}
		
		

		return a;
		/********** END **********/
	}
}
