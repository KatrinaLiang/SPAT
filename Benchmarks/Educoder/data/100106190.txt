package Step1;
import java.lang.*;
public class UnicodeDemo {
     public String Encoding(String s) {
       char[] b = s.toCharArray();
         for(int i=0; i<b.length; i++){//将字符数组的每一项都转换为16进制整型字符串，不足4位的要补0
  			System.out.print("\\u"+Integer.toHexString(b[i]));
         }
         System.out.println(); 
         return s;
     }
}
