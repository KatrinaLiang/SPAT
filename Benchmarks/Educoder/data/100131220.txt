package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int s=0;
		for(int i=num1;i<=num2;i++){
        	int j=i;
            while(j>0){
            	s=s+j%10;
                j=j/10;
            }
        }
		return s;
		
		/********** END **********/
	}
}
