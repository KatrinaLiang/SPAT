package step1;
import java.util.Scanner;

public class Yard {
	int length,width;
	public Yard(){
		
	}
	public Yard(int length,int width){
		this.length=length;
		this.width=width;
	}
    public float getPerimeter(){
    	return length*2+width*2;
    }
    
    public float getAcreage(){
    	return length*width;
    }
}
