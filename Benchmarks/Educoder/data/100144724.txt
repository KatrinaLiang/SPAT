package step1;

public class Task {
	public String inversion(String str){
		/********* Begin *********/
		int s=str.length();
		char []c = new char [s];
        int j=0;
		for(int i=s-1;i>=0;i--)
		{
			c[j]=str.charAt(i);
            j++;
		}
        String q=String.valueOf(c);
        return q;
		/********* End *********/
	}
}
