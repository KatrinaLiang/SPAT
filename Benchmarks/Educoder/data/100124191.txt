package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin	********/
		int a[]=new int[100];
		a[0]=1;
		a[1]=1;
		int sum=2;
		for(int i=2;i<=19;i++) {
			a[i]=a[i-1]+a[i-2];
			sum+=a[i];
			//System.out.println(a[i]);
		}
		System.out.print("前20个元素的和为"+sum);

	/********end********/
	}
}
