package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int arr[] = new int[20];
        arr[0] = 1;
        arr[1] = 1;
        int sum = arr[0] + arr[1];
        for(int i=0;i < arr.length-2;i++)
        {
            arr[i+2] = arr[i] + arr[i+1];
            sum = sum + arr[i+2];
        }
        System.out.print("前20个元素的和为"+sum);        
	/********end********/
	}
}
