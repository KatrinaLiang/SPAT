public class GetScores {

    public int getLostScores(int[] nums){
       //请在此补全你的代码
    int i,j;
		int k=0;
		int[] data=new int[3];
		for (i=1;i<=10000;i++){	
			for (j=0;j<nums.length;j++){
				if (nums[j]==i){
					break;
				}
			}
			if (j>=nums.length){
				data[k]=i;
				k++;
			}
		}
		String s="";
		s=s+data[0];
		s=s+data[1];
		s=s+data[2];
		long sum=Long.parseLong(s);
		return (int)(sum % 11);
	}
