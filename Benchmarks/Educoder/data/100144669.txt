package step1;

public class Task {
	public String inversion(String str){
		/********* Begin *********/
		char[] chars = str.toCharArray();
	    String reverse = "";
	    for (int i = chars.length - 1; i >= 0; i--) {
	      reverse += chars[i];
	    }
	    return reverse;

		
		
		/********* End *********/
	}
}
