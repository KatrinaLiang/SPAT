package step3;

public class Task {

	 public int countOfShips(char[][] ferry) {
	  /**************BEGIN************/
	 int count=0;
     for(int i=0;i<ferry.length;i++){
         for(int j=0;j<ferry[i].length;j++){
             if(ferry[i][j]=='+'){
                 if(j==ferry[i].length-1){
                     //ä¸‹
                     for(int k=i;k<ferry.length&&ferry[k][j]=='+';k++){
                         ferry[k][j]='0';
                     }
                     count++;
                 }else if(i==ferry.length-1){
                          for(int k=j;k<ferry[i].length&&ferry[i][k]=='+';k++){
                         ferry[i][k]='0';
                     }
                     count++;
                 }else if(ferry[i][j+1]=='+'){
                      for(int k=j;k<ferry.length&&ferry[i][k]=='+';k++){
                         ferry[i][k]='0';
                     }
                     count++;
                 }else if(ferry[i+1][j]=='+'){
                     for(int k=i;k<ferry.length&&ferry[k][j]=='+';k++){
                         ferry[k][j]='0';
                     }
                     count++;
                 }else{
                     ferry[i][j]='0';
                     count++;
                 }

             }   
         }
     }
         
	 
         return count;
      /***********END ***************/
           }
         
     }
