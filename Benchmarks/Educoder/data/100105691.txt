package step1;

public class Yard {
    private int length;
    private int width;
    private float x,y;
    public Yard (int length,int width){
        this.length=length;
        this.width=width;
    }
    public float getPerimeter (){
        x=2*(length+width);
        return x;
    }
    public float getAcreage(){
        y=length*width;
        return y;
    }
}
