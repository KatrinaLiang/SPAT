package step3;

public class HelloWorld {
	public static void main(String[] args) {
		int count= 0;	//定义变量存储6的倍数出现的次数
		/*****start*****/
		int i=6;
        do{
            int s=i%6;
            if(s==0){
                count++;
            }
            i++;
        }
        while (i<101);
		

		
		/*****end*****/
		System.out.println("6的倍数出现的次数为：" + count);
	}
}
