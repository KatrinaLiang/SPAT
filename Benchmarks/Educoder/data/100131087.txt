package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int bai,shi,ge,sum=0;
		for(int i=num1;i<=num2;i++){
            bai=i/100;
            shi=i%100/10;
            ge=i%10;
            sum=sum+bai+shi+ge;
        }return sum;
        
		
		
		/********** END **********/
	}
}
