package step1;

public class Task {
	public String inversion(String str){
		/********* Begin *********/
		
		   int i, j;
        i = 0;
        j = str.length() - 1;
        char[] s = str.toCharArray();
        while (i < j) {
            char c = s[i];
            s[i] = s[j];
            s[j] = c;
            i++;
            j--;
        }
        return new String(s);
		
		/********* End *********/
	}
}
