package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
		MyThread t=new MyThread();
        t.N(num);
        t.start();
        try
        {
            t.join();
        }catch(Exception e){
            e.printStackTrace();
        }
        
		
		
	}
}


class MyThread extends Thread{
    int s,num;
    public void N(int num){
        this.num=num;
        if(num==1)
        {    
            this.s=1;
        }
        else if(num==2)
        {
            this.s=1;
        }
        else
        {
        	int a=1,b=1;
            s=0;
   		 	for(int i=3;i<=num;i++)
         	{
        		s=a+b;
        		a=b;
        		b=s;
         	}
        }
    }
    public void run(){		
        System.out.println("子线程计算结果为："+s);       
    }

	
	

}
