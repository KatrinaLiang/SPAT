package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
	public InterRectangle(){
    	this.length=6;
        this.width=5;
    }
    public InterRectangle(double length,double width){
    	this.length=length;
        this.width=width;
    }
    double perimeter(){
    	return 2*(this.length+this.width);
    }
    double area(){
    	return this.length*this.width;
    }

	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
	//	string leng=null, wid=null;
        double l,w;
        InterRectangle r=null;
        double totalArea=0,totalPeri=0;
        for(int i=1;i<=total;i++){
        	l=scanner.nextDouble();
            w=scanner.nextDouble();
            r=new InterRectangle(l,w);
            totalArea=totalArea+r.area();
            totalPeri=totalPeri+r.perimeter();
        }
        System.out.println("总共有"+total+"块土地\n土地总面积为:"+totalArea+"平米\n土地总周长为:"+totalPeri+"米");

		/*********end*********/
	}

}
