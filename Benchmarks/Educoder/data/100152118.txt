package step1;

public class Task {

	 public double solve(double x1, double y1, double r1,
              double x2, double y2, double r2) {
		  /********* Begin *********/
		    double area=0;
		  double d = 0;
		  double absX = Math.abs(x2 - x1);
		  double absY = Math.abs(y2-y1);	 
		  d = Math.sqrt(absX*absX + absY*absY );
		  if(d >r1 + r2 ){
			  area = Math.PI*(r1*r1+r2*r2);
		  }
		  else if(d < Math.abs(r2 - r1)){
			  double max = r2>r1 ? r2 : r1;
			  area = Math.PI*max*max;
		  }
		  else{
			  double A1,A2,s1,s2;    
			  A1 = 2*Math.acos((d*d+r1*r1-r2*r2)/(2*d*r1)); //r1中夹角
	          A2 = 2*Math.acos((d*d+r2*r2-r1*r1)/(2*d*r2)); //r2中夹角
	          s1 = 0.5*r1*r1*Math.sin(A1);
              s2 = 0.5*r2*r2*Math.sin(A2);
              area = Math.PI*(r1*r1+r2*r2);
	          area=area -(A1/2*r1*r1-s1)-(A2/2*r2*r2-s2);
		  }
		  return area;
		  
		  
		  
		  /********* End *********/
	  }
}
