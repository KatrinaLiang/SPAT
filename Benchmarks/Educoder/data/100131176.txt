package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		if(num1 > num2) {
			int t = num1;
			num1 = num2;
			num2 = t;
		}
		
		int sum = 0;
		for(int i=num1; i<=num2; i++) {
			int temp = i;
			while(temp != 0) {
				sum += temp % 10;
				temp /= 10;
			}
		}
		return sum;
		
		
		/********** END **********/
	}
}
