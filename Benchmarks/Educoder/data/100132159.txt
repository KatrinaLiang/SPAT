package step1;

public class Task {
	public int[] dailyTemps(Integer[] temperatures){
		int []all = new int[temperatures.length];
		
		int first = 0 ;
		for(int i=0;i<temperatures.length;i++){
			
			int p = 0;
			int isOK = 0;
			for (int j =i+1;j<temperatures.length;j++){
				p++;
				if(temperatures[j]>temperatures[i]){
					isOK = 1;
					break;
				}
			}
			if(isOK == 1){
				all[first++] = p;
			}else{
				all[first++] = 0;
			}
		}
		
		return all;
		
	}
	
}
