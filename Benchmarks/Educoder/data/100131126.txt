package step1;
import java.util.*;
public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum=0;
		for(int i=num1;i<=num2;i++){
            int a1=i/100;
            int a2=i%100;
            int b1=a2/10;
            int b2=a2%10;
            sum=sum+a1+b1+b2;
        }
		
        return sum;
		
		/********** END **********/
	}
}
