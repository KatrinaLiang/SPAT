package step5;

public class LeapYear {
	public static void main(String[] args) {
		/*********begin***********/
		int i=2000,count=0;
		while(i<=2100){ 
        	if(i%4==0&&i%100!=0||i%400==0){
                count++;
                System.out.print(i+" ");           
            }
            if (count%10==0){
                System.out.print("\n");
            }
            i++;
        }
		
		/*********end*************/

	}

}
