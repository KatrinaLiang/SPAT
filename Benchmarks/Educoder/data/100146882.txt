package step1;

public class GetScores {

	/********* Begin *********/    
    public int getLostScores(int[] nums){
       //请在此补全你的代码
    	long n = 0;
		int[] num = new int[10001];
		int[] a = new int[3];
		for(int i = 0; i < 3; i++){
			a[i] = 0;
		}
		for(int i = 0; i < nums.length; i++){
			num[nums[i]] = 1; 
		}
		for(int i = 1; i < num.length; i++){
			if(num[i] == 0){
				if(a[0] == 0) a[0] = i;
				else if(a[1] == 0) a[1] =i;
				else if(a[2] == 0) a[2] =i;
			}
		}
		for(int i = 0; i < 3; i++){
			for(int j = 2; j > 0; j--){
				if(a[j] < a[j-1]){
					int t =a[j];
					a[j] = a[j-1];
					a[j-1] =t;
				}
			}
		}
		String s = "";
		for(int i = 0; i < 3; i++){
			s += a[i];
		}
		for(int i = 0; i < s.length(); i++){
			n *= 10;
			n += s.charAt(i) - '0';
		}
		int m = 0;
		m = (int) (n % 11);
		return m;
	}

	/********* End *********/
}
