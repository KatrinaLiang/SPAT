package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		
          for(int k = 1;k<=9;k++)
        {         //外循环用于控制行数                
            for(int j = 1;j<=k;j++){                        
                System.out.print(j+"*"+k+"="+(j*k)+"\t");    
                              }                    
                System.out.println();    }    
		
		
		
		
		
		/*****end*****/
	}
}
