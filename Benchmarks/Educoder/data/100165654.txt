package step3;
import step1.Date;
public class Course {
    private int code; //课程编号
    private String courseName; //课程名称
    private Date testTime;//考试日期
    private int score;//考试成绩
   /******Begin**************/
    public Course(){
        this.code=0;
        this.courseName="";
        testTime=new Date();
        this.score=0;
    }
    public Course(int c,String cn,Date d,int s){
        this.code=c;
        this.courseName=cn;
        this.testTime=d;
        this.score=s;
    }
    public Course(int c,String cn,int y,int m,int d,int s){
        this.code=c;
        this.courseName=cn;
        this.testTime=new Date(y,m,d);
        this.score=s;
    }
    public String toString(){
        return code+" "+courseName+" "+testTime+" "+score; 
    }





   /******End************/

}
