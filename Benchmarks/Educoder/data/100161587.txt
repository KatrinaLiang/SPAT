package Pid_108;
import java.util.*;
public class Main {

	public static void main(String[] args) {
		// write your code here
        Scanner s = new Scanner(System.in);
		int a = s.nextInt();
		int count = 0;
		for(int i=1;i<a;i++){
			if(a>=i){
				a = a - i;
				count++;
			}
			else
				break;
		}
		System.out.print(count);
	}
}
