package step3;
import java.util.*;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
		System.out.println("请输入一个五位数字：");
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int p = n;
        int x;
        int temp;
        x = 0;
        for(int i = 0;i <= 5; i++)
        {
            temp = n%10;
            x += temp;
            n /= 10;
        }
        System.out.println(p+"的各位数字之和是："+x);
		/******End******/
	}
}
