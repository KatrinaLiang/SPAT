package step1;

public class Rectangle{
	private double width;
	private double length;
	
	/*********begin*********/
	public void zhouchang(int i,double w,double l){
        
        this.width=w;
        this.length=l;
        double zc=2*(this.width+this.length);
        System.out.println("r"+i+"的周长："+zc);
       
    }
    public void mianji(int i,double w,double l){
       
        this.width=w;
        this.length=l;
        double mj=width*length;
        System.out.println("r"+i+"的面积："+mj);
        
    }
	/*********end*********/


	public static void main(String[] args) {
		/*********begin*********/
       
        Rectangle a=new Rectangle();
		a.zhouchang(1,5,6);
        a.mianji(1,5,6);
        a.zhouchang(2,10,20);
        a.mianji(2,10,20);
		
		/*********end*********/
	}

}
