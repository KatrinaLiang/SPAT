package task3;

public class Box {

    double width;
    double height;
    double depth;

    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public double volume(int boxType) {
        //TODO write your code here
        switch (boxType) {
            case 1:return this.width*this.height*this.depth;
            case 2:return (this.width*this.width)*3.14*this.height;
        }
        return 1;
    }

}
