package step4;		//在这里添加包名  step4

	
        //创建类 添加属性和方法
	class WuMingFen{
        String theMa;
        int quantity;
        boolean likeSoup;
        
		public WuMingFen(String n,int q,boolean l){
            this.theMa=n;
            this.quantity=q;
            this.likeSoup=l;
   		}
        public WuMingFen(String n,int q){
            this.theMa=n;
            this.quantity=q;
        }
        public WuMingFen(){
            this.theMa="酸辣";
            this.quantity=2;
            this.likeSoup=true;
        }
        void check(){
            System.out.println("面码："+theMa+"，粉的份量："+quantity+"两"+"，是否带汤："+likeSoup);
        }
  	}
