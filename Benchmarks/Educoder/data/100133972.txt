		// 实例化子类对象s，调用talk()方法打印信息

Student a = new Student();
		
		a.voice("哈佛大学","张三，",18);

class Person {
	/********* begin *********/
 String name;
	 int age;
	 public void voice(String name,int age) {
			this.name=name;
			this.age=age;
		}
	 public String geteat() {
			return name;
			
			
		}
	 public int geteat2() {
			return age;
			
		}
	/********* end *********/
}

class Student extends Person {
	/********* begin *********/
private String school;


public void voice(String school,String name,int age) {
	super.voice(name, age);
	this.school=school;
	
}
public String get() {
	return school;
	
}
public void test() {
	System.out.println("我是："+super.geteat()+"今年："+super.geteat2()+"岁，我在"+this.get()+"上学");
}
	/********* end *********/
}
