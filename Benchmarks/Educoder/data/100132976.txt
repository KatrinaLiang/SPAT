package step1;

public class Task {
	
	public void sortArray(int[] arr){
		for(int i = 0;i < arr.length-1;i++){
			for(int j = i;j>=0;j--){
				if(arr[j+1]<arr[j]){
					int temp = arr[j+1];
					arr[j+1] = arr[j];
					arr[j] = temp;
				}
				if(arr[j+1]>arr[j])
					continue;
			}
			for(int k = 0;k < arr.length;k++){
				System.out.print(arr[k]+" ");
			}
            System.out.println();
		}
		
	}
}
