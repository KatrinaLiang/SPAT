package step4;
		//创建类 添加属性和方法
public class WuMingFen{
    String theMa;
    int quantity;
    boolean likeSoup;
    public WuMingFen(String t,int q,boolean l){
        theMa=t;
        quantity=q;
        likeSoup=l;
    }
    public WuMingFen(String t,int q){
        theMa=t;
        quantity=q;
    }
    public WuMingFen(){
        theMa="酸辣";
        quantity=2;
        likeSoup=true;
    }
    public void check(){
        System.out.println("面码："+theMa+"，粉的份量："+quantity+"两，是否带汤："+likeSoup);
    }
        
}
