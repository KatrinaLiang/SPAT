package step1;

public class Person {
    private String name; //姓名
    private String number; //身份证号
    private Date birthday; //出生日期 
   /************Begin*************/
    public Person() {
    	Date d2=new Date();
    	birthday=d2;
    }
    public Person(Date birthday){
        this.birthday=birthday;
    }
	public Person(String name,String num,Date birthday){
    	this.name=name;
        this.number=num;
        this.birthday=birthday;
    }
    public Person(String name,String num,int year,int month,int day){
    	this.name=name;
        this.number=num;
        Date d3=new Date(year,month,day);
        //d3.setDate(year, month, day);
        birthday=d3;
    }
	public String name(){
    	return name;
    }
    public String num(){
    	return number;
    }
    public Date birthday(){
    	return birthday;
    }
    public String toString(){
        return "name="+name+" number=" +number+" birthday:"+birthday;
    }

   /************End*************/
}
