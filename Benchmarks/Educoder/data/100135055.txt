class Person {
	/********* begin *********/
    private String name;
    private int age;
    public void set(String name,int age){
        this.name=name;
        this.age=age;
    }
    public String getname(){
        return this.name;
    }
    public int getage(){
        return this.age;
    }
	/********* end *********/
}

class Student extends Person {
	/********* begin *********/
    public String school;
    public void setschool(String s){
        this.school=s;
        super.set("张三",18);
    }
    public String getschool(){
        return this.school;
    }
	/********* end *********/
}
