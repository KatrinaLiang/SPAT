package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		
		int temp;
		int n=matrix.length;
		if(n%2==0) {
			for(int i=0;i<(n+1)/2;i++) {
				for(int j=0;j<(n+1)/2;j++) {
					temp=matrix[i][j];
					matrix[i][j]=matrix[n-j-1][i];
					matrix[n-j-1][i]=matrix[n-i-1][n-j-1];
					matrix[n-i-1][n-j-1]=matrix[j][n-i-1];
					matrix[j][n-i-1]=temp;
				}
			}
		}
		else {
			for(int i=0;i<(n+1)/2-1;i++) {
				for(int j=0;j<(n+1)/2;j++) {
					temp=matrix[i][j];
					matrix[i][j]=matrix[n-j-1][i];
					matrix[n-j-1][i]=matrix[n-i-1][n-j-1];
					matrix[n-i-1][n-j-1]=matrix[j][n-i-1];
					matrix[j][n-i-1]=temp;
				}
			}
		}

		/********* End *********/
	}
}
