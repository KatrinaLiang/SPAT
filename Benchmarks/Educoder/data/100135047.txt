class Person {
	/********* begin *********/
	String name;
    int age;
    String school;
	/********* end *********/
}

class Student extends Person {
	/********* begin *********/
	public Student(String name,int age,String school)
    {
        this.name=name;
        this.age=age;
        this.school=school;
    }
    public void print()
    {
        System.out.println("姓名："+this.name+"，年龄："+this.age+"，学校："+this.school);
    }
	/********* end *********/
}
