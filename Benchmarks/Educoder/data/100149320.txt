    public boolean fitHeart1(double x, double y) {
		double res = Math.pow(x, 2) + Math.pow((5 * y / 4.0 - Math.sqrt(Math.abs(x))), 2) - 1; 
		return res < 0;
	}	
	public boolean fitHeart(double x, double y) {
		double res = Math.pow((Math.pow(x, 2) + Math.pow(y, 2) - 1), 3) - 
				Math.pow(x, 2) * Math.pow(y, 3); 
		
		return res >= 0;
	}
