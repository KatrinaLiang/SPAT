#include<stdio.h>
#include<iostream>
using namespace std;
int cf(int x){
	int m=1;
	for(int i=0;i<x;i++){
		m*=3;
	}
	return m;
}
int main(){
	int k,max,n;
	cin>>k;
	//scanf("%d",&k);
	if(k==1) max=0;
	else if(k==2) max=1;
	else if(k==3) max=2;
	else if(k==4) max=4;
	else{
		if(k%3==1){
			n=k/3-1;
			max=cf(n)*4;
		}
		else if(k%3==2){
			n=k/3;
			max=cf(n)*2;
		}
		else{
			n=k/3;
			max=cf(n);
		}
	}
	printf("%d",max);
}
