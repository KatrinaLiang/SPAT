public class HelloWorld {
    public static void main(String[] args) {


        w();// 调用无参的方法

        z("educoder");// 调用带有一个字符串参数的方法

        zh(666);// 调用带有一个整型参数的方法
        
    }
    
    public static void w()// 无参print方法的定义
    {
        System.out.println("无参的print方法");
    }
    
     public static void z(String z)
    {
        System.out.println("带有一个字符串参数的print方法，参数值为："+z);
    }// 字符串print方法的定义
    
    
    
     public static void zh(int zh)
    {
        System.out.println("带有一个整型参数的print方法，参数值为："+zh);}// 整型print方法的定义
    
}
