package step1;

public class A< T extends Comparable >{
	private T t1,t2; 
/*************BEGIN**************/
	public A(T a1, T a2){
        this.t1 = a1;
        this.t2 = a2;
    }
    public T compare(){
        int judge = this.t1.compareTo(this.t2);
        if(judge > 0){
            return this.t1;
        }
        else if(judge < 0){
            return this.t2;
        }
        else{
            return this.t1;
        }
    }




/*************END*****************/

}
