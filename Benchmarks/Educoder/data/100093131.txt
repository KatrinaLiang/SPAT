package step3;

public class SeasonJudge {
    public static String SPRING = "春季";

    public static String SUMMBER = "夏季";

    public static String AUTUMN = "秋季";

    public static String WINTER = "冬季";
    // 根据月份判断季节
	public String SeasonJudge(int month) {
        String result = "";
        switch (month) {
            case 1:
                result = WINTER;
                break;
            case 2:
                result = WINTER;
                break;
            case 3:
                result = SPRING;
                break;
            case 4:
                result = SPRING;
                break;
            case 5:
                result = SPRING;
                break;
            case 6:
                result = SUMMBER;
                break;
            case 7:
                result = SUMMBER;
                break;
            case 8:
                result = SUMMBER;
                break;
            case 9:
                result = AUTUMN;
                break;
            case 10:
                result = AUTUMN;
                break;
            case 11:
                result = AUTUMN;
                break;
            case 12:
                result = WINTER;
                break;
             default:
                 result = "月份错误";

        }
        return result;
	}
}
