package step3;

public class PalindromeJudge {
	// 判断一个五位正整数是否是回文数
	public boolean PalindromeJudge(int num){
	  int a1,a2,a3,a4,a5;
        a1=num%10;
        num/=10;
         a2=num%10;
        num/=10;
         a3=num%10;
        num/=10;
         a4=num%10;
        num/=10;
         a5=num%10;
        num/=10;
        if(a1==a5&&a2==a4){
            return true;
        }
        else{
        return false;
        }
}
}
