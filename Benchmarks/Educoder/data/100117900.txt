package step3;

import java.util.Scanner;

public class HelloWorld {
    
    public static void main(String[] args) {
        HelloWorld getMax=new HelloWorld();
		Scanner sc = new Scanner(System.in);
		int i = sc.nextInt();   //获取第一个输入的整数
        int j = sc.nextInt();	//获取第二个输入的整数

		//在这里调用方法获取返回值
int max;
        max=getMax.lalala(i,j);

		System.out.println( i+"和"+j+"比较，最大值是：" +max   );

    }

	/*在这里创建getMax方法，以两个整数作为参数，返回两个整数中较大的值*/
    public int lalala (int a,int b){
   int max1=a;
   if(a<b){
     max1=b;
   }
   return max1;

    }
}
