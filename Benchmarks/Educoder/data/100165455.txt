package step2;
import step1.Person;
import step1.Date;
public class Student extends Person{
    private int reg_num; //学号
    private Date reg_day; //注册日期
   // TO DO YOU CODE HERE
   public Student(){
	   super();
	   reg_num = 0;
	   reg_day = new Date();
   }
   public Student(String name, String num, int year1, int month1, int day1,
   				int regNum, int year2, int month2, int day2){
		super(name, num, year1, month1, day1);
		this.reg_num = regNum;
		this.reg_day = new Date(year2, month2, day2);
	}
	public Student(String name, String num, Date date1, int regNum, Date date2){
		super(name, num, date1);
		this.reg_num = regNum;
		this.reg_day = date2;
	}
	public void setinfo(String name, String num, int year1, int month1, int day1,
				int regNum, int year2, int month2, int day2){
		super.setinfo(name, num, year1, month1, day1);
		this.reg_num = regNum;
		this.reg_day = new Date(year2, month2, day2);
	}
	public void setinfo(String name, String num, Date date){
		super.setinfo(name, num, date);
		
	}
	public int getreg_num(){
		return this.reg_num;
	}
	public Date getreg_day(){
		return this.reg_day;
	}
	public String toString(){
		return super.toString()+
		"reg_num:"+this.reg_num+"reg_day:"+this.reg_day.getYear()+"."+this.reg_day.getMonth()+
		"."+this.reg_day.getDay();
	}
	





}
