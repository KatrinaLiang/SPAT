package step3;

public class HelloWorld {
	public static void main(String[] args) {
		int count= 0,i=6;	//定义变量存储6的倍数出现的次数
		/*****start*****/
		do{
			i = i + 6;
			count++;
		}while((i>=0&&i<100) && i % 6 == 0);
		

		
		/*****end*****/
		System.out.println("6的倍数出现的次数为：" + count);
	}
}
