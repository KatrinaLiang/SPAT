//创建Person对象，并创建两种构造方法
class Person{
    String name;
    String sex;
    public Person(String name,String sex){
        this.name=name;
        this.sex=sex;
    }
    public Person(){
        System.out.println("一个人被创建了");
    }
    void p(String name,String sex){
        System.out.println("姓名："+this.name+"，性别："+this.sex+"，被创建了");
    }
}
