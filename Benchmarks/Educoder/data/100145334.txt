package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int s1;
        int sum=0;
        int i=0;
        while(i<1){
           s1=n%2;
            sum=sum+s1;
           n=(n-s1)/2;
        if(n==1){
           sum++;
            break;
        }
        }
		//System.out.println(sum);
		return sum;
		
		
		/********* End *********/
	}
	
}
