package step4;

import java.util.Scanner;

public class Test {
	
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		String theMa = sc.next();
		int quantity = sc.nextInt();
		boolean likeSoup = sc.nextBoolean();
         WuMingFen f1=new WuMingFen(theMa,quantity,likeSoup);
        WuMingFen f2=new WuMingFen(theMa,quantity);
        WuMingFen f3=new WuMingFen();
		//使用三个参数的构造方法创建WuMingFen对象  取名 f1
		//使用两个参数的构造方法创建WuMingFen对象  取名 f2
		//使用无参构造方法创建WuMingFen对象  取名 f3
		//分别调用三个类的 check方法
		f1.check();
        f2.check();
        f3.check();
        
	}
}
class WuMingFen{
    String theMa="酸辣";
    int quantity=2;
    boolean likeSoup=true;
    WuMingFen(String theMa, int quantity, boolean likeSoup)
    { 
        this(theMa,quantity);
        this.likeSoup=likeSoup;
    }
    WuMingFen(String theMa, int  quantity)
    {
        this.quantity=quantity;
        this.theMa=theMa;
        this.likeSoup=false;
    }
    WuMingFen()
    {
         
    }
    public void check()
    {
        System.out.println("面码："+theMa+"，粉的份量："+quantity+"两"+"，是否带汤："+likeSoup);
    }
}
