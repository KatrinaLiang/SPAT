package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int count = 0;
		for (int i = num1; i <= num2; i++) {
			String a = String.valueOf(i);
			for(int j = 0;j<a.length();j++) {
				String charAt = a.charAt(j)+"";
				int nu1 = Integer.parseInt(charAt);
				count+=nu1;
			}
		}
		return count;
		
		
		/********** END **********/
	}
}
