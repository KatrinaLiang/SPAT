package step1;

public class DecomposeFactor {
	public void decompose(int score) {
		/********** Begin *********/
        int x = score;
        double ans;
        if(x>=90) ans=4.0;
        else if(x>=80) ans=3.5;
        else if(x>=70) ans=3;
        else if(x>=60) ans=2.5;
        else {
        	ans=0.0;
        }
        System.out.print(ans);
		/********** End *********/
	}
}
