package step2;

class Task{
	    public int getNum(int n) {
        if(n == 0) {
            return 0;
        }
        else if(n == 1) {
            return 1;
        }
        if(n % 2 == 0) {
            return getNum(n /2);
        }
        else {
            return 1 + getNum(n / 2);
        }

    }
}
