package chapter1.step2;

public class Area{	
public static void main(String[] args) {
	    /********** Begin *********/
	    //声明一个常量PI的值为3.14
	    float pi=3.14f;
	    //变量radius为2
		int radius=2;
	    //计算半径为2的圆面积并输出语句`The area  of the circle is 12.56`
		float area=pi*radius*radius;
		System.out.print("The area  of the circle is "+area);	
		/********** End *********/
}
}
