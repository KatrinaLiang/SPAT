package task4;

public class Box {

    double width;
    double height;
    double depth;

    int boxType;

    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public Box(int boxType){
       this.boxType= boxType;
    }
    public double volume() {
        //TODO your code
        double b=0;
       switch(boxType){
           case 1:b=width*height*depth;
               break;
           case 2:b=width*height*depth*3.14;
               break;
       }
        return b;
    }

}
