package step1;
import java.util.Scanner;

public class AreaOfRec {
	public static void main(String[] args) {
		/******Begin******/
		int length = 3;     //长方形的长度
        int width = 4;      //长方形的宽度
        int area = length*width;    //长方形的面积计算公式
        Scanner w = new Scanner(System.in);
        System.out.println("请输入长方形的长："); //请输入长
        length = w.nextInt();
        System.out.println("请输入长方形的宽："); //请输入宽
        width=w.nextInt();
        area = length*width; 
        System.out.println("长方形的面积是："+area);
		/******End******/
	}

}
