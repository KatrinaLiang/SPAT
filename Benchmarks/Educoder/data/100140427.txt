package step3;

public class HelloWorld {

	public void judge(String fileName,String email){
        
		if(fileName.lastIndexOf(".java")>0&&fileName.lastIndexOf(".java")+5==fileName.length())
            System.out.println("Java文件名正确");
        else
            System.out.println("Java文件名无效");
        if(email.indexOf('@')>0&&email.indexOf('.')>email.indexOf('@')&&email.substring(email.indexOf('.')).indexOf('@')<0)
            System.out.println("邮箱名正确");
        else
            System.out.println("邮箱名无效");
        
	}
	
	
}
