package step1;

public class Yard {
    private int length,width;
    public Yard(int a,int b){
        length=a;
        width=b;
    }
    public double getPerimeter(){
        double x;
        x=(length+width)*2.0;
        return x;
    }
    public double getAcreage(){
        double y;
        y=length*width/1.0;
        return y;
    } 
}
