package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
		
		//在这里打印出正三角形 
		for(int q=0;q<10;q++) {
			for(int p=0;p<q;p++) {
				System.out.print("*");
			}
			System.out.println("*");
		}
		
		
		
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
		for(int q=0;q<10;q++) {
			for(int p=9;p>q;p--) {
				System.out.print("*");
			}
			System.out.println("*");
		}
		
		
		
		
		/*****end*****/
	}
}
