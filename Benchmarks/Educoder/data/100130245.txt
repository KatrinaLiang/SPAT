package step4;
		//创建类 添加属性和方法
class WuMingFen {
    String theMa;
    int quantity;
    boolean likeSoup;
    WuMingFen(String theMa, int quantity,  boolean likeSoup) {
        this.theMa = theMa;
        this.quantity = quantity;
        this.likeSoup = likeSoup;
    }
    WuMingFen(String theMa, int quantity) {
        this.theMa = theMa;
        this.quantity = quantity;
        this.likeSoup = false;
    }
    WuMingFen() {
        this.theMa = "酸辣";
        this.quantity = 2;
        this.likeSoup = true;
    }
    void check() {
        System.out.println("面码："+theMa+"，粉的份量："+quantity+"两，是否带汤："+likeSoup);
    }
}
