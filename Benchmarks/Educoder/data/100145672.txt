package step4;

public class Task {
	
	public String getDate(String releaseDate, int day) {
		int year = Integer.valueOf(releaseDate.substring(0, 4));
		int month = Integer.valueOf(releaseDate.substring(5, 7));
		int d = Integer.valueOf(releaseDate.substring(8, 10));
		boolean isRun = false;
		int add = 0;
		for (int i = 1; i < day + 1; i++) {
			add = 0;
			if (year % 4 == 0 && year % 100 != 0)
				isRun = true;
			else if (year % 400 == 0)
				isRun = true;
			else
				isRun = false;
			if (isRun)
				add = 1;
			int[] days = { 31, 28 + add, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
			d = d + 1;
			if (d > days[month - 1]) {
				d = 1;
				if (month < 12)
					month += 1;
				else {
					year += 1;
					month = 1;
				}
			}
		}
		return String.format("%04d", year) + "-" + String.format("%02d", month) + "-" + String.format("%02d", d);
	}
	
	
}
