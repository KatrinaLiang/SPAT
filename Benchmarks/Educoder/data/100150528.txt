import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		/******Begin******/
	    int i=sc.nextInt();
		int j=i/10000;
        int n=i%10000/1000;
        int m=i%1000/100;
        int k=i%100/10;
        int c=i%10;
        int sum=j+n+m+k+c;
        System.out.println("请输入一个五位数字：");
      
          System.out.println(i+"的各位数字之和是："+sum);
		/******End******/
	}
}
