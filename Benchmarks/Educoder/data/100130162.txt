package step4;//在这里添加包名  step4

public class WuMingFen{
    String theMa="酸辣";
    int quantity=2;
    boolean likeSoup=true;
    public WuMingFen(String ma,int q,boolean l){
        this.theMa=ma;
        quantity=q;
        likeSoup=l;
    }
    public WuMingFen(String ma,int q){
        this.theMa=ma;
        quantity=q;
        likeSoup=false;
    }
    public WuMingFen(){
        
    }
    public void check(){
        System.out.println("面码："+theMa+"，粉的份量："+quantity+"两，是否带汤："+likeSoup);
    }
}//创建类 添加属性和方法
