package step1;

public class Yard {
    float length;
    float width;
	public Yard(int length, int width)
    {
        this.length = (float)length;
        this.width = (float)width;
    }
    public float getPerimeter()
    {
		return 2 * (this.length + this.width);
    }
    public float getAcreage()
    {
        return this.length * this.width;
    }
}
