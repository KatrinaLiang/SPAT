package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int sum = 2;
		int[] arr = new int[21];
		arr[0] = 0;
		arr[1] = 1;
		arr[2] = 1;
		for(int i = 3; i <= 20;i++){
			arr[i] = arr[i-1] + arr[i-2];
			sum += arr[i];
		}
		System.out.print("前20个元素的和为" + sum);	

	/********end********/
	}
}
