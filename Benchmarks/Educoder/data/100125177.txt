package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
	public InterRectangle(double a, double b){
          width = a;
          length = b;
      }
  
      public double perimeter(){
          return width * 2 + length * 2;
      }
  
      public double area(){
          return width * length;
      }

	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		int total=scanner.nextInt();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
        double q = 0, w = 0;
		for (int i = 0; i < total; ++i){
        	double a=scanner.nextDouble();
            double b = scanner.nextDouble();
            InterRectangle l = new InterRectangle(a, b);
            q += l.area();
            w += l.perimeter();
        }
		System.out.println("总共有" + (double)total + "块土地");
        System.out.println("土地总面积为:" + q + "平米");
        System.out.println("土地总周长为:" + w + "米");
		/*********end*********/
	}

}
