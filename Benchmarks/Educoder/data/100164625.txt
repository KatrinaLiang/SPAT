            this.c=c;//YOUR CODE HERE
		}
       
        public double getX1(){
            double x;
            x=(-b+Math.sqrt((b*b)-(4*a*c)))/(2*a);
            return x;
        }
        public double getX2(){
            double x2;
            x2=(-b-Math.sqrt((b*b)-(4*a*c)))/(2*a);
            return x2;
        }
        public double deta(){
            double number=(b*b)-(4*a*c);
            return number;
        }
