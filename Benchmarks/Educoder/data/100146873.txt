package step1;
import java.util.Arrays;
public class GetScores {

	/********* Begin *********/
    public int getLostScores(int[] nums){
     
        
           Arrays.sort(nums);
	    int[] num=new int[3];
	    int id=0;
		for(int i=1;i<nums.length;i++) {
			int a=nums[i-1];
			int b=nums[i];
			while(b-a>1) {
				num[id]=a+1;
				a=num[id];
				id++;
			}
		}
		Arrays.sort(num);
		Long res;
		String str="";
		for(int i=0;i<3;i++)
		{
			str+=num[i];
			//System.out.println(num[i]+"***");
		}
		res= new Long(str);
		long ans= res%(long)11;
		return (int)ans;
	}

	/********* End *********/
}
