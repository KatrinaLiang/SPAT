package step3;

public class PalindromeJudge {
	// 判断一个五位正整数是否是回文数
	public boolean PalindromeJudge(int num){
		int a1=num%10;
        num/=10;
        int a2=num%10;
        num/=10;
        int a3=num%10;
        num/=10;
        int a4=num%10;
        num/=10;
        int a5=num%10;
        num/=10;
        
        if(a1==a5&&a2==a4)
            return true;
        else
            return false;
}
}
