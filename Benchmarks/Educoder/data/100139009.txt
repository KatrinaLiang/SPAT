package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
        MyThread a=new MyThread();
		System.out.println("子线程计算结果为："+a.sum(num));
		
		
	}
}


class MyThread     {
	public	int sum(int num)
        {
            
            if(num==1||num==2)
                return 1;
            else
            {
            	return sum(num-1)+sum(num-2);
            }
        
        }

	
	

}
