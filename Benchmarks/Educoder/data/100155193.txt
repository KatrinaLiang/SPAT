package task2;

public class Box {
      double width;
      double height;
      double depth;

      public static void main(String[] args){
    	  Box box=new Box();
    	  box.width=10;
    	  box.height=20;
    	  box.depth=10;
    	  double x=box.volume();
    	  System.out.println("the volumn is:"+x);
      }
      public double volume() {
    	  double v=width*height*depth;
    	  return v;
      } 
}
