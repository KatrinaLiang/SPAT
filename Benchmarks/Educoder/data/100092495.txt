package step3;
import java.util.Scanner;

public class FindMaxNum {
    // 找出三个整数的最大值
    public int FindMaxNum(int a, int b, int c){
        // 请在此添加实现代码
	int max=0;
	if(a>b){
		max=a;
	}else{
		max=b;
	}
	if(c>max){
		max=c;
	}
	return max;
}
public static void main(String[] args) {
	Scanner sc=new Scanner(System.in);
	int a=sc.nextInt();
	int b=sc.nextInt();
	int c=sc.nextInt();
	FindMaxNum fin=new FindMaxNum();
	System.out.println(fin.FindMaxNum(a, b, c));
	}
}
