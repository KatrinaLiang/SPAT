package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		/********** BEGIN **********/
        boolean isOdd = ((n&1) == 1);
        int startNum = isOdd?1:2;
        double mon = isOdd?4.0:3.0;
        double sum = 0;
        
        for(int i=startNum;i<=n;i+=2){
            sum += (mon/i);
        }
        System.out.println(sum);

        
		/********** END **********/
	}
}
