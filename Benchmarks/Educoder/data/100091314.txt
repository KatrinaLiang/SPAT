package step2;
import java.util.Arrays;

public class TriangleJudge{

    // 判断三条边能否构成一个三角形
    public boolean TriangleJudge(int a, int b, int c) {
        // 请在此添加实现代码
        int []d=new int [4];
        d[0] =a;
        d[1] =b;
        d[2] =c;
        Arrays.sort(d);
        if (a+b>c&&a-b<c)
            return true;
        else
            return false;
        
    }
}
