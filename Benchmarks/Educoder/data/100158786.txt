package Q103;
import java.util.*;
public class Main {

	public static void main(String[] args) {
		Scanner in = new Scanner (System.in);
        int a=in.nextInt();
        System.out.print(zhuan(a));
	}
    
    public static int zhuan(int n){
    if(n==1||n==0)
        return 1;
    else
        return zhuan(n-1)+zhuan(n-2);        
    }
    

}
