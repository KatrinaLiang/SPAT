#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int i,j,k;
    char ch;
    int sum=0;
    int T;
    int a[10000];
    int shu;
    int n=0;
    memset(a,0,sizeof(int));
    while(scanf("%d",&shu)){
       a[n]=shu;
       n++;
       ch=getchar();
       if(ch=='\n') break;
    }
    while(scanf("%d",&T)!=EOF){
    	sum=0;
    }
    for(i=0;i<=n-3;i++){
        for(j=i+1;j<=n-2;j++){
            for(k=j+1;k<=n-1;k++){
                if(a[i]+a[j]+a[k]<T) sum++;
            }
        }
    }
    printf("%d",sum);
}
