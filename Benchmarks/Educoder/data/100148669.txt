/********* Begin *********/
//定义站台类，实现卖票的功能。
public class Station extends Thread{	 
	static int ticket=20;
    static Object obj=new Object();
    public void ticket(){
    	System.out.println("卖出了第"+ticket+"张票");
        ticket--;
    }
    public void run(){
    	while(ticket>0){
        	synchronized(obj){
            	if(ticket>0){
                    ticket();
                }
            }
            if(ticket==0){
            	System.out.println("票卖完了");
            }
            try{
             	Thread.sleep(10);   
            }catch(Exception e){
            	e.printStackTrace();
            }
        }
    }
	
}
/********* End *********/
