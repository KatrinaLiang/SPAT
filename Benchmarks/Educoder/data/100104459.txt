package step2;

public class ConverseNumber {
	public void Converse (int n) {
		/********** Begin *********/
        int a=0;
        while(n>10){
            a*=10;
            a+=n%10;
            n/=10;
        }
        a*=10;
        a+=n;
        System.out.print(a);

        /********** End *********/
	}
}
