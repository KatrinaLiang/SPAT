package step3;

public class Factorial {
	public void factorial(int n) {
		/********** Begin *********/

  int b = 1;
  for(int a = 1; a<=n;a++){
	  b = b*a;
  }
  System.out.print(b);
    }
     public static void main(String[] args) {
         Factorial c = new Factorial();
         int d = 4;
         int e = 7;
         c.factorial(d);
         c.factorial(e);

		/********** End *********/
	}
}
