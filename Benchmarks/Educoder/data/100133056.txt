	public static void birthdayCandle(int sum){
		int start;
        int count;
        int end=0;
        if(sum>36){
            a:for(start=18;start<(sum+1)/2;++start){
                for(count=1;count<(sum+1)/2;++count){
                    if(  (start*count)+(count*(count-1))/2==sum){

                         end=start+count-1;
                        break a;
                    }
                }
            }
        }
        else{
            start=sum;
            end=sum;
        }

	}
