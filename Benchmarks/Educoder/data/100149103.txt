package step2;

public class Task {
    
	public int virusArea(int n,int m,char[][] area){
		int num=0;
		if(area==null||area.length==0) {
			return 0;
		}
		
		for(int i=0;i<n;i++) {
			for(int j=0;j<m;j++) {
				if(area[i][j]=='o') {
					area[i][j]='x';
					dfs(area,i,j,n,m);
					++num;
				}
				
			}
		}
		
		return num;
	}
	public static void dfs(char[][] area,int i,int j,int n,int m) {
		
		if(i-1>=0&&area[i-1][j]=='o') {
			area[i-1][j]='x';
			dfs(area,i-1,j,n,m);
		}
		if(i+1<n&&area[i+1][j]=='o') {
			area[i+1][j]='x';
			dfs(area,i+1,j,n,m);
		}
		if(j-1>=0&&area[i][j-1]=='o') {
			area[i][j-1]='x';
			dfs(area,i,j-1,n,m);
		}
		if(j+1<m&&area[i][j+1]=='o') {
			area[i][j+1]='x';
			dfs(area,i,j+1,n,m);
		}
	}
}
