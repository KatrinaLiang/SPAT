package case1;
import java.util.Scanner;
abstract class Pet{
    public String name;
    public String sex;
    Pet(){}
    Pet(String name,String sex){}
    abstract void talk();
    abstract void eat();    
	
}   
    class Dog extends Pet{
        public String color;
        Dog(){}
        Dog(String name,String sex,String color){
            this.name=name;
            this.sex=sex;
            this.color=color;
        }

    	void talk(){
            System.out.println("名称："+this.name+"，性别："+this.sex+"，颜色："+this.color+"，汪汪叫");
        }
        
        void eat(){
            System.out.println(this.name+"吃骨头！");
        }
    } 

    class Cat extends Pet{
        double weight;
        Cat(){}
        Cat(String name,String sex,double weight){
            this.name=name;
            this.sex=sex;
            this.weight=weight;
           
        }

    	void talk(){
             System.out.println("名称："+this.name+"，性别："+this.sex+"，体重："+this.weight+"kg，喵喵叫");
        }
        
        void eat(){
            System.out.println(this.name+"吃鱼！");
        }
    }
    
    public class Task1{
    public static void main(String[] args){
    	String name;
    	String sex;
    	String color;
    	double weight;
    	Dog dog = new Dog();
    	Cat cat = new Cat();
        Scanner input = new Scanner(System.in);
    	dog.name = input.next();
    	dog.sex = input.next();
    	dog.color = input.next();
    	cat.name = input.next();
    	cat.sex = input.next();
    	cat.weight = input.nextDouble();
    	dog.talk();
    	dog.eat();
    	cat.talk();
    	cat.eat();
    	
    }

  }
