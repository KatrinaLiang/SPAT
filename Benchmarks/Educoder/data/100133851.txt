class Animal {
	/********* begin *********/
    private String name;
    private int age;
    public String getName(){
        return name;
    }
    public void setName(String name){
        this.name=name;
    }
    public int getAge(){
        return age;
    }
    public void setAge(int age){
        this.age=age;
    }
	/********* end *********/

}

class Cat extends Animal {
	// 定义Cat类的voice()和eat()方法
	/********* begin *********/
    String name="大花猫";
    int age=6;
    void voice(){
        System.out.println(name+"喵喵叫");
    }
    void eat(){
        System.out.println(name+"吃鱼");
    }
	/********* end *********/
}
