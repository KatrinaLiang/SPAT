package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int ji;
		for(int i=1;i<10;i++){
			for(int j=1;j<=i;j++){
				ji=i*j;
				System.out.print(j+"*"+i+"="+ji+"\t");
			}
			System.out.println();
		}
		
		
		/*****end*****/
	}
}
