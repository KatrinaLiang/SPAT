package step2;

import java.util.Scanner;
import java.util.Arrays;

public class MarkStatistics {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int[] class1Grade = new int[5];
        int[] class2Grade = new int[5];
        int[] gradeCount = new int[class1Grade.length + class2Grade.length];
        /********** Begin *********/
        for (int i = 0;i < 5; i++)
        {
            //class1Grade[i] = scanner.nextInt();
            int a = scanner.nextInt();
            class1Grade[i] = a;
        }
      /*  for (int i = 0;i<5; i++)
        {
            System.out.print(class1Grade[i]);
        }
        System.out.println();*/
        for (int i = 0;i < 5;i++)
        {
            //class2Grade[i] = scanner.nextInt();
            int a = scanner.nextInt();
            class2Grade[i] = a;
        }
     /*           for (int i = 0;i<5; i++)
        {
            System.out.print(class2Grade[i]);
        }
        System.out.println();*/
        for (int j = 0;j < 10;j++)
        {
            if (j<=4)
            {
                gradeCount[j] = class1Grade[j];
            }
            else{
                gradeCount[j] = class2Grade[j-5];
            }
        }
        
        Arrays.sort(gradeCount);
        /********** End *********/
        for (int i = 0; i < gradeCount.length; i++) {
            System.out.print(gradeCount[i]);
            if (i != gradeCount.length - 1) {
                System.out.print(' ');
            }
        }
    }
}
