public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        int t=a;
        int c=a;
        int x=1;
		while(x<n){
            t=t*10+a;
            c=t+c;
            x=x+1;
        }
		System.out.print(c);

		/********** End *********/
	}
}
