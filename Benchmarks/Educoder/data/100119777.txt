package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
		
		//在这里打印出正三角形 
		int i;
        int k;
		for(i=1;i<=10;i++){
            for(k=1;k<=i;k++) System.out.print("*");
            System.out.println();
        }
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
		for(i=1;i<=10;i++){
            for(k=1;k<=(11-i);k++) System.out.print("*");
            System.out.println();
        }
		/*****end*****/
	}
}
