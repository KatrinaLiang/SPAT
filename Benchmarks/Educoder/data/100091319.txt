package step2;

public class TriangleJudge{

    // 判断三条边能否构成一个三角形
    public boolean TriangleJudge(int a, int b, int c) {
        // 请在此添加实现代码
        boolean rt = true;
        if (a+b<=c)
            rt = false;
        if (b+c<=a)
            rt = false;
        if (a+c<=b)
            rt = false;
        return rt;
                                
    }
}
