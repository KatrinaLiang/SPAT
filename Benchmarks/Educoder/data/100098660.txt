package step1;

class Yard {
    private double chang;
    private double kuan;
    private double perimeter;
    private double area;
    
    public Yard(double chang,double kuan)
    {
        this.chang=chang;
        this.kuan=kuan;
    }
    public double getPerimeter(){
         return 2*(this.chang+this.kuan);
    }
    public double getAcreage(){
         return this.chang*this.kuan;
    }
    

}
