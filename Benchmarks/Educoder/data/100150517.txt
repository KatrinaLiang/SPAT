package step3;
import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
		  Scanner scan=new Scanner(System.in);
		System.out.println("请输入一个五位数字：");
		int a,b;
        a=scan.nextInt();
        b=a;
        int sum = 0;
        while(a!=0){
        sum += a%10;
        a = a/10;
        }
        System.out.println(b+ "的各位数字之和是："+sum);
		/******End******/
	}
}
