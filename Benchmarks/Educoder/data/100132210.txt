package step1;
import java.util.Arrays;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
		int[] day_count = new int[temperatures.length];
        for(int i = 0; i < temperatures.length; i++){
            int count = 0;
            for(int j = i; j < temperatures.length; j++){
                if(temperatures[j] > temperatures[i]) {
                    count = j - i;
                    break;
                }
            }
            day_count[i] = count;
        }
        // System.out.println(Arrays.toString(day_count));
        return day_count;
		
		/********** END **********/
	}
}
