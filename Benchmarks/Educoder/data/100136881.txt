class Demo {
	private int num1;

	public Demo(int num1) {
		this.num1 = num1;
	}

	public boolean equals(Object obj) // Object obj = new Demo()
	{
		if (!(obj instanceof Demo)) // 判断obj是否和Demo是同类
			System.out.println("false");
        else
            System.out.println("true");
		Demo d1 = (Demo) obj; // 将父类的引用(Object)向下转换为子类(Demo)
		return this.num1 == d1.num1;
        
	}

	public String toString() {
		return "Demo:" + num1; // 返回对象的值（每一个对象都有自己的特定的字符串）
	}
    
}
