package step4;
import java.util.Scanner;
public class TestJavaKJ15_2 {

	public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
		int x,y,sum;
		sum=0;
		System.out.println("请输入一个正整数：");
		y=x=sc.nextInt();
		while(x>=0){
			sum=sum+=x;
			x--;
		}		
		System.out.println(y+"到1的和是："+sum);
	}

}
