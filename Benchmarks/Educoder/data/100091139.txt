package step3;
public class IsPowerOfTwo{

    // 判断一个整数是否是2的幂
    public boolean IsPowerOfTwo(int n){
        // 请在此添加实现代码
        int b=1,a;
        for( a=2;b<=n;)
        {
            
            b*=a;
            if(b==n) return true;
            
        }
        
        return false;
    }
}
