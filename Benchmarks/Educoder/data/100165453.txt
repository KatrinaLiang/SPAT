package step2;

public class Student extends Person{
    private int reg_num; //学号
    private Date reg_day; //注册日期
   // TO DO YOU CODE HERE
	public void setReg_num(int reg_num){
        this.reg_num = reg_num;
    }
    public void setReg_day(Date reg_day){
        this.reg_day = reg_day;
    }
    public String toString(){
        return "name="+super.getName()+" number="+super.getNum()+" birthday:"+super.getBirth()+"reg_num:"+reg_num+"reg_day:"+reg_day;
    }
    public Date getreg_day(){
        return reg_day;
    }
    public Student(){
        super();
        Date day = new Date();
        setReg_day(day);
    }
    public Student(String name,String number,int y,int m,int d,int reg_num,int y1,int m1,int d1){
        super.setName(name);
        super.setNum(number);
        Date birthday = new Date(y,m,d);
        setReg_num(reg_num);
        super.setBirth(birthday);
        Date reg_day = new Date(y1,m1,d1);
        setReg_day(reg_day);
    }
    public Student(String name,String number,Date birthday,int reg_num,Date reg_day){
        super.setName(name);
        super.setNum(number);
