class MyRunnable implements Runnable{
    private int b;
    public MyRunnable(int b){this.b=b;}
    public void run(){
        int result=1;
        for(int i=1;i<=b;i++)
            result*=i;
        try{
		Thread.sleep(400);
        }catch(InterruptedException e){
            e.printStackTrace();}
        System.out.println(result);
    }
}
