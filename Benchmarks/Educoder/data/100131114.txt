package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int len = num2-num1+1;
		String str = "";
		int num=num1-1;
		for(int i = 0 ; i < num2-num1+1 ; i++) {
			num+=1;
			str+=num;
		}
		int sum = 0;
		for(int i = 0 ; i < str.length() ; i++) {
			String charAt = str.charAt(i)+"";
			sum+=Integer.parseInt(charAt);
		}
		return sum;
		
		
		/********** END **********/
	}
}
