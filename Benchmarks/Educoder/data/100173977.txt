package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int i,j,mul;
		for(i=1;i<=9;i++){
			for(j=1;j<=i;j++){
                mul=j*i;
				System.out.print(j+"*"+i+"="+mul+"\t");
                
			}
			System.out.print("\n");
		}
		
		/*****end*****/
	}
}
