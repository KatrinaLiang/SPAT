package step1;

public class LetterChange {
    public char LetterChange(char ch){
        switch(ch)
        {
            case 'a':return 'A';
            case 'b':return 'B';
            case 'c':return 'C';
            case 'd':return 'D';
            case 'e':return 'E';
        }
        return ch;
    }
}
