package step1;

public class Yard {
    int length,width;
     public Yard(int length,int width)
    {
        this.length=length;
        this.width=width;
    }
    public double getPerimeter()
    {
        return (length+width)*2;
    }
    public double getAcreage()
    {
        return length*width;
    }

}
