// 定义学生类，其包含2 个属性：学号，姓名
/********* Begin *********/
class Stude
{
	private String name;
	private String ID;
	public String getName() {
		return name;
	}
	public void setName(String name) {
		this.name = name;
	}
	public String getID() {
		return ID;
	}
	public void setID(String ID) {
		this.ID = ID;
	}
	public Stude() {}
	public Stude(String name,String ID) 
	{
		this.name=name;
		this.ID=ID;
	}
	
}

class university extends Stude
{
	private String zhuany;

	public String getZhuany() {
		return zhuany;
	}

	public void setZhuany(String zhuany) {
		this.zhuany = zhuany;
	}

	
	public String toString() {
		return "学号："+this.getID()+"，姓名："+this.getName()+"，专业："+this.zhuany;
	}
	
	
}
