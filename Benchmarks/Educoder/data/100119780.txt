package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
		for(int a=1;a<=10;a++)
        {
            for(int b=1;b<a;b++)
            {
                System.out.print("*");
            }
            System.out.println("*");
        }
	System.out.println("——————————我是华丽的分界线——————————");
		  for(int a=10;a>=1;a--)
        {
            for(int b=1;b<a;b++)
            {
                System.out.print("*");
            }
            System.out.println("*");
        }
		
		
		
		
		//在这里打印出倒三角形
		
		
		
		
		
		/*****end*****/
	}
}
