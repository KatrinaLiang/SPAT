package step2;

import java.io.File;

public class Task {
	
	/********** BEGIN **********/
    private static int a=0;
	public void showDirStructure(File file)	{
		if(a==0){
            System.out.println("+--"+file.getName());
            a++;
        }
        File[] fls = file.listFiles();
        for(File f : fls) {
            for(int i= a;i>0;i--){
                System.out.print("  ");
            }
            if(f.isDirectory()) {
                System.out.println("+--"+f.getName());
                a++;
                showDirStructure(f);
            }
            else if(f.isFile()) {
                System.out.println("--"+f.getName());
            }
        }
        a--;





	}
	


	/********** END **********/
}
