package step1;

public class Person {
    private String name; //姓名
    private String number; //身份证号
    private Date birthday; //出生日期
    
   /************Begin*************/
    public void setName(String name){this.name = name;}
    public void setNumber(String number){this.number = number;}
    public void setDate(Date birthday){this.birthday = birthday;}
    public Person(){
        this.birthday = new Date();
    }
    public Person(String name,String num,Date birthday){
        this.name = name;
        this.number = num;
        this.birthday = birthday;
    }
    public Person(String name,String num,int year,int mouth,int day){
        this.name = name;
        this.number = num;
        this.birthday =new Date(year,mouth,day);
    }
    public void setinfo(String name,String num,Date birthday){
        this.name = name;
        this.number = num;
        this.birthday = birthday;
    }
    public String getName(){return name;}
    public String getNumber(){return number;}
    public Date getDate(){return birthday;}
    public String toString(){
        return "name="+name+" number="+number+" birthday:"+birthday;
    }


    
    
    
    
   /************End*************/
}
