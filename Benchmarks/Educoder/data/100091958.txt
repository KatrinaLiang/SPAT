package step1;

public class IsOfVowels {
	// 判断输入字符是否是元音字母
	public boolean IsOfVowels(char ch){
		// 请在此添加实现代码
        String str = "aeiouAEIOU";
        for(int i = 0; i < str.length(); i++ )
        {
            if(ch == str.charAt(i))
                return true;
        }
            
        return false;
	}
}
