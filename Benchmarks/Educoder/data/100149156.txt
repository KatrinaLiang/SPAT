package step2;

public class Task {
    
	public int virusArea(int n, int m, char[][] area) {
		int numOfVirus = 0;
		for (int i = 0; i < n; ++i) {
			for (int j = 0; j < m; ++j) {
				if (area[i][j] == 'o') {
					DFS(area, i, j);
					++numOfVirus;
				}
			}
		}
		return numOfVirus;
	}

	public void DFS(char[][] area, int x, int y) {
		if (x < 0 || x >= area.length || y < 0 || y >= area[x].length || area[x][y] != 'o')
			return;
		area[x][y] = 'x';
		DFS(area, x - 1, y);
		DFS(area, x + 1, y);
		DFS(area, x, y - 1);
		DFS(area, x, y + 1);
	}

}
