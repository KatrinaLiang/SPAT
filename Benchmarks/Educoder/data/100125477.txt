package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
    public InterRectangle(double width,double length){
        this.width=width;
        this.length=length;
    }
	public double z(){
        return (width+length)*2;
    }
    public double m(){
        return width*length;
    }
	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
        InterRectangle s=new InterRectangle(0,0);

        double m=0,z=0;
        double width;
        double length;
        double d=total;
        while(total>0){            
            width=scanner.nextDouble();
            length=scanner.nextDouble();
            //s=new InterRectangle(width,length);
            s.width=width;
            s.length=length;
            m+=s.m();
            z+=s.z();
            total--;
        }
		System.out.println("总共有"+d+"块土地");
        System.out.println("土地总面积为:"+m+"平米");
        System.out.println("土地总周长为:"+z+"米");
		/*********end*********/
	}

}
