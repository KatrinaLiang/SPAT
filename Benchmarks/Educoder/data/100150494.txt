package step3;
import java.util.*;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
		Scanner s = new Scanner(System.in);
        System.out.println("请输入一个五位数字：");
        int num = s.nextInt();
        int ans = num/10000 + num%10 + num / 10 % 10 + num / 100 % 10 + num / 1000 %10;
        System.out.println(num + "的各位数字之和是：" + ans);
		/******End******/
	}
}
