package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
		int sum=0,c=0;
        
        while(n>0){
            
            sum=sum*10+a;
            c=c+sum;
            n--;
        }
		System.out.print(c);

		/********** End *********/
	}
}
