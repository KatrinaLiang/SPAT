package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum = 0;
		for (int i = num1; i <= num2; i++) {
        	sum += getValue(i);
        }
		
		return sum;
		/********** END **********/
	}
    
    public static int getValue(int num) {
    	return (num / 100) + (num % 10) + (num % 100 / 10);
    }
}
