package step3;

public class Task {
	public static void rotate(int[][] b) {
	    int[][] temp = new int[b[0].length][b.length];  
        for(int i = 0; i < b.length; i++) {  
            for (int j = 0; j < b[i].length; j++) {  
                temp[i][j]=b[b[i].length-j-1][i];  
            }  
        }
        for(int i = 0; i < b.length; i++) {  
            for (int j = 0; j < b[i].length; j++) {  
            b[i][j] = temp[i][j];
            }
            }
            }
	}
