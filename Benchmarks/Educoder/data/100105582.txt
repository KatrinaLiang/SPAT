package step1;

public class Yard {
        double a;
        double b;
        public Yard(double length,double width)
        {
             a=length;
             b=width;
        }
        public  double getPerimeter(){
            double sum1=0;
            sum1=2*a+2*b;
            return sum1;
        }
        public double getAcreage(){
            double sum2=0;
            sum2=a*b;
            return sum2;
        }
}
