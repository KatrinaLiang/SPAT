package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum = 0;
        String begin = null;
        for ( int i = num1; i <= num2; i++ )
        {
            begin = i + "";
            char[] nums = begin.toCharArray();
            for ( int j = 0; j < nums.length; j++ )
            {
                sum += Integer.parseInt(nums[j] + "");
            }
        }

        return sum;
		
		
		/********** END **********/
	}
}
