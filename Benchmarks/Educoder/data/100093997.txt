package chapter1.step3;

import java.util.Scanner;
import java.text.DecimalFormat;
public class FarheheitToCentigrade {

	public static void main(String[] args) {
		 DecimalFormat df=new DecimalFormat("#.00");
  		 Scanner sc=new Scanner(System.in);
  		 double tempfarenheit = sc.nextDouble();   //华氏度
    	 double Centigrade=(tempfarenheit-32)/9*5;
    	 String ret = df.format(Centigrade);
    	 System.out.print("If the Farenheit is "+tempfarenheit+",the Centigrade is "+ ret);
		
	}
}
