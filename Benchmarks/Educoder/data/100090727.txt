package step1;
public class ListNode {
	public ListNode next;
	public int data;

	// 创建一个空的链表节点
	public ListNode() {
		next = null;
		data = Integer.MIN_VALUE;
	}

    // 为一个新创建的链表节点赋值
    public ListNode(int new_data) {
        // 请在此添加实现代码
        next = null;
        data  = new_data;
    }

	// 返回当前节点数据对应的字符串
	public String toString () {
		return Integer.toString(data);
	}
}
