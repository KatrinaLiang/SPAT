package step1;
import java.util.Scanner;
public class ShiXinSanJiaoXing
{
	public static void main(String[] args) 
	{
		Scanner in=new Scanner(System.in);
		int n,i,j,k;
		n=in.nextInt();
		for(i=0;i<n;i++)
		{
			for(j=1;j<=n-i-1;j++)
			{
				System.out.print(" ");
			}
			for(k=0;k<n+i-j+1;k++)
			{
				System.out.print("*");
			}
			System.out.print("\n");
		}		
	}
}
