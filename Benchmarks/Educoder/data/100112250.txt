package chapter2.step7;

import java.util.Scanner;

public class HelloWorld{
	public static void main(String[] args){
        String name;	//存储姓名的变量
        int age;		//存储年龄的变量
        String sex;		//存储性别的变量
        float weight;	//存储体重
        String addr;	//存储地址
        String marriage;//存储婚否
        //声明一个Scanner,用来接收从键盘输入的内容
        Scanner scanner = new Scanner(System.in);
        
        //下面是提示输入内容,并接收
        System.out.println("请录入嫦娥个人信息");
        System.out.println("请输入姓名：");
        name = scanner.next();
        System.out.println("请输入年龄：");
        age = scanner.nextInt();
        System.out.println("请输入性别：");
        sex = scanner.next();
        System.out.println("请输入体重：");
        weight = scanner.nextFloat();
        System.out.println("请输入地址：");
        addr = scanner.next();
        System.out.println("请输入是否已婚：");
        marriage = scanner.next();
        
        //接收之后,进行打印输出
        System.out.println("信息如下：");
        System.out.println("\t姓名：" + name);
        System.out.println("\t年龄：" + age + "岁");
        System.out.println("\t性别：" + sex);
        System.out.println("\t体重：" + weight + "kg");
        System.out.println("\t地址：" + addr);
        System.out.println("\t婚否：" + marriage);
        
	}
}
