package step1;

public class Task {

	
		public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum =0;
        for(int i=num1;i<=num2;i++)
        {
            sum+=getSum2(i);
        }
		
		return sum;
		/********** END **********/
	}
	public int getSum2(int num){
    	int sum=0,number = num;
        while(number!=0){
        	sum+=number%10;
            number/= 10;
        }
        return sum;
    }
}
