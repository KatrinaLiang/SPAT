package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum=0;
        for(int i=num1;i<=num2;++i)
        { sum+=de(i);}
        return sum;
		
		/********** END **********/
	}
    public int de(int n)
    {int n1,n2,n3;
     n1= n%10;
     n2 = (n/10)%10;
     n3 = n/100;
     return n1+n2+n3;}
}
