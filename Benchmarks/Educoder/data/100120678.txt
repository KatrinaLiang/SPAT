package step4;

import java.util.Scanner;

public class ForPractice4 {
	public static void main(String[] args) {
		/*****start*****/
        Scanner input = new Scanner(System.in);
		System.out.println("欢迎使用中国人民银行ATM取款机");
        int allMoney = 1000;
        int choose=2;
        int quMoney;
       do{
       		System.out.print("输入取款金额：");
           quMoney=input.nextInt();
           if(allMoney<quMoney){
				System.out.print("\n目前余额："+allMoney+"无法满足您的取款需求！");           
           }else{
               allMoney=allMoney-quMoney;
           		System.out.print("\n剩余金额："+allMoney+"，是否继续（'1'：结束，'2'：继续）：");
                choose=input.nextInt();                 
           }
       		System.out.println();
       }while(choose==2);
        
		System.out.println("取款结束！");
		
		
		/*****end*****/
	}
}
