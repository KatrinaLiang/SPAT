
package step3;
import java.util.Scanner;
public class Sum{
	public static void main(String args[]){
		Scanner out=new Scanner(System.in);
		int num = out.nextInt();
        int result = recursionSum(num);
        System.out.println(result);
        out.close();
	}
	
    public static int recursionSum(int mum){
    	int Result=0;
    	for(int i=1;i<=mum;i++) {
    		Result=Result+i;
    	}
    	return Result;
    }  
}
