package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int i;
        int[] f=new int [20];
        f[0]=1;
        f[1]=1;
        int s;
        s=2;
        for(i=2;i<f.length;i++)
            f[i]=f[i-1]+f[i-2];
        for(i=2;i<20;i++) 
            s=s+f[i];
            
   System.out.print("前20个元素的和为"+s);

	/********end********/
	}
}
