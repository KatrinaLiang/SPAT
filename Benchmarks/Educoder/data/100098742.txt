package step1;

public class Yard {
    public int length;
    public int wigth;
	public Yard(int length,int wigth) {
		this.length=length;
		this.wigth=wigth;
		}
		public double getAcreage() {
			double  ac=length*wigth;
			return ac;
		}
		public double getPerimeter() {
		double  pe=(length+wigth)*2;
			return pe;
		}
}
