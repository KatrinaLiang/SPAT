package step2;

public class SwitchToGrade {

    // 请用switch来将学生成绩转换成对应等级
    public String SwitchToGrade(int score) { 
        // 请在此提供实现代码  
	   String str ="";
        if (score>=90&score<101)
             str = "等级：A";
      else if (score>=70&score<90)
             str = "等级：B";
         else if (score>=60&score<70)
             str = "等级：C";
     else if (score<60&score>=0)
             str = "等级：D";
     else
         str ="输入成绩非法";
        return str;
        
      
    }
}
