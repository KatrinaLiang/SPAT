package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int a,b;
		for(a=1;a<=9;a++){
            for(b=1;b<a;b++){
                 System.out.print(b+"*"+a+"="+(a*b)+"\t");
			}
			System.out.println(b+"*"+a+"="+(a*b)+"\t");
		}
		/*****end*****/
	}
}
