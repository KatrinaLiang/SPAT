package step2;

public class ForPractice2 {
	public static void main(String[] args){
     int i=0,j;
        for(;i<10;i++){
            for(j=0;j<=i;j++){
            System.out.print("*");}
            System.out.print("\n");
        }
		
		
		
		
		System.out.println("——————————我是华丽的分界线——————————");
		for(;i>0;i--){
            for(j=0;i>j;j++){
            System.out.print("*");}
            System.out.print("\n");
        }
		
		
		
		
		/*****end*****/
	}
}
