package step3;
import java.util .Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
		Scanner A= new Scanner(System.in);
        System.out.println("请输入一个五位数字：");
        int a =A.nextInt(),t=0;
        for(int i=1;i<=10000;i*=10)
        {
            t+=(a/i)%10;
        }
      System.out.println(a+"的各位数字之和是："+t);
        
        
		/******End******/
	}
}
