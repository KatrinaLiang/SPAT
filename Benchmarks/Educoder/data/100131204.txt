package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
     int count = 0;
		for(int i = num1 ; i <= num2 ; i++) {
			int sum = 0;
			int x = i;
			while(x>0) {
				sum = sum + x%10;
				x = x/10;
			}
			count+=sum;
		}
		return count;

		/********** END **********/
	}
}
