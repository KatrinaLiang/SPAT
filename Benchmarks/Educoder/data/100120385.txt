package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
        int x;
        for(int i=1;i<=9;i++){
            for(int j=1;j<=i;j++){
                x=i*j;
                System.out.print(j+"*"+i+"="+x+"\t");
            }
            System.out.print("\n");
        }
		
		
		
		
		
		
		/*****end*****/
	}
}
