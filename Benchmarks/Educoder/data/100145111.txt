package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		int n= matrix.length;
        int [][]arr=new int[n][n];
        for(int i=0;i<n;i++)
            for (int j=0;j<n;j++){
                arr[j][i]=matrix[n-i-1][j];
            }

       
         for (int i=0;i<n;i++) 
            for (int j = 0; j < n; j++) 
                matrix[i][j] = arr[i][j];
		
		/********* End *********/
	}
}
