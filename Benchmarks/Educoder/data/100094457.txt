package step1;

public class DecomposeFactor {
	public void decompose(int score) {
		/********** Begin *********/
        double p=0;
        if(score>=90){
        p=4.0;
        }
        else if(score>=80){
        p=3.5;
        }
        else if(score>=70){
        p=3.0;
        }
        else if(score>=60){
        p=2.5;
        }
		else{
        p=0.0;
        }
        System.out.print(p);
        /********** End *********/
	}
}
