package step3;

import java.util.Arrays;

public class Task {

	public void matrixSum(int[][] arr){
		/********** BEGIN **********/
        int c=0;
		for(int a=0;a<arr.length;a++)
            for(int b=0;b<arr[a].length;b++)
                if(a==b) c+=arr[a][b];
        System.out.println(c);
		



		/********** END **********/
	}
	
	
}
