package step3;

public class Task {
 
    public static void rotate(int[][] metrix){
		int len=metrix.length;
		int b [][]=new int [len][len];
		for(int i=0;i<len;i++){
			for(int j=0;j<len;j++){
				b[j][len-1-i]=metrix[i][j];
			}
		}
		for(int i=0;i<len;i++)
			for(int j=0;j<len;j++)
				metrix[i][j]=b[i][j];
	}

}
