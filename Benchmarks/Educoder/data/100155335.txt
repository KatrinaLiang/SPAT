package task3;

public class Box {

    double width;
    double height;
    double depth;

    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public double volume(int boxType) {
        int a=boxType;
        double b,c;
        if(a==1)
        {    b=width*height*depth;
        return b;}
        if(a==2)
        {
            c=3.14*width*height*depth;
            return c;
        }
        return 0;
    }

}
