package step1;

public class Yard {
    int length;
    int width;
    public Yard(int a,int b){
    length=a;
    width=b;}
public  float getPerimeter(){
    float sum;
    sum=(float)(length+width)*2;
    return sum;}
public float getAcreage(){
  float sum;
    sum=(float)(length*width);
    return sum;}
}
