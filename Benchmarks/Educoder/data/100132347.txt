package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
		int len = temperatures.length;
		int a[] = new int[len];
        for(int t=0;t<len;t++){
        	a[t] = 0;
        }
		for(int i=0;i<temperatures.length-1;i++){
        	for(int j=i+1;j<temperatures.length;j++){
            	if(temperatures[i]<temperatures[j]){
                	a[i] = j-i;
                    break;
                }
            }
		}
		
		

		return a;
		/********** END **********/
	}
}
