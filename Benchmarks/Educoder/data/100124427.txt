package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int[] fi = new int[20];	
	fi[0]=1;
	fi[1]=1;
    int sum = 2;
    for(int i=2;i<fi.length;i++){
        fi[i]=fi[i-1]+fi[i-2];
        sum +=fi[i];
    }
    System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
