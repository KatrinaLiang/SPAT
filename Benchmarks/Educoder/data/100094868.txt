package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        int sum=0;
		while(n>0){
            int i=n;
            int b=a;
            while(i>1){
                b=b*10;
                b+=a;
                i--;
            }
            sum+=b;
            n--;
        }
        System.out.printf(""+sum);
		/********** End *********/
	}
}
