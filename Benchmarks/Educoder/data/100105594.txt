package step1;

public class Yard {
    int x;
    int y;
    
    public Yard(int x,int y){
        this.x=x;
        this.y=y;
    }
    
    public double getPerimeter(){
        double p;
        p=(x+y)*2.0;
        return p;
    }
    
    public double getAcreage(){
        double s;
        s=x*y*1.0;
        return s;
    }
}
