package step1;

public class Yard {
    int length;
    int width;
    float x,y;
    public Yard(int length,int width)
    {
   this.length=length;
    this.width=width;}
    public float getPerimeter()
    {
    return x=2*(length+width);
    }
    public float getAcreage()
    {
    return y=length*width;
    }

}
