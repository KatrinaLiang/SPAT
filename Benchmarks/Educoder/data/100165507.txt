package step2;
import step1.Person;
import step1.Date;
public class Student extends Person{
    private int reg_num; //学号
    private Date reg_day; //注册日期
   // TO DO YOU CODE HERE
	public Student(){
    	super();
        reg_num=0;
        reg_day=new Date();
    }
    public Student(String name,String number,int x1,int x2,int x3,int reg_num,int y1,int y2,int y3){
      	super(name,number,x1,x2,x3);
        this.reg_num=reg_num;
        reg_day=new Date(y1,y2,y3);
    }
    public Student(String name,String number,Date d1,int reg_num,Date d2){
    	super(name,number,d1);
        this.reg_num=reg_num;
        reg_day=d2;
    }
 
    public int getreg_num(){
    	return reg_num;
    }
    public Date getreg_day(){
    	return reg_day;
    }
    public String toString(){
    	String na=super.getname();
    	String num=super.getnumber();
        Date t=super.getdate();
        return "name="+na+" number="+num+" birthday:"+t+"reg_num:"+reg_num+"reg_day:"+reg_day;
    }
}
