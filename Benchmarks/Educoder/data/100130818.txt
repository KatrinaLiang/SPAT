package step5;

import java.util.Scanner;


public class Task {
	
	public int getDepth(TreeNode tree){
		/********** BEGIN **********/
		if(tree == null)
            return 0;

        //分别计算左子树和右子树的深度
        int leftDepth = getDepth(tree.left) + 1;
        int rightDepth = getDepth(tree.right) + 1;

        return leftDepth > rightDepth ? leftDepth: rightDepth;

		/********** END **********/
		
	}
	
}
