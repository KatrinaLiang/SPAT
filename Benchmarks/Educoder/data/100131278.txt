package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int i,n,u,m;
    	int sum=0;
    	int mid;
		mid = num1;
    	for(i=0;i<=num2-num1;i++)
    	{  
        	n = mid; 
       		u = mid;
       		m = mid;
			sum+=n/100;
			sum+=u%100/10;
			sum+=m%10;
			mid++;

    	}
    		return sum;
		
		
		/********** END **********/
	}
}
