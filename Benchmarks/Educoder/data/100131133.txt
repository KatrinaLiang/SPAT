package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum = 0;
        
        for(int i = num1; i <= num2; i++) {
         sum += sumEachNumber(i);   
        }
		
		return sum;
		/********** END **********/
	}
    
    private int sumEachNumber(int num) {
     	int sum = 0;
        
        while (num != 0) {
            int t = num % 10;
            sum += t;
            num /= 10;
        }
        
        return sum;
    }
}
