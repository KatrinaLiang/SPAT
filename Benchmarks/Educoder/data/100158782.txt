package Q103;
import java.util.Scanner;
public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		long[] dp = new long[1001];
		 dp[1] = 1;dp[2] = 2;
			int n = sc.nextInt();
		for(int i = 3; i <= n; i++)
		dp[i]=dp[i-1]+dp[i-2];
		System.out.print(dp[n]);
	}

}
