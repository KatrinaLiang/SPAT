package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
		System.out.println("子线程计算结果为："+MyThread.run(num));
		
		
	}
}


class MyThread   {
    public static int run(int num){
        if(num<=2)
            return 1;
        else
            return run(num-1)+run(num-2);
    }
}
