package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int sum;
		for(int i=1;i<10;i++){
			for(int j=1;j<=i;j++){
			sum=j*i;
			System.out.print(j+"*"+i+"="+sum);
			System.out.print("\t");
			}
			System.out.println();
		}
		
		
		
		
		
		/*****end*****/
	}
}
