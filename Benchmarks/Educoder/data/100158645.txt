#include<stdio.h>
int main()
{
	char c;
	int i,j,k;
	int n,m;
	int num[1000];
	n=0;
	int count=0;
	while(1)
	{
		scanf("%d",&num[n]);
		n++;
		c=getchar();
		if(c=='\n')
		{
			break;
		}
	}
	int t;
	scanf("%d",&t);
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			for(k=0;k<n;k++)
			{
				if((i!=j)&&(i!=k)&&(j!=k)&&(i<j)&&(j<k))
				{
					if(num[i]+num[j]+num[k]<t)
					{
						count++;
					}
				}
			}
		}
	}
	printf("%d",count);
	return 0;
}
