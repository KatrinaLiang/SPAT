package task4;

public class Box {

    double width;
    double height;
    double depth;

    int boxType;

    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;
	public Box(int boxType) {
    	this.boxType=boxType;
    }
    
    public double volume() {
        //TODO your code
        double v=0;
        if(boxType==1){
        	v=width*height*depth;
            return v;
        }else{
        	v=3.14*width*width*height;
            return v;
        }
    }

}
