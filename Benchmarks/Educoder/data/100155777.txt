    public String getUnit(){
    return unit;
    }
    public void setUnit(String a){

    //实现 getUnit、setUnit方法
	public void printVolumn(){

        double h=super.getHeight();
        if(a==1){
        double volume=w*h*d;
         System.out.println("the volumn is:"+volume+unit);
        }
        else{
        double volume=w*w*3.14*h;
           System.out.println("the volumn is:"+volume+unit);
        }
    }
