package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int s=0;
        for(int i=num1;i<=num2;i++){
		int a=i/100;
        int b=(i/10)%10;
        int c=i%10;
        s=s+a+b+c;    
        }
		return s;
		/********** END **********/
	}
}
