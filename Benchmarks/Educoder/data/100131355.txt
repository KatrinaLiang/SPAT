package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum=0;
		for(int a=num1;a<=num2;a++)
        {
            int i=a%10,b=a/10%10,c=a/10/10%10;
            int d=i+b+c;
            sum+=d;
        }
        return sum;
		/********** END **********/
	}
}
