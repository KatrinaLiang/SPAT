package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
		int sum=0;
		for (int i = 0; i < n; i++) {
			int x=0;
			for(int j=n-1-i;j>=0;j--)
			{
				x+=Math.pow(10,j)*a;
			}
			sum+=x;
		}
		System.out.println(sum);


		/********** End *********/
	}
}
