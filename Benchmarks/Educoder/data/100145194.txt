package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		
		int result = 0;
		while (true) {
			if (n % 2 != 0) {

				n--;
				n /= 2;
				result++;
			} else
				n /= 2;
			if (n == 0)
				break;
		}

		return result;
		
		
		
		/********* End *********/
	}
	
}
