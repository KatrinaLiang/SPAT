package step1;

public class Yard {
    double length;
    double width;
    public Yard(double length,double width){
        this.length=length;
        this.width=width;
    }
    public double getPerimeter(){
        double p;
        p=2*(length+width);

        return p;
    }
    public double getAcreage(){
        double k;
        k=length*width;
        return k;
    }
}
