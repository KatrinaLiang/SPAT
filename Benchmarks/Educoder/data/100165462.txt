public class Student extends Person{
    private int reg_num; //学号
    private Date reg_day; //注册日期
   // TO DO YOU CODE HERE
    public Student()
    {
        super();
        reg_num=0;
        reg_day=new Date(1999,9,9);
    }
    public Student(String name,String number,int y1,int m1,int d1,int num,int y2,int m2,int d2)
    {
        super(name,number,y1,m1,d1);
        reg_num=num;
        reg_day=new Date(y2,m2,d2);
    }
    public Student(String name,String number,Date d1,int num,Date d2)
    {
        super(name,number,d1);
        reg_num=num;
        reg_day=d2;
    }
    public int getreg_num()
    {
        return reg_num;
    }
    public Date getreg_day()
    {
        return reg_day;
    }
    public String toString(){
        return "name="+getName()+" number=" +getNumber()+" birthday:"+getBirthday()+"reg_num:"+reg_num+"reg_day:"+reg_day;
    }
