package step1;

public class Yard {
    int length;
    int width;
    public Yard(int length,int width)
    {
        this.length=length;
        this.width=width;
           
    }
    public double getPerimeter()
    {
        double a=(length+width)*2;
        return a;
    }
    public double getAcreage()
    {
       double s=length*width;
        return s;
    }
        
        

}
