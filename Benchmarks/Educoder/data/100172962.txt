package step3;

public class HelloWorld {
	public static void main(String[] args) {
		int count= 0;	//定义变量存储6的倍数出现的次数
		/*****start*****/
		do{
			count++;
		}
		while(count*6<=100);

		count=count-1;
		/*****end*****/
		System.out.println("6的倍数出现的次数为：" + count);
	}
}
