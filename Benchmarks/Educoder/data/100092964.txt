package step2;

public class SwitchToGrade {

    // 请用switch来将学生成绩转换成对应等级
    public String SwitchToGrade(int score) { 
        // 请在此提供实现代码  
        if(score<=100&&score>=0)
		{
				switch(score/10)
			{
				case 9:return"等级：A";
				case 8:return"等级：B";
				case 7:return"等级：B";
				case 6:return"等级：C";
				default:return"等级：D";
			}
		}
		else
		{
			return "输入成绩非法";
		}
	}
}
