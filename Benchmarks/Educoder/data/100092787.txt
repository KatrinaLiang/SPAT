package step1;

public class LetterChange {

    // 使用 switch把小写字母转为大写字母
    public char LetterChange(char ch){
        // 请在此提供实现代码
        switch(ch)
        {
            case 'q':
                return 'q';
            case 'w':
                return 'w';
            case 'e':
                return 'E';
            case 'r':
                return 'r';
            case 't':
                return 't';
            case 'y':
                return 'y';
            case 'u':
                return 'u';
            case 'i':
                return 'i';
            case 'o':
                return 'o';
            case 'p':
                return 'P';
            case 'a':
                return 'A';
            case 's':
                return 's';
            case 'd':
                return 'D';
            case 'f':
                return 'f';
            case 'g':
                return 'g';
            case 'h':
                return 'h';
            case 'j':
                return 'j';
            case 'k':
                return 'k';
            case 'l':
                return 'l';
            case 'z':
                return 'z';
            case 'x':
                return 'x';
            case 'c':
                return 'C';
            case 'v':
                return 'v';
            case 'b':
                return 'B';
            case 'n':
                return 'n';
            case 'm':
                return 'm';
        }
        return 0;
        
    }
}
