package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum=0;
		for(int i=num1;i<=num2;i++){
            if(i<100){
                int x=i%10;
            int y=i/10;
            sum+=x+y; 
            }
           if(i>=100){
               int x=i%10;
               int y=i/10%10;
               int z=i/100;
               sum+=x+y+z;
           }
        }
       return sum;
		/********** END **********/
    }
}
