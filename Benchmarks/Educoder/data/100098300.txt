package step3;

import java.util.Scanner;

public class IO {
    public static void main(String[] args) {
        int number1=0;
        int number2=0;
        int number3;
        /********** Begin *********/
		Scanner sc= new Scanner(System.in);
        if(sc.hasNextInt()){
            number1 = sc.nextInt();
        }
        if(sc.hasNextInt()){
            number2 = sc.nextInt();
        }
        number3 = number1+number2;
        System.out.print(number3);

        /********** End *********/
    }
}
