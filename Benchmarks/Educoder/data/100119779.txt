public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
		
		//在这里打印出正三角形 
		for(int i=0;i<=10-1 ;i++){
            for(int j=0;j<=i;j++)
        System.out.print("*");
        System.out.print("\n");}
		
		
		
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
		for(int a=10;a>=1;a--){
        for(int b=9;b>=10-a;b--)
        System.out.print("*");
        System.out.print("\n");}
		
		
		
		
		/*****end*****/
	}
}
