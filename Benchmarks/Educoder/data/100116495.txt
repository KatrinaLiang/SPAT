package step3;

public class HelloWorld {
	public static void main(String[] args) {
		int count= 0;	//定义变量存储6的倍数出现的次数
		/*****start*****/
        int j=1;
        do{
            j++;
            if(j%6==0){
                count++;
            }
        }while(j<=100);
		
		

		
		/*****end*****/
		System.out.println("6的倍数出现的次数为：" + count);
	}
}
