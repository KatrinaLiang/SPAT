package step3;

import java.io.File;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

public class DeleteFiles {
    public static void start() throws IOException {
        StringBuilder data = new StringBuilder("");
        /********** Begin *********/
        //读取readme.txt文件 10 = /n 
		File f = new File("src/step3/readme.txt");
        FileReader fr = new FileReader(f);
        File fn = new File("src/step3/readmeCopy.txt");
        fn.createNewFile();
        FileWriter fw= new FileWriter(fn,false);
        int word = fr.read();
        //写入readmeCopy.txt文件
		while(word != -1){
        	fw.write((char)word);
          	word = fr.read();
        }

        //删除readme.txt文件
		fw.close();
        fr.close();
        f.delete();

        /********** End *********/
    }
}
