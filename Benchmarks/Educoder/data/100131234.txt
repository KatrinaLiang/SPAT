package step1;

public class Task {
	int sum = 0;
    public void getNum(int n){
        while(n/10 != 0){
        	sum += n%10;
        	n /= 10;
        }
        sum += n;
    }
	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		for(int i = num1;i <= num2;i++){
        	getNum(i);
        }
		return sum;
		/********** END **********/
	}
}
