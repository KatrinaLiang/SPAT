package step2;
public class Task {

	public int getNum(int n){
		/********* Begin *********/
		
            int num,re;
            for(num=0,re=n;re>1;){
                if(re%2==1){
                    re=(re-1)/2;
                    num++;
                }
                else re/=2;
             }
        if (re==1) num++;
        return num;

		
		/********* End *********/
	}
	
}
