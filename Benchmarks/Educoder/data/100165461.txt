package step2;
import step1.Person;
import step1.Date;
public class Student extends Person{
    private int reg_num; //学号
    private Date reg_day; //注册日期
   // TO DO YOU CODE HERE
	public Student(String name, String number, int y, int m, int d,
                  int reg_num, int ry, int rm, int rd) {
    	super(name, number, y, m, d);
        this.reg_num = reg_num;
        this.reg_day = new Date(ry, rm, rd);
    }
    public Student(String name, String number, Date d1,
                  int reg_num, Date r) {
    	super(name, number, d1);
        this.reg_num = reg_num;
        this.reg_day = r;
    }
    public Student() {
    	super();
        this.reg_num = 0;
        this.reg_day = new Date(1999, 9, 9);
    }
    
    public Date getreg_day() {
    	return this.reg_day;
    }
    public int getreg_num() {
    	return reg_num;
    }
    public String getname() {
    	return super.getname();
    }

	public String toString() {
    	return "name="+super.getname()+" number="+super.getnum()
            +" birthday:"+super.getbirthday().toString()
            +"reg_num:"+getreg_num()
            +"reg_day:"+getreg_day().toString();
    }

}
