package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int a;
        a=0;
		for(int j=n;j>=1;j=j/2){
            if(j%2==0){
				continue;
			}
			if(j==1){
				a++;
				break;
			}			
			else{
				j=j-1;
				a++;
			}
		}
		return a;	
		/********* End *********/
	}	
}
