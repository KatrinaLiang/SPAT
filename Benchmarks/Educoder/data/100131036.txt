package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum=0;
		
		for (int i = num1; i <= num2; i++) {
			char[] num1Str=String.valueOf(i).toCharArray();
			for (int j = 0; j < num1Str.length; j++) {
				sum=sum+Integer.valueOf(String.valueOf(num1Str[j]));
			}
		}
		
		return sum;
		/********** END **********/
	}
}
