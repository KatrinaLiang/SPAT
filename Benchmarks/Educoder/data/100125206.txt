package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
    public InterRectangle(){
    
    }
    public InterRectangle(double length,double width){
		this.length = length;
        this.width = width;
    }
    public double mj(){
    	return width * length;
    }
	public double zc(){		
        return (width + length) * 2;
    }

	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
		System.out.println("总共有" + total + "块土地");
        double le;
        double wi;
        InterRectangle ag;
        double t_mj = 0,t_zc = 0;
        for(int i = 0; i < total; i++){
        	le = scanner.nextDouble();
        	wi = scanner.nextDouble();
            ag = new InterRectangle(le,wi);
            t_mj += ag.mj();
            t_zc += ag.zc();
        }
        System.out.println("土地总面积为:" + t_mj + "平米");
        System.out.println("土地总周长为:" + t_zc + "米");
		/*********end*********/
	}

}
