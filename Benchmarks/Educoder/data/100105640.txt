package step1;

public class Yard {
	private float length;
    private float width;
    
    public Yard() {
    }
    
    public Yard(int length, int width) {
        this.length = length;
        this.width = width;
    }
    
    float getPerimeter() {
        return (length + width) * 2;
    }
    
    float getAcreage() {
        return length * width;
    }
    
    
}
