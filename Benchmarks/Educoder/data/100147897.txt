package step4;

//注意文件名，这里应该写MyThread类
import java.lang.Thread;

public class MyThread extends Thread {
	private int num;
	
	public MyThread() {
		this(0);
	}
	
	public MyThread(int num) {
		this.num=num;
	}
	
    @Override
    public void run() {    	
    	int result = 1;    	 
		for(int fate=num;fate>1;fate--)
            result = result * fate;
        try {
            Thread.sleep(2000);
        } catch (InterruptedException e) {
            e.printStackTrace(); 
        }
    	System.out.println(result);
    }
}
