package step1;

public class Task {
	public String inversion(String str){
		/********* Begin *********/
		char []p=str.toCharArray();
		int n=str.length();
		for(int i=0;i<n/2;i++){
			char temp=p[n-1-i];
			p[n-1-i]=p[i];
			p[i]=temp;
		}
		String s=new String(p);
		return s;
		
		
		
		/********* End *********/
	}
}
