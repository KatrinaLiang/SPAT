package step1;

public class Task {
	public String inversion(String str){
		/********* Begin *********/
		// String string="123456789";
	    int length = str.length();
	    String reverse = ""; 
	    for (int i = 0; i < length; i++){
	        reverse = str.charAt(i) + reverse;
	    }
	    return(reverse);
		
		
		/********* End *********/
	}
}
