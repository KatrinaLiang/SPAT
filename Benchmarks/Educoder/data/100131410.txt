package step2;

import java.io.File;

public class Task {
	
	/********** BEGIN **********/
	public void showDirStructure(File file)	{
		dfs(file,0);

	}
	private void dfs(File file,int n){
		for(int i=0;i<n;i++) {
			System.out.print("  ");
		}
		if(file.isDirectory()) {
			System.out.println("+--"+file.getName());
			File[] myFile=file.listFiles();
			for(File f:myFile) {
				dfs(f,n+1);
			}
		}else {
			System.out.println("--"+file.getName());
		}
	}
	


	/********** END **********/
}
