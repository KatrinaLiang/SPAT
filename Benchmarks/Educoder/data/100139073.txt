package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
		MyThread t=new MyThread(num);
        new Thread(t).start();
		
		
	}
}


class MyThread  implements Runnable   {
    int num;
    public MyThread(int num){
    this.num=num;
    }
public void run(){
	int temp1=1;
    int temp2=1;
    int temp=0;
    if(num<=2)
        temp=1;
    else{
    for(int i=3;i<=num;i++){
    	temp=temp1+temp2;
        temp1=temp2;
        temp2=temp;
    }
    }
    System.out.println("子线程计算结果为："+temp);
	}

	
	

}
