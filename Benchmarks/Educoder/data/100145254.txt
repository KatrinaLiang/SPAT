package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int count = 0;
		while (true) {
			if (n == 1) {
				count++;
				break;
			}
			if (n % 2 == 0) {
				n = n / 2;
			} else {
				count++;
				n = (n - 1) / 2;
			}
		}
		/********* End *********/
		return count;
	}
	
}
