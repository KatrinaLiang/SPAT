package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum=0,n=num2-num1;
        if(num1>=100||num2>=100){
            for(int i=0;i<=n;i++){
            sum=sum+num1/100+num1/10%10+num1%10;
            num1++;
        }
        }else{
            
        for(int i=0;i<=n;i++){
            sum=sum+num1/10+num1%10;
            num1++;
        }
        }
		return sum;
		
		
		/********** END **********/
	}
}
