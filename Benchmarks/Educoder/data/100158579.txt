#include "stdio.h"
int main()
{
    char s1[100],s2[100];
    scanf("%s",s1);
    scanf("%s",s2);
    int i,j,k,m,n,max=0;
    for(i=0;s1[i]!='\0';i++)
        for(j=0;s2[j]!='\0';j++)
          {
            k=0;
            m=i;
            n=j;
            while(s1[m]!='\0'&&s2[n]!='\0')
            {
          	    while(s1[m]==s2[n])
              		{
                  k++;
                  m++;
                  n++;
                  max=max>k?max:k;
              		}
                if(s1[m]!=s2[n])
                  break;
          	}
        }
	printf("%d",max);
}
