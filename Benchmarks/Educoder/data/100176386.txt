package step3;

import java.util.Scanner;

public class HelloWorld {
	public static int getMax(int a,int b){
       int temp=b;
	   if (a>b) temp=a;
	   return temp; 
		}
		
	
    public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int i = sc.nextInt();   
         int j=sc.nextInt();
		 int c=getMax(i,j);
        System.out.println( i+"和"+j+"比较，最大值是："  + c   );
		
    }

}
