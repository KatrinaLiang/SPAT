package step1;

public class Person {
    private String name; //姓名
    private String number; //身份证号
    private Date birthday; //出生日期
    
   /************Begin*************/
   public Person(){
	   birthday =new Date();
   }
   public Person (String name,String num,Date birthday){
this.name=name;this.number=num;this.birthday=birthday;
   }
   public Person (String name,String number,int year,int month,int day){
	   this.name=name;this.number=number;this.birthday=new Date(year,month,day);
   }
   public void setinfo(String name,String num,Date birthday){
	   this.name=name;this.number=num;this.birthday=birthday;
   }
public String getname(){
	return name;
}
public String getnumber(){
	return number;
}
public Date getbirthday(){
	return this.birthday;
}
public String togetbirthday(){
	return this.getbirthday().getYear()+"."+this.getbirthday().getMonth()+"."+this.getbirthday().getDay();
}

public String toString(){
	return "name="+getname()+" number="+getnumber()+" birthday:"+getbirthday();

}
    
    
    
    
   /************End*************/
}
