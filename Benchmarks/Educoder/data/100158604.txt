#include <iostream>
#include <string>
using namespace std;
int main()
{
	string a, b;
	cin >> a >> b;
	int ans = 0;
	int n=a.length();
	for (int k = n; k > 0; k--)
	{
		for (int l = 0; l + k <= n; l++)
		{
			string tmp = a.substr(l, k);
			if (b.find(tmp) != string::npos)
			{
				ans = k;
				break;
			}
		}
		if (ans) break;
	}
	cout << ans;
	return 0;
}
