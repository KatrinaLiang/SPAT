public class HelloWorld {
    public static void main(String[] args) {
HelloWorld c=new HelloWorld();

        // 调用无参的方法
c.print();
        
        // 调用带有一个字符串参数的方法
c.print("educoder");
        
        // 调用带有一个整型参数的方法
       c.print(666); 
    }
    
    // 无参print方法的定义
void print(){
System.out.println("无参的print方法");}
    
    // 字符串print方法的定义
  void print(String a){
      System.out.println("带有一个字符串参数的print方法，参数值为："+a);
  }  
    
    // 整型print方法的定义
    void print (int b){
          System.out.println("带有一个整型参数的print方法，参数值为："+b);
    }
}
