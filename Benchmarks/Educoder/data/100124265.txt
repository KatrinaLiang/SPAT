package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] a=new int[20];
         a[0]=1;
         a[1]=1;
        int sum=0;
        for(int i=2;i<a.length;i++){
a[i]=a[i-1]+a[i-2];
        }
for(int c=0;c<a.length;c++){
    sum+=a[c];
}
System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
