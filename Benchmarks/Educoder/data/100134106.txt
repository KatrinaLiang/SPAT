Student s=new Student("张三",18,"哈佛大学");
        s.talk();
		/********* end *********/

class Person {
	/********* begin *********/

    String name ;
    int age;
    public String talk(String name,int age){
        this.name=name;
        this.age=age;
        String n="我是："+ name + "，"+ "今年："+age+"岁，";
        
    return n;
    }
	/********* end *********/
}

class Student extends Person {
	/********* begin *********/

    String school;
    String n;
    public Student(String name ,int age,String school){
        n=super.talk(name,age);
        this.school=school;
            
    }
    public void talk(){
    
        System.out.println(n+"我在"+school+"上学");
    }
	/********* end *********/
}
