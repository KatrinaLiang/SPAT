package Step3;

public class MultiplyTable {
	public static void main(String args[]){
		/********** Begin *********/
		
		for(int i=1;i<10;i++)
        {
			for(int j=1;j<=i;j++)
            {
				System.out.print(" "+i+"×"+j+"="+i*j);
				//排版   低于两位数时多占一个空格
				if(i*j<10)
                {
					System.out.print("");
                }
			
			}
			System.out.println("");
		}


		 /********** End *********/
	}
}
