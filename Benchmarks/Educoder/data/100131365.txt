package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int count=0;
        int ans=0;
        for(int i=num1;i<=num2;i++){
            int n=i;
            while(n>=1){
                count=n%10;
                n=n/10;
                ans+=count;
            }
        }
        return ans;
		
		
		/********** END **********/
	}
}
