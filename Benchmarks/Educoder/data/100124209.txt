package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int a[]=new int[20];
	int s=0;
	a[0]=1; a[1]=1;
	for(int i=0;i<18;i++)
	{
		a[i+2]=a[i+1]+a[i];
	}
	for(int i=0;i<20;i++) {
		s=s+a[i];
	}
	System.out.print("前20个元素的和为"+s);	

	/********end********/
	}
}
