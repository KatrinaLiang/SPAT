package step2;

public class TriangleJudge{

    // 判断三条边能否构成一个三角形
    public boolean TriangleJudge(int a, int b, int c) {
        // 请在此添加实现代码
        if(a<=0 || b<=0 || c<=0)
            return false;
        if(a+b<=c)
            return false;
        if(a+c<=b)
            return false;
        if(b+c<=a)
            return false;
        return true;
    }
}
