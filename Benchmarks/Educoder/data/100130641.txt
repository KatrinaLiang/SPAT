package step3;

import java.util.Arrays;

public class Task {

	public void matrixSum(int[][] arr){
		/********** BEGIN **********/
        int i=0,j=0,sum=0;
		while(i!=arr.length)
        {
            sum=sum+arr[i][j];
            i++;
            j++;
        }
		
	System.out.println(sum);


		/********** END **********/
	}
	
	
}
