package step4;

public class HelloWorld {
	public static void main(String[] args) {
		/********** Begin **********/
        int[][] scores={
            {92,85},
            {91,65},
            {90,33}
        };
        for(int a=0;a<3;a++)
            for(int b=0;b<2;b++)
                System.out.println(scores[a][b]);
        for(int m=0;m<3;m++)
            for(int n=0;n<2;n++){
                if(n==0)
                    scores[m][n]=1;
                else
                    scores[m][n]=2;
                System.out.println(scores[m][n]);}	
		/********** End **********/
	}
}
