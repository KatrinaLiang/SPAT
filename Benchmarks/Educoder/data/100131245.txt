package step1;
	 class Task{

		public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int a[] = new int[1000];
        int j = 0;
        for(int i=num1;i<=num2;i++,j++){
            a[j] = i;
        }
        int sum = 0;
        for(int i = 0;i<num2;i++){
        sum += (a[i]/100+(a[i]%100)/10+(a[i]%10));
        }
        return sum;
		/********** END **********/
	}
    }
