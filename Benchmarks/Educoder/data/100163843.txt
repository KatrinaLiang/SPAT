package step2;
import java.util.Scanner;
public class SumOfDigit {
		public static void main(String[] args) {
			int number;
			int sum=0;
			Scanner scan1=new Scanner(System.in);
	        number=scan1.nextInt();
			/******Begin******/
	        int i=10000;
	        int temp=0;
	        while(i>=1) {
	        	temp=number/i;
	        	sum+=temp;
	        	number-=i*temp;
	        	i/=10;
	        }
	        System.out.print(sum);
           

	         
	        
	                    
			/******End******/
		}
}
