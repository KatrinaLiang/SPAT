package step3;
public class NarcissisticNumber {

    //判断一个三位数是否是水仙花数
    public boolean NarcissisticNumber(int n){
        int cnt = 0, record = n;
        for(int i = 0; i < 3; ++i){
            int temp = n % 10, r = 1;
            for(int j = 0; j < 3; ++j){
                r *= temp;
            }
            n /= 10;
            cnt += r;
        }
        return cnt == record;
        // 请在此添加实现代码
    }
}
