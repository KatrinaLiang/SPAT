package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int b=0;
		for(int i=n;i>=1;i=i/2){
			if(i==1){
				b++;
				break;
			}
			if(i%2==0){
				continue;
			}
			else{
				i=i-1;
				b++;
			}
			
		}
		return b;
	

		
		
		
		
		/********* End *********/
	}
	
}
