package step2;

import java.util.Scanner;

public class SystemUpdate {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String name = scanner.next();
        String birth = scanner.next();
        String position = scanner.next();
        double salary = scanner.nextDouble();
        Salary employee = new Salary(name, birth, position, salary);
        employee.introduction();
    }
}

/********** Begin *********/
class Employee {
     String name;
     String birth;
     String position;
    public Employee(String a,String b,String c){
    	name=a;
        birth=b;
        position=c;
    }
}

class Salary extends Employee {
    private double salary;
     Salary(String a,String b,String c,double d){
    	super(a,b,c);
        salary=d;
    }
     void introduction(){
    	System.out.print("员工姓名："+super.name+" 出生年月："+super.birth+" 职位："+super.position+" 薪水："+salary);
    }
}
/********** End *********/
