package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		int result[][] = new int[matrix[0].length][matrix.length];
		for (int i = 0; i < matrix.length; i++) {
			for (int j = 0; j < matrix[i].length; j++) {
				result[j][result[i].length - i - 1] = matrix[i][j];
			}
		}
		for (int i = 0; i < matrix.length; i++) {
			for (int j = 0; j < matrix[i].length; j++) {
				matrix[i][j]=result[i][j];
			}
		}
			
		

		
		
		
		/********* End *********/
	}
}
