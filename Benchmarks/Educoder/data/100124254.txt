package step2;
public class Fibonacci{
	public static void main(String args[]){
		/*********begin********/
	        int sum=2;
			int[] args1=new int[20];
	        args1[0]=1;
	        args1[1]=1;
	        for(int i=2;i<20;i++)
	        {
	            
	            args1[i]=args1[i-1]+args1[i-2];
	            sum=sum+args1[i];
	            
	        }
	        System.out.print("前20个元素的和为"+sum);
		/********end********/
		}
}
