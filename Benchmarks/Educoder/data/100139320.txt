package step3;

public class MyThread implements Runnable {   
	    private String name;
    private Object prev;
    private Object self;
    private MyThread(String name,Object prev,Object self){
     this.name=name;
        this.prev=prev;
        this.self=self;
    }
    public void run(){
        int count=5;
        while(count>0){
            synchronized(prev){
                synchronized(self){
                   
                        System.out.print(name);
                        count--;
                        self.notify();
                    }
                    try{
                        prev.wait();
                    }catch(InterruptedException e) {
                        e.printStackTrace();
                    }
                }
                
            }
            
        System.exit(0);//退出jvm
    }   
    public static void main(String[] args) throws Exception {   
        Object e=new Object();
        Object d=new Object();
        Object u=new Object();
        MyThread te=new MyThread("E",u,e);
        MyThread td=new MyThread("D",e,d);
        MyThread tu=new MyThread("U",d,u);
        
        
        new Thread(te).start();
        Thread.sleep(100);//确保按照顺序E、D、U执行
        new Thread(td).start();
        Thread.sleep(100);
        new Thread(tu).start();
      
    }   
}
