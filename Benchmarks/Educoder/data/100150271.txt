package step2;
import java.util.*;
public class AreaOfCircle {
	public static void main(String[] args) {
		/******Begin******/
        final float  PI=3.14f;
		float r;
		float s;
		Scanner scan = new Scanner(System.in);
		System.out.println("请输入半径r的值：");
		r=scan.nextFloat();
		s=PI*r*r;
		System.out.println("圆的面积是："+String.format("%.2f",s));
		/******End******/
	}
}
