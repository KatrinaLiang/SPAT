package Pid_108;
import java.util.Scanner;
import java.util.Arrays;

public class Main {

	public static void main(String[] args) {
		// write your code here
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int row = 0;
        while (n > 0) {
            ++row;
            n = n - row;
        }
        System.out.print(row - 1);
	}

}
