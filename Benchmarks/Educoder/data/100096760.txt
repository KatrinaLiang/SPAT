package step1;
/*
 * 类MyThread有一成员变量name，补充完成该类的构造方法，初始化成员变量name的值。
在run()方法中填写相应的代码，利用格式”运行: …” 输出它的 name信息，格式参见测试样例。
注意，这里“运行“后的冒号“:”是英文字符。
 */
public class MyThread extends Thread {
    //成员变量name
	private String name;
    
    //初始化成员变量name的值
    public MyThread(String name) {
        /********** Begin *********/
    	this.name = name;
        /********** End *********/
    }
    public void run() {
        /********** Begin *********/
    	System.out.println("运行:"+name);
        /********** End *********/
    }  
}
