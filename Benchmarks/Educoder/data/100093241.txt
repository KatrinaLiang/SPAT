package step3;

public class SeasonJudge {

    // 根据月份判断季节
	public String SeasonJudge(int month) {
        // 请在此提供实现代码
        String mon;
   if(month<1||month>12)
     mon="月份错误";
    else{
        switch(month){
            case 3:
            case 4:
            case 5:mon="春季";
                break;
            case 8:
            case 6:
            case 7:mon="夏季";
                break;
            case 1:
            case 2:
            case 12:mon="冬季";
                break;
            default:mon="秋季";
                break;
                    }
   }
        return mon;
                    }
      
	}
