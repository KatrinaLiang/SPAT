package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		int [][]a=new int[matrix[0].length][matrix.length];
		int dst=matrix.length-1;
		for(int i=0;i<matrix.length;i++,dst--){
			for(int j=0;j<matrix[0].length;j++){
				a[j][dst]=matrix[i][j];
			}
		}
        dst=matrix.length-1;
        for(int i=0;i<matrix.length;i++,dst--){
			for(int j=0;j<matrix[0].length;j++){
				matrix[j][dst]=a[j][dst];
			}
		}
		/********* End *********/
	}
}
