package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
		int i = 1;
		int sum = 0;
		int k = 1;
		int temp = 0;
		int y = 0;

		while (i <= n) {

			k = i;
			y = a;
			while (k - 1 > 0) {
				y = y * 10;
				k--;
			}
			temp = temp + y;

			sum = sum + temp;
			i++;

		}
		System.out.print(sum);
	}
}
