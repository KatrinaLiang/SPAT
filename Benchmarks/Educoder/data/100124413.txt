package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] a = new int[1000];
        int sum=0;
        a[0]=1;a[1]=1;
        for(int i=2;i<a.length;i++)
        {
            a[i]=a[i-1]+a[i-2];
        }
        for(int j=0;j<20;j++){
            sum +=a[j];
        }
        System.out.print("前20个元素的和为"+sum);

	/********end********/
	}
}
