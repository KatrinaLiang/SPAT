package step1;
import java.util.Scanner;
public class AreaOfRec {
	public static void main(String[] args) {
		/******Begin******/
	Scanner sc=new Scanner(System.in);
	int chang,kuan;
	int area;
	System.out.println("请输入长方形的长：");
	chang=sc.nextInt();
	System.out.println("请输入长方形的宽：");
	kuan=sc.nextInt();
	area=chang*kuan;
	System.out.println("长方形的面积是："+area);
		/******End******/
	}

}
