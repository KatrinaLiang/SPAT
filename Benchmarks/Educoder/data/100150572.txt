package step3;
import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
        int s ,a,b,c,d,e,S;
        Scanner scan = new Scanner (System.in);
       System.out.print("请输入一个五位数字：");
        s=scan.nextInt();
        a=s%10;
        b=s/10%10;
        c=s/100%10;
        d=s/1000%10;
        e=s/10000%10;
        System.out.print("\n");
        S=a+b+c+d+e;
        System.out.println(s+"的各位数字之和是："+S);
		/******End******/
	}
}
