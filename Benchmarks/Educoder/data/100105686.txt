package step1;

public class Yard {
    float length;
    float width;
	public Yard(float length,float width){
    	 this.length=length;
         this.width=width;
    
    }
    public float getPerimeter(){
        float zc;
        zc=2*(this.length+this.width);
    return zc;
    }
    
    public float getAcreage(){
        float mj;
        mj=this.length*this.width;
    return mj;
    }
    
}
