package step3;

public class MyThread implements Runnable {   
	private String name;   
    private Object prev;   
    private Object self;   
    private Object last;
    private MyThread(String name, Object prev, Object self,Object last) {   
        this.name = name;   
        this.prev = prev;   
        this.self = self;
        this.last=last;
    }   
    public void run() {   
        int count = 5;   
        while (count > 0) {   
            synchronized (prev) {   
                synchronized (self) {
                    synchronized (last) {
                    System.out.print(name);   
                    count--;  

                    self.notify(); 
                    }
                }   
                try {   
                    prev.wait();   
                } catch (InterruptedException e) {   
                    e.printStackTrace();   
                }   
            }   

        }  
        System.exit(0);//退出jvm
    }   

    public static void main(String[] args) throws Exception {   
        Object a = new Object();   
        Object b = new Object();
         Object c = new Object();   
       
        MyThread A=new MyThread("E",c,b,a);
       MyThread B=new MyThread("D",b,a,c);
       MyThread C=new MyThread("U",a,c,b);
        
        new Thread(A).start();
        Thread.sleep(100);  //确保按顺序A、B执行
        new Thread(B).start();
        Thread.sleep(100);  
        new Thread(C).start();
        Thread.sleep(100);
    }   
}
