package task3;

public class Box {

    double width;
    double height;
    double depth;

    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public double volume(int boxType) {
        double sum;
        if(boxType==1){
            sum =width*height*depth;
            return sum;
        }
        if(boxType==2){
            sum =width*depth*height*3.14;
            return sum;
        } return 0;
    }
        

}
