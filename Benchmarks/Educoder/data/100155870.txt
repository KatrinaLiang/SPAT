//编写一个方法，计算并返回1~50的偶数和
public class TestJava23_4 {
	public static void main(String[] args) {
	int sum=TestJava23_4.SumOfEven();
		System.out.println("1~50之间偶数之和是："+sum);
	}
	public static int SumOfEven(){
        int x,sum=0;
        for(x=1;x<=50;x++){
            if(x%2==0){
                sum+=x;
            }
           
        }
        return sum;
    }
}
