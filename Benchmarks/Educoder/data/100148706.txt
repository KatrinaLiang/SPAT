package step1;

public class Task {
	
    public void sort(Integer[] arr) {
        /**
         * ******* Begin 
         */
        int i, count = 0, j;
        Integer k;

        for (i = 0; i < arr.length - 1; ++i) {
            if (arr[i] < arr[i + 1]) {
                ++count;
            }
        }
        if (count == arr.length - 1) {
            for (i = 0; i < arr.length; ++i) {
                System.out.print(arr[i] + " ");
            }
            System.out.println();
        } 
        else {
            while (count < arr.length - 1) {
                count = 0;
                for (i = 0; i < arr.length - 1; ++i) {
                    if (arr[i] > arr[i + 1]) {  //4,6,2,7
                        k = arr[i];
                        arr[i] = arr[i + 1];
                        arr[i + 1] = k;
                        for (j = 0; j < arr.length; ++j) {
                            System.out.print(arr[j] + " ");
                        }
                        System.out.println();
                    } else {
                        ++count;
                    }
                }
                for (i = arr.length - 1; i > 0; --i) {
                    if (arr[i] < arr[i - 1]) {
                        k = arr[i];
                        arr[i] = arr[i - 1];
                        arr[i - 1] = k;
                        for (j = 0; j < arr.length; ++j) {
                            System.out.print(arr[j] + " ");
                        }
                        System.out.println();
                    }
                }
            }
        }
        /**
         * ******* End ********
         */
    }
}
