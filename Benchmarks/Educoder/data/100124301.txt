package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
        int sum=2,a=1,b=1,c;
		for(int i=0;i<18;i++){
            c=a+b;
            sum=sum+c;
            a=b;
            b=c;
        }
        System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
