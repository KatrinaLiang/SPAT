package Pid_106;
import java.util.Scanner;
public class Main {

	public static void main(String[] args) 
	{
		Scanner in=new Scanner(System.in);
        String s;
        s=in.nextLine();
        for(int i=0;i<s.length();i++)
        {
        	if(s.charAt(i)==' ')
        	{
        		System.out.print("%10");
        		continue;
        	}
        	System.out.print(s.charAt(i));
        }
    }

}
