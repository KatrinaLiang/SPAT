package step3;

import java.util.Scanner;

public class HelloWorld {
    public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int t = sc.nextInt();   //获取第一个输入的整数
        int l = sc.nextInt();	//获取第二个输入的整数

		//在这里调用方法获取返回值
        HelloWorld tt = new HelloWorld();


		System.out.println( t+"和"+l+"比较，最大值是：" + tt.getMax(t,l));

    }

	/*在这里创建getMax方法，以两个整数作为参数，返回两个整数中较大的值*/
    public int getMax(int t,int l){
        int max = 0;
        if(t>max) max = t;
        if(l>t) max = l;
        return max;
    }

    
}
