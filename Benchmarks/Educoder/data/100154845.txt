package step3;

public class TestJava19_5 {
	//第一种形式
	public void first() {
		/********* Start ********/
	for(int a=1;a<=9;a++){
			for(int b=1;b<=a;b++){
				System.out.print((a*b)+"\t");
			}
			System.out.println("");
		}
		System.out.println("");
		
		/********* End ********/
	}

	// 第二种形式
	public void second() {
		/********* Start ********/
	for(int x=1;x<=9;x++){
			for(int y=1;y<=x;y++){
				System.out.print(x+"*"+y+"="+(x*y)+"\t");
			}
			System.out.println("");
		}
		
		
		/********* End ********/
	}

}
