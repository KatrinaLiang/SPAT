     */
		public static void printRect(int a,int b){
            for(int i=0;i<a;i++)
            {
                for(int j=0;j<b;j++)
                 System.out.print("*");

            }
            
        }

	 */
		public static void sortArr(int a[]){
            for(int i=0;i<a.length;i++)
            {
                for(int j=i+1;j<a.length;j++)
                 			if(a[i]>a[j])
                			{
                    			int q=a[i];
                    			a[i]=a[j];
                    			a[j]=q;
                			}
                
            }
            for(int i=0;i<a.length;i++)
                System.out.println(a[i]);
            
        }
