    public static int gcd(int a,int b) {
		int count=0;
		int x;
		int r;
		if(a<b) {
			x=a;
			a=b;
			b=x;//保证a>b
		}
		if(a%b==0||a==b) {
			count=b;
		}
		else {
			while(a%b!=0) {
				r=a%b;
				a=b;
				b=r;
				
			}
			count=b;
		}
		return count;
	}
