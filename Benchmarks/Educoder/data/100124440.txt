package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
        int [] a;
        a=new int[20];
        int i,s=0; 
        a[0]=1;
        a[1]=1;
        for(i=0;i<a.length;i++){
            if(i>1)
                a[i]=a[i-1]+a[i-2];
            s=s+a[i];
        }
        System.out.print("前20个元素的和为"+s);
        
	/********end********/
	}
}
