package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
        int sum = 2;
        
        int[] arr = new int[20];
        arr[0] =1;
        arr[1] =1;
        for(int i = 0 ; i<18; i++){
            arr[i+2] = arr[i]+arr[i+1];
            sum += arr[i+2];
        }
		System.out.print("前20个元素的和为"+sum);	

	/********end********/
	}
}
