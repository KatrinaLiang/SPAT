package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		/********** BEGIN **********/
	       if(n%2==0)
        {
     
            double b=0;
      
            double c;
            for(int i=1;i<=(n/2);i++)
            { 
                c=2*i;
                b=b+(3/c);
                
               
            }
            System.out.println(b);
        
        }
         
        
        if(n%2==1)
        {
     
            double b=0;
      
            double c;
            for(int i=1;i<=((n+1)/2);i++)
            { 
                c=2*i-1;
                b=b+(4/c);
                
               
            }
            System.out.println(b);
        
        }
		
        
		
		
		
		/********** END **********/
	}
}
