package step1;

public class Task {
    String target="AA";
	public static void main(String[] args) throws Exception {
		/********* Begin *********/
		//在这里创建线程， 开启线程
        Task t1=new Task();
	MyThread thread1=new MyThread(t1,"AA");
    MyThread thread2=new MyThread(t1,"BB");
    MyThread thread3=new MyThread(t1,"CC");
        thread1.start();
        thread2.start();
        thread3.start();
		/********* End *********/
	}
}

class MyThread extends Thread {
	/********* Begin *********/
	Task task;
	String threadName;
    
	public MyThread(Task t,String name){
		this.task=t;
      threadName=name;
	}
    

	public void run() {
	String order=threadName;
		int count = 5;
		synchronized(task){
		while(count>0){
		  while(task.target != order){
              try{
                   task.wait();
              }catch(Exception e ){
              }
          }
		
           System.out.println("Java Thread"+this.threadName);
            count--;
            switch(order){
                    case"AA":
                    task.target="BB";
                    break;
                    case"BB":
                    task.target="CC";
                    break;
                    case"CC":
                    task.target="AA";
                    break;
            }
            task.notifyAll();
		}
      }
    }
}
