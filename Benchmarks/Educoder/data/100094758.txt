package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        int i,x,j,b;
        i=x=0;
        while(i<n)
        {
            j=b=0;
            while(j<=i)
            {
                b=b*10+a;
                j++;
            }
            x=x+b;
            i++;
        }
        System.out.print(x);
		/********** End *********/
	}
}
