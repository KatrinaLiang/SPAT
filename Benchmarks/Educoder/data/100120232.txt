package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int i,j;
		for(j=1;j<=9;j++)
        {	
            for(i=1;i<=j;i++){
				System.out.print(i+"*"+j+"=" + i*j +"\t");
            }
            System.out.println();
        }
		/*****end*****/
	}
}
