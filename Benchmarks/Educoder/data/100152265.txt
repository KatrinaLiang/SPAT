public class QueryListTest {
    
    private static QueryList queryList = new QueryList(1,10,0);
    private static Query req = new Query(2,1.5, Query.Direction.UP);
    
    @Test
    public void testAppend() {
        if (0 <= req.getTime()) {
            if (!(req.getTarget() == 1 && req.getDirection() == Query.Direction.DOWN)
                    && !(req.getTarget() == 10 && req.getDirection() == Query.Direction.UP)) {
                assertEquals(true,queryList.append(req));
            }
        }
    }
}
