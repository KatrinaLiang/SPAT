package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int result = 0;
		
		for (int i = num1; i <= num2; i++) {
			char[] chs = (i+"").toCharArray();
			int len = chs.length;
			int[] valus = new int[len]; 
			for (int j = 0; j < len; j++) {
				valus[j] = chs[j] - '0' + 0;
				result += valus[j];
			}
		}			
		return result;
		
		
		/********** END **********/
	}
}
