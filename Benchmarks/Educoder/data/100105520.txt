package step1;

public class Yard {
    int length, width;
    Yard(int len, int wid){
        length =len;
        width = wid;
    }
    float getPerimeter(){
         return (length + width)*2;
     }
    float getAcreage(){
         return length * width;
     }
    public static void main(String[] args){
        Yard a = new Yard(2,3);
        System.out.println(a.getPerimeter());
        System.out.println(a.getAcreage());
    }

}
