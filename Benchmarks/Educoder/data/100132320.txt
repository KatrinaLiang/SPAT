package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
		int i;
        int j;
        int num;
        int p[]=new int[temperatures.length];
        for(i=0;i<=temperatures.length-2;i++)
        {
            num=0;
            for(j=i+1;j<=temperatures.length-1;j++)
            	{
                if(temperatures[j]<=temperatures[i]) num++;
                else break;
            	}
            if(j>temperatures.length-1) num=-1;
            p[i]=num+1;
        }
        p[temperatures.length-1]=0;
		
		

		return p;
		/********** END **********/
	}
}
