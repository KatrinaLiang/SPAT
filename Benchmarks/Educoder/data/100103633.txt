package step3;

import java.io.File;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

public class DeleteFiles {
    public static void start() throws IOException {
        StringBuilder data = new StringBuilder("");
        /********** Begin *********/
        //读取readme.txt文件

   File file1 = new File("src/step3/readme.txt");
   FileReader readmeReader = new FileReader(file1);
   int read = readmeReader.read();
    while (read != -1) {
    data.append((char) read);
    read = readmeReader.read();
}
readmeReader.close();

        //写入readmeCopy.txt文件
File file2 = new File("src/step3/readmeCopy.txt");
   if (!file2.exists()) {
  file2.createNewFile();
  }
   FileWriter fileWriter = new FileWriter(file2, false);
   fileWriter.write(data.toString());
   fileWriter.close();
        //删除readme.txt文件
        
if (file1.exists()) {   //如果文件存在
    file1.delete();   //删除该文件，并输出布尔值
}

        /********** End *********/
    }
}
