package step2;

public class Point {
	private double x;
	private double y;

	public Point() {
		this(0, 0);
	}

    public double getX()
    {
    	return this.x;
    }
    
    public double getY()
    {
    	return this.y;
    }
    
	public Point(double x, double y) {
		this.x = x;
		this.y = y;
		System.out.println("Point constructor:" + this.toString());
	}

	public String toString() {
		return "[" + this.x + "," + this.y + "]";
	}

}
