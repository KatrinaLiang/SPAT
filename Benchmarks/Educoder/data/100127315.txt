package step2;

public class Circle extends Point {
	private double radius;

	/*********begin*********/
	//编写两个构造方法，无参函数中的半径默认为10，x和y均为0;
    Circle(){
        super();
        this.radius=10;
        System.out.print("Circle constructor:Center="+this.tostring());
        System.out.println("radius="+this.radius);
    }
    Circle(double b){
        super(10,10);
        this.radius=b;
        System.out.print("Circle constructor:Center="+this.tostring());
        System.out.println("radius="+this.radius);
    }
    //在构造方法中输出Circle constructor:  加上 toString方法输出的结果
	
	
	//编写toString方法
	public String tostring(){
        return "[" + this.getx() + "," + this.gety() + "]";
    }
	/*********end*********/

	public static void main(String[] args) {

		System.out.println("第一个圆的信息：");
		/*********begin*********/
		//构造第一个圆cir1
        Circle cir1=new Circle();
		System.out.println("第二个圆的信息：");
		//构造第二个圆cir2，x为5，y为10，半径为10
		Circle cir2=new Circle(5);	
		/*********end*********/

	}
}
