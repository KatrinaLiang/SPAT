package step3;

public class Task {

	 public int countOfShips(char[][] ferry) {
	  /**************BEGIN************/
     int   number=0;
         if(ferry==null||ferry.length==0||ferry[0].length==0)number=0;
         for(int i=0;i<ferry.length;i++)
             for(int j=0;j<ferry[i].length;j++)
                 if(ferry[i][j]=='+'){
                   for(;j+1<ferry[i].length&&ferry[i][j+1]=='+';j++){}
                     if(i-1==-1||ferry[i-1][j]!='+')number++;
                 }
                 
         return number;
      /***********END ***************/
	 }
}
