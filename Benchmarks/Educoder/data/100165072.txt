package task5;

public class Box {

    double width;
    double height;
    double depth;

    //TODO your code
    public Box(){};
    public Box(double width,double height,double depth){
        this.width = width;
        this.height = height;
        this.depth = depth;
    }
    public double volume(){
        double vol;
        vol = width * height * depth;
        return vol;
    }


}
