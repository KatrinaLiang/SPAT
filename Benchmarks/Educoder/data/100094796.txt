package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        double sum=0,temp=0;
        for(int i=0;i<n;i++){
            temp=temp+a*Math.pow(10,i);
            sum=sum+temp;
        }
        int s=(int)sum;
        System.out.print(s);       



		/********** End *********/
	}
}
