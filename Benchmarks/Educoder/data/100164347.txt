		   int total,number;

	public static void Josecount(int personNumber ,int number) { //显示最后那个人的原始编号
		    //显示最后那个人的原始编号
                int ima = personNumber;
        int check = 0;
        	int[] persons = new int[personNumber];
        	while (ima != 1)
            {
                for (int i = 0; i < personNumber; i++)
                {
                    if (persons[i] != 1145141919)
                    {
                        persons[i] = check + 1;
                        if (persons[i] == number)
                        {
                            persons[i] = 1145141919;
                            check = 0;
                        }
                       	else check++;
                    }
                    else continue;

                }
                ima = 0;
                for (int i = 0; i < personNumber; i++)
                {
                   
                    if (persons[i] != 1145141919)
                        ima++;
                }
            }
       for (int i = 0; i < personNumber; i++)
