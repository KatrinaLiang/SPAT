package step4;

public class DayJudge {

    // 根据输入年、月、日判断这一天是当年的第几天
    public String DayJudge(int year, int month, int day){
        // 请在此提供实现代码
        String l;
        if(year%4==0&&year%100!=0||year%400==0)
            if(month>2)
                day=day+1;
        switch(month)
        {
            case 12:day=day;
            case 11:day=day+31;
            case 10:day=day+30;
            case 9:day=day+31;
            case 8:day=day+31;
            case 7:day=day+30;
            case 6:day=day+31;
            case 5:day=day+30;
            case 4:day=day+31;
            case 3:day=day+28;
            case 2:day=day+31;
            case 1:day=day;
        }
        if(month==12)
            day=day+30;
        l=year+"年的第"+day+"天";
        return l;
            
	}
}
