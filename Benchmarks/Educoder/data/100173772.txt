package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
	int i,j;	
	for(i=1;i<=10;i++){
	    for(j=1;j<=i;j++){
	    System.out.print("*");	//在这里打印出正三角形 
		}	
	   System.out.println();	
	}
		
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
	int k,l;	
	for(k=11;k>1;k--){
	    for(l=k;l>1;l--){
	    System.out.print("*");	
		}	
	System.out.println();	
	}	
		
		
		
		
		/*****end*****/
	}
}
