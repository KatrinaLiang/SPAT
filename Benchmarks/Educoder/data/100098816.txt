package step1;

public class Yard {
private  int Perimeter;
    private int Acreage;
    public Yard(int x,int y){
        this.Perimeter=x;
        this.Acreage=y;
    }
    public double getPerimeter(){
        return (Perimeter+Acreage)*2;
    }
    public double getAcreage(){
			return Perimeter*Acreage;
    }
}
