package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
        int[]arr=new int[20];
        arr[0]=1;
        arr[1]=1;
        for(int i=1;i<=18;i++){
        arr[i+1]=arr[i]+arr[i-1];
        }
        int sum =0;
        for(int i=0;i<=19;i++){
        sum=sum+arr[i];
        }
System.out.print("前20个元素的和为"+ sum );
	/********end********/
	}
}
