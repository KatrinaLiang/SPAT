package step3;
import java.util.Scanner;
public class NarcissisticNumber {
    public static void main(String[] args) {

        // 从命令行读入一个三位数
        // 这个整数取自测试集的输入
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        NarcissisticNumber number = new NarcissisticNumber();
        System.out.println(number.NarcissisticNumber(n));
    }

    //判断一个三位数是否是水仙花数
    public String  NarcissisticNumber(int n){
        // 请在此添加实现代码
        String s;
        int a,b,c;
        a=n%10;
        b=n/10%10;
        c=n/100;
        if(((a*a*a)+(b*b*b)+(c*c*c))==n)
            s="true";
        else 
            s="false";
  return s;  }
}
