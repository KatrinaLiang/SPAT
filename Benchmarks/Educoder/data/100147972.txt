public class ThreadHelper{
    static public void calcOnNewThread(int c){
        new Thread(new Runnable() {
    	    @Override public void run() {
    	    	int result = 1;
    			for(int i=1;i<=c;++i)
                {
                    result=result * i;
                }
    	    	System.out.println(result);
    	    }		
    	}).start();
    }
}
