package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum=0,hundred,ten,bits;
        for(int i=num1;i<=num2;i++){
        	hundred=i/100;
            ten=i%100/10;
            bits=i%10;
            sum=hundred+ten+bits+sum;
        }
        return sum;
		
		
		/********** END **********/
	}
}
