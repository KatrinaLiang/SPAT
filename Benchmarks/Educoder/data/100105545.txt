package step1;

public class Yard {
int length;
    int width;
    String str1;
    String str2;
    public Yard(int length,int width){
        this.length=length;
        this.width=width;
    }
    public String getPerimeter(){
        int num =2*length+2*width;
        str1 = String.valueOf(num);
        str1+=".0";
        return str1;
    }
      public String getAcreage(){
          int num =length*width;
        str2 = String.valueOf(num);
        str2+=".0";
        return str2;
      }
}
