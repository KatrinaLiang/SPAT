package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
		int i = 0;
        int j = a;
        while(n > 0){
        	i = j + i;
            j = j * 10 + a;
            n--;
        }
		System.out.print(i);
		/********** End *********/
	}
}
