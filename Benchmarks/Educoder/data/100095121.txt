package step3;
import java.util.Scanner;
public class Factorial {
	public void factorial(int n) {
		/********** Begin *********/
int s=1;
for(int i=1;i<=n;i++){
    s=s*i;
}
System.out.print(""+s);
	}
	public static void main(String[] args) {
		Scanner scanner=new Scanner(System.in);
        int n=scanner.nextInt();
Factorial a=new Factorial();
a.factorial(n);
	}
		/********** End *********/
}
