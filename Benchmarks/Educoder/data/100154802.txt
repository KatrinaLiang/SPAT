package step3;

public class TestJava19_5 {
	//第一种形式
	public void first() {
		for (int n = 1; n <= 9; n++) {
			for (int z = 1; z <= n; z++) {
				System.out.print(n*z+"\t");
			}
			System.out.println();
		}
        System.out.println();
	}

	// 第二种形式
	public void second() {
		for (int n = 1; n <= 9; n++) {
			for (int z = 1; z <= n; z++) {
				System.out.print(n+"*"+z+"="+n*z+"\t");
			}
			System.out.println();
		}

	}
}
