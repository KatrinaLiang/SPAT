package step3;
import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
		Scanner in=new Scanner(System.in);
        int number;
        System.out.println("请输入一个五位数字：");
		number=in.nextInt();
		int result=0;
		int sum;
		do{
			sum=number%10;
			result=result+sum;
			number=number/10;
		}while(number>0);
		System.out.println("85698的各位数字之和是："+result);
        
		/******End******/
	}
}
