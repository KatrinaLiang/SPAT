package step3;

public class MyThread implements Runnable {   
	private String name;   
    private Object prev;   
    private Object self;
    private Object after;
    private MyThread(String name, Object prev, Object self,Object after) {   
        this.name = name;   
        this.prev = prev;   
        this.self = self;  
        this.after=after;
    }   
    public void run() {   
        int count = 5;   
        while (count > 0) {   
            synchronized (prev) {   
                synchronized (self) {
                	synchronized(after){
                    System.out.print(name);   
                    count--;  
                    self.notify();   
                }
            }
               try {   
                   prev.wait();   
               } catch (InterruptedException e) {   
                   e.printStackTrace();   
               }   
            }   
        }  
        System.exit(0);
    }   
    public static void main(String[] args) throws Exception {   
        Object e = new Object();  
        Object d = new Object();
        Object u = new Object();
        MyThread ta = new MyThread("E",d,u,e);   
        MyThread tb = new MyThread("D",u,e,d);
        MyThread tc = new MyThread("U",e,d,u);
        new Thread(ta).start();
        Thread.sleep(100);
        new Thread(tb).start();
        Thread.sleep(100); 
        new Thread(tc).start();
        Thread.sleep(100);
    }
}
