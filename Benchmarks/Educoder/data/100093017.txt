package step2;

public class SwitchToGrade {

    // 请用switch来将学生成绩转换成对应等级
    public String SwitchToGrade(int score) { 
        // 请在此提供实现代码 
        int score0=score/10;
        char ch='a';
        switch(score0){
            case 9:ch='A';break;
            case 8:case 7:ch='B';break;
            case 6:ch='C';break;
            case 5:case 4: case 3: case 2: case 1: case 0:ch='D';break;
            default:break;
        }
        if(ch=='A')
           return "等级："+ch;
        else if(ch=='B')
           return "等级："+ch;
        else if(ch=='C')
           return "等级：C";
        else if(ch=='D')
           return "等级：D";
        else
        	return "输入成绩非法";
	}
}
