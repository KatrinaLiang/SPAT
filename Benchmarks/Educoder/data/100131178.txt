package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int max,min,sum=0;
        if(num1>num2)
        {
            max=num1;
            min=num2;
        }
        else
        {
            max=num2;
            min=num1;
        }
        
        for(int i = min;i <= max; i++)
        {
          int num = i;
          while(num!=0)
          {
              sum += (num%10);
              num=num/10;
          }
        }
        return sum;
		/********** END **********/
	}
}
