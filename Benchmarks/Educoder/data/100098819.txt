package step1;

public class Yard {
  
    public Yard(double l ,double w)
    {
        length=l;
        width=w;
    }
   public double getPerimeter()
    {
        return length + width + length + width;}
   public double getAcreage()
    {
        return length*width;}
     private  double length;
    private double width;
    
}
