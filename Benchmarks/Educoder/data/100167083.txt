package step1;

public class A< T extends Comparable >{
 private T t1,t2; 
/*************BEGIN**************/
 public A(T x,T y){
 	this.t1=x;
    this.t2=y;
 }
 public T compare(){
 	if(t1.compareTo(t2)>=0){
    	return t1;
    }else 
        return t2;
 }




/*************END*****************/

}
