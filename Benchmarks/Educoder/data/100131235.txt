package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int num=(num1>num2)?num2:num1;
		int number=0;
        int end=(num1>num2)?num1:num2;
        for(;num<=end;num++){
            int nums=num;
           	while(nums>0){
                int n=nums%10;
                number+=n;
                nums/=10;
        }
        }
        return number;
		
		/********** END **********/
	}
}
