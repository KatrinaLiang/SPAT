package step1;

public class Yard {
	private int  length;
    private int  width;
    public float getPerimeter(){
	return (length+width)*2;
    }
    public float getAcreage(){
    return length*width;
    }
    public Yard(int length,int width)
    {
        this.length=length;
        this.width=width;
    }
       public Yard(){}
    
    	
}
