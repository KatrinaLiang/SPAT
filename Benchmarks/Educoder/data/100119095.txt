package setp17;

import java.util.Arrays;
import java.util.Scanner;
public class HelloWorld {
	public static void main(String[] args) {
		
		Scanner input=new Scanner(System.in);
	    int a=input.nextInt();
	    int b=input.nextInt();
	    double f=input.nextDouble();
	    double l=input.nextDouble();
	    double z=input.nextDouble();
	    int h=input.nextInt();
	    int g=input.nextInt();
	    int[] m=new int[5];
	    for(int as=0;as<m.length;as++)
	    {
	        m[as]=input.nextInt();
	    }
	    int p=getSum(a,b);
	    double k=getAvg(f,l,z);
	    System.out.println(p);
	    System.out.println(k);
	    printRect(h,g);
	    sortArr(m);
	    Print99();
	}
	public static int getSum(int a,int b){
	    int d=a+b;
	    return d;
	}
	public static double getAvg(double a,double b,double c){
	    double j=(a+b+c)/3;
	    return j;
	}
	public static void printRect(int a,int b){
	    for(int i=1;i<=a;i++){
	        for(int n=1;n<=b;n++){
	            System.out.print("*");
	        }
	        System.out.println();
	    }
	}
	public static void sortArr(int f[]){
	    Arrays.sort(f);
	    for(int i=0;i<5;i++) {
		    	System.out.println(f[i]);
		    	
		    }
	        
	}


	    public static void Print99() {
	        for (int i=1;i<=9;i++   ) {
	            for(int n=1;n<=i;n++   ) {
	                System.out.print(n+ " * " + i + " = " + n* i+ "\t");
	            }
	            System.out.println();
	        }
	    }

	    
	    
	    
	}
