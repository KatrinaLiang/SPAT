package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
        int sum = 0;
        int j = 1; 	//第二个
        int k = 0;	//前一个
        for(int i = 0; i < 20 ; i++){
            sum += j;
            j = j + k;
            k = j - k;
        }
		System.out.print("前20个元素的和为"+sum);

	/********end********/
        
         
             
	}
}
