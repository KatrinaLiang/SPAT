public class MyThread extends Thread {
	private int num;
	public MyThread() {
		this(0);
	}
	public MyThread(int num) {	
	this.num=num;
	}
	 @Override
    public void run() {    	
    	int result = 1;
		for(int i=2;i<=num;i++){
        result*=i;
        }
        try{
          Thread.sleep(500);
        }catch(InterruptedException e){
        e.printStackTrace();
        }
    	System.out.println(result);
    }
}
