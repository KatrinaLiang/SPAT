package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		/********** BEGIN **********/
		double ans = 0;
        int t;
		if(n%2==0) {
            t = 2;
            while(true) {
                ans += 3.0/t;
                t += 2;
                if(t>n) break;
            }
        } else {
            t = 1;
            while(true) {
                ans += 4.0/t;
                t += 2;
                if(t>n) break;
            }
        }
		System.out.println(ans);
		
		
		
		/********** END **********/
	}
}
