package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int x=1,y=1,z=0;
		int sum=2;
		int[] arr=new int[18];
		for(int i=1;i<=20-2;i++)
		{
			z=y;
			y=x+y;
			x=z;
			arr[i-1]=y;
			sum=sum+arr[i-1];
		}
		System.out.print("前20个元素的和为"+sum);

	/********end********/
	}
}
