package step1;

public class LetterChange {

    // 使用 switch把小写字母转为大写字母
    public char LetterChange(char ch){
        // 请在此提供实现代码
        char ch1;
        switch(ch)
        {
            case 'a':ch1='A';break;
            case 'b':ch1='B';break;
            case 'c':ch1='C';break;
            case 'd':ch1='D';break;
            case 'e':ch1='E';break;
            default:ch1=ch;
        }
        return ch1;
  }
}
