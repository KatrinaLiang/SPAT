public class MyThread extends Thread{
    private int a;
    public MyThread(){
    	this(0);
    }
    public MyThread(int a){
    	this.a=a;
    }
    public void run(){
    	try{
            Thread.sleep(5000);
        }catch(Exception e){}
        int result = 1;
        for(int i=1;i<=a;i++){
        	result=result*i;
        }
        System.out.println(result);
    }
}
