    public static  int sort( int i   ){
    	int[] arr={79,52,98,81};
    for( i = 0; i < arr.length - 1; i++) {
    int min = i;
    for(int j = i +1;j < arr.length; j++) {
     if(arr[j] < arr[min]) {
      min = j;
     }
    }
     if(i != min) {
     int temp = arr[i];
     arr[i] = arr[min];
     arr[min] = temp;
    }
   
    }
    System.out.println(Arrays.toString(arr));   

  
	return i;
    }     // 打印出数组
    // 数组从小到大排序处理
