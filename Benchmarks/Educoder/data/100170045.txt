package step1;
import java.util.Scanner;

public class Cal {
	public static void main(String[] args) {
		/*********start*********/
		Scanner input = new Scanner(System.in);
		System.out.println("请输入第一个整数");
		int a=input.nextInt();
		System.out.println("请输入第二个整数");
		int b=input.nextInt();
		int sum=a+b;
		int red=a-b;
		int mul=a*b;
		int div=a/b;
		int i=a%b;
		System.out.print("两数相加的结果为："+sum+"\n");
		System.out.print("两数相减的结果为："+red+"\n");
		System.out.print("两数相乘的结果为："+mul+"\n");
		System.out.print("两数相除的结果为："+div+"\n");
		System.out.print("两数取余数的结果为："+i+"\n");


		/*********end*********/
	}

}
