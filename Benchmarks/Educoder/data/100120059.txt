package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		for(int i=1;i<=9;i++)
        {
            for(int z=1;z<=i;z++)
            {
                System.out.print(z+"*"+i+"="+(i*z));
                if(z==i)
                    System.out.print("\t");
                else
                    System.out.print("\t");
            }
            System.out.print("\n");
        }
		
		
		
		
		/*****end*****/
	}
}
