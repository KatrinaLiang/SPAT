package step2;

public class MyRunnable implements Runnable {
	private int num;
	
    public MyRunnable() {
    	this(0);
    }
    
    public MyRunnable(int num) {
    	/***begin your code here***/
    	int i;
        for(i=1;num!=1;num--)
            i=i*num;
        this.num=i;
    	/***end your code here***/
    }
	
	@Override
	public void run() {
		//重写run方法，在子线程中想要执行的代码写在run方法中
    	int result = 1;//result保存计算出的结果
    	
        /***begin your code here***/ 
		result =this.num;
		/***end your code***/
    	
    	//直接输出结果
    	System.out.println(result);
	}

}
