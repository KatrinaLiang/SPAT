package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum=0;
        for(int i=num1;i<=num2;i++){
             if(i<100){
             sum+=i/10+i%10;
            }else if(i<1000){
             sum+=i/100+i%10+i%100/10;
        }
        }
        return(sum);
		
		
		/********** END **********/
	}
}
