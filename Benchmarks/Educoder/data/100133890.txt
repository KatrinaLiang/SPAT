class Animal {
	/********* begin *********/
    String name;
    int age;
     public String getName(){
        return name;
    }
    public void setName(String animalName){
        name=animalName;
    }
    public int getAge(){
        return age;
    }
    public void setAge(int animalAge){
        if(animalAge>0)
        age=animalAge;
    }
	/********* end *********/

}

class Cat extends Animal {
	// 定义Cat类的voice()和eat()方法
	/********* begin *********/
	public void voice(){
        System.out.println(name+"喵喵叫");
    }
    public void eat(){
        System.out.println(name+"吃鱼");
    }
	/********* end *********/
}
