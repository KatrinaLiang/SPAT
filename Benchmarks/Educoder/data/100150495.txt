package step3;
import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
		System.out.println("请输入一个五位数字：");
			Scanner t=new Scanner(System.in);
			int i=t.nextInt();
			int j=0;
			while(i%10!=0)
			{
				j=j+i%10;
				i/=10;
			}
			System.out.println("85698的各位数字之和是："+j);
		/******End******/
	}
}
