package Pid_108;
import java.util.Scanner;
public class Main {

	public static void main(String[] args) {
		// write your code here
        Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int m = 0;
		int num = 0;
		for(int i=1;m<n;i++)
		{
			m = m+i;
			num++;
		}
		num--;
		System.out.print(num);
	}

}
