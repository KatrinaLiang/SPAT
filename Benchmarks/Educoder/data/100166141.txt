package step1;

public class Person {
    public String name; //姓名
    public String number; //身份证号
    public Date birthday; //出生日期
    
   /************Begin*************/
   public Person(){
       birthday=new Date();}
   public Person(String name,String num,Date birthday){
           this.name=name;
           number=num;
           this.birthday=birthday;}
   public Person(String name,String num,int year,int month,int day){
           this.name=name;
           number=num;
           birthday=new Date(year,month,day);}
    public void setinfo(String name,String number,Date d1){
        this.name=name;
        this.number=number;
        birthday=d1;}
    public String toString(){
        return "name="+name+" number="+number+" birthday:"+birthday.toString();}
       


    
    
    
    
   /************End*************/
}
