public class GetScores {

    public int getLostScores(int[] nums){
       //请在此补全你的代码
    int f[] = new int[10010];
    int f1[] = new int[10];
    int flag=1,res=0,l1=1,l2=1;
	
	for(int i=0;i<10000;i++) f[i]=0;
	for(int i=0;i<9997;i++) f[nums[i]]=1;
	for(int i=1;i<=10000;i++)
		if(f[i]==0)	f1[flag++]=i;
		
	int k1=f1[1],k2=f1[2],k3=f1[3];
	
	flag=0;
	while(k2!=0){	flag++;k2=k2/10;}
	for(int i = 1;i <= flag;i++)	l1 *= 10;
	 
	flag=0;
	while(k3!=0){	flag++;k3=k3/10;}
	for(int i = 1;i <= flag;i++)	l2 *= 10;

	l2=l2*l1; 
	res = (f1[3]%11*l2+f1[2]%11*l1+f1[1]%11)%11;
    return res;

    }
