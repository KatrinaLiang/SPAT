package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temps){
		/********** BEGIN **********/
        int[] s=new int[temps.length];
		for(int i=0;i<temps.length;i++){
            int k=0;
            int j=i;
       while(j<temps.length){ 
           if(temps[j]>temps[i]) break;
           j++;
       }
            if(j==temps.length)
                s[i]=0;
       else s[i]=j-i;
       }

		return s;
		/********** END **********/
	}
}
