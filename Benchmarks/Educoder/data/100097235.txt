package step2;

// 利用 Runnable 接口实现
/********** Begin *********/
public class MyThread implements Runnable {
/********** End *********/
	private String name;
    public MyThread(String name) {
        /********** Begin *********/
             this.name=name;
		/********** End *********/
      
    }  
    public void run() {
        /********** Begin *********/
          System.out.println("运行:" +this.name);
		/********** End *********/
        
    }
    
}
