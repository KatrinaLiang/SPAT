package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int a = 0;
		while (n >= 1) {
			if (n % 2 == 0) {
				n = n / 2;
			} else {
				a++;
				n = (n - 1) / 2;
			}
		}
		if (n == 1) {
			a++;
		}
		return a;
		
		
		
		
		/********* End *********/
	}
	
}
