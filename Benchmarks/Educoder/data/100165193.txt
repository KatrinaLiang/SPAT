public class Box {

    private double height;
    private double depth;

  /********* Begin *********/        
	public double setWidth(double width){
        this.width=width;
    	return width;
    }

        this.height=height;
        return height;

    public double getHeight(){
        return height;
    }
    public double setDepth(double depth){
        this.depth=depth;
        return depth;
    }
    public double getDepth(){
        return depth;
    }



  /********* End *********/        

}
