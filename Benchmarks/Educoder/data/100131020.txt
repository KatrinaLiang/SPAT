package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int res=0;
		int i=0;
		for(int temp=num1;temp<=num2;temp++) {
			i=temp;
			res+=i/100;
			i%=100;
			res+=i/10;
			i%=10;
			res+=i;
		}
		return res;
		/********** END **********/
	}
}
