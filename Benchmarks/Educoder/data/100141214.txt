package step3;

public class Task {
	public static void main(String[] args) {
		double a = 78.5;
		//将基本类型a转换为字符串
		Double d1 = new Double(a);
		String str = d1.toString()     ;
        
		System.out.println("str + 12  的结果为： "+(str + 12));
		
		String str1 = "180.20";
		Double d2 = new Double(str1);
	    // 将字符串str1转换为基本类型
		Double d =  d2.doubleValue()    ;
	
		System.out.println("d + 100 的结果为： "+ (d + 100));

	} 
}
