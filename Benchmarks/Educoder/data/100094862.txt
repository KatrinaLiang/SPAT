package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        int sum=0;
        
		while(n>0){
            int i=0;
            int m=0;
			for(i=n;i>0;i--){
             	m+=a*Math.pow(10,i-1);  
            }
            sum+=m;
            m=0;
            n--;
        }
        System.out.print(sum);


		/********** End *********/
	}
}
