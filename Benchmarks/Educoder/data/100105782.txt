package step1;

public class Yard {
    private double width;
    private double length;
    
    public Yard(double width,double length){
        this.width = width;
        this.length = length;
    }
        
    public double getPerimeter(){
        return (width+length)*2;
    }
    public double getAcreage(){
        return length*width;
    }
    

}
