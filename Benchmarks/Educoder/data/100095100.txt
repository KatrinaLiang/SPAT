package step3;
import java.util.Scanner;
public class Factorial {
	public void factorial(int n) {
        int sum=1;
		for (int i=n;i>1;i--){
           sum=sum*i;
        }
        System.out.print(sum);
	}
    public static void main(String[] args){
       Scanner sa =new Scanner(System.in);
       int n;
       n=sa.nextInt();
       Factorial se = new Factorial();
       se.factorial(n);
    }
}
