package step1;

public class Rectangle{
	private double width;
	private double length;
	
	/*********begin*********/
    public Rectangle()
    {
        this(6.0,5.0);
    }
    public Rectangle(double length,double width)
    {
        this.length=length;
        this.width=width;
    }
	public double getC()
    {
        return 2.0*(width+length);
    }
    public double getS()
    {
        return 1.0*width*length;
    }
	/*********end*********/


	public static void main(String[] args) {
		/*********begin*********/
		Rectangle r1=new Rectangle();
        Rectangle r2=new Rectangle(20,10);
		System.out.println("r1的周长："+r1.getC());
        System.out.println("r1的面积："+r1.getS());
       	System.out.println("r2的周长："+r2.getC());
        System.out.println("r2的面积："+r2.getS());
		/*********end*********/
	}

}
