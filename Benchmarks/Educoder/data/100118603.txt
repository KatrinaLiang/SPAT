public class HelloWorld {
    public static void main(String[] args) {

		defin();
        // 调用无参的方法
		defin("educoder");
        // 调用带有一个字符串参数的方法

        // 调用带有一个整型参数的方法
        defin(666);
    }
    
    // 无参print方法的定义
	public static void defin(){
        System.out.println("无参的print方法");
    }
    
    // 字符串print方法的定义
    	public static void defin(String a){
        System.out.println("带有一个字符串参数的print方法，参数值为："+a);
    }
    
    // 整型print方法的定义
    	public static void defin(int a){
        System.out.println("带有一个整型参数的print方法，参数值为："+a);
    }
    
}
