package step3;

public class Task {
    
	public boolean isPowerCut(String date){
			int [] d = new int [100];
		int i=1,f=1,cnt=0;
		while(i<=92){
			d[i]=1;
			i=i+f+1;
			f++;
		}
 		int y1=date.charAt(0)-'0';
		int d1;
		int t=date.length();
		if(t==3)
			d1=date.charAt(2)-'0';
		else 
			d1=(date.charAt(2)-'0')*10+date.charAt(3)-'0';
		if(y1==7) {
			cnt=d1;
		}
		else if(y1==8) {
			cnt=31+d1;
		}
		else {
			cnt=62+d1;
		}
		if(d[cnt]==1)
			return true;
		else
			return false;
	}
}
