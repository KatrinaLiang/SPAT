package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
	    for(int a=1;a<=9;a++){
            for(int i=1;i<=a;i++){
            System.out.print(i+"*"+a+"="+(a*i));
            if(a==i) System.out.println("\t");
            else System.out.print("\t");
            }
        }	
		/*****end*****/
	}
}
