package step3;
import step1.Date;
public class Course {
    private int code; //课程编号
    private String courseName; //课程名称
    private Date testTime;//考试日期
    private int score;//考试成绩
   /******Begin**************/

public Course(int c, String cn, int y, int m, int d, int sc){
        code = c;
        courseName = cn;
        testTime = new Date(y, m, d);
        score = sc;
    }

    public Course(int c, String cn, Date d, int sc){
        code = c;
        courseName = cn;
        testTime = d;
        score = sc;
    }

    public Course(){
        testTime = new Date();
    }

    public String toString(){
        if(code == 0) return "0  " + testTime.toString() + " " + score;
        return code + " " + courseName + " " + testTime.toString() + " " + score;
    }

    public void setCourse(){

    }

    public int getCode() {
        return code;
    }

    public String getCourseName() {
        return courseName;
    }

    public Date getTestTime() {
        return testTime;
    }

    public int getScore() {
        return score;
    }



   /******End************/

}
