package step1;

public class GetScores {

	/********* Begin *********/
    public int getLostScores(int[] nums){
       //请在此补全你的代码
    	Boolean[] arr = new Boolean[10000];
		int[] temp = new int[3];
		int p = 0;
		int max = 0, min = 0,mid = 0;
		for (int i = 0; i < nums.length; i++) {
			arr[nums[i] - 1] = true;
		}
		for (int i = 0; i < 10000; i++) {
			if (arr[i] == null || arr[i] != true) {
				temp[p] = i + 1;
				p++;
			}
		}
		
		for (int i = 1; i < 3; i++) {
			if (temp[i] > temp[max])
				max = i;
			if (temp[i] < temp[min])
				min = i;
		}
		
		for (int i = 0; i < 3; i++) {
			if (i != max && i != min) {
				mid = i;
				break;
			}
		}
		String str = "";
        str+=String.valueOf(temp[min])+String.valueOf(temp[mid])+String.valueOf(temp[max]);
        long l = Long.valueOf(str).longValue();
        int returnId=Integer.parseInt(String.valueOf(l%11));
        return returnId;

	}

	/********* End *********/
}
