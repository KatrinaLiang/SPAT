public class Task {
	public static String inversion(String s){
		/********* Begin *********/
		int len=s.length();
        String str="";
        for(int i=0;i<len;i++){
            str=s.charAt(i)+str;
        }
        return str;
    }
    public static void main(String[] args) {
        String str="hello world";
        System.out.println(Task.inversion(str));
    }
