package step1;

public class Yard {
    private int length = 0;
    private int width = 0;
	public Yard(int length, int width) {
        this.length = length;
        this.width = width;
    }
    
    public double getAcreage() {
        return this.length*this.width;
    }
    
    public double getPerimeter() {
        return 2*(this.length+this.width);
    }
            
}
