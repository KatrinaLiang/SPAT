class MyThread extends Thread{
    private int num;
    public MyThread(int num) {
		this.num = num;
	}
    
    public void run(){
        int result = 1;
        for(int i=1;i<=num;i++){
            result= result * i;
        }
        try{
            Thread.currentThread().sleep(200);
        }catch(Exception e){}
            System.out.println(result);
    }
}
