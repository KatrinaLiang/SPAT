package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
        int a=0;
		for(int i=1;i<=9;i++){
            for(int j=1;j<=i;j++){
                a=j*i;
                System.out.print(j+"*"+i+"="+a+"\t");
            }
            System.out.print("\n");
        }
		
		
		
		
		/*****end*****/
	}
}
