package step2;

import java.io.File;
import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

public class Task {
	
	public void task() throws IOException{
		/********* Begin *********/
		FileInputStream fs = new FileInputStream("src/step2/input/task.txt"); //创建输入流
        byte[] b = new byte[8];  //定义一个字节数组
        fs.read(b);		//将输入流的数据读入到字节数组
        String str = new String(b);  //将字节转换成字符
        System.out.println(str);   
        File dir = new File("src/step2/output/");  // 创建文件路径
        if(!dir.exists()){
            dir.mkdir();  //创建文件夹
        }
        FileOutputStream fos = new FileOutputStream("src/step2/output/output.txt"); //创建输出流
        String out = "learning practice";
        byte[] outByte = out.getBytes();    //将字符串转换成字节
        fos.write(outByte);                    //写数据
   
        fs.close();
        fos.close();  //释放资源
        
		
		/********* End *********/
	}
	
}
