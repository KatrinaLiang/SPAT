public class Box {

    double width;
    double height;
    double depth;

    int boxType;
    public Box(){};
	public Box(int boxType) {
    	this.boxType=boxType;
    }
    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public double volume() {
        //TODO your code
        double vol=0.0;
        switch(boxType){
            case 1:
                vol=width*depth*height;
                break;
            case 2:
                vol=3.14*width*width*height;
            	break;
        }
        return vol;
    }

}
