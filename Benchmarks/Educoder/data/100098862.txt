package step1;

public class Yard{
    int length;
    int width;
    public Yard(int length,int width){
        this.length=length;
        this.width=width;
    }
   double getPerimeter(){
        int a=length;
        int b=width;
        double c=(double)a+a+b+b; 
       return c;
      }
    double getAcreage(){   
        int a=length;
        int b=width;
        double s=(double)a*b;
        return s;
      }
}
