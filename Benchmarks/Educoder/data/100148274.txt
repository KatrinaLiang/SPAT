package step1;

public class Task {
	public static void main(String[] args) throws Exception {
		/********* Begin *********/
		//在这里创建线程， 开启线程
		MyThread t1 = new MyThread("AA");
		MyThread t2 = new MyThread("BB");
		MyThread t3 = new MyThread("CC");
		t1.start();
		t2.start();
		t3.start();

		/********* End *********/
	}
}

class MyThread extends Thread {
	/********* Begin *********/
	
	private static String str = "CC";
	String threadName;
	public MyThread(String threadName){
		this.threadName = threadName;
		
	}

	public void run() {
		int count = 5;
			while(count > 0){
				synchronized (Task.class) {
					if(str == "CC"&&threadName == "AA"){
						System.out.println("Java Thread" + threadName);
						count--;
						str =threadName;
					}else if(str == "AA"&&threadName == "BB"){
						System.out.println("Java Thread" + threadName);
						count--;
						str =threadName;
					}else if(str == "BB"&&threadName == "CC"){
						System.out.println("Java Thread" + threadName);
						count--;
						str =threadName;
					}
				}
            }
    }

	/********* End *********/
}
