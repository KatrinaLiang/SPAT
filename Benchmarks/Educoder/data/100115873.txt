public class HelloWorld {
    static final int MAX_INDEX=100;
	public static void main(String[] args) {
		int n=MAX_INDEX;
        int sum=0;
        while(n>0){
        sum+=n;
            n--;
        
        
        }
		System.out.println("1到100相加的结果为"+sum);
		
		
		
		
	}
}
