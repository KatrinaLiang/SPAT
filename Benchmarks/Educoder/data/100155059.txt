package task1;

public class Box {
      double width;
      double height;
      double depth;

      public static void main(String[] args){
    	  Box box=new Box();
    	  box.width=10;
    	  box.height=20;
    	  box.depth=10;
    	  box.volume();
      }
      public void volume(){
    	  double v=width*height*depth;
    	  System.out.println("the volumn is:"+v);
      } 
}
