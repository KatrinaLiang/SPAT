        double a=sc.nextInt();
        double b;
        System.out.println("请输入顾客所付钱数：");
        double c=sc.nextInt();
        if(a>20){
         	b=0.85*a;   
        }else if(a>15&&a<=20){
         	b=0.9*a;   
        }else if(a>10&&a<=15){
