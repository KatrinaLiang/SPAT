package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
	int num = 0;
		int sum = 0;
		for (int j = 0; j < n; j++) {
			num = 0;
			for (int i = 1; i <= Math.pow(10, j); i *= 10) {
				num += a * i;
				
			}
			sum += num;
		}
		System.out.print(sum);


		/********** End *********/
	}
}
