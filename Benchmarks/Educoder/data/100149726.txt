package step1;

import java.sql.*;

public class QueryPass {
	
	// 查询数据代码不用上实验报告
	public static void queryDB() {
		
		/********* Begin *********/

		// 第一步：加载驱动
		
		

		// 第二步：建立连接, "root"和"123123"是针对MySQL设置了用户名(root)和密码(123123)的情况
		// 127.0.0.1:3306是mysql服务器地址及端口   数据库编码格式设置为utf-8
			
		

		// 第三步：建立statement对象
		
		

		// 第四步：查询数据
		ResultSet res = null;
		Statement sta = null;
        Connection con = null;
        

		// 第五步：关闭statement对象和连接对象
		try{
            Class.forName("com.mysql.jdbc.Driver");
            con = DriverManager.getConnection("jdbc:mysql://127.0.0.1:3306/tsgc?useUnicode=true&characterEncoding=utf-8", "root", "123123");
			sta = con.createStatement();
            res = sta.executeQuery("select * from employee");
            while(res.next()){
                 System.out.println("no:" + res.getString("no") + "\t" + "name:" + res.getString("name") + "\t" + "password:" + res.getString("password") + "\t" + "sex:"+ res.getString("sex") + "\t" + "salary" + res.getFloat("salary"));
            }
        }catch(ClassNotFoundException e){
            e.printStackTrace();
        }catch(SQLException e){
            e.printStackTrace();
        }finally{
            try{
                res.close();
            	sta.close();
            	con.close();
            }catch(SQLException e){
                e.printStackTrace();
            }
        }
    
		/********* End *********/
	}
}
