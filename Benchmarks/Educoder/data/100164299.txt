package step3;

import java.util.Scanner;
public class Number_Judge {
		public static void main(String[] args) {
			Scanner scan1=new Scanner(System.in);
			/******Begin******/
			int aim=scan1.nextInt();
            int b=0;
            if(aim%2==0)
                System.out.print("偶数");
            else{
            	for(int i=2;i<aim;i++){
                	if(aim%i==0)
                        b=1;
                }
                if(b==1)
                    System.out.print("奇数");
                else
                	System.out.print("素数");
                
            }


	         
			/******End******/
		}
}
