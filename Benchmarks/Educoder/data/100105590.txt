package step1;

public class Yard {
	public Yard(){
		
	}

	
	public Yard(int length,int width) {
		this.length=length;
	        this.width=width;   
	}
	private int length;
	private int width;
		      
	 
public float getPerimeter(){
	        
	       int perimeter=2*length+2*width;
	        return perimeter;
	    }
float getAcreage(){
	        float acreage=length*width;
	        return acreage;	  
	        }

}
