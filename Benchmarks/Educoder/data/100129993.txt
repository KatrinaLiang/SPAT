package step4;
		//创建类 添加属性和方法
public class WuMingFen
{
 	String theMa;
    int quanlity;
    boolean likeSoup;
    public WuMingFen()
    {
        this("酸辣",2,true);
    }
    public WuMingFen(String theMa,int quanlity)
    {
        this.theMa=theMa;
        this.quanlity=quanlity;
        likeSoup=false;
    }
    public WuMingFen(String theMa,int quanlity,boolean likeSoup)
    {
        this.theMa=theMa;
        this.quanlity=quanlity;
        this.likeSoup=likeSoup;
    }
    public void check()
    {
       System.out.println("面码："+theMa+"，粉的份量："+quanlity+"两，是否带汤："+likeSoup); 
    }
}
