package step3;

import java.util.Arrays;
import java.util.Scanner;

public class RecordRecovery {
    public static void main(String[] args) {
		Scanner scanner = new Scanner(System.in);
		int[] record = new int[10];
		int num = 0;
		int numint ;
		for (int i = 0; i < 10; i++) {
			numint = scanner.nextInt();
			boolean flag = true;
			for(int j = 0 ;j<num;j++) {
				if(record[j] == numint) {
					flag = false;
				}
			}
			if(flag)record[num++] = numint;
			flag = true;
		}
		Arrays.sort(record, 0, num);
		for (int i = 0; i < num; i++) {
			System.out.print(record[i]);
			if (i != num - 1) {
				System.out.print(' ');
			}
		}
	}
}
