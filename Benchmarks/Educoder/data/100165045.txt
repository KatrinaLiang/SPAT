package task5;

public class Box {

    double width;
    double height;
    double depth;

    //TODO your code
    public Box(double w,double h,double d)
    {
        width=w;
        height=h;
        depth=d;
    }
    
    public double volume(){
    	double vol=0;
        vol=width*depth*height;
        return vol;
    }



}
