package step1;
import java.util.Scanner;
public class LetterChange {

    // 使用 switch把小写字母转为大写字母
    public char LetterChange(char ch){
        char a=(char)(ch-32);
        if('a'<=ch&&ch<='e')
        return a ;
        else
            return ch;
        // 请在此提供实现代码
    }
    public  void main(String[]args)
    {
        Scanner sc=new Scanner(System.in);
        String s=sc.next();
        char ch=s.charAt(0);
        LetterChange(ch);
    }
}
