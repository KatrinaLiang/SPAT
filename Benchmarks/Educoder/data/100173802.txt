package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
		for ( int i =1 ; i<=10;i++){
			for (int n=1; n<=i;n++){
				System.out.print("*");
			}
			System.out.println("");
		}
		//在这里打印出正三角形 
		
		
		
		
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
		for ( int a =10 ; a>=1;a--){
			for (int b=10; b>(10-a);b--){
				System.out.print("*");
			}
			System.out.println("");
		}
		
		
		
		/*****end*****/
	}
}
