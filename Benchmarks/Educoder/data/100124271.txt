package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
        int i;
        int sum = 2;
       int[] l=new int[20];
       l[0]=1;       
        l[1]=1;
        for(i=2;i<l.length;i++)
        {
            l[i]=l[i-1]+l[i-2];
            sum = sum + l[i];
        }
        System.out.print("前20个元素的和为" + sum);
 
	/********end********/
	}
}
