package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temps){
		/********** BEGIN **********/
		int a=temps.length;
        int []all=new int[a];
        for(int i=0;i<temps.length-1;i++){
        	for(int j=i+1;j<temps.length;j++){
            	if(temps[i]<temps[j]){
                	all[i]=j-i;
                    break;
                }else{
                all[i]=0;
                }
            }
        }
		
     
		return all;
		/********** END **********/
	}
}
