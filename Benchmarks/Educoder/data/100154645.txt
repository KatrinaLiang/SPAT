package step3;

import java.util.Scanner;

public class TestJava20_6 {
	static final Scanner sc=new Scanner(System.in);
	public void print() {
		int m;
		System.out.println("请输入菱形上半部的行数：");
		m=sc.nextInt();   //m表示行数
		/********* Start ********/
        int i,j;
//输出上半部
		for(i=1;i<=m;i++){
			for(j=1;j<=m-i;j++){
				System.out.print(" ");
			}
			for(j=1;j<=2*i-1;j++){
				System.out.print("*");
			}
			System.out.println("");
		}
		for(i=m-1;i>=1;i--){
			for(j=1;j<=m-i;j++){
				System.out.print(" ");
			}
			for(j=1;j<=2*i-1;j++){
				System.out.print("*");
			}
			System.out.println("");
		}
        
        
		
		
		/********* End ********/
	}

}
