//在这里定义Dog类
class Dog{
    String name,color,variety;
    public void set(String name,String color,String variety){
        this.name=name;this.color=color;this.variety=variety;
    }
    public void eat(){
        System.out.println("啃骨头");
    }
    public void run(){
        System.out.println("叼着骨头跑");
    }
    public void out(){
        System.out.println("名字："+name+"，毛色："+color+"，品种："+variety);
    }
}
