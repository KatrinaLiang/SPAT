package step1;
import java.util.Scanner;

public class TestJava19_2 {
	static final Scanner sc=new Scanner(System.in);
	public void print(){
		int m,n;
		System.out.println("请输入行数：");
		m=sc.nextInt();
		System.out.println("请输入列数：");
		n=sc.nextInt();
		/*********Start********/
		for (int i = 1; i <= m; i++) // 外循环变量i控制行数
		{
			for (int j = 1; j <= n; j++) // 内循环变量j控制列数
			{
				System.out.print("*"); // 每列输出一个*号
			}
			System.out.println(""); // 每行*号输出完成后换行
		}
		
		
		
		/*********End********/
	}

}
