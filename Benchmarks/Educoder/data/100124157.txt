package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] a = new int[20];
        a[0] = 1;
        a[1] = 1;
        for (int i = 2; i < 20; ++i){
        	a[i] = a[i - 1] + a[i -2];
        }
		int all = 0;
        for (int i = 0; i < 20; ++i){
        	all += a[i];
        }
        System.out.print("前20个元素的和为" + all);
	/********end********/
	}
}
