package step3;

public class ForPractice3 {
	public static void main(String[] args) {
				/*****start*****/
				int layer =9;
		        		for(int n=1; n<=layer;n++){
		        			for(int j=1;j<=n;j++){
		        			System.out.print(j+"*"+n+"="+j*n+"\t");
		        			}
		        			for(int j=1;j<=(layer-n);j++){
		        				System.out.print("");
		        			}
		        			System.out.println();
		        		}
	}
}
