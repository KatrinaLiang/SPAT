#include<algorithm>
#include<stdio.h>
#include<iostream>
#include<string.h>
using namespace std;
main()
{
    int n,m,c=0,z,x;
    char a[100],b[100];
    cin>>a;
    n=strlen(a);
    cin>>b;
    m=strlen(b);
    //cout<<a<<' '<<b<<endl;

    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++)
    {
        if(a[i]==b[j])
        {
            for(z=i,x=j;a[z]==b[x];z++,x++);
            c=max(c,z-i);
        }
    }
    printf("%d",c);
}
