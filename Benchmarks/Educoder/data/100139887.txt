package step2;

import java.util.Scanner;

public class NameSearch {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        /********** Begin *********/
        String data=new String();
        String name=new String();
        data=scanner.nextLine();
        name=scanner.nextLine();
        int oldPostion=-1;
        int firstPostion=data.indexOf(name);
        while(firstPostion>-1){
            oldPostion+=firstPostion+1;
            System.out.println(oldPostion);
            if(firstPostion+name.length()>data.length())
                break;
            data=data.substring(firstPostion+name.length()-1);  
            firstPostion=data.indexOf(name);}  
    
        /********** End *********/
    }
}
