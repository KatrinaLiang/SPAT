    public static StringBuffer start(String date) {
        
        StringBuffer ans = new StringBuffer();
        /********** Begin *********/
        String[] an=date.split(" ");
        //StringBufffer a=new
        for(int i=0;i<an.length; i++){
        	StringBuffer en=new StringBuffer(an[i]);
            en.reverse();
            ans.append(en);
            ans.append(" ");
            
        }
        
        //ans.reverse();
        /********** End *********/
        return ans;
    }
