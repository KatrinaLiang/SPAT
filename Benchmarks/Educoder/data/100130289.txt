public class Task {

	public void sortArray(int[] arr, String sort){
		/********** BEGIN **********/
        if(sort.equals("asc")){
			Arrays.sort(arr);
            for(int i=0;i<arr.length;i++)
                System.out.println(arr[i]);
        }
           if(sort.equals("desc")){
			Arrays.sort(arr);
            for(int j=arr.length-1;j>=0;j--)
                System.out.println(arr[j]);

        }         	
		/********** END **********/
	}
}
