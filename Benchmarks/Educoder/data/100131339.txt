package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum = 0;
		for(int i = num1; i <= num2; i++) {
            int n = i;
            while(n > 0) {
                sum += (n % 10);
                n /= 10;
            }
        }
		
		return sum;
		/********** END **********/
	}
}
