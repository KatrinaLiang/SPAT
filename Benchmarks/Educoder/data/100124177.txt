package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int[] a=new int[20];
	int sum=0;
	for(int i=0;i<20;i++)
	{
		int s=1;
		if(i<2)
		{
			a[i]=s;
		}
		else
		{
			a[i]=a[i-1]+a[i-2];
		}
		sum+=a[i];
	}
	System.out.print( "前20个元素的和为"+sum);
	/********end********/
	}
}
