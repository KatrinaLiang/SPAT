package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		int[][] j=new int[matrix.length][matrix[0].length];
        for(int k=0;k<matrix.length;k++){
        	for(int l=0;l<matrix[0].length;l++){
            	j[l][matrix.length-1-k]=matrix[k][l];
            }
        }
		for(int k=0;k<matrix.length;k++){
        	for(int l=0;l<matrix[0].length;l++){
            	matrix[l][k]=j[l][k];
            }
        }

		
		
		
		/********* End *********/
	}
}
