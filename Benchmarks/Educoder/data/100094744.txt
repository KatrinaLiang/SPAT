package step2;
import java.util.Scanner;
public class AdditionCalculator {
	public static void addition(int a, int n) {
		/********** Begin *********/
		int sum = 0, rule = 0;
		
		for(int i = 1 ; i <= n ; i ++) {
			
			rule = rule * 10 + a;
			sum += rule;
	
		}
		
		System.out.print(sum);
		/********** End *********/
	}
	
	public static void main(String[] args) {
		
		Scanner sc = new Scanner(System.in);
		int a = sc.nextInt();
		int n = sc.nextInt();
		addition(a, n);
	}
}
