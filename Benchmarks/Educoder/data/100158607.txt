#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
    char str1[100];
    char str2[100];
	
	
	int sum =0;
	scanf("%s",&str1);
	scanf("%s",&str2);
	int n1=0;
	int n2=0;
	do{
		n1++;
	}while(str1[n1] != '\0');
	do{
		n2++;
	}while(str2[n2] != '\0');
	
	
	for(int i=0;i<n1;i++){
		for(int j=0;j<n2;j++){
			if(str1[i]==str2[j]){
				int l=j;
				int m=1;
				for(int k = i+1;k<n1;k++){
					
					l++;
                    if(str1[k]==str2[l]){
							m++;
							
						}
					else{	
					    break;
					}
					
					}	
				if(m>sum){
				     sum=m;
					 }
				
			}
		}
	}
	printf("%d",sum);
	
}
