package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
	InterRectangle(double w,double l){
		this.width=w;
		this.length=l;
	}

	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
		InterRectangle [] rs=new InterRectangle[10];
		for(int i=0;i<total;i++){
			double w=scanner.nextDouble();
			double l=scanner.nextDouble();
			rs[i]=new InterRectangle(w,l);
		}
		System.out.println("总共有"+total+"块土地");
		double s=0;
		double d=0;
		for(int i=0;i<total;i++){
			s=s+rs[i].length*rs[i].width;
			d=d+rs[i].length*2+rs[i].width*2;
		}
		System.out.println("土地总面积为:"+s+"平米");
		System.out.println("土地总周长为:"+d+"米");
		scanner.close();

		/*********end*********/
	}

}
