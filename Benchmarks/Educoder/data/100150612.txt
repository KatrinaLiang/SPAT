package step3;
import java.util.*;
public class SumOfDigit {
	public static void main(String[] args) {
		int number,sum=0,tmp;
        Scanner scan1=new Scanner(System.in);
        System.out.println("请输入一个五位数字：");
        number=scan1.nextInt(); 
        tmp=number;
        for (int i=0;i<5;i++){
	          	sum=sum+number%10;
	        	number=number/10;}
        System.out.println(tmp+"的各位数字之和是："+sum);
        /******End******/
	}
}
