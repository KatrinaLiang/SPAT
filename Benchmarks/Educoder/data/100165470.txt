package step2;
import step1.Person;
import step1.Date;
public class Student extends Person{
    private int reg_num; //学号
    private Date reg_day; //注册日期
   public Student(String name,String number,Date d1,int a,Date d2){
   		super(name,number,d1);
       	reg_num=a;
       reg_day=d2;
   }
	public Student(String name,String number,int year1,int month1,int day1,int a,int year2,int month2,int day2){
    	super(name,number,year1,month1,day1);
        reg_num=a;
        reg_day=new Date(year2,month2,day2);
    }
	public Student(){
    	super();
        reg_num=0;
        reg_day=new Date(1999,9,9);
    }
	public String toString(){
    	return "name="+super.getname()+" number="+super.getnum()+" birthday:"+super.getBirthday().toString()+"reg_num:"+
            reg_num+"reg_day:"+reg_day.toString();
    }
	public Date getreg_day(){
    	return reg_day;
    }
    public int getreg_Num(){
    	return reg_num;
    }

}
