package step3;

import java.util.Arrays;

public class Task {

	public void matrixSum(int[][] arr){
		/********** BEGIN **********/
		int ans = 0;
		for (int i = 0; i < arr.length; i++) {
			for (int j = 0; j < arr[0].length; j++) {
				if(i==j) {
					ans += arr[i][j];
				}
			}
		}
		System.out.println(ans);
		



		/********** END **********/
	}
	
	
}
