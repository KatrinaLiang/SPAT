package step1;

public class Yard {
    
    int len,wid;
    public Yard(int length,int width){
        len=length;
        wid=width;
    
       
    }
    
    
   
    public double getPerimeter()
    {
        int a;
        a=len*2+wid*2;
        return a;
    }
    public double getAcreage()
    {
        int b;
        b=len*wid;
        return b;
    }
        

}
