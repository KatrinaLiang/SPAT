package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int reng = 0;
        while(true){
        	if(n%2==0){
            	n/=2;
            }else{
            	n=(n-1)/2;
                reng+=1;
            }
        	if(n==0)break;
        }
		
		return reng;
		
		
		/********* End *********/
	}
	
}
