#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
char s1[1010],s2[1010];
int main()
{
    scanf("%s",s1);
    scanf("%s",s2);
    int l1=strlen(s1);
    int l2=strlen(s2);
    int ans=0;
    for(int i=0;i<l1;i++)
    {
        for(int j=0;j<l2;j++)
        {
            int a=1;
            if(s1[i]==s2[j])
            {
                for(int k1=i+1,k2=j+1;k1<l1&&k2<l2;k1++,k2++)
                {
                    if(s1[k1]!=s2[k2]) break;
                    a++;
                }
                ans=max(ans,a);
            }
        }
    }
    printf("%d",ans);
}
