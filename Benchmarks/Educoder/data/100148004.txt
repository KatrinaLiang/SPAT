public class MyThread extends Thread
{
	int number;
    MyThread(int number)
    {
    	this.number = number;
    }
	public void run()
    {	
    	int i,result = 1;
        for(i=2;i<=number;i++)
        	result *= i;
        try
            {
            	Thread.sleep(200);
             }catch(InterruptedException is){}
        System.out.println(result);
     }
}
