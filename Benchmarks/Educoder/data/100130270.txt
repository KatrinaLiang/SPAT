class WuMingFen{
    public String theMa;
    public int quantity;
    public boolean likeSoup = false;
    public WuMingFen(String mianma,int quantity,boolean likesuop){
        theMa = mianma;
        this.quantity = quantity;
        likeSoup = likesuop;
    }
    
public WuMingFen(String mianma,int quantity){
        theMa = mianma;
        this.quantity = quantity;
    }
public WuMingFen(){
        theMa = "酸辣";
        this.quantity = 2;
        likeSoup = true;
    }
    public void check(){
        System.out.println("面码："+theMa+"，粉的份量："+quantity+"两，是否带汤："+likeSoup);
    }
    
}
