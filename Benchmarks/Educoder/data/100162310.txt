/**************** begin *****************/
// 在此定义学生类名
class Student
/**************** end *****************/
{
	/**************** begin *****************/
	 // 将姓名name和年龄age属性定义为私有的
private String name;
private int age;
	/**************** end *****************/


	/**************** begin *****************/
	public Student()
	{
	}
	public Student (String a)
	{
		name = a;
	}
	public Student (String x, int y)
	{
		name = x;
		age = y;
	}
	/**************** end *****************/
	
	void speak() {
		System.out.println("我的名字是" + name + ",今年" + age + "岁");
	}
}

public class Task04{
	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);

		/**************** begin *****************/
		Student S1 =new Student();
		S1.speak();

		/**************** end *****************/
		

		/**************** begin *****************/
	

		/**************** end *****************/


		/**************** begin *****************/
	String a = scan.next();
	Student S2 =new Student (a);
	S2.speak();
String b = scan.next();
int c = scan.nextInt();
	Student S3 =new Student (b,c);
	S3.speak();
		/**************** end *****************/
	}
}
