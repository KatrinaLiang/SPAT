package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int i,a,b,c,temp=0,sum=0;
		for(i=num1;i<=num2;i++) {
			if(i>=100) {
				a=i/100;
				b=(i-a*100)/10;
				c=(i-a*100)%10;
			    temp=a+b+c;
			}else if(i>=10&&i<100) {
				a=i/10;
				b=i%10;
				temp=a+b;
			}else if(i<10) {
				temp=i;
			}
			sum=sum+temp;
		}
		return sum;
	
		/********** END **********/
	}
}
