package step3;
public class Task {

	 public int countOfShips(char[][] ferry) {
	  /**************BEGIN************/
    	 int row = ferry.length;
         if (row == 0){
         return 0;
         }
      	 int line = ferry[0].length;
         boolean ifrow = false;
         for (int i = 0; i < row; i++){
         	for (int j = 0; j < line-1; j++){
            	if (ferry[i][j] == '+' && ferry[i][j+1] == '+'){
                	ifrow = true;
                    i = row-1;
                    break;
                }
            }
         }
         int num = 0;
         if (ifrow){
         	for (int i = 0; i < row; i++){
                if (ferry[i][0] == '+'){
                	num++;
                }
            	for (int j = 1; j < line; j++){
                	if (ferry[i][j] == '+' && ferry[i][j-1] != '+'){
                    	num++;
                    }
                }
            }
         } else{
         	for (int i = 0; i < line; i++){
            	if (ferry[0][i] == '+'){
                	num++;
                }
                for (int j = 1; j < row; j++){
                    if (ferry[j][i] == '+' && ferry[j-1][i] != '+'){
                    	num++;
                    }
                }
            }
         }
         return num;
      /***********END ***************/
	 }
}
