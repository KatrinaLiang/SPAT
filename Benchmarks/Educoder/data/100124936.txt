package step1;

public class Rectangle{
	private double width;
	private double length;
	
	/*********begin*********/
	public Rectangle(){
	this.length = 6;
	this.width = 5;
}
public Rectangle(double a,double b){
	this.length = a;
	this.width = b;
}
public double  perimeter(){
	double c;
	c = (this.length + this.width)*2;
	return c;
}
public double  area(){
	double c;
	c = this.length * this.width;
	return c;
}
	/*********end*********/


	public static void main(String[] args) {
		/*********begin*********/
		Rectangle anlie1 = new Rectangle();
		Rectangle anlie2 = new Rectangle(20,10);
		System.out.println("r1的周长："+anlie1.perimeter());
		System.out.println("r1的面积："+anlie1.area());
		System.out.println("r2的周长："+anlie2.perimeter());
		System.out.println("r2的面积："+anlie2.area());

		/*********end*********/
	}

}
