package task4;

public class Box {

    double width;
    double height;
    double depth;

    int boxType;
    
    public Box(int boxType) {
        this.boxType = boxType;// your code
    }

    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public double volume() {
        double volume=0;
        switch (boxType) {
            case 1:
                volume = width * depth * height;
                break;
            case 2:
                volume = 3.14 * width * width * height;
                break;
        }
        return volume;
        //TODO your code
    }

}
