package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temps){
		/********** BEGIN **********/
		int len=temps.length;
		int[] res=new int[len];		
		for(int i=0;i<len-1;i++) {
			int j=i+1;
			while(temps[j]<=temps[i]) {
				j++;
				if(j==len) break;
			}
			if(j==len) res[i]=0;
			else res[i]=j-i;
		}
		res[len-1]=0;
		return res;
		/********** END **********/
	}
}
