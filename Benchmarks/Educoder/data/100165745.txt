package step3;
import step1.Date;
public class Course {
    private int code; //课程编号
    private String courseName; //课程名称
    private Date testTime;//考试日期
    private int score;//考试成绩
   /******Begin**************/
	public Course(){
    	testTime=new Date();   
    }
    public Course(int code,String courseName,int year,int month,int day,int score){
    	this.courseName=courseName;
        this.score=score;
        this.code=code;
        testTime=new Date(year,month,day);
    }
    public Course(int code,String courseName,Date testTime,int score){
    	this.courseName=courseName;
        this.score=score;
        this.code=code;
        this.testTime=testTime;
    }
    public void setCourse(int code,String courseName,int year,int month,int day,int score)
    {
    	this.courseName=courseName;
        this.score=score;
        this.code=code;
        testTime.setDate(year,month,day);
    }
    public int getcode()
    {
    	return code;
    }	    
    public String getCoureseName()
    {
    	return courseName;
    }
    public int getScore()
    {
    	return score;
    }	    
    public Date gettestTime()
    {
    	return testTime;
    }
	public String toString()
    {
        if(courseName!=null){
        	return code+" "+courseName+" "+testTime+" "+score;
        }else{
        	return code+"  "+testTime+" "+score;
        }
    }
   /******End************/

}
