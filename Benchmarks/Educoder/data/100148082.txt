package task1;

public class Task1 {

	public static void main(String[] args) {
		/********* Begin *********/
	  for (int i = 101; i < 1000; i = i + 2) {
            // isPrime(i)判断i是否为素数
            // isPrime(i+2)判断与i相邻的奇数是否为素数
            if (isPrime(i) && isPrime(i + 2)) {
                System.out.println((i) + "和" + (i + 2) + "是姐妹素数");
            }
        }
		
		/********* End *********/
	}

	// 判断一个数是否为素数的方法：返回true表示是素数，返回false表示不是素数
	/********* Begin *********/
    private static boolean isPrime(int n) {
        boolean flag = true;
        for (int i = 2; i < n; i++) {
            if (n % i == 0) {
                flag = false;
                break;
            }
        }
        return flag;
    }
	/********* End *********/
}
