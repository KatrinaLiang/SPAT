package step2;
import java.util.Scanner;
public class AreaOfCircle {
	public static void main(String[] args) {
		float s,r;
        final float PI=3.14f;
        Scanner sc=new Scanner(System.in);
        System.out.println("请输入半径r的值：");
        r=sc.nextFloat();
        s=PI*r*r;
        System.out.println("圆的面积是："+s+"0");
	}
}
