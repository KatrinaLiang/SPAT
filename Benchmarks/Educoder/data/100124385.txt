package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int fib[]=new int[20];
        fib[0]=1;
        fib[1]=1;
        int i;
        int sum=0;
        for(i=2;i<fib.length;i++){
            fib[i]=fib[i-1]+fib[i-2];
        }
        for(i=0;i<fib.length;i++){
            sum=sum+fib[i];
        }
        System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
