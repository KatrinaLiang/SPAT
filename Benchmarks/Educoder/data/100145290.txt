package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int cont = 0;
		x: while (n != 0) {
			if (n % 2 == 0) {
				n = n / 2;
				continue x;
			} else {
				n = (n - 1) / 2;
				cont += 1;
				continue x;
			}
		}
		return cont;
		
		
		
		
		/********* End *********/
	}
	
}
