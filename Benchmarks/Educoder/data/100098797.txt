package step1;

public class Yard {
	
   	public Yard(int length, int width){
	  		  this.width = width;
	          this.length = length;
	    }
	    
	    public double getPerimeter(){
	    	double p = 2 * (this.width + this.length);
	        return p;
	    }
	    
	    public double getAcreage(){
	    	double a = this.width * this.length;
	    	return a;
	    }
	    
	    private int width;
	    private int length;
    
}
