package step3;

public class HelloWorld {
	public static void main(String[] args) {
		int count= 0;	//定义变量存储6的倍数出现的次数
		/*****start*****/
        int sun=1;
        do {
		     if  (sun%6==0) count++;
		     sun++;
        }while(sun<=100);
            sun++;
		/*****end*****/
		System.out.println("6的倍数出现的次数为：" + count);
	}
}
