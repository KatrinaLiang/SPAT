package step2;
	import java.util.Scanner;
public class AreaOfCircle {
	public static void main(String[] args) {
		/******Begin******/
		Scanner sc=new Scanner(System.in);
        final float PI=3.14f;
        float r;
        double area;
        System.out.println("请输入半径r的值：");
        r=sc.nextFloat();
        area=PI*r*r;
        System.out.print("圆的面积是："+area+"0"+"\n");    
		/******End******/
	}
}
