

package step4;

public class MyThread extends Thread{
	int a,result=1;
	public MyThread(int a) {
		this.a = a;
	}
	public void run() {
		try {
			sleep(3000);
		} catch (Exception e) {
			// TODO: handle exception
		}
		for (int i = 1; i <= a; i++) {
			result*=i;
		}
		System.out.println(result);
		
	}
	
}
