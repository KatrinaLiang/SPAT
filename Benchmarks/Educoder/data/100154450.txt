package step5;
import java.util.Scanner;
public class TestJava18_5 {
	static final Scanner sc=new Scanner(System.in);
	public void jisuan(){
		int x=1,y ,z=0;
		for(;true;x++){
			System.out.println("请输入第"+x+"门课程成绩：");
			y=sc.nextInt();
			if(y<1)	{			
				break;			
			}
			z=z+y;		  			
		}
		System.out.println("您一共输入了"+(x-(-y))+"门课程成绩，总分为："+z+".00，平均分为："+(z+1)/(x-1)+".00");
	}

}
