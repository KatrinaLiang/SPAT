package step3;

import java.util.Scanner;

public class TestJava20_6 {
	static final Scanner sc=new Scanner(System.in);
	public void print() {
		int m;
		System.out.println("请输入菱形上半部的行数：");
		m=sc.nextInt();   //m表示行数
		/********* Start ********/
for(int a=1;a<=m;a++){
        	for(int b=1;b<=5-a;b++){
        		System.out.print(" ");
            }
            for(int c=1;c<=2*a-1;c++){
                System.out.print("*");
            }System.out.println("");
        }
        for(int p=4;p>=1;p--){
            for(int d=1;d<=5-p;d++){
                System.out.print(" ");
            }
            for(int f=1;f<=2*p-1;f++){
                System.out.print("*");
            }System.out.println("");
        }
        
        
		
		
		/********* End ********/
	}

}
