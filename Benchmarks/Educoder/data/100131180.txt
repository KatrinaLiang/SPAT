package step1;
import java.util.Scanner;
public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum=0;
		for(int i=num1;i<=num2;i++){
        	int a=i/100;
            int b=(i-100*a)/10;
            int c=i-100*a-10*b;
            sum=sum+a+b+c;
        }
		return sum;
       
		/********** END **********/
	}
}
