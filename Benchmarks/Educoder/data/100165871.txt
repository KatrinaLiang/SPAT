package step4;
import step1.Date;
import step2.Student;
import step3.Course;

public class Report {
    private Student s;//学生基本信息
    private String major; //所学专业  
    private Course[] selCourse;//所选课程
   /************Begin****************/
    public Report(){
		s=new Student();
		selCourse=new Course[10];
	}
	public Report(Student s,String major,Course[] selCourse){
		this.s=s;
		this.major=major;
		this.selCourse=selCourse;
	}
	public void setReport(Student s,String major,Course[] selCourse){
		this.s=s;
		this.major=major;
		this.selCourse=selCourse;
	}
	public Student getStudent(){
		return this.s;
	}
	public String getmajor(){
		return this.major;
	}
	public String getCourse(){
		String courseInfo="";
		for(int i=0;i<selCourse.length;i++){
			courseInfo += selCourse[i]+"\n";
		}
		return courseInfo;
	}

	public String toString(){
		return s.toString()+" major:"+this.major+"\n"+this.getCourse();
	}




    /************End****************/

  
}
