package step3;
public class NarcissisticNumber {

    //判断一个三位数是否是水仙花数
    public boolean NarcissisticNumber(int n){
        boolean sum;
        int a,b,c;
        a=n%10;
        b=(n%100)/10;
        c=n/100;
        int add;
        add=a*a*a+b*b*b+c*c*c;
        if (add==n){
           sum=true;
        }
        else{
           sum=false;
        }
        return sum;
    }
}
