	public static void main(String[] args) {

        Scanner in=new Scanner(System.in);
		int m=in.nextInt();
        System.out.print(mark(m));
	
    }
   static int mark(int x){
       		if(x==2) return 1;
      		else if(x==3) return 2;
       		else if(x%3==1){
                return (int)Math.pow(3,x/3-1)*4;
            }else if(x%3==2){
                return (int)Math.pow(3,x/3)*2;
            }else{
                return (int)Math.pow(3,x/3);
            }
   }
