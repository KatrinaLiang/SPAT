package step3;

import java.util.Scanner;
public class Number_Judge {
		public static void main(String[] args) {
			Scanner scan1=new Scanner(System.in);
			/******Begin******/
			int usr = scan1.nextInt();
			if(usr%2==0){
				System.out.print("偶数");
				}
			else if(isPrime(usr)){
				System.out.print("素数");
			}else{
				System.out.print("奇数");
			}

		}
		public static boolean isPrime(int n){
			if(n<=0||n==1||(n%2==0&&n!=2)){
				return false;
			}
			else if(n==2){
				return true;
			}else{
				for(int i = 3;i<=Math.sqrt(n);i++){
					if(n%i==0){
						return false;
					}
				}
			}
			return true;
		}
}
