package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] arr=new int[20];
        arr[0]=1;arr[1]=1;
        int i,sum=2;
        for( i=2;i<arr.length;i++)
        {
            arr[i]=arr[i-1]+arr[i-2];
            sum+=arr[i];
        }
        System.out.print("前"+i+"个元素的和为"+sum);

	/********end********/
	}
}
