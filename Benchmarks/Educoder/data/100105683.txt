package step1;

public class Yard {
	private int length;
    private int width;
    
    public Yard(int l,int w)
    {
    	this.length = l;
        this.width = w;
    }
    
    public float getPerimeter()
    {
    	return 2.0f * (this.length + this.width);
    }
    
    public float getAcreage()
    {
    	return this.length * this.width * 1.0f;
    }
}
