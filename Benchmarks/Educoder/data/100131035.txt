package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int count = 0,a = 0,b = 0,c = 0;
		for(int i = num1; i <= num2; i++){
			a = i / 100;
			b = i / 10 - a * 10;
			c = i % 10;
			count = count + a + b + c;
		}
		return count;
		
		
		/********** END **********/
	}
}
