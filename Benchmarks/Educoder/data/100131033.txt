package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int num,n,d;
		num = 0;
		
		for(int i = num1;i <= num2;i++){
			d = 0;
			String s = ""+i;
			char a[] = s.toCharArray();
			for(int j = 0;j < a.length;j++){
				String st = a[j]+"";
				n = Integer.parseInt(st);
				d = d + n;
			}
			num = num + d;
		}
		return num;
		
		
		/********** END **********/
	}
}
