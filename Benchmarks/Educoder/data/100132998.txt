package step1;

public class Task {
	
	public void sortArray(int[] arr){
		for(int i=1;i<arr.length;i++){
            int temp=arr[i];
            for(int j=i-1;j>=0;j--){
                if(arr[j]>temp){
                    arr[j+1]=arr[j];
                    arr[j]=temp;
                    temp=arr[j];
                }
            }
            for(int j:arr){
                System.out.print(j+" ");
            }
             System.out.println("");
        }
		
		
	}
}
