package step1;

public class Test {
	public static void main(String[] args) {
		//创建Dog对象
		//设置Dog对象的属性
		Dog dog = new Dog();
        dog.setName("五花肉");
        dog.setColor("棕色");
		dog.setVariety("阿拉斯加");
		//输出小狗的属性
		System.out.println("名字：" +dog.name+ "，毛色：" +dog.color+ "，品种：" +dog.variety);
		//调用方法
		dog.eat();
        dog.run();
	}
}

//在这里定义Dog类
class Dog{
	public Dog(){
    }
    void setName(String name){
        this.name = name;
    }
    void setColor(String color){
        this.color = color;
    }
    void setVariety(String variety){
        this.variety = variety;
    }
    void eat(){
        System.out.println("啃骨头");
    }
    void run(){
        System.out.println("叼着骨头跑");
    }
    String name;
    String color;
    String variety;
}
