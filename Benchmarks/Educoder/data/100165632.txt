package step3;
import step1.Date;
public class Course {
    private int code; //课程编号
    private String courseName; //课程名称
    private Date testTime;//考试日期
    private int score;//考试成绩
   /******Begin**************/

	public Course(){
		testTime=new Date();
		code=0;
	}

	public Course(int num,String name,int year,int month,int day,int score){
		Date date=new Date(year,month,day);
		testTime=date;
		code=num;
		courseName=name;
		this.score=score;
	}

	public Course(int num,String name,Date date,int score){
		testTime=date;
		code=num;
		courseName=name;
		this.score=score;
	}

	public void setCourse(){
		
	}

	public int getCode(){
		return code;
	}

	public String getCourseName(){return courseName;}

	public Date getTestTime(){ return testTime;}

	public int getScore(){return score;}

	public String toString(){
		if(this.code==0){
			return code+"  "+getTestTime().getYear()+"."+getTestTime().getMonth()+
			"."+getTestTime().getDay()+" 0";
		}
		return code+" "+getCourseName()+" "+getTestTime().getYear()+"."+getTestTime().getMonth()+
			"."+getTestTime().getDay()+" "+getScore();
	}

   /******End************/

}
