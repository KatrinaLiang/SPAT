package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
        int flg;
		for(int i=1;i<=9;i++)
        {
        for(int j=1;j<=i;j++)
        {
            flg=i*j;
        System.out.print(j+"*"+i+"="+flg+"\t");
            
        }
        System.out.println();    
        }
		
		
		
		
		
		/*****end*****/
	}
}
