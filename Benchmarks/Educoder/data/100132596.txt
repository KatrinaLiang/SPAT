package step2;

import java.util.Scanner;

public class Task {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int length = sc.nextInt();	//长
		int width = sc.nextInt();	//宽
		/********** BEGIN **********/
		
		int s = 0;
		int t = 0;
		int i = 1;
		
		for(i=1;i<=width;i++){
			if(width%i==0){
				s = width/i;
				if(length%s==0){
					t = length/s;
					break;
				}
				else{
					continue;
				}
			}
		}
		System.out.println(s);
		System.out.println(i*t);
		
		
		
		
		
		/********** END **********/
	}
}
