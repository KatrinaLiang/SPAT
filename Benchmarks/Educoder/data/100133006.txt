package step1;
import java.util.*;
public class Task {
	
	public void sortArray(int[] arr){
        for(int i=1;i<arr.length;i++){
        	int[] a=new int[i+1];
            for(int j=0;j<i+1;j++){
            	a[j]=arr[j];
            }
            Arrays.sort(a);
            for(int j=0;j<i+1;j++){
            	System.out.print(a[j] +" ");
            }
            for(int k = (a.length) ;k<arr.length;k++){
            	System.out.print(arr[k]+" " );
            }
            System.out.println();
        }

		
		
	}
}
