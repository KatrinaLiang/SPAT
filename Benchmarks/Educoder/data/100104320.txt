package step3;

import java.util.*;

public class GcdAndGcm {
	public static void main(String[] args) {
		/********** Begin *********/
		//键盘输入数据
		//do—while循环，辗转相除法
		//关掉输入流
        /********** End *********/
        Scanner input = new Scanner(System.in);
        
        int a = input.nextInt();
        int b = input.nextInt();
        
        int i = a;
        int j = b;
        
        if(a < b)
        {
        	int flag = a;
            a = b;
            b = flag;
        }
        
        int yue = 0;
        while(b != 0)
        {
        	yue = a % b;
            a = b;
            b = yue;
        }
        
        System.out.println("最大公约数：" + a);
        System.out.println("最小公倍数：" + (i * j) / a);
	}
}
