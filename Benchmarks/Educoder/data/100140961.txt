package step2;

public class Task {
	public static void main(String[] args) {

		// 定义int类型变量，值为67
		int heipi=67;

		// 创建Integer包装类对象，表示变量score的值
		Integer heipi1 = new Integer(heipi);

		// 将Integer包装类转换为double类型
		double heipi2 =heipi1.doubleValue(); 

		// 将Integer包装类转换为float类型
		float heipi3 =heipi1.floatValue();

		// 将Integer包装类转换为int类型
		int heipi4 =heipi1;

		System.out.println("Integer包装类：" + heipi1);
		System.out.println("double类型：" + heipi2);
		System.out.println("float类型：" + heipi3);
		System.out.println("int类型：" + heipi4);

	}
}
