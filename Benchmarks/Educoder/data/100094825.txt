package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        int[] results = new int[n];
        //if(a<1||n<1) return 0;
        int temp=0;
        for(int i=1;i<=n;i++){
            temp = temp*10+a;

            results[i-1]=temp;
       }
       int sum=0;

       for(int i=0;i<results.length;i++){
           sum+=results[i];
       }
       System.out.print(sum);


		/********** End *********/
	}
}
