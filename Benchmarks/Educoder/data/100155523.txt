package task4;

public class Box {

    double width;
    double height;
    double depth;

    int boxType;
	
    /**
     * 代表长方体
     */
    public Box(int boxType){
        this.boxType=boxType;
        
        
    }
    
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public double volume() {
        //TODO your code
        if(boxType==1){
            return height*depth*width;
        }else{
            return 3.14*depth*height*width;
        }
        
    }

}
