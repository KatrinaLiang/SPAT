package step1;

public class Yard {
              int length;
     int width;
    public Yard(int length,int width){
     
        this.length=length;
        this.width=width;
    }

public  float getPerimeter(){
    int s=(this.length+this.width)*2;
    return s;
}
     public float getAcreage(){
        int m=this.length*this.width;
        return m;
    }
    

}
