//在这里定义Dog类
class 	Dog
{
    Dog(String name,String color,String variety)
    {
        this.name = name;
        this.color = color;
        this.variety = variety;
    }

   public void set_name(String na)
    {
        this.name = na;
    }
    public void set_color(String na)
    {
        this.color = na;
    }
    public void set_variety(String na)
    {
        this.variety = na;
    }
    public String get_name()
    {
        return this.name;
    }
    public  String get_color()
    {
        return this.color;
    }
    public  String get_variety()
    {
        return this.variety;
    }
    public void eat()
    {
        System.out.println("啃骨头");
    }
    public void run()
    {
        System.out.println("叼着骨头跑");
    } 

    private String name;
    private String color;
   private  String variety;

}
