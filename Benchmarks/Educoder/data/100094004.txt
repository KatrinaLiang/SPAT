package chapter1.step3;

import java.util.Scanner;
import java.text.DecimalFormat;
public class FarheheitToCentigrade {

	public static void main(String[] args) {
			DecimalFormat df=new DecimalFormat("#.00");
			Scanner sc=new Scanner(System.in);
			double tempfarenheit = sc.nextDouble();   //华氏度
	        /********** Begin *********/
			//计算摄氏度
       		 double tem=tempfarenheit;
			tempfarenheit=(tempfarenheit-32)*5/9;
	        
	        //利用DecimalFormat保留摄氏度的两位小数并输出
			System.out.print("If the Farenheit is "+tem+",the Centigrade is "+df.format(tempfarenheit));

	        /********** End *********/
		
	}
}
