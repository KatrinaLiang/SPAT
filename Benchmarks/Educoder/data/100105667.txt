package step1;

public class Yard {
    Yard(double length, double width) {    //构造方法
        this.length = length;
        this.width = width;
    }
    public double getPerimeter() {    //返回周长
        return (length + width) * 2;
    }
    public double getAcreage() {    //返回面积
        return length * width;
    }
    private double length;    //成员变量（长度）
    private double width;

}
