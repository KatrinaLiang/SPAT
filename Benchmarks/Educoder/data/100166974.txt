package step1;
public class A< T extends Comparable >{
private T t1,t2;
public A(T t1,T t2){
    this.t1=t1;
    this.t2=t2;
}
public T getT1(){
    return t1;
}
public T getT2(){
    return t2;
}
public void setT1(T t1){
    this.t1=t1;
}
public void setT2(T t2){
    this.t2=t2;
}
public T compare(){
  if(t1.compareTo(t2)>=0)
     return t1;
  else
     return t2;
  }
 public String toString(){
     return " "+t1+ " "+t2;
 }
}
