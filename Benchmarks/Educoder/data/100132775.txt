package step3;

public class Task {

	 public int countOfShips(char[][] ferry) {
	  /**************BEGIN************/
	int sum=0;
		int rowLength=ferry.length;//行长度
         if(rowLength==0){
            return 0;
        }
		int colLength=ferry[0].length;//列长度
		
		for(int i=0;i<rowLength;i++) {
			for(int j=0;j<colLength;j++) {
				if(ferry[i][j]=='+') {
					sum++;
					if(i==rowLength-1&&j<colLength-1) {
						if(ferry[i][j+1]=='+') {
							sum--;
						}
					}else if(i<rowLength-1&&j==colLength-1) {
						if(ferry[i+1][j]=='+') {
							sum--;
						}
					}else if(i==rowLength-1&&j==colLength-1) {
						
					}else {
					if(ferry[i+1][j]=='+'||ferry[i][j+1]=='+') {
						sum--;
					}
				}
			}
			}
		
		
		
		
		}
		
		return sum;
         
      /***********END ***************/
	 }
}
