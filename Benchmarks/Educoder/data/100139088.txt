package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
		MyThread my = new MyThread(num);
		my.start();
		try{
            my.join();
		}catch(Exception e){
            e.printStackTrace();
        }
}
}

class MyThread  extends Thread {
	int num;
    int pro = 1;
    int pri = 1;
    int sum = 0;
    int t;
    MyThread(int num){
        this.num = num;
    }
    public void run(){
    	if(num == 1){
             System.out.println("子线程计算结果为："+pri);
        	}else if(num == 2){
             System.out.println("子线程计算结果为："+pro);
    }else if(num > 2){
        	for(int i = 1;i <= num; i++){
            sum = pro+pri;
            t = pro;
            pro = pri;
            pri = sum;
            }
            System.out.println("子线程计算结果为："+t);
	}
}
}
