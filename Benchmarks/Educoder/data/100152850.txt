package step1;

import java.util.Scanner;
public class TestJava12_2 {

    public static void main(String[] args) {
        /******Begin******/
String days="";
        Scanner sc=new Scanner(System.in);
        int month;
        System.out.println("请输入月份：");
        month=sc.nextInt();
        switch (month){
            case 3:
            case 4:
            case 5:
                days="春季";
                break;
            case 6:
            case 7:
            case 8:
                days="夏季";
                break;
            case 9:
            case 10:
            case 11:
                days="秋季";
                break;
            case 12:
            case 1:
            case 2:
                days="冬季";
        }if (month<=12){
        System.out.println(month+"月份属于"+days);
        }else {
            System.out.println("输入的月份有误！");
        }
        /******End******/
    }

}
