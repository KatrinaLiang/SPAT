package step2;

import java.util.Scanner;

public class SystemUpdate {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String name = scanner.next();
        String birth = scanner.next();
        String position = scanner.next();
        double salary = scanner.nextDouble();
        Salary employee = new Salary(name, birth, position, salary);
        employee.introduction();
    }
}

/********** Begin *********/
class Employee {
    private String name;
    private String birth;
    private String position;
    public void setName(String name){
        this.name = name;
    }
    public void setBirth(String birth){
        this.birth = birth;
    } 
    public void setPosition(String position){
        this.position = position;
    }
    public String getName(){
        return name;
    }
    public String getBirth(){
        return birth;
    }
    public String getPosition(){
        return position;
    }
    public Employee(String name,String birth,String position){
        setName(name);
        setBirth(birth);
        setPosition(position);
    }
}

class Salary extends Employee {
    private double salary;
    public Salary(String name,String birth,String position,double salary){
        super(name,birth,position);
        setSalary(salary);
    }
    public void setSalary(double salary){
        this.salary = salary;
    }
    public double getSalary(){
        return salary;
    }
    public void introduction(){
        System.out.print("员工姓名：" + super.getName() + " 出生年月：" + super.getBirth() + " 职位：" + super.getPosition() +  " 薪水：" + this.getSalary());
    }
}
/********** End *********/
