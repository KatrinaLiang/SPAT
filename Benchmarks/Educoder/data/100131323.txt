package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int result = 0;
		for(int i = num1;i<=num2;i++){
        	result += i%10;
            result += (i/10)%10;
            result += (i/100)%10;
        }
		return result;
		
		/********** END **********/
	}
}
