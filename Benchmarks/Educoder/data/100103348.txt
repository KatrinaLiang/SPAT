package step3;

import java.io.File;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

public class DeleteFiles {
    public static void start() throws IOException {
       
        /********** Begin *********/
            File file1 = new File("src/step3/readmeCopy.txt");   ////file对象指向src文件夹下的step1文件夹下的Hello World.txt文件（此时此文件不存在）
    if (!file1.exists()) {   //如果文件不存在
       file1.createNewFile();   //创建文件
    }
        //读取readme.txt文件
        File file2 = new File("src/step3/readmeCopy.txt");
FileWriter fileWriter = new FileWriter(file2, true);
        //...................//
File file3 = new File("src/step3/readme.txt");
FileReader fileReader = new FileReader(file3);
    int read = fileReader.read();
    while (read != -1) {   //循环读取，直到read的值为-1，表示文件读到末尾
        
        fileWriter.write((char)read);//输出读取的数据，使用(char)方法对read的值进行类型转换
        read = fileReader.read();   //读取下一个字符
    }
            fileReader.close();
            fileWriter.close();
        //写入readmeCopy.txt文件


        //删除readme.txt文件
    File file4 = new File("src/step3/readme.txt");   //file对象指向src文件夹下的step1文件夹下的Hello World.txt文件
    if (file4.exists()) {   //如果文件存在
        file4.delete();   //删除该文件，并输出布尔值
    }

        /********** End *********/
    }
     
}
