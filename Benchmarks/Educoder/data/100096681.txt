package step3;

import java.util.Arrays;
import java.util.Scanner;

public class RecordRecovery {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int[] record = new int[10];
        int[] recordRecovery = new int[10];
        /********** Begin *********/
        for (int i = 0;i < 10;i++)
        {
            int a = scanner.nextInt();
            record[i] = a;
        }
        Arrays.sort(record);
        for (int i = 0;i < 10;i++) //创建recordRecovery数组
        {
            int x = 0;
            int y = record[i];
            for (int j = 0;j < 10;j++)
            {
                if (record[j] == y)
                {
                    x++;
                }
            }
            if (x > 1){
                record[i] = 0;
            }
        }
        int abc = 0;
        for (int i = 0;i<10;i++)
        {
            if(record[i] != 0)
            {
                recordRecovery[abc] = record[i];
                abc++;
            }
        }

    //    Arrays.sort(recordRecovery);
        int num = abc;
        /********** End *********/
        for (int i = 0; i < num; i++) {
            System.out.print(recordRecovery[i]);
            if (i != num - 1) {
                System.out.print(' ');
            }
        }
    }
}
