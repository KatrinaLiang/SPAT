package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum = 0;
        for(int i = num1;i <= num2;i++){
            int index = i;
            while(index != 0){
                sum += index%10;
//                System.out.println("sum:" + sum);
                index = index/10;
//                System.out.println("INDEX " + index);
            }
        }
        return sum;		
		
		
		/********** END **********/
	}
}
