package step2;

import java.io.*;

public class CatchException {
    public static void main(String[] args) {
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
        String str;
        /********** Begin *********/
        //补全try/catch块，异常类型为IOException
        try {
        	int letter=0;
        	int space=0;
        	int number=0;
        	int others=0;
        	str=reader.readLine();
        	char[] ch = str.toCharArray();
        	
        	for(int i=0 ; i<str.length() ; i++) {
        		if ( ch[i]>='0' && ch[i]<='9' )
        			number++;
        		else if ( ch[i]>='a' && ch[i]<='z' )
        			letter++;
        		else if ( ch[i]==' ')
        			space++;
        		else others++;
        	}
        	
        	System.out.print("letter="+letter+",space="+space+",number="+number+",other="+others);
        }
        catch(IOException e) {
        	System.out.print("捕获的异常信息为："+e.getMessage());
        }
        /********** End *********/
    }
}
