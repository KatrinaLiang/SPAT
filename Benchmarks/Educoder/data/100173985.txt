package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int sum;
		for(int a=1;a<=9;a++)
		{
			for(int b=1;b<=a;b++)
			{
				
                sum=a*b;
				System.out.print(b+"*"+a+"="+sum);
				System.out.print("\t");

			}
			System.out.println();
		}
		
		
		
		
		/*****end*****/
	}
}
