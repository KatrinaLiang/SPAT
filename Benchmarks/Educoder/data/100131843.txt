package step3;

import java.io.File;

public class Task {
	
	/********** BEGIN **********/
	public void showDirStructure(File dir)	{
		showFile(0,dir);
	}
    public void showFile(int tab,File file) {

		if(file.isFile()) {
			String name = file.getName();
			String fs = name.substring(name.lastIndexOf('.')+1);
			if(fs.equals("jpg")||fs.equals("png")||fs.equals("bmp")) {
				for(int i=0;i<tab;i++) {
					System.out.print("  ");
				}
				System.out.println("--"+name);				
			}
			//System.out.println("--"+name);
		}else {
			for(int i=0;i<tab;i++) {
				System.out.print("  ");
			}
			System.out.println("+--"+file.getName());
			File childFiles[] = file.listFiles();
			for(File child: childFiles) {
				showFile(tab+1,child);
			}
		}
	}
	

	/********** END **********/
}
