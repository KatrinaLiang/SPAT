package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		
		int reslut = 0;
		int start = n;
		int drop =0;
		
		while(start>0){
			if (start%2 == 0) {//可等分
				start = start/2;
			}
			else{//不可等分
				drop++;
				start = (start-1)/2;
			}
			
		}
		
		
		return drop;

		
		
		/********* End *********/
	}
	
}
