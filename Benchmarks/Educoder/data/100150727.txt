package step1;
import java.util.Scanner;

public class TestJava8_1 {

	public static void main(String[] args) {
		/******Begin******/
		int i=0;
        System.out.println("请输入一个整数：");
        Scanner sc=new Scanner(System.in);
		int w = sc.nextInt();
        if(w>0){
            i=w;
            System.out.println("输入的整数为："+w);
            System.out.println("其绝对值为："+i);
        }
        else if(w<0){
            i=0-w;
            System.out.println("输入的整数为："+w);
            System.out.println("其绝对值为："+i);
        }
		/******End******/
	}
}
