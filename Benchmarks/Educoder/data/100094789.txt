package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
		int sum=0;
        while(n>0){
            int flag=n;
            int ss=0;
            while(flag>0){
           		ss=(int) (a*(Math.pow(10,flag-1))+ss);
                flag--;
            }
            sum=sum+ss;
            n--;
        }
        System.out.print(sum);


		/********** End *********/
	}
}
