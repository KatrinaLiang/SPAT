package step1;

public class Yard {
double width = 0;
double height = 0;
    
   public Yard(int h, int w){
       this.width =w;
       this.height = h;
   }
    
   public double getPerimeter(){
       return 2*width+2*height;
   }
    
    public double getAcreage(){
       return width*height;
    }
   
}
