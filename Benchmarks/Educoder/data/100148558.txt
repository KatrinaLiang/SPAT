/********* Begin *********/
//定义站台类，实现卖票的功能。
public class Station extends Thread   {	 
		static int t=20;
        static Object s=new Object();
        public void t()
        {
            System.out.println("卖出了第"+t+"张票");
            t--;
        }
        public void run()
        {
            while(t>0)
            {
                synchronized(s)
                {
                    if(t>0)
                    {
                        t();
                    }
                }
                if(t==0)
                {
                    System.out.println("票卖完了");
                }
                try{
                    Thread.sleep(100);
                }
                catch(Exception e)
                {
                }
            }
        }
	
}
/********* End *********/
