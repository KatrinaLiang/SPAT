package step1;

public class Task {
	int sum=0;
	public int getSum(int num1, int num2) {
        
		/********** BEGIN **********/
        for(int i=num1;i<=num2;i++){
            sum=sum+(i/100)+(i%100/10)+(i%10);
        }
        //System.out.println(sum);
		return sum;
		/********** END **********/
	}
}
