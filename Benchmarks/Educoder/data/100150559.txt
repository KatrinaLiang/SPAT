package step3;
import java.util.Scanner;

public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
		
        Scanner sc=new Scanner(System.in);
		int num = sc.nextInt();
        System.out.println("请输入一个五位数字：");
        int n=0;
        int sum=0;
        while( num!= 0 ) { //未取尽
			n = num%10;  //取个位数
			sum += n;  //增加
			num = (num-n)/10;  //剩余数字
		}
        System.out.println("85698的各位数字之和是："+sum);
		/******End******/
	}
}
