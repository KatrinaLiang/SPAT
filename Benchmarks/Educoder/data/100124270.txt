package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int first=1,second=1,next=0,sum=2;
        for(int i=3;i<=20;i++){
        	next=first+second;
            sum+=next;
            first=second;
            second=next;
        }
        System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
