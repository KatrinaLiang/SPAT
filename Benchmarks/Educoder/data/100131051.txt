package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		//对变量的分离
        int number1=0;
		
		int weishu=0;
		int sum=0;
		if((num2>=1 && num2<=999) &&(num1>=1 && num1<=999)) {
			
			

			//需要双重循环：
			
			for(int i=num1;num1<=num2;num1++) {
				for(number1=num1;number1!=0;) {
					weishu=number1%10;//个位数
					number1=number1/10;
					sum+=weishu;
					
				}			
			}
				
				}
		

		return sum;
		
		
		/********** END **********/
	}
}
