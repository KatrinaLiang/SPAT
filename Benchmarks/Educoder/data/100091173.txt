public class AbsoluteOfTwo {

    // 求两个整数差的绝对值
    public int AbsoluteOfTwo(int x, int y) {
        // 请在此添加实现代码
        int  ab=0;
		if(x>y)
        {
            ab=x-y;
        return ab;
        }
        else
        {
            ab=y-x;
            return ab; 
        }
    }
}
