#include <iostream>
using namespace std;
int main()
{
    
	int a[50]={0};
    int num=0;
    int i=0;
    for(i=0;i<50;i++){
    	scanf("%d",&a[i]);
    	char ch;
    	ch=getchar();
    	if(ch==10){
			break;
		}
    }
    int t;
    scanf("%d",&t);
    int b,j,k;
    for(b=0;b<i+1;b++){
    	for(j=b+1;j<i+1&&b<j;j++){
        	for(k=j+1;k<i+1&&k>j;k++){
            	if(b!=j&&j!=k&&b!=k&&a[b]+a[j]+a[k]<t){
                	num++;
                }
            }
        }
    }
    printf("%d",num);
    //cout << "Hello, world!" << endl;
    return 0;
}
