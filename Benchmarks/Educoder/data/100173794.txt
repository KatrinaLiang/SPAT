package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
	int a=0,b=0;
		//在这里打印出正三角形 
		for(a=1;a<=10;a++){
			for(b=0;b<a;b++){
				System.out.print("*");
			}
         System.out.print("\n");
		}
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
		for(a=0;a<10;a++){
			for(b=1;b<=(10-a);b++){
				System.out.print("*");
			}
			System.out.print("\n");
		}
		
		
		
		
		/*****end*****/
	}
}
