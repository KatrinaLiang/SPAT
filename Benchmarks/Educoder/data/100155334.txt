package task3;

public class Box {

    double width;
    double height;
    double depth;
	
    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public double volume(int boxType) {
        //TODO write your code here
        double v=0;
        switch (boxType){
            case 1:v=width*height*depth;break;
            case 2:v=height*width*width*3.14;break;
            default:break;
        }
        return v;
    }
    public static void main(String[] args){
        Box box = new Box();
        box.width = 10;
        box.height = 20;
        box.depth = 10;
        double volume = box.volume(Box.CYLINDER);
        System.out.println("the volumn is:" + volume);
    }

}
