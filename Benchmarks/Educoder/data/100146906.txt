package step1;

public class GetScores {

	/********* Begin *********/
    public int getLostScores(int[] nums){
       //请在此补全你的代码
    	int[] presentTbl = new int[10000];
		String conNum = new String();
		for(int i = 0; i < 10000; ++i){presentTbl[i] = 0;}
		for(int ele : nums){presentTbl[ele - 1] = 1;}
		for(int i = 0; i < 10000; ++i)
		{
			if(presentTbl[i] == 0)
			{
				conNum += Integer.toString(i + 1);
			}
		}
		long sumNum = Long.parseLong(conNum);
		long modNum = 11;
//		return 0;
		return new Long(sumNum % modNum).intValue();
	
	}

	/********* End *********/
}
