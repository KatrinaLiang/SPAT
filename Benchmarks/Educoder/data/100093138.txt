package step3;
import java.util.Scanner;
public class SeasonJudge {
    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);
        int month = sc.nextInt();
        
        System.out.println(SeasonJudge(month));
    }
	public static String SeasonJudge(int month) {
        String n;
        switch(month)
        {
            case 3 : n="春季";break;
                case 4 : n="春季";break;
                case 5 : n= "春季";break;
                case 6 : n ="夏季";break;
                case 7 : n= "夏季";break;
                 case 8 : n= "夏季";break;
                 case 9 : n= "秋季";break;
                 case 10 : n= "秋季";break;
                 case 11 : n= "秋季";break;
                 case 12: n= "冬季";break;
                case 2: n= "冬季";break;
                case 1: n= "冬季";break;
            default:n= "月份错误";                
        }
		return n;
	}
}
