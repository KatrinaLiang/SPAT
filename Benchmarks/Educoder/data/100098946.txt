package step1;

public class Yard {
    int length,width;
    public Yard(int length,int width){
      this.length=length;
        this.width=width;
      }
     public double getPerimeter(){
        double perimeter;
            perimeter=2*length+2*width;
        return perimeter;
    }
    public double getAcreage(){
        double acreage;
            acreage=length*width;
     return acreage;   
    }

}
