package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		/********** BEGIN **********/
        double sum=0;
        if(n%2==0) {
        	for(double i=2;i<n+2;i+=2) {
        		sum+=3/i;
        	}
        }
        else {
        	for(double i=1;i<n+2;i+=2) {
        		sum+=4/i;
        	}
        }
		System.out.println(sum);
		
		
		/********** END **********/
	}
}
