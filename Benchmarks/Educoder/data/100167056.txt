package step1;

public class A< T extends Comparable >
{
 private T t1,t2; 
/*************BEGIN**************/
 public A(){}
 public A(T a1,T a2)
 {
     t1=a1;
     t2=a2;
 }
  public  T compare()
  {
      if((t1.compareTo(t2))>0)
      {
          return t1;
      }
      else
          return t2;
  }



/*************END*****************/

}
