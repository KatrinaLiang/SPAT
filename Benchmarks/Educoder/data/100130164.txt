package step4	;	//在这里添加包名  step4

		//创建类 添加属性和方法
 class WuMingFen{
    String name;
    int sum = 0 ;
    boolean io = false ;
    public WuMingFen(String name , int sum , boolean io){
    	this.name = name;
        this.sum = sum;
        this.io = io;
    }
    public WuMingFen(String name , int sum){
    	this.name = name;
        this.sum = sum;
        this.io = false;
    }
    public WuMingFen(){
    	this.name = "酸辣";
        this.sum = 2;
        this.io = true;
    }
    public void check(){
     System.out.println("面码："+this.name+"，粉的份量："+this.sum+"两，是否带汤："+this.io);
    }
}
