package step1;

public class A< T extends Comparable >{
 private T t1,t2; 
/*************BEGIN**************/
public A(T a,T b)
{
    this.t1=a;
    this.t2=b;
}
public T compare()
{
    if(t1.compareTo(t2)>=0)
    {
        return t1;
    }
    else
    {
        return t2;
    }
}





/*************END*****************/

}
