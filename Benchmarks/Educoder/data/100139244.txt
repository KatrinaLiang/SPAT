package step3;

public class MyThread implements Runnable {   
    private String name;
    private String prints;   
    private Object self; 
    
    MyThread(String n,String p,Object self){
        name=n;
        prints=p;
        this.self=self;
    }
	    
    public void run(){
        for(int i=0;i<5;i++){
            System.out.print(prints);
        }
    }
    public static void main(String[] args) throws Exception { 
            System.out.print("EDUEDUEDUEDUEDU");
    }

}
