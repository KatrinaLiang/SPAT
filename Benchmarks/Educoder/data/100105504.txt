package step1;

public class Yard {
    double chang;
    double kuan;
    double circle;
    double s;
    public Yard(double chang,double kuan){
        this.chang=chang;
        this.kuan=kuan;
    }
    double getPerimeter(){
        circle=chang*2+kuan*2;
        return circle;
    }
    double getAcreage(){
        s=chang*kuan;
        return s;
    }
}
