package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
	public InterRectangle(){
		
	}
	
	public InterRectangle(double length,double width){
		this.length = length;
		this.width = width; 
	}
	
	public double Perimeter(double length,double width){
		double perimeter = (this.length + this.width) * 2;
		return perimeter;
		
	}
	 public double area(double length,double width){
	        double area = (this.width * this.length);
	        return area;
	     
	 }   

	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
		 System.out.println("总共有" + total + "块土地");
		 
		
		 double totalPerimeter = 0;
		 double totalArea = 0 ;
		 for(int i = 0 ;i < total ;i++){
//			 System.out.println("请输入长与宽：");
			 double length = scanner.nextDouble();
			 double width = scanner.nextDouble();
			 InterRectangle ir = new InterRectangle(length,width);
			 totalPerimeter += ir.Perimeter(length, width);
			 totalArea += ir.area(length, width);
			
			
			 
		 }
		System.out.println("土地总面积为:" + totalArea + "平米");
		System.out.println("土地总周长为:" + totalPerimeter + "米");

		/*********end*********/
	}

}
