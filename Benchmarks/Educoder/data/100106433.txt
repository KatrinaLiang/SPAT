package Step3;

public class MultiplyTable {
	public static void main(String args[]){
		/********** Begin *********/
		for(int a=1;a<=9;a++){
            for(int b=1;b<=a;b++){
                System.out.print(" "+a+"×"+b+"="+a*b);// \t 跳到下一个TAB位置
            }
            System.out.println();
        }
		 /********** End *********/
	}
}
