package step2;
public class Fibonacci{
	 public static void main(String args[]){
    /*********begin********/
        int i;
        int sum = 2;
        int[] F=new int[20];
        F[0]=1;
        F[1]=1;
        for(i=2;i<F.length;i++){
            F[i]=F[i-1]+F[i-2];
            sum = sum + F[i];
        }
        System.out.print("前20个元素的和为" + sum);
    /********end********/
    }
}
