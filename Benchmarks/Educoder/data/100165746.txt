package step3;
import step1.Date;
public class Course {
    private int code; //课程编号
    private String courseName; //课程名称
    private Date testTime;//考试日期
    private int score;//考试成绩
   /******Begin**************/
    public Course(){
        testTime=new Date();
        courseName="";
    }
    public Course(int code,String courseName,int y,int m,int d,int score){
        testTime=new Date(y,m,d);
        this.code=code;
        this.courseName=courseName;
        this.score=score;
    }
    public Course(int code,String courseName,Date testTime,int score){
        this.testTime=testTime;
        this.code=code;
        this.courseName=courseName;
        this.score=score;
    }
	public void setCourse(String courseName){
        this.courseName=courseName;
    }
    public int getcode(){return code;}
    public String getCourseName(){return courseName;}
    public Date getTestTime(){return testTime;}
    public int getscore(){return score;}
    public String toString(){
        return getcode()+" "+getCourseName()+" "+getTestTime()+" "+getscore();
    }
        




   /******End************/

}
