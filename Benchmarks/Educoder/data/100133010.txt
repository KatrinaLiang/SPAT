package step1;

public class Task {
	
	public void sortArray(int[] arr){
        for(int i = 1 ;i < arr.length;i++){
            int s = i;
            for (int j = i-1; j >=0;j--){
               
                if(arr[j] > arr[s]){
                    int temp = arr[j];
                    arr[j] = arr[s];
                    arr[s] = temp;
                }
                  s--;  
             }
            for(int k = 0;k<arr.length;k++){
                    System.out.print(arr[k]+" ");
                }
            System.out.println();
        }
        
        
    }

		
		
	}
