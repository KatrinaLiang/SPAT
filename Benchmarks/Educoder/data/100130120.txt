package step4;
class WuMingFen
{
	String theMa;
    int quantity;
    boolean likeSoup;
    public WuMingFen(String a,int b,boolean c)
    {
        theMa=a;
        quantity=b;
        likeSoup=c;
    }
    public WuMingFen(String a,int b)
    {
        theMa=a;
        quantity=b;
        likeSoup=false;
    }
    public WuMingFen()
    {
        theMa="酸辣";
            quantity=2;
            likeSoup=true;
    }
    public void check()
    {
        System.out.println("面码："+theMa+"，粉的份量："+quantity+"两，是否带汤："+likeSoup);
    	
    }
}
