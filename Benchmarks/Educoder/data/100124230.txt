package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int[] a=new int[25];
		a[1]=1;
		a[2]=1;
		for(int i=3;i<=20;i++) {
			a[i]=a[i-1]+a[i-2];
		}
		int sum=0;
		for(int i=1;i<=20;i++) {
			sum+=a[i];
			//System.out.println(a[i]);
		}
		System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
