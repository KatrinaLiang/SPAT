package step1;

public class Person {
    private String name;//姓名
    private String num;//身份证号
    private Date birthday = new Date();//生日
    public Person(){}
    public Person(String name,String number,Date birthday){
        this.name = name;
        num = number;
        this.birthday = birthday;
    }
    public Person(String name,String number,int year,int month,int day){
        this.name = name;
        num = number;
        birthday.setDate(year,month,day);
    }
    public void setinfo(String name, String num, Date birthday){
        this.name = name;
        this.num = num;
        this.birthday = birthday;
    }
    public String getnum(){
        return num;
    }
    public String getname(){
        return name;
    }
    public Date getBirthday(){
        return birthday;
}
    @Override
    public String toString(){
        return "name="+name+" number="+num+" birthday:"+birthday.toString();
   }

   /************End*************/
}
