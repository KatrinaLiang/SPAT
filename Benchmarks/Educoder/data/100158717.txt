#include <stdio.h>
#include <math.h>
int main()
{
	int a,b,c,n;
	scanf("%d",&a);
	c=a%3;
	
	if(a==2)
	{
		printf("2");
	}
	else if(a==1)
	{
		printf("1");
	}
	else
	{
		if(c==1)
		{
			n=(a-1)/3;b=4*pow(3,n-1);
			printf("%d",b);
		}
		if(c==0)
		{
			n=a/3;b=pow(3,n);
			printf("%d",b);
		}
		
		if(c==2)
		{
			n=(a-2)/3;b=2*pow(3,n);
			printf("%d",b);
		}
	}
	return 0;
}
