package step1;

public class Yard {
     int length,wideth;
       public Yard(int length,int wideth)
       {
           this.length=length;
       this.wideth=wideth;
       }
    float getPerimeter(){
        float perimeter;
            perimeter=2*(length+wideth);
            return perimeter;
    }
    float getAcreage(){
        float acreage;
         acreage=length*wideth;
            return acreage;
    }
}
