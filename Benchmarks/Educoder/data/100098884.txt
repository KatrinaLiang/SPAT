package step1;

public class Yard {
     private float length;
     private float width;
     public Yard(float length, float width){
        this.length = length;
        this.width = width;
      }
      public float getPerimeter(){
     	return   (length +  width) * 2;
     }
     public float getAcreage(){
      	return   (length *  width);
     }
}
