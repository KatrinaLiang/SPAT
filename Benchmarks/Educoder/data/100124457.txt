package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		
 int[]ids=new int[20];
    int s=2,i=2;
    ids[0]=1;
    ids[1]=1;
    while(i<20)
    {
    ids[i]=ids[i-1]+ids[i-2];
    s=s+ids[i];
    i++;    }
    System.out.print("前20个元素的和为"+s);
	/********end********/
	}
}
