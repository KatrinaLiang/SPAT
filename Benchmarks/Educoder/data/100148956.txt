package step3;

public class Task {
	
	public int concaveValley(Integer[] arr){
		/********* Begin *********/
		int state=0;
		int maxlen=0;
		int temp=0;
		int first=0;
		for(int i=0;i<arr.length-1;i++){
			if(arr[i]<=arr[i+1]){
				first++;
			}else{
				break;
			}
		}
		while(first<arr.length-1){
			while(first<arr.length-1&&(arr[first]>arr[first+1])){
				temp++;
				first++;
				
			}
			while(first<arr.length-1&&(arr[first]<arr[first+1])){
				temp++;
				first++;
				if((first==arr.length-1)||(first<arr.length-1&&arr[first]>=arr[first+1])){
					temp++;
				}
			}
			maxlen=temp>maxlen?temp:maxlen;
			if(first<arr.length-1&&arr[first]==arr[first+1]){
				break;
			}
			temp=0;
		}
		return maxlen>2?maxlen:0;
		
		/********* End *********/	
	}
}
