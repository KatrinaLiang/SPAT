package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		if(n==2 || n==1){
        	return 1;
        }
		else{
        	if(n%2==0)
                return getNum(n/2);
            else
                return getNum(n/2)+1;
        }
		
		
		
		/********* End *********/
	}
	
}
