package step4;//在这里添加包名  step4

		//创建类 添加属性和方法
class WuMingFen{
    String theMa;
    int quantity;
    boolean likesoup;
    public WuMingFen(String ch_theMa,int ch_quantity,boolean ch_likesoup){
        theMa=ch_theMa;
        quantity=ch_quantity;
        likesoup=ch_likesoup;
    }
    public WuMingFen(String ch_theMa,int ch_quantity){
        theMa=ch_theMa;
        quantity=ch_quantity;
        likesoup=false;
    }
    public WuMingFen(){
        theMa="酸辣";
        quantity=2;
        likesoup=true;
    }
    
    public void check(){
         System.out.println("面码："+theMa+"，粉的份量："+quantity+"两，是否带汤："+likesoup);   
        }
   
}
