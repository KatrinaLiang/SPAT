package step2;

public class Task {

	public int getNum(int n){
       int num=0;
       
           while(n>0){
               if(n%2==0){
                   n=n/2;
               }else{
                  n=n-1;
                  num++;
                  n=n/2;
               } 
           }
           return num;
       }

	
}
