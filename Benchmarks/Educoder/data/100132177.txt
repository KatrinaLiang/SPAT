package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		int[] newtemp = new int[temperatures.length];
		for (int i = 0; i < temperatures.length; i++) {
			int mark=0;
			for (int j = i; j < temperatures.length; j++) {
				if (temperatures[j]>temperatures[i]&&mark==0) {
					mark=j;
				}
			}
			if (mark==0) {
				newtemp[i]=mark;
			}else {
				newtemp[i]=mark-i;
			}
		}
		return newtemp;
	}
}
