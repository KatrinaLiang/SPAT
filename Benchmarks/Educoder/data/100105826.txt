package step1;

public class Yard {
    public int length;
    public int width;
    public Yard(int length,int width)
    {
        this.length=length;
        this.width=width;
    }
    public double getPerimeter()
    {
        double l=(length+width)*2;
        return l;
    }
	public double getAcreage()
    {
    	double s=length*width;
        return s;
    }
}
