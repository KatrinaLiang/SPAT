package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
        int[] newtem=new int[temperatures.length];
        int n=0;
        while(n<temperatures.length){
            int count=n+1;
            int flag=0;
            while(flag<temperatures.length-n-1){
            if(temperatures[count]>temperatures[n]){
                 newtem[n]=count-n;
                 break;
            }
            else{
                count++;
                flag++;
            }
            }
           n++;
        }
		return newtem;
		/********** END **********/
	}
}
