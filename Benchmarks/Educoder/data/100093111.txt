package step2;

public class SwitchToGrade {

    // 请用switch来将学生成绩转换成对应等级
    public String SwitchToGrade(int score) { 
        // 请在此提供实现代码
        String ch;
        int cj=score/10;
        switch(cj){
           
            case 10:
            case 9:
              ch = "等级：A";break;
            case 8:
            case 7:
                 ch = "等级：B";break;
            case 6:
                 ch = "等级：C";break;
            case 5:
            case 4:   
            case 3:   
            case 2:    
            case 1:   
                ch = "等级：D";break;
            default:
                ch = "输入成绩非法";
        }return ch;
	}
}
