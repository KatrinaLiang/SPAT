package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int a=0,b=0,c=0;
		int temp = 0;
		for(int i=num1;i<=num2;i++) {
			
			if(i>=0&&i<10) {
				a = i;
				temp = temp + a;
			}
			if(i>=10&&i<100) {
				a = i%10;
				b = i/10;
				temp = temp+a+b;
			}
			if(i>=100&&i<=999) {
				a = i % 10;
				b = i/10%10;
				c = i/100;
				temp = temp+a+b+c;
			}
			
		}
		return temp;
		
		/********** END **********/
	}
}
