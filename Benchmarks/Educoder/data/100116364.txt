package step3;

public class HelloWorld {
	public static void main(String[] args) {
		int count= 0;	//定义变量存储6的倍数出现的次数
		/*****start*****/
        int n=1;
		do{
            count+=(n%6==0) ? 1:0;
            n++;
        }while(n<101);
     
		
		/*****end*****/
		System.out.println("6的倍数出现的次数为：" + count);
	}
}
