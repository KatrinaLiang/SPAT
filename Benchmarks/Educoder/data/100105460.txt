package step3;

public class Reverse {
    public static StringBuffer start(String data) {
        StringBuffer ans = new StringBuffer();
        /********** Begin *********/
		String[] tems=data.split(" ");
		for(int i=0;i<tems.length;i++)
        {
         	StringBuffer tem=new StringBuffer(tems[i]);
            ans.append(tem.reverse());
            ans.append(" ");
        }
        /********** End *********/
        return ans;
    }
}
