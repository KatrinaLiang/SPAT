package step1;

public class Yard {
    public Yard(int length,int width){
        this.length=length;
        this.width=width;
        
    }
    double getPerimeter(){
        return 2*(length+width)*1.0;
    }
     double getAcreage(){
        return length*width*1.0;
    }

    
    
    private int length;
    private int width;
}
