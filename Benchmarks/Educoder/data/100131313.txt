public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int[] a=new int[3];
        int num=0;
        for(int j=num1;j<num2+1;j++){
            a[0]=j/100;
            a[1]=j%100/10;
            a[2]=j%10;
            for(int i=0;i<3;i++){
            num+=a[i];
            }
            }
		return num;
        
		/********** END **********/
	}
}
