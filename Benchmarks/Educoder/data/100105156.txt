package step3;
public class Reverse {
    public static StringBuffer start(String data) {
        StringBuffer ans = new StringBuffer();
        /********** Begin *********/
         StringBuffer mid = new StringBuffer();
        for(int i = 0 ;i<data.length();i++)
        {

            if(data.charAt(i)==' ')
            {
                mid = mid.reverse();
                String S = mid.toString();
                ans.append(S);
                ans.append(' ');
                mid = mid.delete(0,mid.length());
                continue;
            }
            mid.append(data.charAt(i));
            if(i ==data.length()-1)
            {
                mid = mid.reverse();
                String S = mid.toString();
                ans.append(S);
                ans.append(' ');
                mid = mid.delete(0,mid.length());
                continue;
                
            }
            
        }
        /********** End *********/
        return ans;
    }
}
