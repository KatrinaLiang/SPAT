package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        int sum = 0;
		while(n > 0)
        {
            int num = 0;
            for(int i = 0; i < n; i++)
            {
                num = num * 10 + a;
            }
            sum += num;
            --n;
        }
		System.out.print(sum);

		/********** End *********/
	}
}
