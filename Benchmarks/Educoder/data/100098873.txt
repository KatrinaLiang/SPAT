package step1;

public class Yard {
    public double length, width;
    public Yard(){};
    public Yard(int a, int b){
        length = a;
        width =  b;
    }
    public double getPerimeter(){
        return length * 2 + width * 2;
    }
    public double getAcreage(){
        return length * width;
    }
}
