package step2;
import step1.Person;
import step1.Date;
public class Student extends Person{
    private int reg_num; //学号
    private Date reg_day; //注册日期
   // TO DO YOU CODE HERE
    public Student(String a,String b,int year,int month,int day,int c,int i,int j,int k){
    super(a,b,year,month,day);
    reg_num=c;
    reg_day=new Date(i,j,k);
}
   public Student(String a,String b,Date c,int d,Date e){
    super(a,b,c);
    reg_num=d;
    reg_day=e;
}
public Student(){
    super();
    reg_num=0;
    reg_day=new Date();
}
 public Date getreg_day(){
     return reg_day;
 }
public String toString(){
    return super.toString()+"reg_num:"+ reg_num +"reg_day:"+reg_day;
}
}
