package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum = 0;
        for(int i = num1; i <= num2; i++){
            int n1 = i / 100;
            int n2 = i % 100 / 10;
            int n3 = i % 10;
            sum += n1 + n2 + n3;
        }
        return sum;
		
		
		/********** END **********/
	}
}
