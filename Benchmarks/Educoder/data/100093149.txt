package step3;
import java.util.Scanner;
public class SeasonJudge {
    public static void main (String[] args)
    {
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        SeasonJudge sea = new SeasonJudge();
        System.out.println(sea.SeasonJudge(a));
    }
    // 根据月份判断季节
	public String SeasonJudge(int month) {
        // 请在此提供实现代码
        String a;
        switch(month)
        {
            case 3:
            case 4:
            case 5:
                a ="春季";break;
            case 6:
            case 7:
            case 8:
                a="夏季";break;
            case 9:
            case 10:
            case 11:
                a="秋季";break;
            case 12:
            case 1:
            case 2:
                a ="冬季";break;
            default:
                a = "月份错误";
        }
        return a;
	}
}
