package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int a,b,c,sum;
        sum=0;
        for(int i=num1;i<=num2;i++){
        	if(i<100){a=i/10;
            b=i%10;
            sum+=(a+b);}
            else{
            	a=i%10;
                b=i/100;
                c=(i%100-a)/10;
                sum+=(a+b+c);
            }
        }
        
        return sum;
		
		
		/********** END **********/
	}
}
