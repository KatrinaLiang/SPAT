package step3;
import java.util.Scanner;
public class Factorial {
	public void factorial(int n) {
		/********** Begin *********/
        int sum=1;
        for(int i=1;i<=n;i++){
            sum*=i;
        }
        System.out.print(sum);
		/********** End *********/
	}
    public static void main(String[] args){
        Factorial f=new Factorial();
        Scanner in=new Scanner(System.in);
        int n=in.nextInt();
        f.factorial(n);
    }
}
