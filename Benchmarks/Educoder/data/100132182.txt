package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temps){
		/********** BEGIN **********/
		//temps = {34,35,33,25,44,19,18,17}
		//        {34,35,33,25,44,19,18,17}
		//[1, 3, 2, 1, 0, 0, 0, 0]
		int length=temps.length;
		int tian[]=new int[length];
		int n=0;
        for(int i=0;i<temps.length;i++){
        	for(int j=i;j<temps.length;j++){
        		if(temps[j]>temps[i]){
        			n=j-i;
        			break;
        			
        			
        		}else
        			n=0;
        	}
        	tian[i]=n;
        }
        
		
		
		

		return tian;
		/********** END **********/
	}
}
