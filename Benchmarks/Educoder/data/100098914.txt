package step1;

public class Yard {
    private int length;
    private int width;
    
	public Yard() {
    
    }
    
    public Yard(int length, int width) {
    	this.length = length;
        this.width = width;
    }
    
    public float getPerimeter() {
    	return (this.length + this.width) * 2;
    }
    
    public float getAcreage() {
    	return this.length * this.width;
    }
}
