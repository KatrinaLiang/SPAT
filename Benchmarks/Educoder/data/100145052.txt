package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		
		
if (matrix.length==0)
			return;
		int start = 0;
		int last = matrix.length - 1;
		while (start < last)
		{
			int[] mid=new int[matrix.length];
			mid=matrix[start];
			matrix[start++]=matrix[last];
			matrix[last--]=mid;
			//swap(q[start++], q[last--]);
		}
		for (int i = 0; i < matrix.length ; ++i){
			for (int j = 0; j < i; ++j)
			{
				int t=0;
				t=matrix[i][j];
				matrix[i][j]=matrix[j][i];
				matrix[j][i]=t;
				//swap(q[i][j], q[j][i]);
			}
		}
		
		
		/********* End *********/
	}
}
