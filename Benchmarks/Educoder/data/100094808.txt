package step2;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/

        int sum = 0;
        for(int i = 1; i <= n; ++i){
            int temp = 0;
            for(int j = 1; j <= i; ++j){
                temp = temp * 10 + a;
            }
            sum += temp;
        }
            
        System.out.print(sum);


		/********** End *********/
	}
}
