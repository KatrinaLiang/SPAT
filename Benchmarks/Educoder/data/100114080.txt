package step1;
import java.util.Scanner;

public class Cal {
	public static void main(String[] args) {
		/*********start*********/
        Scanner result = new Scanner(System.in);
		System.out.println("请输入第一个整数");
        int i = result.nextInt();
        System.out.println("请输入第二个整数");
        int j = result.nextInt();
        int add = i+j;
        int xj = i-j;
        int xc = i*j;
        int xx = i/j;
        int qy = i%j;
        System.out.println("两数相加的结果为："+add);
        System.out.println("两数相减的结果为："+xj);
        System.out.println("两数相乘的结果为："+xc);
        System.out.println("两数相除的结果为："+xx);
        System.out.println("两数取余数的结果为："+qy);
        
        
		
		
		/*********end*********/
	}

}
