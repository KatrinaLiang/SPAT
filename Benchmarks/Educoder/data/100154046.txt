package step1;

public class ShuShu {

	// 第一题
	public void first() {
		/*********Start********/
        int a=1;
        while(true){
            if(a>100){
                break;
            }
            if(a%10!=0){
            System.out.print(a+"	");
            }
            a++;
        if(a%10==0){
            System.out.println(a+"	");
        }
        }
		
		/*********End********/
	}
	
	// 第二题
	public void second() {
		/*********Start********/
        System.out.println();
        int a=1;
        do{
            if(a%10!=0){
            System.out.print(a+"	");
            }
            if(a%10==0){
                System.out.println(a+"	");
            }
            a++;
        }while(a<=100);
		
		/*********End********/
	}
	
	// 第三题
	public void third() {
		/*********Start********/
        System.out.println();
        for(int a=1;a<=100;a++){
            if(a%10!=0){
            System.out.print(a+"	");
            }
            if(a%10==0){
                System.out.println(a+"	");
            }
        }
		
		/*********End********/
	}
}
