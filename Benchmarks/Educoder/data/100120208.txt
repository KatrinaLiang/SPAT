public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		for(int i=1;i<=9;i++){
            for(int m=1,n=i;m<=i;m++){
                System.out.print(m+"*"+n+"="+(m*n));
                System.out.print("\t");}
                 System.out.println();}                										
		/*****end*****/
	}
}
