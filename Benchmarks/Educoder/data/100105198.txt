package step3;

public class Reverse {
    public static StringBuffer start(String data){
        /********** Begin *********/
        StringBuffer ans = new StringBuffer();
        
        String[]b=data.split(" ");
        for(int a=0;a<b.length;a++)           
        {
            StringBuilder sbtmp = new StringBuilder(b[a]);
            sbtmp.reverse();
            ans.append(sbtmp).append(" ");
        }
        return ans;
        /********** End ********/
    }
}
