package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int sum=0;
        while(n!=1)
        {
            if(n%2!=0)
            {
                sum=sum+1;
            }
            n=n/2;
        }
        sum=sum+1;
        return sum;
		
		
		
		
		/********* End *********/
	}
	
}
