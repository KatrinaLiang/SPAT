package step1;

public class Person {
    private String name; //姓名
    private String number; //身份证号
    private Date birthday; //出生日期
    
   /************Begin*************/
	public Person(){
     	name = new String("null");
        number = new String("null");
        birthday = new Date("1999.9.9");
    }
    public Person(String name,String number,Date birthday){
     	this.name = name;
        this.number = number;
        this.birthday = birthday;
    }
	public Person(String name,String number,int y,int m,int d){
     	this.name = name;
        this.number = number;
        birthday = new Date(y,m,d);
    }
    public String getname(){
     	return name;   
    }
    public String getnumber(){
     	return number;   
    }
    public Date getDate(){
     	return birthday;   
    }
    public void setname(String name){
     	this.name = name;   
    }
    public void setnumber(String number){
     	this.number = number;   
    }
    public void setDate(Date birthday){
     	this.birthday = birthday;   
    }
    public String toString(){
     	return "name="+name+" number="+number+" birthday:"+birthday;   
    }
   /************End*************/
}
