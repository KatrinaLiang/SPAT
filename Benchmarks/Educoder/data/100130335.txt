package step1;
public class Task {

	public void sortArray(int[] arr, String sort){
		/********** BEGIN **********/
        int temp=-1;
        if(sort.equals("asc"))
        {
        	for(int i=0;i<arr.length;i++)
            {
            	for(int j=i+1;j<arr.length;j++)
                {
                	if(arr[i]>arr[j])
                    {
                    	temp=arr[i];
                        arr[i]=arr[j];
                        arr[j]=temp;
                    }
                }
            }
            
        }
        if(sort.equals("desc"))
        {
        	for(int m=0;m<arr.length;m++)
            {
            	for(int n=m+1;n<arr.length;n++)
                {
                    if(arr[m]<arr[n])
                    {
                    	temp=arr[m];
                        arr[m]=arr[n];
                        arr[n]=temp;
                    }
            }
        }
            
            
        }
		for(int k=0;k<arr.length;k++){
            	System.out.println(arr[k]);
            }

		
		/********** END **********/
	}
}
