package step3;

public class NarcissisticNumber {

    //判断一个三位数是否是水仙花数
    public boolean NarcissisticNumber(int n){
        // 请在此添加实现代码
        if((n / 100)*(n / 100)*(n / 100) + (n/10 - 10*(n / 100))*(n/10 - 10*(n / 100))*(n/10 - 10*(n / 100)) + (n % 10)*(n % 10)*(n % 10) == n)
        {
        	return true;
        }
        return false;
    }
}
