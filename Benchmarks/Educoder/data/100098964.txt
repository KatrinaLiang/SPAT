package step1;

public class Yard {
	int length, width;
    public Yard(){};
    public Yard(int length, int width){
  	    this.length = length;
    	this.width = width;

    }
    public double getPerimeter(){

        double a = (length + width) * 2;
        return a;
    }
    public double getAcreage(){
        double p = length * width;
        return p;
    }
}
