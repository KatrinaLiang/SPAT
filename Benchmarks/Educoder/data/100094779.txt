package step2;
import java.util.Scanner;
import java.lang.Math;


public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
       // static pow(double z,double w);
        long sum=0;
		long t=0;
		while(n>0){
			for(int i=0;i<n;i++) {
				sum=sum+(long)java.lang.Math.pow(10,i)*a;
			}
			t=t+sum;
			n--;
			sum=0;
		}
		System.out.print(t);
	}

		/********** End *********/
	}
