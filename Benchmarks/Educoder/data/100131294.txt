package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum = 0;
        for(int i=num1;i<=num2;i++){
            String string = i+"";
            for(int k=0;k<string.length();k++){
                String current = String.valueOf( string.charAt(k));
                int a = Integer.parseInt(current);
                sum = sum + a;
            }
        }
        return sum;
		
		/********** END **********/
	}
}
