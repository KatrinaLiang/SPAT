package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int num = n;
        int result = 0;
        while(num!=0){
            if(num%2 == 1){
                result = result+1;
                num = num -1;
            }
            num = num/2;
        }
        return result;
		
		
		
		
		/********* End *********/
	}
	
}
