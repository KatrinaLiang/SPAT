package task4;

public class Box {

    double width;
    double height;
    double depth;
    double volume;
    int boxType;

    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public Box(int x){
    	boxType = x;
    }
        
    public double volume() {
        //TODO your code
        if (boxType == 1){
            volume = height*width*depth;
        }
        else 
            volume = width*width*3.14*height;
        return volume;
    }

}
