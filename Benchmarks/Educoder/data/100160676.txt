package step3;
import java.io.File;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;
import org.jsoup.Jsoup;
import org.jsoup.nodes.Document;
import org.jsoup.nodes.Element;
import org.jsoup.select.Elements;
public class Task {
    
    //获取“http://hotels.ctrip.com/”Docment对象
	public Document getDoc1(String url) throws IOException{
        Document doc=Jsoup.parse(new File("./backups/hotel.ctrip.com.txt"),"utf-8","http://hotels.ctrip.com/");
		return doc;
	}

	//获取“http://hotels.ctrip.com/domestic-city-hotel.html”的Docment对象
	public Document getDoc2(String url) throws IOException{
        Document doc1=Jsoup.parse(new File("./backups/hotels.ctrip.com_domestic-city-hotel.txt"),"utf-8","http://hotels.ctrip.com/domestic-city-hotel.html");
		return doc1;
	}

	//获取所有链接
	public List<String> getLinks(Document doc){
		List list=new ArrayList();
        Elements e=doc.select("a[href]");
        for(Element e1:e){
            String lianjie=e1.attr("abs:href");
            String wb=e1.text();
            String s="a$"+lianjie+"("+wb+")";
            list.add(s);
        }
	    return list;
	}
	
	//获取图片
	public List<String> getMedia(Document doc){
		List list1=new ArrayList();
		Elements e2=doc.select("img[src]");
        for(Element e3:e2){
            String photo=e3.attr("abs:src");
            String rels=e3.attr("rel");
            String output="img$"+photo+rels;
            list1.add(output);
        }
        return list1;
	}
	
	//获取link[href]链接
	public List<String> getImports(Document doc){
		List list2=new ArrayList();
        Elements e4=doc.select("link[href]");
        for(Element e5:e4){
            String e6=e5.attr("abs:href");
            String wbs=e5.attr("rel");
            String s="link$"+e6+"("+wbs+")";
            list2.add(s);
        }
        return list2;
	}
        
	
}
