package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		for(int i=1;i<10;i++){
            int k=1;
            for(int j=i;j>0;j--){
                
                System.out.print(k+"*"+i+"="+(k*i)+"\t");
                k++;
            }
            System.out.println();
        }
		
		
		
		
		
		
		
		/*****end*****/
	}
}
