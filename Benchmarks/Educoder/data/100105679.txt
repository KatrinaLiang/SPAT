package step1;

public class Yard {
    private double length;    //成员变量（长度）
    private double width;    //成员变量（宽度）
    Yard(double length, double width) {    //构造方法
        this.length = length;
        this.width = width;
    }
    public double getPerimeter(){
        return 2*(length+width);
    }
        public double getAcreage(){
        return (length*width);
    }
}
