package step2;

public class Task {
    boolean[][] visit;
	public int virusArea(int n,int m,char[][] area){
		/********* Begin *********/
		visit = new boolean[n][m];
        int num = 0;
        for(int i = 0;i < n;i++){
            for(int j = 0;j < m;j++){
                if(area[i][j] == 'o' && visit[i][j] == false) {
                    //System.out.println(i + " " + j + " " + Oil[i][j]);
                    DFS(n,m,i, j,area);
                    num +=1;
                }
            }
        }
		return num;
		/********* End *********/	
	}
    void DFS(int n,int m,int x,int y,char[][] area){
        if(x < 0 || y < 0 || x > n-1 || y > m-1)
            return;
        else if(area[x][y] == 'x' || visit[x][y] == true)
        	return;
        else {
            visit[x][y] = true;
            int[][] t = new int[][]{{-1,0},{0,-1},{1,0},{0,1}};
            for(int i = 0;i < 4;i++){
                DFS(n,m,x+t[i][0],y+t[i][1],area);
            }
        }
    }
}
