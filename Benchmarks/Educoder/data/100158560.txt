package Pid_203;
import java.util.Scanner;
public class Main {

	public static void main(String[] args) {
		// write your code here
        Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int m = n/3;
		int k = n%3;
		long x = (long)Math.pow(3,m);
		if(k==0) {
			System.out.print(x);
		}else if(k==1) {
			System.out.print(x/3*4);
		}else {
			System.out.print(x*2);
		}
	}

}
