		Scanner sc=new Scanner(System.in);
		System.out.println("请输入菱形上半部的行数：");
		int n=sc.nextInt();
		
		for(int i=0;i<n;i++){
			for(int x=i+1;x<n;x++){
				System.out.print(" ");
			}
			for(int j=1;j<=i*2+1;j++){
			System.out.print("*");	
			
			
			
			}
			System.out.println();
			}
		int z=n-1;
		for(int i=0;i<z;i++){
			for(int b=-1;b<i;b++){
				System.out.print(" ");
			}
			for(int j=1;j<=2*z-i*2-1;j++){
				System.out.print("*");
			}
			System.out.println();
		}
