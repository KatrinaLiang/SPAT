package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
	public InterRectangle(double width, double length){
    	this.width = width;
        this.length = length;
    }
    public double perimeter()
    {
        return 2 * (width + length);
    }
    public double area(){
    	return width * length;
    }

	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
        System.out.println("总共有" + total + "块土地");
        double TotalEara = 0;
        double TotalPerimeter = 0;

        double length1;
        double width1;
        for(double i=0; i < total; i++){

            length1 = scanner.nextDouble();

            width1 = scanner.nextDouble();
            
            InterRectangle r = new InterRectangle(width1, length1);
            TotalEara = TotalEara + r.area();
            TotalPerimeter = TotalPerimeter + r.perimeter();
        }
		System.out.format("土地总面积为:%.1f平米\n", TotalEara);
        System.out.format("土地总周长为:%.1f米\n", TotalPerimeter);

		/*********end*********/
	}

}
