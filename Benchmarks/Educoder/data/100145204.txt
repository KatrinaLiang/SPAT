package step2;

public class Task {

	public int getNum(int n){
        /********* Begin *********/
        int thowNumber = 0;
        while (n != 1){
            if (n%2 == 0){
                n = n/2;
            }
            else {
                thowNumber++;
                n = (n-1)/2;
            }
        }
        thowNumber++;

        return thowNumber;


        /********* End *********/
    }
	
}
