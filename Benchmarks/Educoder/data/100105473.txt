package step3;

public class Reverse {
    public static StringBuffer start(String data) {
        StringBuffer ans = new StringBuffer();
        /********** Begin *********/
   		String[] a =data.split(" ");
        for(int i = 0; i < a.length; i++)
        {
            StringBuffer s = new StringBuffer(a[i]);
            s = s.reverse();
            ans.append(s+" ");
        }
        /********** End *********/
        return ans;
    }
}
