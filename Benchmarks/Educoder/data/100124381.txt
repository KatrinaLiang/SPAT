package step2;
public class Fibonacci{

	public static void main(String args[]){
	/*********begin********/
        int[] b=new int[20];
        int c=2;
        b[0]=1;
        b[1]=1;
        for(int i=2;i<20;i++){
            b[i]=b[i-1]+b[i-2];
            c+=b[i];
        }
        System.out.print("前20个元素的和为"+c);  
		

	/********end********/
	}
}
