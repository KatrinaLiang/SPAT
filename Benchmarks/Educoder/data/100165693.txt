package step3;
import step1.Date;
public class Course {
    private int code; //课程编号
    private String courseName; //课程名称
    private Date testTime;//考试日期
    private int score;//考试成绩
   /******Begin**************/
	public Course(){testTime = new Date(); courseName = "";}
	public Course(int code,String courseName, int year, int month, int day,int score){
		this.code = code;
		this.score = score;
		this.courseName = courseName;
		this.testTime = new Date(year,month,day);
	}
    public Course(int code,String courseName,Date d1, int score){
        this.code = code;
        this.courseName = courseName;
        testTime = d1;
        this.score = score;
    }
	public void set_course(int code2, int score2, String courseName2, int year, int month, int day){
		code = code2;
		score = score2;
		courseName = courseName2;
		this.testTime = new Date(year,month,day);
	}
	public int getcode(){
		return code;
	}
	public String getCourseName(){
		return courseName;
	}
	public Date getTestTime(){
		return testTime;
	}
	public int getscore(){
		return score;
	}
	public String toString(){
		return code+" "+courseName+" "+testTime+" "+score;
	}




   /******End************/

}
