#include <iostream>
using namespace std;
int main()
{
    int k,max1=0;
    scanf("%d",&k);
    int n=k/2;
    int x=k;
    int max[k];
    for(int j=0;j<k;j++)
    {max[j]=1;}
    for(int i=n;i>1;i--)
    {
        x=k;
        for(;x>=i;x=x-i)
        {
        	max[i]=max[i]*i;
        }
        if(x==1){max[i]=(max[i]/i)*(i+1);}
        if(x>1){max[i]=max[i]*x;}
        if(max[i]>max1) max1=max[i];
    }
    printf("%d",max[3]);
}
