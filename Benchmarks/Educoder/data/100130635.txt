package step3;

import java.util.Arrays;

public class Task {

	public void matrixSum(int[][] arr){
		/********** BEGIN **********/
       int i,j,sum=0;
        for(i=0;i<arr.length;i++){
            for(j=0;j<arr.length;j++){
                if(i-j==0)
        			sum+=arr[i][j];
           }
        }
        System.out.println(sum);       
    


		/********** END **********/
	
        
    }
}
