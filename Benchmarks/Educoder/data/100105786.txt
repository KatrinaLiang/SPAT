package step1;

public class Yard {
    int length;
    int width;
   public Yard (int length,int width){
       this.length = length;
       this.width = width;
   }
   double getPerimeter(){
      return (length*2+width*2);
        
   }
   double getAcreage(){
       return (length*width);
       
   }
}
