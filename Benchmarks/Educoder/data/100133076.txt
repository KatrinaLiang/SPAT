package step1;

public class Task {

	public static void birthdayCandle(int num){
        int b=0,a=0,c=0;
		for(int i=18;i<100;i++){
            for(int j=i;j<100;j++){
            	b=b+j;
            	if(b>num) break;
                if(b==num){
                	a=i;
                    c=j;
                    break;
                }
            }
            if(a!=0)	break;
        	b=0;
        }
		
		System.out.println("小明今年：" +  c  + "岁");
		System.out.println("小明从" +  a  + "岁开始每年举办派对");
	}
}
