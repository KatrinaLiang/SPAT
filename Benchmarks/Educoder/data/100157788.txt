
import java.util.Scanner;
public class Main {

	public static void main(String[] args) {
		// write your code here
        Scanner in=new Scanner(System.in);
        int x=in.nextInt();
        System.out.print(num(x));
	}
	static int num(int x){
        if(x==1) return 1;
        else if(x==2) return 2;
        else{
            return num(x-1)+num(x-2);
        }
    }
}
