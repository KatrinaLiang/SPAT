#include <stdio.h>
using namespace std;
int main()
{
	int maxn=1e6+10;
    int a[maxn];
    int n=0;
    while(scanf("%d",&a[n])!=EOF){
    	n++;
    }
    n--;
    int T;
    T=a[n];
    
    int i,j,k;
    int num=0;
    for(i=0;i<n;i++){
    	for(j=i+1;j<n;j++){
    		for(k=j+1;k<n;k++){
    			if(a[i]+a[j]+a[k]<T)
    				num++;
    		}
    	}
    }
    printf("%d",num);
}
