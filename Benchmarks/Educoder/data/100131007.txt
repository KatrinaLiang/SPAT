package step1;
import java.util.*;
public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum=0;

		for(int tmp=num1;tmp<=num2;tmp++) {
			int t=tmp;
			while(t!=0) {
			sum+=t%10;
			t=t/10;
			}
		}
		
		return(sum);

		
		
		/********** END **********/
	}
}
