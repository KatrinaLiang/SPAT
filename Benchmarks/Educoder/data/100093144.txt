package step3;

public class SeasonJudge {

    // 根据月份判断季节
	public String SeasonJudge(int month) {
        // 请在此提供实现代码
        if(month>=3 && month<=5)
            return "春季";
        else if(month>=6 && month<=8)
            return "夏季";
        else if(month>=9 && month<=11)
            return "秋季";
        else if(month>=1 && month<=2 || month==12)
            return "冬季";
        else
            return "月份错误";
        
	}
}
