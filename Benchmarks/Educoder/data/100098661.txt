package step1;

public class Yard {
	int length,width;
	
	public Yard(int length,int width) {
		this.length=length;
		this.width=width;
	}

	public float  getAcreage() {
		// TODO 自动生成的方法存根
		float C;
		C=length*width;
		return C;
	}

	public float getPerimeter() {
		// TODO 自动生成的方法存根
		float S;
	    S=2*(length+width);
		return S;
	}
}
