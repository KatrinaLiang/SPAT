package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
		
		String yuce = "";
		for (int i = 0; i < temperatures.length; i++) {
			Integer integer = temperatures[i];
			int day = 0;
			two:for (int j = i+1; j < temperatures.length; j++) {
				Integer integer1 = temperatures[j];
				if (integer1 > integer) {
					day = j-i;
					break two;
				}
			}
			yuce = yuce+ " " + day;
			
		}
		yuce = yuce.replaceFirst(" ", "");
		String[] split = yuce.split(" ");
		int[] rt = new int[split.length];
		for (int i = 0; i < rt.length; i++) {
			rt[i] = Integer.parseInt(split[i]);
		}
		
		return rt;
		/********** END **********/
	}
}
