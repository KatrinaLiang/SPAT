package step1;

public class Yard {
    
static   int length;
static int width;
    
    public Yard(int length,int  width){
    this.length=length;
    this.width=width;
    }
    public double getPerimeter(){
    return 2*(length+width);
    }
    public double getAcreage(){
    return length*width;
    }
}
