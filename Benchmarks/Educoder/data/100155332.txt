package task3;

public class Box {

    double width;
    double height;
    double depth;

    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public double volume(int boxType) {
        double volume=0;
        switch(boxType){
            case RECTANGLE:volume=width*height*depth;break;
            case CYLINDER:volume=width*width*height*3.14;break;
        }
        return volume;
    }

}
