package step3;

public class FindMaxNum {
    // 找出三个整数的最大值
    public int FindMaxNum(int a, int b, int c){
        // 请在此添加实现代码
        int result = 0;
        if(a > b) {
        	result = a;
        } else {
        	result = b;
        }
        
        if(result < c) {
        	result = c;
        }
        
        return result;
        
	}
}
