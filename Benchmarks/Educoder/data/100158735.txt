#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
    int m,n,i,j,res=1;
    scanf("%d",&m);
    i=m/3;
    j=m-i*3;
    for(n=0;n<i;n++)
    {
    	res*=3;
    }
    if(j==1)
    	res=res/3*4;
    else
    	res*=j;
    if(m!=2)
    	printf("%d",res);
    else
    	printf("1");
    return 0;
}
