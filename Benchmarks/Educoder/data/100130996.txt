package step1;
public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum=0,i=0,j=0;
        for(i=num1;i<num2+1;i++){
            for(j=i;j>0;j/=10)
            	sum+=j%10;
        }
        //System.out.println(sum);
       return sum;
		/********** END **********/
    }
}
