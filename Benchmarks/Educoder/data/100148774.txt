package step1;

public class Task {
	
	public void sort(Integer[] arr){
		/********* Begin *********/
		int i=arr.length;
for(int j=0;j<=i;j++){
	int t=0,n=0;
	if(j%2==0){
		for(int k=0;k<i-1;k++){
			if(arr[k]>arr[k+1]){
			    t=arr[k];
				arr[k]=arr[k+1];
				arr[k+1]=t;
				for(int y=0;y<arr.length;y++){
					System.out.print(arr[y]+" ");
					if(y==arr.length-1)
						System.out.println();
				}	
			}
			else 
				n++;
		}
			if(j==0&&n==i-1){
				for(int y=0;y<arr.length;y++){
					System.out.print(arr[y]+" ");
					if(y==arr.length-1)
						System.out.println();
				}
			    }
	}
	else{
	for(int k=i-1;k>0;k--){
		if(arr[k]<arr[k-1]){
			     t=arr[k];
				arr[k]=arr[k-1];
				arr[k-1]=t;
            
				for(int y=0;y<arr.length;y++){
					System.out.print(arr[y]+" ");
					if(y==arr.length-1)
						System.out.println();
				}
			}
	
		}
				
	}
		
}
		
		/********* End *********/
	}
}
