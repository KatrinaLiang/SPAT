package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum = 0;
    	for(int i= num1; i<=num2; i++){
    		char[] ch = (i+"").toCharArray();
    		int part = 0;
    		for(int j =0; j< ch.length;j++){
    			part = part + (ch[j] - '0');
    		}
    		sum+=part;
    	}
    	return sum;
		
		/********** END **********/
	}
}
