package step3;
import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
        Scanner sc=new Scanner(System.in);
        int a=sc.nextInt();
        int temp=0,b=0;
        temp=temp+a%10;
        b=a/10;
        while(b>0){
            temp=temp+b%10;
            b=b/10;
        }
        System.out.println("请输入一个五位数字：");
        System.out.println("85698的各位数字之和是："+temp);
		/******End******/
	}
}
