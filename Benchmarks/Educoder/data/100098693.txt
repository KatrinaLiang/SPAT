package step1;

public class Yard {
    
      private int  length,width; //数据成员私有
      public Yard( ) //构造方法，无返回类型，方法名和类名相同
           { length=1 ;   width= 1;    }     
      public Yard(int l,int w) //构造方法无返回类型
           { length=l;   width=w;}
    
      public double getPerimeter( ) {  return (length+width)*2; }
      public double getAcreage( )  {  return length*width;  }
    
    
    
    
}
