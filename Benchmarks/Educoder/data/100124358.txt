package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
        int sum =0;
	int[] fibonacc;
	fibonacc= new int[20];
	fibonacc[0]=fibonacc[1]=1;
	for(int i=2;i<20;i++) {
   fibonacc[i]=fibonacc[i-1]+fibonacc[i-2];
    sum+=fibonacc[i];
	}      
	sum+=fibonacc[0]+fibonacc[1];
        System.out.print("前20个元素的和为"+sum);

		

	/********end********/
	}
}
