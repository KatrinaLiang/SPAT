package step3;
import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
		Scanner sc = new Scanner(System.in);
        int a, b, c, sum = 0;
        System.out.println("请输入一个五位数字：");
        a = sc.nextInt();
        c = a;
        while (c != 0){
            b = c % 10;
            sum = sum + b;
            c = c / 10; }
        System.out.println(a + "的各位数字之和是：" + sum);
		/******End******/
	}
}
