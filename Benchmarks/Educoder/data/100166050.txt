package step1;

public class Person {
    private String name; //姓名
    private String number; //身份证号
    private Date birthday; //出生日期
    
   /************Begin*************/
	public Person() {
			birthday = new Date();
		}
		public Person(String name, String number, Date birthday) {
			this.name = name;
			this.number = number;
			this.birthday = birthday;
		}
		public Person(String name, String number, int year, int month, int day) {
			birthday = new Date(year, month, day);
			this.name = name;
			this.number = number;
		}
    public void setName(String name) {
			this.name = name;
		}
		public void setNumber(String number) {
			this.number = number;
		}
		public void setInfo(String name, String number, int year, int month, int day) {
			this.name = name;
			this.number = number;
			birthday = new Date(year, month, day);
		}
		public void setinfo(String name, String number, Date birthday) {
			this.name = name;
			this.number = number;
			this.birthday = birthday;
		}
		public String getName() {
			return name;
		}
		public String getNumber() {
			return number;
		}
	public String toString() {
			return "name="+ name +" number="+ number +" birthday:"+birthday;
		}
    
    
    
    
   /************End*************/
}
