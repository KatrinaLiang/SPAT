package step1;

public class Yard {
	public double length;
    public double width;
    public Yard(double length,double width)
    {
        this.length=length;
        this.width=width;
    }
    public double getPerimeter()
    {
        double perimeter=2*(length+width);
        return perimeter;
    }
    public double getAcreage()
    {
        double acreage=length*width;
        return acreage;
    }
}
