package step2;

import java.io.File;

public class Task {
	
	/********** BEGIN **********/
	public  void showDirStructure(File file)	{
		System.out.println("+--"+file.getName());
        show(file," ");
	}
    public static void show(File file,String interval){
        File[] files=file.listFiles();
        interval+=" ";
        for(int i=0;i<files.length;i++){
			file=files[i];
            if(!file.isDirectory())
            {
                System.out.println(interval+"--"+file.getName());
            }
            else{
                System.out.println(interval+"+--"+file.getName());
				show(file,interval+" ");
                
            }
        }
    }

	/********** END **********/
}
