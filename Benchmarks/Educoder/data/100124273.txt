package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] con = new int[20];
        int sum = 0;
        con[0]=1;
        con[1]=1;
        sum = con[1]+con[0] ;
        for(int o=2;o<con.length;o++){
        	con[o]=con[o-1]+con[o-2] ;
            sum=con[o]+sum;
        }
	System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
