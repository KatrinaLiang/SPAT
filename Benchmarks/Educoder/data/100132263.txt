package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
	int len = temperatures.length;
		int[] temInc = new int[len];
		for(int i=0;i<len;i++)
		{
			int days=0;
			boolean flag=false;
			for(int j=i+1;j<len;j++)
			{
				days++;
				if(temperatures[j]>temperatures[i])
				{
					flag=true;
					break;
				}
			}
			if(flag)
				temInc[i] = days;
			else
				temInc[i] = 0;
		}
		return temInc;
		/********** END **********/
	}
}
