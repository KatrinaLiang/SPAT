package step2;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import com.alibaba.fastjson.JSON;
import com.alibaba.fastjson.JSONArray;
import com.alibaba.fastjson.JSONObject;
import java.io.*;
public class Task {
	
	/**
	 * 使用fastjson解析数据
	 * @param hotelResult	已经为你解析的所需json数据
	 * @return
	 */
	public List<Hotel> getHotle(String hotelResult){
		/**********   Begin   **********/
        List<Hotel> hotels = new ArrayList<Hotel>();
        JSONObject hotel= JSONObject.parseObject(hotelResult);
        List<Hotel> pages = JSON.parseArray(hotel.getString("hotelPositionJSON"), Hotel.class);
        JSONArray price = hotel.getJSONArray("htllist");
        if (price != null && !price.isEmpty()) {
            for (int j = 0; j < pages.size(); j++) {
                JSONObject price1 = price.getJSONObject(j);
                if (price1 != null && !price1.isEmpty()) {
                    Hotel hotel1 = pages.get(j);
                    String hotelId = price1.getString("hotelid");
                    double price2 = price1.getDoubleValue("amount");
                    if (hotel1.getId().equals(hotelId)) {
                        hotel1.setPrice(price2);
                    }
                }
            }
        }
        hotels.addAll(pages);
        return hotels;

        /**********   End   **********/
	}
