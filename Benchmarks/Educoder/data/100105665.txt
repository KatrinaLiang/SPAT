package step1;

public class Yard {
  int length;
       int width;
       public Yard(){
    	   
       }
       public Yard(int length,int width){
    	   this.length=length;
    	   this.width=width;
       }
      public float getPerimeter(){
    	  return (2*length+2*width);
      }
      public float getAcreage(){
    	  return (length*width);
      }
}
