package step1;

public class Task {

	
	public static int getSum(int num1, int num2) {
		/********** BEGIN **********/
        if(num1 > num2){
            int temp = num1;
            num1 = num2;
            num2 = temp;
        }
        int sum = 0;
		for(; num1 <= num2; num1++) {
			int num = num1;
            while(num > 9) {
                sum += num%10;
                num /=10;
            }
            sum += num;
        }
		return sum;
		/********** END **********/
	}
}
