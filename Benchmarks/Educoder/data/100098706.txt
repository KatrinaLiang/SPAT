public class Yard {
    int length,width;
   public Yard(int a_length,int a_width){
       length=a_length;
       width=a_width;
   }
      public float getPerimeter(){
	        float temp=2*(length+width);
	        return temp;
	            }
	         public float getAcreage(){
	             return length*width;
	            }

}
