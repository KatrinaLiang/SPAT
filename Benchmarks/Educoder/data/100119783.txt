package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
		
		//在这里打印出正三角形 
		for(int i=1;i<=10;i++){
			for(int x=0;x<i;x++){
			System.out.print("*");
    }
    		System.out.println();
}
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
		for(int n=10;n>=1;n--){
			for(int m=n;m>=1;m--){
			System.out.print("*");
            }
			System.out.println();
        }
		/*****end*****/
	}
}
