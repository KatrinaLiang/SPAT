package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int m ;
		for(int i = 1;i<10;i++){
			for(int j=1;j<=i;j++){
				m = i*j;
				System.out.print(+j+"*"+i+"="+m);
				System.out.print("\t");
			}
			System.out.println();
		}
		/*****end*****/
	}
}
