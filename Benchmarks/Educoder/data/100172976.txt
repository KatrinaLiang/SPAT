package step3;

public class HelloWorld {
	public static void main(String[] args) {
		int count= 0;	//定义变量存储6的倍数出现的次数
		/*****start*****/
		int x=6;
		do {
		    count++;
			x=x+6;
	    }while(x<=100);
		/*****end*****/
		System.out.println("6的倍数出现的次数为：" + count);
	}
}
