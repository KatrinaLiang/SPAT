package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int a[] = new int[20];
        int sum=0;
        for(int i=0;i<20;i++){
        	if(i==1||i==0)
                a[i]=1;
            else
            a[i]=a[i-1]+a[i-2];
        }
        for(int i:a){
           sum+=i;
        }
        System.out.print( "前20个元素的和为"+sum);
	/********end********/
	}
}
