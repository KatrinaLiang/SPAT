package step3;
public class MyThread implements Runnable {   
    private String name;   
    private Object prev;   
    private Object self;   
    private MyThread(String name, Object prev, Object self) {   
        this.name = name;   
        this.prev = prev;   
        this.self = self;   
    }   
    public void run() {   
        int count = 5;   
        while (count > 0) {   
            synchronized (prev) {   
                synchronized (self) {   
                    System.out.print(name);   
                    count--;  
                    self.notify();   
                }   
                try {   
                    prev.wait();   
                } catch (InterruptedException e) {   
                    e.printStackTrace();   
                }   
            }   
        }  
        System.exit(0);
    }   
    public static void main(String[] args) throws Exception {   
        Object x = new Object();   
        Object y = new Object();
        Object z = new Object();   
        MyThread ta = new MyThread("E", z,x);   
        MyThread tb = new MyThread("D", x, y);
        MyThread tc = new MyThread("U", y, z);
        new Thread(ta).start();
        Thread.sleep(100); 
        new Thread(tb).start();
        Thread.sleep(100);
        new Thread(tc).start();
        Thread.sleep(100);
        }   
}
