package step2;

public class ForPractice2 {
	public static void main(String[] args) {
        for(int i=0;i<10;i++){
            for(int s=0;s<=i;s++){
                System.out.print("*");
            }
            System.out.println();
        }
       
		/*****start*****/
		
		//在这里打印出正三角形 
		
		
		
		
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
		for(int i=0;i<10;i++){
            for(int s=10;s>i;s--){
                System.out.print("*");
            }
            System.out.println();
        }
		
		
		
		
		/*****end*****/
	}
}
