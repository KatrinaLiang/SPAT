	public Map<String, Integer> getMap(String str){

            map.put("will",13);
            map.put("arguments",3);

        }
        else{
            map.put("through",7);
            map.put("the",15);

            map.put("world",41);

            map.put("show",47);
            map.put("their",15);
            map.put("wants",58);
            map.put("let",33);
            map.put("we",0);
        }




		return map;
	}
