package step1;

public class Test {
	public static void main(String[] args) {
		//创建Dog对象
		//设置Dog对象的属性
		Dog dog = new Dog();

		// 输出小狗的属性
		System.out.println("名字：" + dog.name + "，毛色：" + dog.color + "，品种："
				+ dog.variety);

		// 调用方法
		dog.eat();
		dog.run();
	}
}

//在这里定义Dog类
class Dog {
	String name = "五花肉";
	String color = "棕色";
	String variety = "阿拉斯加";
	public void eat(){
		System.out.println("啃骨头");
	}
	public void run(){
		System.out.println("叼着骨头跑");
	}
}
