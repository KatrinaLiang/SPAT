public class Rectangle{
	private double width;
	private double length;
	 public double sum;
	/*********begin*********/
     Rectangle(double length,double width)
		 {
			 this.length=20;
			 this.width=10;
		 }
     Rectangle(){
         
         this.length=6;
         this.width=5;
     }
	public double  perimeter(){
        sum=(this.length+this.width)*2;
        return sum;
    } 
    public double area(){
        sum=this.length*this.width;
        return sum;
    }
    
	/*********end*********/
