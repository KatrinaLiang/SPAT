class MyRunnable extends Thread {
    private int num;
    public MyRunnable(int num){
     	this.num = num;   
    }
 	public void run(){
        int result = 1;
     	for(;this.num>0;this.num--){
         	result  *= num;   
        }
        try{
            sleep(250);
        }catch(InterruptedException e){
         	   
        }
        System.out.println(result);
    }
}
