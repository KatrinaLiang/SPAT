package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int s = 0;
		for(int i=1;i<=9;i++){
			for(int j=1;j<=i;j++){
				s=i*j;
				System.out.print(j + "*" + i + "=" + s +"\t");
			}
			System.out.println();
		}
		
		
		
		
		
		/*****end*****/
	}
}
