package step2;
// 利用 Runnable 接口实现
/********** Begin *********/
public class MyThread implements Runnable{
/********** End *********/
	private String name;
    public MyThread(String name) {
        /********** Begin *********/
     this.name=name;
		/********** End *********/
    }  
    public void run() {
        /********** Begin *********/
   System.out.println("运行:"+name);
		/********** End *********/
    }
public static void main(String[]args){
    new Thread(new MyThread("A")).start();
    new Thread(new MyThread("B")).start();
}
}
