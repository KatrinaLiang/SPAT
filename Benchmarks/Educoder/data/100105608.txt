package step1;

public class Yard {
    int length;
    int width;
    public Yard()
    {
    }
    public Yard(int l,int w)
    {
        length=l;
        width=w;
    }
    double getPerimeter()
    {
        return 2*length+2*width;
    }
    double getAcreage()
    {
        return length*width;
    }

    public static void mian(String[ ] args)
    {
        Yard yard=new Yard(3,4);
        System.out.println(yard.getPerimeter()+' '+yard.getAcreage());
    }
}
