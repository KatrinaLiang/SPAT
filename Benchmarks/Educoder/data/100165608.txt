package step3;
import step1.Date;
public class Course {
    private int code; //课程编号
    private String courseName; //课程名称
    private Date testTime;//考试日期
    private int score;//考试成绩
   /******Begin**************/
	public Course(){testTime=new Date();}
	public Course(int code, String name, int y,int m,int d,int score){
		this.code=code;
		this.courseName=name;
		this.testTime=new Date(y,m,d);
		this.score=score;
	}
	public Course(int code, String name, Date time,int score){
		this.code=code;
		this.courseName=name;
		this.testTime=time;
		this.score=score;
	}

	public int getcode(){return code;}
	public String getCourseName(){return courseName;}
	public Date getTestTime(){return testTime;}
	public int getscore(){return score;}


	public String toString() {
		if(courseName==null)
			return code+"  "+testTime+" "+score;
		else return code+" "+courseName+" "+testTime+" "+score;
	}



   /******End************/

}
