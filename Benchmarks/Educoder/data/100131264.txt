package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int s=0;
		for(;num1<=num2;num1++){
            if(1<=num1&&num1<=99){
             s+=num1/10+num1%10;
            }else if(num1<999){
                s+=num1%10+num1/10%10+num1/100;
            }
        }
		return s;
		
		/********** END **********/
	}
}
