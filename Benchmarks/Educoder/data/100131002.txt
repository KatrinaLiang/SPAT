package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum=0,t,i,a,b,c;
        if(num1>num2)
        {t=num1;num1=num2;num2=t;}
        for(i=num1;i<=num2;i++)
        {a=i/100;b=(i-a*100)/10;c=i%10;
         sum=sum+a+b+c;}
		return sum;
		/********** END **********/
	}
}
