package step3;
import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
			Scanner sc=new Scanner(System.in);
		System.out.println("请输入一个五位数字：");
		int r=sc.nextInt();
		int s=0;
		int n=r;
		while(n>0){
			int m=n%10;
			s=s+m;
			n=n/10;
		}
		System.out.println(r+"的各位数字之和是："+s);
		/******End******/
	}
}
