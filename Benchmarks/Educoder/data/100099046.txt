package step2;

import java.util.Scanner;

public class SelfIntroduction {
    public void setName(String name) {
        this.name = name;
    }
    public void setAge(int age) {
        this.age = age;
    }
    public void setEducation(String education) {
        this.education = education;
    }
    public void setPosition(String position) {
        this.position = position;
    }
    public void introduction() {
        System.out.println("大家好！我是" + name + "，我今年" + age + "岁，" 
                           + education + "学历，" + "目前职位是" + position);
    }
    public void setInfo(String name,int age,String education,String position){
       this.name=name;
       this.age=age;
       this.education=education;
       this.position=position;
    }
    private String name;
    private int age;
    private String education;
    private String position;

    public static void main(String[] args) {
		Scanner scanner = new Scanner(System.in);
        String name = scanner.next();
		int age = scanner.nextInt();
		String education = scanner.next();
		String position = scanner.next();

        /********** Begin *********/
        SelfIntroduction person=new SelfIntroduction();
        person.setInfo(name,age,education,position);
        person.introduction();
        
        /********** End *********/
    }
}
