package Step2;

import java.util.Scanner;

public class RankingBySwitch {
	public static void main(String args[]){
		/********** Begin *********/

		//输入课程的分数
        Scanner i = new Scanner(System.in);
        int j=i.nextInt();
        switch(j/10){
            case 10:
                System.out.println("成绩为"+j+"分，评定为优秀");
                break;
            case 9:
                System.out.println("成绩为"+j+"分，评定为优秀");
                break;
            case 8:
                System.out.println("成绩为"+j+"分，评定为良好");
                break;
            case 7:
                System.out.println("成绩为"+j+"分，评定为中等");
                break;
            case 6:
                System.out.println("成绩为"+j+"分，评定为及格");
                break;
            default:
                System.out.println("成绩为"+j+"分，评定为不及格");
                break;
        }
        /*90~100分：优秀
          80~89分：良好
          70~79分：中等
          60~69分：及格
          0~59分：不及格*/
        
		
		//switch-case语句

		/********** End *********/
	}
}
