package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
        int n,m;
        n=matrix.length;
        m=matrix[0].length;
		int[][] A= new int[n][m];
		for(int i=0;i<n;i++)
        {
           for(int j=0;j<m;j++)
           {
               A[n-j-1][n-i-1]=matrix[i][n-1-j];
           } 
        }
       	for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                matrix[i][j]=A[i][j];
            }
        }
		
		/********* End *********/
	}
}
