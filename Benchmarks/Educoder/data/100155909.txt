package step2;

//编写一个方法，计算并返回1~50的偶数和
public class TestJava23_4 {
	public static void main(String[] args) {
		/*******Start*********/
		//此处调用方法SumOfEven，并将1~50的偶数和输出
		
        System.out.println("1-50之间偶数之和是："+TestJava23_4.SumOfEven());
       
		
		
		
		/*******End*********/
		
	}
	/*******Start*********/
	//此处定义一个方法，名为，实现计算并返回1~50的偶数和的功能
	public static int SumOfEven(){
		int s=0;
		for(int a=1;a<=50;a++){
			if(a%2==0){
			s+=a;
			}
		}
		return s;
	}
	
	
	/*******End*********/

}
