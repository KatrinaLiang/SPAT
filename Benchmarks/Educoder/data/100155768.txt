package task6;

public class ScaleBox extends Box        {  //继承Box类

    //TODO 定义私有变量unit
    private String unit; 

    public ScaleBox(int boxType) {
        super(boxType);
    }
	public void setWidth(double n)
    {
    	super.setWidth(n);
    }
    public void setHeight(double n)
    {
    	super.setHeight(n);
    }
    public void setDepth(double n)
    {
    	super.setDepth(n);
    }
    //实现 getUnit、setUnit方法
	public void setUnit(String unit)
    {
    	this.unit=unit;
    }
	public String getUnit()
    {
    	return unit;
    }
	//实现printVolumn()方法，在该方法中调用父类的方法计算体积 最后输出体积
	public void printVolumn()
    {
    	System.out.println("the volumn is:"+super.volume()+unit);
    }
}
