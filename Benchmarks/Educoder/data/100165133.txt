package task4;

public class Box {

    private double width;
    private double height;
    private double depth;

  /********* Begin *********/        
	double  getWidth(){
		return this.width;
	}
	void setWidth(double width){
		this.width = width;
	}
	double  getHeight(){
		return this.height;
	}
	void setHeight(double height){
		this.height = height;
	}
	double  getDepth(){
		return this.depth;
	}
	void setDepth(double depth){
		this.depth = depth;
	}



  /********* End *********/        

}
