package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
		
		int cum = n;
        int num = 0;
        while(true){
            if(cum==0){
                break;
            }
            if (cum == 1){
                num++;
                break;
            }
            if (cum % 2 != 0){
                cum--;
                num++;
            }else{
                cum = cum >> 1;
            }

            

        }
        return num;
		
		
		
		/********* End *********/
	}
	
}
