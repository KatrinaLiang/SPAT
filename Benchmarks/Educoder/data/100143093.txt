import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

public class Task {
	
	public void task() throws IOException{
		FileReader a=new FileReader("src/step4/input/input.txt");			
	FileReader b=new FileReader("src/step4/input/input.jpg");
        FileWriter c=new FileWriter("src/step4/output/output.txt");
        FileWriter d=new FileWriter("src/step4/output/output.jpg");
        char[] e=new char[1024]; 
        char[] f=new char[1024]; 
        int l=0;
        int m=0;
        while((l=a.read(e))!=-1){
        
        c.write(e,0,l);
        }
        while((m=a.read(f))!=-1){
        
        d.write(f,0,m);
        }
        a.close();
        b.close();
        c.close();
        d.close();
	}
}
