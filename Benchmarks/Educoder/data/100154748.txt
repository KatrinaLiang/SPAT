package step2;
import java.util.Scanner;

public class TestJava19_4 {

	static final Scanner sc=new Scanner(System.in);
	public void print(){
		System.out.println("请输入行数：");
		int m=sc.nextInt();
		for(int a=1;a<=m;a++){
            for(int b=1;b<=a;b++){
                System.out.print("*");
            }
            System.out.println();
        }
	}

}
