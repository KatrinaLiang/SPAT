package step2;

public class Task {

	public int getNum(int total){
		/********* Begin *********/
        int sheng=0;
		while(true){
    			if(total%2 == 0){
    				total /= 2;
    			}else{
    				total = (total-1)/2;
    				sheng +=1;
    			}
    			if(total == 0){
    				break;
    			}
    		}
    		return(sheng);
		/********* End *********/
	}
	
}
