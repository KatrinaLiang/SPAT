public class HelloWorld {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		System.out.println("请给定一个自然数N：");

		int sum = 1;int i;		
		/*****start*****/
		 for(i=1;i<=N;i++)
        {
            sum=sum*i;

		System.out.println("自然数N的阶乘为" + sum);
	}
}
