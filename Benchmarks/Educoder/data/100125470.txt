package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
    InterRectangle(){
       
    }
	InterRectangle(double width,double length){
        this.width=width;
        this.length=length;
    }
    double getPerimeter(){
        return (width+length)*2;
    }
    double getArea(){
        return width*length;
    }
	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
		double width,length;
        double sum_area=0,sum_perimeter=0;
        for(int i=0;i<total;i++){           
            width=scanner.nextDouble();
            length=scanner.nextDouble();
            InterRectangle inter=new InterRectangle(width,length);
            sum_area+=inter.getArea();
            sum_perimeter+=inter.getPerimeter();
        }
        System.out.println("总共有"+total+"块土地");
        System.out.println("土地总面积为:"+sum_area+"平米");
        System.out.println("土地总周长为:"+sum_perimeter+"米");
		/*********end*********/
	}

}
