package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int i1=0,i2=0,i3=0,sum=0;
		for(int i=num1;i<=num2;i++)
        {
        	i1=i%10;
            i2=i/10%10;
            i3=i/100%10;
            sum+=i1+i2+i3;
        }
		return sum;
		
		/********** END **********/
	}
}
