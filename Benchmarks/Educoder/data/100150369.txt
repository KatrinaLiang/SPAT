package step2;
import java.util.Scanner;
public class AreaOfCircle {
	public static void main(String[] args) {
		/******Begin******/
        Scanner input=new Scanner(System.in);
        System.out.println("请输入半径r的值：");

        float s=PI*r*r;

		/******End******/
	}
}
