package task3;

public class Box {

    double width;
    double height;
    double depth;

    /**
     * 代表长方体
     */
    public static final int RECTANGLE = 1;

    /**
     * 圆柱体
     */
    public static final int CYLINDER = 2;

    public double volume(int boxType) {
        switch(boxType)
        {
            case RECTANGLE:return width*height*depth;
            case CYLINDER:return width*width*height*3.14;
            default:throw new IllegalArgumentException("不支持的几何类型");
        }                           
        //TODO write your code here
    }

}
