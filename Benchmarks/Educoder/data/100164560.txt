public class Equation {
		double a, b, c; //一元二次方程的三个系数
		public Equation(){}
		public Equation(double a,double b,double c){
            this.a = a;
		this.b = b;
		this.c = c;
			//YOUR CODE HERE
		}
		/*************Begin*******************/
	    //YOUR CODE HERE
public double deta() {
		return b*b-4*a*c;
	}
	public double getX1() {
		double d ;
		d = b*b-4*a*c;
		return (-b+(Math.sqrt(d)))/(2*a);
	}
	public double getX2() {
		double d ;
		d = b*b-4*a*c;
		return (-b-(Math.sqrt(d)))/(2*a);
	}
