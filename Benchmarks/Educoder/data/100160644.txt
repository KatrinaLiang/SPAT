package step3;
import java.io.File;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;
import org.jsoup.Jsoup;
import org.jsoup.nodes.Document;
import org.jsoup.nodes.Element;
import org.jsoup.select.Elements;
public class Task {
    
    //获取“http://hotels.ctrip.com/”Docment对象
	public Document getDoc1(String url) throws IOException{
        File in =new File("./backups/hotel.ctrip.com.txt");
        Document doc=Jsoup.parse(in,"utf-8",url);
		return doc;
	}

	//获取“http://hotels.ctrip.com/domestic-city-hotel.html”的Docment对象
	public Document getDoc2(String url) throws IOException{
	    File in =new File("./backups/hotels.ctrip.com_domestic-city-hotel.txt");
	    Document doc=Jsoup.parse(in,"utf-8",url);
		return doc;
	}

	//获取所有链接
	public List<String> getLinks(Document doc){
		//a$http://www.ctrip.com(携程旅行网)
		List list=new ArrayList<String>();
		Elements e=doc.select("a[href]");
		for(Element el:e) {
		String link=el.attr("abs:href");
		String wb=el.text();
		String jg="a$"+link+"("+wb+")";
		list.add(jg);
		}
		return list;
		
	}
	
	//获取图片
	public List<String> getMedia(Document doc){
		List list=new ArrayList<String>();
		Elements p = doc.select("img");
		for(Element el:p) {
		String 	a=el.attr("abs:src");
		String jg="img$"+a;
		list.add(jg);}
        return list;
	}
	
	//获取link[href]链接
	public List<String> getImports(Document doc){
		List list=new ArrayList<String>();
		Elements h=doc.select("link[href]");
		for(Element el:h) {
		String hl=el.attr("abs:href");
		String l=el.attr("rel");
		String jg="link$"+hl+"("+l+")";
		list.add(jg);}
        return list;
	}
	
}
