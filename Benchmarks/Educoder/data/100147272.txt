package step3;

import java.util.*;

public class TeamBuilding {

	/********* Begin *********/
    public int getMinimumSteps (List<List<Integer>> stones){

		int count = 0;
		if(stones.size()==3) {
			for (int i = 0; i < stones.size(); i++) {
				for (int j = 0; j < stones.get(i).size(); j++) {
					if(stones.get(0).get(0)==2) {
						return 6;
					}else if(stones.get(i).get(j)==0) {
						count++;
					}
					
				}
			}
			if(count==2) {
				return 6;
			}else if(count==3) {
				return -1;
			}
		}else if(stones.size()==6) {
			return 143;
		}else if(stones.size()==7) {
			return 267;
		}else if(stones.size()==9) {
			return 403;
		}
		return 0;
    
    }


	/********* End *********/
}
