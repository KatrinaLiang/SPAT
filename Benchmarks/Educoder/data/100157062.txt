package step2;
import java.util.Scanner;

//定义一个数组，数组一共有5个元素，接收用户输入的5个数，找出最大值和最小值，求总和、平均值并输出。
public class Array {

	public static void main(String[] args) {
			Scanner sc=new Scanner(System.in);
		int[] num=new int[5];	//定义数组num，存放5个整数
		int k,max,min,sum=0;	//max、min、sum分别存放最大值、最小值、总和
		float avg=0.0f;	
		
		
		 for(k=1;k<=num.length;k++){
				System.out.println("请输入第"+k+"个整数：");
		            num[k-1]=sc.nextInt();
		            sum=sum+num[k-1];
		        }
					//此处，求数组num中最大值、最小值、总和、平均值
				if(num[0]>=num[1]){
		            max=num[0];
		        }else if(num[1]>=num[2]) {
		        	max=num[1];
		        }else if(num[2]>=num[3]) {
		        	max=num[2];
		        }else if(num[3]>=num[4]){
		        	max=num[3];
		        }else {
		        	max=num[4];
		        }
				
				
				
				if(num[0]>=num[1]){
		            min=num[1];
		        }else if(num[1]>=num[2]) {
		        	min=num[2];
		        }else if(num[2]>=num[3]) {
		        	min=num[3];
		        }else if(num[3]>=num[4]){
		        	min=num[4];
		        	}else {
		        		min=num[0];
		        	}
				avg=sum/num.length;	
	
				for(k=1;k<num.length;k++) {
					System.out.print(num[k-1]+"，");
				
					
				}
				System.out.println(num[(num.length)-1]+"中的最大值是"+max+"，最小值是："+min+"，总分是："+sum+"，平均值是："+avg);
	}
}
