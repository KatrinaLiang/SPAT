package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum = 0;
        for(int i = num1; i <= num2; i++){
        	sum += singleSum(i);
        }
		return sum;
		
		/********** END **********/
	}
    private int singleSum(int n){
    	String num = String.valueOf(n);
        int sum = 0;
        for(int i = 0; i < num.length(); i++){
        	sum += num.charAt(i) - '0';
        }
        return sum;
    }
}
