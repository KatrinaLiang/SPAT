package step3;

import java.io.File;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

public class DeleteFiles {
    public static void start() throws IOException {
        StringBuilder data = new StringBuilder("");
        /********** Begin *********/
        //读取readme.txt文件
        
		File file2 = new File("src/step3/readme.txt");
		FileReader fileReader = new FileReader(file2);
        File file1 = new File("src/step3/readmeCopy.txt");
		FileWriter fileWriter = new FileWriter(file1, true);
        int read = fileReader.read();
		while (read != -1) {   //循环读取，直到read的值为-1，表示文件读到末尾
    	fileWriter.write((char)read);   //输出读取的数据，使用(char)方法对read的值进行类型转换
    	read = fileReader.read();   //读取下一个字符
        
}

        //写入readmeCopy.txt文件
		fileReader.close();
		fileWriter.close();
        //删除readme.txt文件
		file2.delete();

        /********** End *********/
    }
}
