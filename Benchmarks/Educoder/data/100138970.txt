package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
	MyThread x=new MyThread(num);
        try{
            x.join();}
        catch(Exception e){
            e.printStackTrace();
        }
	System.out.println("子线程计算结果为："+x.run(num));
}
}
class MyThread  extends Thread{
	int sum,d;
    public MyThread(int num){
        d=num;
    }
	public int run(int num){
		if(num==1){
			sum=num;
		}
		if(num==2){
			sum=1;
		}
		if(num>2){
			sum=run(num-1)+run(num-2);
		}
		return sum;
	}
	
	
   
}
