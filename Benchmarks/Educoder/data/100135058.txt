class Person {
	/********* begin *********/
    String name;
    int age;
	/********* end *********/
}

class Student extends Person {
	/********* begin *********/
    String name;
    int age;
    String school;
    public void f2(String name,int age,String school){
        super.name=name;
        super.age=age;
        this.school=school;
        System.out.println("姓名："+super.name+"，年龄："+super.age+"，学校："+this.school);
    }
	/********* end *********/
}
