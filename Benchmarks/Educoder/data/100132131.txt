package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temps){
		/********** BEGIN **********/
       
        int d=temps.length;
        int[] arr=new int[d];
		for(int i=0;i<temps.length;i++)
        { int a=0;
            for(int b=i;b<temps.length;b++)
        {
            if(temps[i]<temps[b])
            {
            a=b-i;
            break;
            }
         }
         arr[i]=a;
        }
       return arr;
		

		
		/********** END **********/
	}
}
