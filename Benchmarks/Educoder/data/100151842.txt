		Scanner sc=new Scanner(System.in);
		int a,b,c;
		System.out.println("请输入第一个整数：");
		a=sc.nextInt();
		System.out.println("请输入第二个整数：");
		b=sc.nextInt();
		System.out.println("请输入第三个整数：");
		c=sc.nextInt();
		int max=a;
		if(b>max&&b>c)
		{
			max=b;
		}
		else if(c>max&&c>b)
		{
			max=c;
		}
		System.out.println(a+"，"+b+"，"+c+"这三个整数中，"+max+"是最大值");
	}

}
