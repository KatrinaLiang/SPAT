package step4;

public class DayCount {
	public void dayCount(int year, int month, int day) {
		/********** Begin *********/
int flag,a=30,b=31,c=28,s=0;
        if(year%4==0&&year%100!=0||year%400==0)
            flag=1;
        else
            flag=0;
        switch(month)
        {
            case 1:s=0;break;
            case 2:s=b;break;
            case 3:s=c+flag+b;break;
            case 4:s=c+flag+b*2;break;
            case 5:s=c+flag+b*2+a;break;
            case 6:s=c+flag+b*3+a;break;
            case 7:s=c+flag+b*3+a*2;break;
            case 8:s=c+flag+b*4+a*2;break;
            case 9:s=c+flag+b*5+a*2;break;
            case 10:s=c+flag+b*5+a*3;break;
            case 11:s=c+flag+b*6+a*3;break;
            case 12:s=c+flag+b*6+a*4;break;
            default:
                break;
        }
        s=s+day;
        System.out.print(s);


		/********** End *********/
	}
}
