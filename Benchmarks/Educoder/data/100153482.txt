package step1;
import java.util.Scanner;

public class TestJava16_7 {
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
	    /*********begin***********/
		System.out.println("请输入一个整数：");
        int n=sc.nextInt();
        int i=1;
        float jc=1;
        while (i<=n)
        {
            jc=jc*i;
            i++;
        }
		System.out.println(n+"的阶乘为："+jc);
		/*********end*************/
	}

}
