package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		int[][] tmpmatrix=new int[matrix.length][matrix[0].length];
		for (int i = 0; i < matrix.length; i++) {
			int []tmp=new int[matrix[0].length];
			for (int j = 0; j < matrix[i].length; j++)
				tmp[j]=matrix[i][j];
			tmpmatrix[i]=tmp;
		}

		for (int i = 0, k = 0; i < tmpmatrix.length && k < tmpmatrix[i].length; i++, k++) {
			int[] v1=new int[matrix[0].length];
			for (int j = tmpmatrix.length - 1,m=0; j >= 0&&m<v1.length; j--,m++) {
				v1[m]=tmpmatrix[j][k];
			}
			//matrix.set(i, v1);
			matrix[i]=v1;
		}
		

		
		
		
		/********* End *********/
	}
}
