/***begin your code here***/
public class MyRunnable extends Thread implements Runnable{
	private int num;
    
    public MyRunnable(int num){
    	this.num=num;
    }
	
    public void run(){
    	try{
        Thread.sleep(500);
        }catch(InterruptedException e){e.printStackTrace();}
        int result=1;
        for(int i=1; i<=num; ++i)
    		result*=i;
    	System.out.println(result);
    }
}
