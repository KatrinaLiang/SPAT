package step1;

public class Task {
	public static void main(String[] args) throws Exception {
		/********* Begin *********/
		//在这里创建线程， 开启线程
	MyThread s1=new MyThread();
        	MyThread s2=new MyThread();
        	MyThread s3=new MyThread();
		s1.setPriority(10);
		s3.setPriority(1);
        s2.setPriority(2);
        s1.threadName="AA";
		s2.threadName="BB";
        s3.threadName="CC";
        s1.start();
        s2.start();
        s3.start();
		/********* End *********/
	}
}

class MyThread extends Thread {
	/********* Begin *********/
	
	
	String threadName;
	

	public void run() {
	
		int count = 5;
		
		while(count > 0){
			
			System.out.println("Java Thread" + this.threadName);
            try{
              Thread.sleep(1000);
            }catch(InterruptedException e){}
			count--;
		}
	}

	/********* End *********/
}
