package step3;
public class NarcissisticNumber {

    //判断一个三位数是否是水仙花数
    public boolean NarcissisticNumber(int n){
        // 请在此添加实现代码
        int  g  = n%10;
        int  s  = n/10%10;
        int  b  = n/100;
        if(g*g*g+s*s*s+b*b*b==n)
            return true;
        else
            return false;
          
    }
}
