package Step1;
import java.lang.*;
import java.util.*;

public class UnicodeDemo {
	  public String Encoding(String s) {
	        //将读入的字符串变量转换为字符数组
		  	char []b=s.toCharArray();
	         //将字符数组的每一项都转换为16进制整型字符串，不足4位的要补0
	         for(int i=0;i<b.length;i++) {
	        	 String unicode = Integer.toHexString(b[i]);
	        	 if(unicode.length()<=2) {
	        		 unicode = "00" + unicode;
	        	 }
	         unicode = "\\u"+unicode;
	         //输出转换后的Unicode编码
	         System.out.print(unicode); 
	         }
          	 System.out.println();
	         return s;
	     }
}
