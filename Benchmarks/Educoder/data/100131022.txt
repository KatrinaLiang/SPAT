package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
				int sum=0;
		for(int i=num1;i<=num2;i++) {
			String cur=i+"";
			int count=cur.length();
			for(int j=0;j<count;j++) {
				sum+=Integer.parseInt(cur.charAt(j)+"");
			}
		}
		return sum;
		
		
		/********** END **********/
	}
}
