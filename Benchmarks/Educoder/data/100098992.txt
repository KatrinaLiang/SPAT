package step1;

public class Yard {
    private double length;
    private double width;
    
    public Yard( double newlength,double newwidth ){
        this.length = newlength;
        this.width = newwidth;
    }
    public double getPerimeter( ){
         return 2*( length+width );
    } 
    public double getAcreage( ){
          return  length*width;
    }  
}
