public class SelectionSort {

    /**
     * 选择排序
     *
     * @param arr
     */
    public static void sort(int arr[]) {
        /********** Begin *********/
    	int i,j=0;
		for(i=0;i<arr.length-1 ;i++){
         	for(j=i+1;j<arr.length;j++)
                if(arr[j]<arr[i])
                {
          			int temp = arr[j] ;
                	arr[j] = arr[i] ;
                	arr[i] = temp ;
                }
            print(arr) ;
        }
        /********** End *********/
    }

    private static void print(int arr[]) {
        for (int i = 0; i < arr.length; i++) {
            System.out.print(arr[i] + " ");
        }
        System.out.println();
    }

}
