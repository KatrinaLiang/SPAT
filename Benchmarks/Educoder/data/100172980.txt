package step3;

public class HelloWorld {
	public static void main(String[] args) {
		int count= 0;	//定义变量存储6的倍数出现的次数
		/*****start*****/
		int i;
		for(i=1;i<=100;i++){
			if(i%6==0){
				count=count+1;
			}
		}

		
		

		
		/*****end*****/
		System.out.println("6的倍数出现的次数为：" + count);
	}
}
