package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
        int i=1,j=1,result;
		for(i=1;i<=9;i++){
        for(j=1;j<=i;j++){
            result =i*j;
        System.out.print(j+"*"+i+"="+result+"\t");
        }
            System.out.print("\n");
        }

		/*****end*****/
	}
}
