package step3;
import java.util.Arrays;

public class Task {

	public void matrixSum(int[][] arr){
		/********** BEGIN **********/
		//实现求矩阵对角线的功能。
		int i,j,sum=0;
		for(i=0;i<arr.length;i++) {
			for(j=0;j<arr[i].length;j++) {
				if(i==j) {
					sum=sum+arr[i][j];
				}
			}
		}
		System.out.println(sum);
		/********** END **********/
	}
	
	
}
