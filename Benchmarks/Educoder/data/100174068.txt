package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
	int q=1;
	for(int m=1;m<10;m++){
		for(int n=1;n<=q;n++){
			System.out.print(n+"*"+q+"="+n*q+"\t");
		}
		q++;
		System.out.print("\n");
	}

		
		
		
		
		
		/*****end*****/
	}
}
