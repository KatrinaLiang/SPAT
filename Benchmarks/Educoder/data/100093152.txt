package step3;

public class SeasonJudge {

 
	public String SeasonJudge(int month) {
        // 请在此提供实现代码
        String year;
        switch(month){
            case 3:
            case 4:
            case 5:
                year = "春季";
                break;
            case 6:
            case 7:
            case 8:
                year = "夏季";
                break;
            case 9:
            case 10:
            case 11:
                year = "秋季";
                break;
            case 12:
            case 1:
            case 2:
                year = "冬季";
                break;
            default:
                year = "月份错误";
        }
        return year;
	}
}
