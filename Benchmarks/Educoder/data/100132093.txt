package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
			int i,j;
       int counter[]=new int[temperatures.length],sum;
        for(i=0;i<temperatures.length;i++)
        {
          sum=0;
            for(j=i;j<temperatures.length;j++)
            {
                if(temperatures[j]>temperatures[i])
                {
                    sum=1;
                   break;
                 }
                counter[i]++;
             }
            if(sum==0)
            counter[i]=0;
        }
           
                return counter;
   
		/********** END **********/
	}
}
