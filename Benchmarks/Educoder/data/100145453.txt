package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
        int sum=0;
        while(n != 0){
            if(n%2 == 0){
                n/=2;
            }else{
                n--;
                sum++;
            }
        }
		return sum;
		
		
		
		
		/********* End *********/
	}
	
}
