public class ScaleBox extends Box{  //继承Box类
    
	private String unit;
    //TODO 定义私有变量unit
    
    public ScaleBox(int boxType) {

    }

    public void setUnit(String unit){this.unit=unit;}
    public String getUnit(){return unit;}//实现 getUnit、setUnit方法

	public void printVolumn(){

    }
}
