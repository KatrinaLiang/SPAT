class MyThread extends Thread {
	private int a;
	public MyThread(int a){
		this.a=a;
	}
	public void run(){
		try{
		//new Thread(new MyRunnable()).join();
		Thread.sleep(1000);}
		catch(InterruptedException e){
			e.printStackTrace();
		}

		int result=1;
		for(int i=1;i<=a;i++)
		result*=i;
		System.out.println(result);
	}
}
