package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
        double sum=0;
		/********** BEGIN **********/
		if(n%2==0){
            for(int i=2;i<=n;){
                sum=sum+(double)3/i;
                i=i+2;
            }
        }else{
            for(int i=1;i<=n;){
                sum=sum+(double)4/i;
                i=i+2;
            }
        }
        System.out.println(sum);
		
		
		
		
		
		/********** END **********/
	}
}
