package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum =0;
		for(int i = num1;i<=num2;i++){
            int shang =i;
            do{
                
                int temp = shang /10;
                int yushu = shang%10; 
                sum +=yushu;
                shang = temp;
            }while(shang!=0);
            //sum+=yushu;
        }
		return sum;
		/********** END **********/
	}
}
