package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int sum = 0;
		
		for(int t = num1;t<=num2;t++)
		{
			int flag = t;
			while(flag!=0)
			{
				sum+=flag%10;
				flag/=10;
			}
		}
		return sum;
		
		
		/********** END **********/
	}
}
