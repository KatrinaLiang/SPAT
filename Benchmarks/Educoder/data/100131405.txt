package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int num = 0;
		for(int i = num1 ; i <= num2; ++i)
        {
            int k = i;
            while(k != 0)
            {
                num += k % 10;
                k /= 10;
            }
        }
		return num;
		
		/********** END **********/
	}
}
