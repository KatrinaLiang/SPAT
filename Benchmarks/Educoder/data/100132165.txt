package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
		int n=temperatures.length;
		int[] ans=new int[n];
		for(int i=0;i<n;i++) {
			for(int j=i+1;j<n;j++) {
				if(j==n-1) {
					ans[i]=0;
				}
				if(temperatures[j]>temperatures[i]) {
					ans[i]=j-i;
					break;
				}
				
			}
		}
		

		return ans;
		/********** END **********/
	}
}
