package step2;

import java.util.Scanner;

public class SelfIntroduction {
    public SelfIntroduction(String name,int age,String education,String position){
        this.name = name;
		this.age = age;
		this.education = education;
		this.position = position;
	}
    
    public void introduction() {
        System.out.println("大家好！我是" + name + "，我今年" + age + "岁，" 
                           + education + "学历，" + "目前职位是" + position);
    }
    private String name;
    private int age;
    private String education;
    private String position;

    public static void main(String[] args) {
		Scanner scanner = new Scanner(System.in);
        String name = scanner.next();
		int age = scanner.nextInt();
		String education = scanner.next();
		String position = scanner.next();

        /********** Begin *********/
        
        SelfIntroduction person= new SelfIntroduction(name,age,education,position);
        person.introduction();
        
        /********** End *********/
    }
}
