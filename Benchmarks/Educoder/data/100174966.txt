			scores[i] = sc.nextInt();
		}
		/********** Begin **********/
		//在这里计算数组scores的平均值和最大值

		int max=scores[0];
		for(int a=1;a<scores.length;a++){
			if(scores[a]>max){
				max=scores[a];
			}
		
		}
		int h=0,j=0;
		for(int b=0;b<scores.length;b++){
		    
			h=scores[b]+h;
			

		}
		
		
		
		
		
		
		System.out.println("平均值："+(double)h/scores.length );
		System.out.println("最大值：" +max );
		/********** End **********/
	}
}
