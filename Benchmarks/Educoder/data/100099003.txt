package step1;

public class Yard {
    int length;
    int width;
    public Yard(int a,int b)
    {
    length=a;
    width=b;
    }
    public double getPerimeter()
    {
    return (double)(2*(length+width));
    }
    public double getAcreage()
    {
    return  (double)(length*width);
    }
}
