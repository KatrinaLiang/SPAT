public class MyThread extends Thread{
    private int a;
    
    public MyThread(int a){
        this.a=a;
    }
    public void run(){
        int result = 1;
        for(int i=1;i<=a;i++){
            result= result * i;
        }
        try{
        Thread.currentThread().sleep(200);
        }catch(Exception e){}
        System.out.println(result);
    }
}
