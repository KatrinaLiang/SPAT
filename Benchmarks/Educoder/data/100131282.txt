package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum=0;
        for(int i=0 ; i<=(num2-num1) ; i++){
        	int unit = 0;
            int decade = 0;
            int hundred = 0;
            unit = (num1 + i)%10;
            decade = (num1 + i)/10%10;
            hundred = (num1 + i)/100%10;
            sum += unit + decade + hundred;
        }
		return sum;
		
		/********** END **********/
	}
}
