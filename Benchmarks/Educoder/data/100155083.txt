package task1;

public class Box {

    double width;
    double height;
    double depth;
    public void volume() {
        double volume;
        volume=width*height*depth;
        System.out.println("the volumn is:"+volume);

    }
    public static void main(){
        Box box=new Box();
        box.width=10;
        box.height=20;
        box.depth=10;
        box.volume();
    }
}
