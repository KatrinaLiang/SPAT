package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int i;
        int sum=0;
        for(i=num1;i<=num2;i++)
        {
        	if(i<10)
            {
            	sum=sum+i;
                continue;
            }
             if(i<100)
            {
            	sum=sum+i%10+i/10;
                continue;
            }
          	else
            {
            	sum=sum+(i/100)+(i%100)/10+(i%10);
                continue;
            }
            
                
        }
        return sum;
		
		
		/********** END **********/
	}
}
