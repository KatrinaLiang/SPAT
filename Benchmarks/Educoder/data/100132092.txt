package step1;

public class Task {
	static int[] dailyTemps(Integer[] temps) {
		int result[]=new int[temps.length];
	
		for (int i=0; i<temps.length;i++) {
			if (i == temps.length - 1)
				continue;
			
			Integer after[]=new Integer [temps.length - i - 1];
			System.arraycopy(temps, i+1, after, 0, (temps.length - i - 1));
	
			result[i]=getUpDay(temps[i], after);
		}
		return result;
		
	}

	static int getUpDay(Integer today, Integer after[]) {
			int i = 0;
		Boolean succ = false;
		for (int temp : after)

		{
			i++;
			if (temp > today) {
				succ=true;
				break;
			}
		}
		if (!succ)
			return 0;
		else
			return i;

	}

}
