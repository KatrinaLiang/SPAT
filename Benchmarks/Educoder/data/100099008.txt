package step1;

public class Yard {

    private int length;
    private int width;
    public Yard(int a,int b)
    {
        length =a;
        width =b;
    }
    
    public double getPerimeter()
    {
        return (length+width)*2;
    }
    
    public double  getAcreage()
    {
        return length*width;
    }
     
    
    
}
