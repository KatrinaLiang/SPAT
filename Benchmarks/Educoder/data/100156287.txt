		int n2=sc.nextInt();
		System.out.println("请输入第3个整数：");
		int n3=sc.nextInt();
		//此处调用GetMaxValue方法，找到n1，n2，n3中的最大数并输出
		int z;
        z=TestJava23_8. Max(n1,n2,n3);
        System.out.println(n1+","+n2+","+n3+"三个数中的最大数是："+z);
		
	}
	/*******Start*********/
	//此处定义一个方法，名为GetMaxValue，实现求三个整数的最大数的功能，并返回
	public static int Max(int num1,int num2,int num3){	
		int z=num1;
		if(z<num2){
			z=num2;
		}
        if(z<num3){
			z=num3;
        }	
        return z;
	}
