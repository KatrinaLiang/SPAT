package step1;

public class DecomposeFactor {
	public void decompose(int score) {
		/********** Begin *********/
        double flag=4.0;
        if(score>=90)
            flag=4.0;
        else if(score>=80)
            flag=3.5;
        else if(score>=70)
            flag=3.0;
        else if(score>=60)
            flag=2.5;
        else
            flag=0.0;
        System.out.print(flag);
		/********** End *********/
	}
}
