package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		int n = matrix.length;
		for(int i = 0; i < n; i++){
			for(int j = i; j < n; j++){
				int tmp = matrix[i][j];
				matrix[i][j] = matrix[j][i];
				matrix[j][i] = tmp;
            }
		}
		for(int i = 0; i < n / 2; i++){
			for(int j = 0; j < n; j++){
				int k = matrix[j][i];
				matrix[j][i] = matrix[j][n-1-i];
				matrix[j][n-1-i] = k;
			}
		}
		

		
		
		
		/********* End *********/
	}
}
