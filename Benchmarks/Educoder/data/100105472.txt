package step3;

public class Reverse {
    public static StringBuffer start(String data) {
        StringBuffer ans = new StringBuffer();
        /********** Begin *********/
        String[] s=data.split(" ");
        for (String x:s){
            StringBuffer xs=new StringBuffer (x);
            xs.reverse();
            ans.append(xs).append(" ");
        }
        /********** End *********/
        return ans;
    }
}
