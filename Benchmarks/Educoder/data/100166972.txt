package step1;

public class A< T extends Comparable >{
 private T t1,t2; 
/*************BEGIN**************/

            public A(T t1_, T t2_){
                t1 = t1_;
                t2 = t2_;
            }

            public T compare(){
                if(t1.compareTo(t2) > 0){
                    return t1;
                }else{
                    return t2;
                }
            }



/*************END*****************/

}
