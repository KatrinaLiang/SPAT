package step1;
public class Yard{
    private int l;
    private int w;
    private float Perimeter;
    private float Acreage;
    public Yard(){}
    public Yard(int l,int w){
        this.l=l;
        this.w=w;
    }
    public float getPerimeter(){return l*2+w*2;}
    public float getAcreage(){return l*w;}

}
