    public static boolean writeFileByChars(Reader pReader, File pFile, boolean pAppend) {
        boolean flag = false;
        try {
            FileWriter fw = new FileWriter(pFile, pAppend);
            IOUtils.copy(pReader, fw);
            fw.flush();
            fw.close();
            pReader.close();
            flag = true;
        } catch (Exception e) {
            LOG.error("灏嗗瓧绗︽祦鍐欏叆锟�?" + pFile.getName() + "鍑虹幇寮傚父锟�?", e);
        }
        return flag;
    }
