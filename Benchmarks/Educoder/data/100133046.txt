package step1;

public class Task {

	public static void birthdayCandle(int num){
		int age,start;
		outer:for(start=18;start<=num;start++) {
			for(age=start;age<=num;age++){
                if((((start+age)*(age-start+1)/2)==num)&&(age<100)){
                	System.out.println("小明今年：" +age+ "岁");
					System.out.println("小明从" + start+ "岁开始每年举办派对");
                    break outer;
                }
            }
        }
	}
}
