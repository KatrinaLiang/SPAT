package step1;

public class Yard {
	double length;
    double width;
    public Yard(double width, double length){
        this.length = length;
        this.width = width;
    }
    public double getPerimeter(){
        double perimeter = 2*(length+width);
        return perimeter;
    }
    public double getAcreage(){
        double Acreage = length * width;
        return Acreage;
    }
    
}
