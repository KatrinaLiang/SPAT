package step1;

public class Yard {
	double  lenth;
    double  width;
    public Yard(double lenth1,double width1){
		this.lenth = lenth1;
        this.width = width1;
    }
    	
    double  getPerimeter(){
        	return lenth*2+width*2;
        }
        double  getAcreage(){
        	return lenth*width;
            
        }
    
}
