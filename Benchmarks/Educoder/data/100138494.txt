package step1;

    
         
class ThreadClassTwo implements Runnable{
    public void run(){
        for(int i=0;i<=10;i++){
            if((i%2)==0){
                System.out.print(i+" ");
            }
        }
    }
} 


public static void main(String[] args){
       
        System.out.println("ThreadClassTwo线程定义正确");
        new ThreadClassTwo().start();
    }
