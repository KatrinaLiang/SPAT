package step3;
public class NarcissisticNumber {

    //判断一个三位数是否是水仙花数
    public boolean NarcissisticNumber(int n){
        // 请在此添加实现代码
        int m=n;
        int sum=0;
        while(m!=0)
        {
            int a=m%10;
            sum+=a*a*a;
            m/=10;
        }
        if(sum==n)
        return true;
        else
            return false;
    }
}
