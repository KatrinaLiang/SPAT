package step2;

import java.util.Scanner;

public class NameSearch {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        /********** Begin *********/
 String data=scanner.nextLine();
        String name=scanner.nextLine();
int oldPosition=-1;
        int fristPosition=data.indexOf(name);
        
       while(fristPosition>-1)
        {
            
        oldPosition+=fristPosition+1;
           System.out.println(oldPosition);
           if(fristPosition+name.length()>data.length())
               break;
           data=data.substring(fristPosition+name.length()-1);
           fristPosition=data.indexOf(name);
           
       }
        /********** End *********/
    }
}
