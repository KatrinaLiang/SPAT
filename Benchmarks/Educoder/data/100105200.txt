package step3;

public class Reverse {
    public static StringBuffer start(String data) {
        StringBuffer ans = new StringBuffer();
        /********** Begin *********/
		String[] strArray = data.split(" ");
		for(int i = 0; i<strArray.length ; i++)
		{
			StringBuffer a = new StringBuffer();
			a.append(strArray[i]);
			a.reverse();
			ans.append(a + " ");
		}
		ans.trimToSize();
        /********** End *********/
        return ans;
    }
}
