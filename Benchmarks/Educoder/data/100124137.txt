package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
        int []array=new int[20];
        array[0]=1;
        array[1]=1;
        int sum = 0;
        for (int i=2;i<array.length;i++ ){
           array[i]=array[i-1]+array[i-2];
        }
        for (int i=0;i<array.length;i++ ){
             sum +=array[i];
        }
        System.out.print("前20个元素的和为"+sum);
    }
}
