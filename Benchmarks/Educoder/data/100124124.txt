package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
int sum = 0;
	  int[] arr = new int[20];
	  arr[0] = arr[1] = 1;
	  int temp = 0;
	  for(int i = 0 ; i < arr.length ; i++)
	  {		  
	      if(i > 1)
	      {
	    	  arr[i] = arr[i-1] + arr[i-2];
	      }
	  }
	  for(int j = 0 ; j < arr.length ; j++)
	  {
		  temp = temp + arr[j];		  
	  }
	  System.out.print("前20个元素的和为" + temp);
       

	/********end********/
	}
}
