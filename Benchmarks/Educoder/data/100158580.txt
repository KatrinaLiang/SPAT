#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int main(){
	string stra,strb;
	int a[1000][1000];
	int ans = 0;
	cin>>stra;
	cin>>strb;
	for(int i=0;i<stra.size();i++){
		for(int j=0;j<strb.size();j++){
			if(stra[i]==strb[j]){
				if(i>0&&j>0){
					a[i][j] = a[i-1][j-1] + 1;
				}
				else{
					a[i][j] = 1;
				}
				ans = max(ans,a[i][j]);
			}
		}
	}
	cout<<ans;
}
