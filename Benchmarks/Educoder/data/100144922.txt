package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		
		
int len = matrix[0].length;
		
		for(int i = 0;i < len;i ++)
			for(int j = 0;j < len;j ++)
				if(i < j){
					matrix[i][j] ^= matrix[j][i];
					matrix[j][i] ^= matrix[i][j];
					matrix[i][j] ^= matrix[j][i];
				}
		for(int j = 0;j < (len / 2); j ++)
		{
			for(int i = 0;i < len;i ++)
			{
				matrix[i][j] ^= matrix[i][len - j - 1];
				matrix[i][len - j - 1] ^= matrix[i][j];
				matrix[i][j] ^= matrix[i][len - j - 1];
				
			}
		}		
		/********* End *********/
	}
}
