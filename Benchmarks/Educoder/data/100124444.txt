package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	 int i;
        int sum = 2;
        int[] f=new int[20];
        f[0]=1;
        f[1]=1;
        for(i=2;i<f.length;i++){
            f[i]=f[i-1]+f[i-2];
            sum = sum + f[i];
        }
        System.out.print("前20个元素的和为" + sum);
	

	/********end********/
	}
}
