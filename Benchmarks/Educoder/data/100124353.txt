public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int [] f = new int [22];
        f[0] = 1;
        f[1] = 1;
        for(int i=2;i<=21;i++){
    
        f[i] = f[i-1] + f[i-2];
       
        }
        System.out.print("前20个元素的和为" + --f[21]);
	/********end********/
	}
}
