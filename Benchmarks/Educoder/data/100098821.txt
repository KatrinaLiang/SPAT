package step1;

public class Yard {
    
    int length,width,perimeter,acreage;
    public Yard (int l,int w)
    {
        length = l;
        width = w;
    }

	public float getPerimeter()
    {
        perimeter = 2 * (length + width);
        return perimeter;
    }

	public float getAcreage()
    {
        acreage = length * width;
        return acreage;
    }

}
