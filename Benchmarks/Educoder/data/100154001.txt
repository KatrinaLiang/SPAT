package step1;

public class ShuShu {

	// 第一题
	public void first() {
        int a=1;
        while(true)
        {
            if(a>100)
            {
                break;
            }
            if(a%10!=0)
            {
                System.out.print(a+"	");
            }
            a++;
            if(a%10==0)
            {
                System.out.println(a+"	");
            }
        }
		/*********Start********/
		
		/*********End********/
	}
	
	// 第二题
	public void second() {
        int a;
        a=1;
        System.out.println();
        do
        {
            if(a%10!=0)
            {
                System.out.print(a+"	");
            }
            if(a%10==0)
            {
                System.out.println(a+"	");
            }
            a++;
        }while(a<=100);
		/*********Start********/
		
		/*********End********/
	}
	
	// 第三题
	public void third() {
        System.out.println();
        int a;
        for(a=1;true;a++)
        {
            if(a>100)
            {
                break;
            }
            if(a%10!=0)
            {
                System.out.print(a+"	");
            }
            if(a%10==0)
            {
                System.out.println(a+"	");
            }
        }
		/*********Start********/
		
		/*********End********/
	}
}
