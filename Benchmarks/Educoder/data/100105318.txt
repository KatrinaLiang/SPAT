package step3;

public class Reverse {
    public static StringBuffer start(String data) {
        StringBuffer ans = new StringBuffer();
        /********** Begin *********/
		for(String str: data.split(" ")) {
        	StringBuffer str1 = new StringBuffer(str);
            ans.append(str1.reverse() + " "); 
        }
        


        /********** End *********/
        return ans;
    }
}
