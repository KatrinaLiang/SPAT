package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temps){
		/********** BEGIN **********/
		int[] a = new int[temps.length];
        int flag = 1;
        int k = 0;
        for(int i = 0;i<temps.length-1;i++){
        for(int j = i+1;j<temps.length;j++){
         if(temps[j]>temps[i]){
             flag = 0;
             a[k] = j-i;
             break;
            }
        }
            if(flag==1){
                a[k]=0;
                flag = 1;
            }
            k++;
        }
		

		return a;
		/********** END **********/
	}
}
