package step2;
public class ForPractice2 {
	public static void main(String[] args) {				
				for(int i=1;i<=10;i++)
        {
            for (int m=i;m>=1;m--){
             System.out.print("*");
            }       
		System.out.println("");}
	
	
	System.out.println("——————————我是华丽的分界线——————————");
		for(int i=1;i<=10;i++)
        {
            for (int m=i;m<=10;m++){
             System.out.print("*");
            }       
              System.out.println("");}
	}
}
