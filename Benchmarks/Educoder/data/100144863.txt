package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		
		int x=matrix[0].length;
        
        for(int i=0;i<=x/2;i++) 
        {
    		for(int j=i;j<x-i-1;j++)
    		{
    			int t=matrix[i][j]; 
    			matrix[i][j]=matrix[x-1-j][i];
    			matrix[x-1-j][i]=matrix[x-1-i][x-1-j];
    			matrix[x-1-i][x-1-j]=matrix[j][x-1-i];
    			matrix[j][x-1-i]=t;
    		}
        }

		
		
		
		/********* End *********/
	}
}
