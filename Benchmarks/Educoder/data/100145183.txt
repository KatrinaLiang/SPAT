package step2;
import java.util.Scanner;
public class Task {

	public int getNum(int n){
		/********* Begin *********/
		int num=0;
		while(true) {
			if(n%2==0) {
				n=n/2;
			}
			else {
				n=(n-1)/2;
				num++;
			}
			if(n==1) {
				num++;
				break;
			}
			if(n<=0) {
				break;
			}
		}
		return num;
		
		
		
		
		/********* End *********/
	}
	
}
