package task5;

public class Box {

    double width;
    double height;
    double depth;

    //TODO your code
    public Box(double widthvol, double heightvol, double depthvol){
        width = widthvol;
        height = heightvol;
        depth = depthvol;
    }
    public double volume(){
        double vol;
        vol = width*height*depth;
        return vol;
    }



}
