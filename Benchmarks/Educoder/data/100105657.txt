package step1;

public class Yard {
    int chang,kuan;
    public Yard(int chang,int kuan)
    {
        this.chang=chang;
        this.kuan=kuan;
    }
    public double getPerimeter()
    {
        return this.chang+this.kuan+this.chang+this.kuan;
    }
    public double getAcreage()
    {
        return this.chang*this.kuan;
    }

}
