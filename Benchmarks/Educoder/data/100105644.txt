package step1;

public class Yard {
int length;
	int width;
	double zc,mj;
     public Yard(int length,int width)
     {
    	 this.length = length;
    	 this.width = width;
     }
     double getPerimeter()
     {
    	 zc = 2*(this.length+this.width);
    	 return zc;
     }
     double getAcreage()
     {
    	 mj = this.length*this.width;
    	 return mj;
     }
}
