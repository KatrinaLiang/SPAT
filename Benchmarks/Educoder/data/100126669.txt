package step4;

import java.util.*;

public class HelloWorld {
	
	
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		//程序会输入三次数据
		/********** Begin **********/
		ArrayList<String> list = new ArrayList<>();
        for(int i = 0; i < 3; i++){
          list.add(sc.next());
        }
        int i = 1;
		for(String e : list ){
            System.out.println("集合的第" + i + "个数据为：" + e);
            i++;
        }
		
		/********** End **********/
	}
	
}
