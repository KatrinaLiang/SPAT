package step1;

public class Task {
	public static void main(String[] args) throws Exception {
		/********* Begin *********/
		//在这里创建线程， 开启线程
		MyThread first = new MyThread("AA");
		MyThread second = new MyThread("BB");
		MyThread third = new MyThread("CC");
		for (int i = 5; i > 0; i--){
        	first.run();
            second.run();
            third.run();
        }
		/********* End *********/
	}
}

class MyThread extends Thread {
	/********* Begin *********/
	String threadName;
	
    MyThread(String threadName){
    	this.threadName = threadName;
    }

	public void run() {
		System.out.println("Java Thread" + this.threadName);
	}
	/********* End *********/
}
