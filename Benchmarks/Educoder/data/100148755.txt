package step1;

public class Task {
	
	public void sort(Integer[] arr){
		/********* Begin *********/
int flag1 = 1, flag2 = 1;
		while(flag1 == 1) {
			flag1 = 0;
			for(int i = 0; i < arr.length - 1; i++) {
				if(arr[i] > arr[i + 1]) {
					int temp = arr[i];
					arr[i] = arr[i + 1];
					arr[i + 1] = temp; 
					for(int j = 0; j < arr.length; j++) {
						System.out.print(arr[j] + " ");
					}
					System.out.println("");
					flag1 = 1;
					flag2 = 0;
				}
			}
			for(int i = arr.length - 1; i > 0; i--) {
				if(arr[i] < arr[i - 1]) {
					int temp = arr[i];
					arr[i] = arr[i - 1];
					arr[i - 1] = temp; 
					for(int j = 0; j < arr.length; j++) {
						System.out.print(arr[j] + " ");
					}
					System.out.println("");
					flag1 = 1;
					flag2 = 0;
				}
			}
		}
		if(flag2 == 1) {
			for(int j = 0; j < arr.length; j++) {
				System.out.print(arr[j] + " ");
			}
			System.out.println("");
		}

		
		/********* End *********/
	}
}
