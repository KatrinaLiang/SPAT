public class Task {
	public String inversion(String str){
		/********* Begin *********/
        char[] a = str.toCharArray();
		String blank = "";
        int l;
        
        for(l=a.length-1;l>=0;l--){
            blank=blank+a[l];
        }
        
		return blank;
        }
		/********* End *********/
    }
