#include <stdio.h>
#include <stdlib.h>
#include <string.h>
/* run this program using the console pauser or add your own getch, system("pause") or input loop */

int main(int argc, char *argv[]) {
	char a[100];
	char b[100];
	memset(a,0,sizeof(a));
	memset(b,0,sizeof(b));
	scanf("%s %s",&a,&b);
	int i,j,k,t,m;
	int length = 0;
	int max = 0;
	for(i=0;i<=100;i++)
	{
		length = 0;
		if(a[i]==0)
			break;
		else
		{
			for(j=0;j<=100;j++)
			{
				length = 0;
				if(b[j]==0)
					break;
				else
				{
					if(b[j] == a[i])
					{
						k = i;
						t = j;
						length++;
						if(length > max)
							max = length;
						while(a[k+1]!=0&&b[t+1]!=0)
						{
							k++;
							t++;
							if(a[k] == b[t])
							{
								length++;
								if(length > max)
									max = length;
							}
							else
								break;
							
						}
						
					}
					else
						continue;
				}
			}
			
		}
	}
	printf("%d",max);
	return 0;

			
}
