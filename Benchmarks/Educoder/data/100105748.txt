package step1;

public class Yard {
    int length,width;
    public Yard(int length,int width){
    
        this.length=length;
        this.width=width;
    
    }
	public double getPerimeter(){
        double a;
        a = length*2+width*2;
        
    	return a;
    }
    public double getAcreage(){
        double b;
    	b = length*width;
        
        return b;
    }
}
