		int score2 = sc.nextInt(); //第二门成绩

        // 调用方法，传入两门课程的成绩
        calaAvg(score1,score2);
    }
         private static void calaAvg(int score1,int score2){
    /*
     * 功能：计算两门课程考试成绩的平均分并输出平均分
     * 定义一个包含两个参数的方法，用来传入两门课程的成绩
    */
             int avg=(score1+score2)/2;
             System.out.println("平均分："+avg);
}
