package step3;
import java.io.File;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;
import org.jsoup.Jsoup;
import org.jsoup.nodes.Document;
import org.jsoup.nodes.Element;
import org.jsoup.select.Elements;
public class Task {
    
    //通过filePath文件路径获取Docment对象
	public Document getDoc(String filePath) throws IOException{
		/**********   Begin   **********/
		File input = new File(filePath);
        Document doc = Jsoup.parse(input,"UTF-8");
        return doc;
        /**********   End   **********/
	}

	//获取所有链接
	public List<String> getLinks(Document doc){
		/**********   Begin   **********/
		Elements links = doc.select("a[href]");
        
        List<String> list1 = new ArrayList<String>();
        for (Element link : links) {
        String linkText=link.text();    
     	String linkHref=link.attr("href");
        String absHref = link.attr("abs:href");
        
        if(linkHref.startsWith("http")){
        list1.add('a'+"\u0024"+linkHref+'('+linkText+')');
        }
        else{
        list1.add('a'+"\u0024"+"http:"+linkHref+'('+linkText+')');
        }
            
		}

		return list1;
		/**********   End   **********/
	}
	
	//获取图片
	public List<String> getMedia(Document doc){
		/**********   Begin   **********/
		Elements links = doc.select("img[src]");
        List<String> list2 = new ArrayList<String>();
        for (Element link : links) {
     	String linkHref=link.attr("src");
        String absHref = link.attr("abs:href");
     	list2.add("img"+"\u0024"+"http:"+linkHref);
		}

		return list2;

		
		/**********   End   **********/
	}
	
	//获取link[href]链接
	public List<String> getImports(Document doc){
		/**********   Begin   **********/
		Elements links = doc.select("link[href]");
        List<String> list3 = new ArrayList<String>();
        for (Element link : links) {
     	String linkHref=link.attr("href");
        String absHref = link.attr("rel");
     	list3.add("link"+"\u0024"+"http:"+linkHref+"("+absHref+")");
		}

		return list3;

        
		/**********   End   **********/
	}
	
}
