package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum = 0;
		for(int i = num1; i <= num2 ; i ++){
            sum += sum(i);
        }
        return sum;
		/********** END **********/
	}
    
    public int sum(int num){
    	int num1 =0;
		int c = 0;
        if(num >= 100){
        	c = num / 100;
        	num1 += c;
            num = num - c * 100; 
        }
        if(num >= 10){
        	c = num / 10;
        	num1 += c;
            num = num - c * 10; 
        }
        num1 += num / 1;
        
        return num1;
    }
}
