package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		
		for(int i=1;i<10;i++) {
			for(int k=1;k<=i;k++) {
			System.out.print(k+"*"+i+"="+(k*i)+"\t");
			}
			System.out.println();
		}
		
		
		
		
		/*****end*****/
	}
}
