package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        if(num1>num2) {
			int num3 = num1;
			num1 = num2;
			num2 = num3;
		}
		int sum=0;
		while(num1<=num2) {
			int f1 = (Integer)num1/100;
			int s1 = (Integer)num1/10-f1*10;
			int l1 = num1-s1*10-f1*100;
			sum = sum+ f1 + s1+ l1;
			num1++;
		}
		return sum;
		/********** END **********/
        
    }
	
}
