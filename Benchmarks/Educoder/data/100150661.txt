package step3;
import java.util.Scanner;
public class SumOfDigit {
	public static void main(String[] args) {
		/******Begin******/
        Scanner scan=new Scanner(System.in);
		System.out.println("请输入一个五位数字：");
        int x,y,s=0,k;
        x=scan.nextInt();
        k=x;
        for(int i=0;i<5;i++){
           y=x%10;
            x=x/10;
            s=s+y;
            y=0;
        }
        System.out.println(k+"的各位数字之和是："+s);
		/******End******/
	}
}
