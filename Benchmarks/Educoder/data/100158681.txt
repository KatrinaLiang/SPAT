#include <iostream>
#include <stdio.h>
using namespace std;
int a[10005];
int main()
{
	int n = 0,t,s=0;
	while(scanf("%d",&a[n])!=EOF)
	{
		n++;
	}
	t = a[n-1];
    for(int i = 0; i < n-1; i++)
    {
    	for(int j = i+1; j < n-1; j++)
    	{
    		for(int k = j+1; k < n-1; k++)
    		{ 			
    				if(a[i]+a[j]+a[k]<t)
    				{
    					s++;
    				}  			
    		}
    	}
    }
    cout<<s;
    return 0;
}
