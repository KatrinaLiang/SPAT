package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		int [] temps=new int[temperatures.length];
        int maxID=0;
        for (int i = 0; i < temperatures.length; i++) {
            for (int j = i + 1; j < temperatures.length; j++) {
                if (temperatures[j] > temperatures[i]) {
                    maxID=j;
                    break;
                } 
            }
            int result=maxID-i;
            if(result<0){
                result=0;
            }
            temps[i]=result;
        }
        return temps;
	}
}
