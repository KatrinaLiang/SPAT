package step3;

public class Reverse {
    public static StringBuffer start(String data) {
        StringBuffer ans = new StringBuffer();
        String[] a=data.split(" ");
        int i;
        for(i=0;i<a.length;i++)
        {
           StringBuffer b=new StringBuffer(a[i]);
           ans.append(b.reverse());
           ans.append(" ");
           
        }
       
        /********** End *********/
        return ans;
    }
    
}
