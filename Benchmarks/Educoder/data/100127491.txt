package step1;
public class DynamicBind {

	/**
	 * @param args
	 */
	public static void main(String[] args) {
		Pear p=new Pear("香梨");
		Banana b = new Banana("仙人蕉");
		Apple a = new Apple("红富士苹果");
		System.out.println(a.show());
		System.out.println(p.show());
		System.out.println(b.show());
	}

}
