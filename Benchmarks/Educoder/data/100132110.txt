package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
		int []time= new int[temperatures.length];
        
        for(int i=0;i<temperatures.length;i++)
        {
            int a=0;
            int j;
            for(j=i+1;j<temperatures.length;j++)
            {
                if(temperatures[i]>=temperatures[j])a++;
                else {
                    a=a+1;
                    break;
                }
            }
            if(j==temperatures.length)time[i]=0;
            else time[i]=a;
        }
        return time;
        
		/********** END **********/
	}
}
