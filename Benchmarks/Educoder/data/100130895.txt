package step4;

import java.util.Arrays;

public class Task {
    
	/********** BEGIN **********/
    public void sortArray(int[][] arr){
		int i,j,k,temp;
        for(i=0;i<arr.length;i++)
        {
            for(j=1;j<arr[i].length;j++)
            {
                temp=arr[i][j];
                k=j-1;
                while(k>=0 && arr[i][k]<temp)
                {
                    arr[i][k+1] = arr[i][k];
                    k--;
                }
                arr[i][k+1]=temp;
            }
         }
        for(i=arr.length-1;i>=0;i--)
        {
            System.out.print("[");
            for(j=0;j<arr[i].length;j++)
            {
                System.out.print(arr[i][j]);
                if(j != arr[i].length-1)
                    System.out.print(", ");
            }
            System.out.println("]");
        }
    }
    
    /********** END **********/
}
