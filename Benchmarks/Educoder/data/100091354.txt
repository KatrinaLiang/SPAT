package step3;
import java.util.Scanner;
public class NarcissisticNumber {
    public static void main(String[] args)
    {
        Scanner sc = new Scanner (System.in);
        int a = sc.nextInt();
        NarcissisticNumber sss = new NarcissisticNumber();
        System.out.print(sss.NarcissisticNumber(a));
    }
    //判断一个三位数是否是水仙花数
    public boolean NarcissisticNumber(int n){
        // 请在此添加实现代码
        boolean q = true;
        int a = n%10;
        int b = n/10%10;
        int c = n/100%10;
        int s = a*a*a+b*b*b+c*c*c;
        if(s == n)
            return q;
        else 
            q = false;
        return q;
    }
}
