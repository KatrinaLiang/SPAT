package step1;

import java.util.Scanner;

public class TestJava20_2 {
	static final Scanner sc = new Scanner(System.in);	
	public void print() {
        System.out.println("请输入倒直角三角形的行数：");
		
			int m=sc.nextInt();
		for(int a=1;a<=m;a++){
            for(int b=m; b>=a; b--){
                System.out.print("*\t");
            }
            System.out.println();
        }
	}

}
