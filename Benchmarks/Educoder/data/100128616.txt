package step1;
class Dog{
    String name,color,type;
    Dog(String _name,String _color,String _type){
        name= _name;
        color = _color;
        type =_type;
    }
    public void b(){
        System.out.println("啃骨头");
    }
    public void runwithB(){
        System.out.println("叼着骨头跑");
    }
        
}
public class Test {
	public static void main(String[] args) {
		//创建Dog对象
        Dog d = new Dog("五花肉","棕色","阿拉斯加");
		//设置Dog对象的属性
		
		
		//输出小狗的属性
		System.out.println("名字：" +d.name    + "，毛色：" + d.color   + "，品种：" +d.type  );
		
		//调用方法
		d.b();
        d.runwithB();
	}
}
