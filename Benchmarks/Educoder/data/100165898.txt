public class Report {

   /************Begin****************/
    public Report(Student s,String major,Course[] selCourse){
        this.s = s;
        this.major =major;
        this.selCourse = selCourse;
    }

	public void setReport(){
        
    }
    public Student getStudent(){
        return s;
    }
    public String getmajor(){
     	return major;   
    }
	public String getCourse(){
        String str = new String("");
        for(int i = 0;i<selCourse.length;i++)
        	str = str + selCourse[i] + "\n";
        return str;
    }
	
    public String toString(){
        return "name=" + getStudent().getname() + " number=" + getStudent().getnum() + " birthday:" + getStudent().getBirthday() + "reg_num:" + getStudent().getreg_Num() + "reg_day:" + getStudent().getreg_day() + " major:" + getmajor() + "\n" + getCourse();
    }
