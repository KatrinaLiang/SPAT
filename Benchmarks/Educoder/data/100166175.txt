package step1;

public class Person {
    private String name; //姓名
    private String number; //身份证号
    private Date birthday; //出生日期
    
   /************Begin*************/
	public Person(){
        Date birthday = new Date();
    	setBirth(birthday);
    }
       
    public void setName(String name){
        this.name = name;
    }
    public void setNum(String number){
        this.number = number;
    }
    public void setBirth(Date birthday){
        this.birthday = birthday;
    }
    public String getName(){
        return name;
    }
    public String getNum(){
        return number;
    }
    public Date getBirth(){
        return birthday;
    }
    public String toString(){
        return "name="+name+" number="+number+" birthday:"+birthday;
    }
    public Person(String name,String number,Date b){
       	setName(name);
        setNum(number);
        setBirth(b);
    }
    public Person(String name,String number,int y,int m,int d){
        setName(name);
        setNum(number);
        Date birthday = new Date(y,m,d);
        setBirth(birthday);
    }

    
    
    
    
   /************End*************/
}
