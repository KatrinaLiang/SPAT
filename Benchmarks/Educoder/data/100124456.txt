package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
        int i,s;
        int [] a=new int[20];
        a[0]=1;
        a[1]=1;
        s=2;
        for(i=2;i<a.length;i++)
        {
            a[i]=a[i-1]+a[i-2];
            s=s+a[i];
        }
        System.out.print("前20个元素的和为"+s);

	/********end********/
	}
}
