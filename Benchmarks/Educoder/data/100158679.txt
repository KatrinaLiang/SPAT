#include<bits/stdc++.h>
#define max 100
using namespace std; 
int main(){
   int a[max];
   int n,m=0,sum,su=0;
   while(cin>>n){
   	a[m++]=n;
   	char q=getchar();
   	if(q=='\n'){
   		break;
   	}
   }
   cin>>sum;
   //cout<<m<<endl;
   for(int i=0;i<m-2;i++){
   	//cout<<"a[i]:"<<a[i]<<endl;
   	for(int j=i+1;j<m-1;j++){
   		//cout<<"a[j]:"<<a[j]<<endl;
	   for(int k=j+1;k<m;k++){
	   	//cout<<"a[k]:"<<a[k]<<endl;
	   	if(a[i]+a[j]+a[k]<sum){
	   		su++;
	   	}
	   }	
   	}
   }
   cout<<su;
   //cout<<a[4];
}
