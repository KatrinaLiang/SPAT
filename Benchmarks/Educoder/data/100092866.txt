package step1;

public class LetterChange {

    // 使用 switch把小写字母转为大写字母
    public char LetterChange(char ch){
        // 请在此提供实现代码
        char x;
        switch(ch)
        {
            case 'a':x='A';break;
            case 'b':x='B';break;
            case 'c':x='C';break;
            case 'd':x='D';break;
            case 'e':x='E';break;
            default:x=ch;break;
        }
    return x;
    }
}
