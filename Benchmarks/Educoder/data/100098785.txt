package step1;

 class Yard {
	public double length,width;
	public Yard(){
    	this.length = 0;
        this.width = 0;
    }
	public Yard(int x, int y){
    	this.length = x;
        this.width = y;
    }
    public double getPerimeter(){
    	return 2*(this.length + this.width);
    }
	public	double  getAcreage(){
    	return this.length*this.width;
    }
}
