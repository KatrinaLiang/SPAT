package step1;

public class DecomposeFactor {
	public void decompose(int score) {
		/********** Begin *********/
        float grade;
        if(score>=90)
            grade=4.0f;
        else if(score>=80)
            grade=3.5f;
        else if(score>=70)
            grade=3.0f;
        else if(score>=60)
            grade=2.5f;
        else
            grade=0.0f;
        System.out.print(grade);

		/********** End *********/
	}
}
