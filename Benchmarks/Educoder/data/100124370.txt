package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] fib = new int[20];
        fib[0]=1;
        fib[1]=1;
        for(int i=2;i<20;i++){
            fib[i]=fib[i-1]+fib[i-2];
        }
        int sum =0;
        for(int j =0;j<20;j++){
            sum+=fib[j];
        }
        System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
