package step2;
public class Fibonacci{
	
	 public static void main(String args[])	
	{		int i;
	    int sum=2;
		int arr[]  = new int [20];		
		arr[0] = 1;
        arr[1] = 1;	
		for(int i1 = 2;i1 < arr.length;i1++)	
		{			
			arr[i1] = arr[i1 - 1] + arr[i1 - 2];
			sum+=arr[i1];
				
		}
		System.out.print("前20个元素的和为" + sum);
	}
	}
