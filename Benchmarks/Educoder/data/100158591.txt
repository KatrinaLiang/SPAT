#include <iostream>
#include <string.h>
using namespace std;
int main()
{
    char s[1000],s0[1000];
    int flag1=0;
    scanf("%s",s);
    scanf("%s",s0);
    for(int i=0;i<strlen(s);i++){
        for(int j=0;j<strlen(s0);j++){
            if(s[i]==s0[j])
            {
                int flag=1;
                for(int k1=i+1,k2=j+1;k1<strlen(s),k2<strlen(s0);k1++,k2++)
                {
                    if (s[k1]!=s0[k2])
                        break;
                    flag++;
                }
                flag1=max(flag,flag1);
            }
            
        }
    }
    printf("%d",flag1);
}
