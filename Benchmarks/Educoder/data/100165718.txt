package step3;
import step1.Date;
public class Course {
    private int code; //课程编号
    private String courseName; //课程名称
    private Date testTime;//考试日期
    private int score;//考试成绩
   /******Begin**************/
	public Course(){
        this.code = 0;
        this.courseName = "";
        this.testTime = new Date();
        this.score = 0;
    }
    public Course(int code, String courseName, int y, int m, int d, int score){
        Date testTime = new Date(y, m, d);
        this.code = code;
        this.courseName = courseName;
        this.testTime = testTime;
        this.score = score;
    }
    public Course(int code, String courseName, Date d1, int score){
        this.code = code;
        this.courseName = courseName;
        this.testTime = d1;
        this.score = score;
    }
    
    public int getcode(){
        return this.code;
    }
    public String getCourseName(){
        return this.courseName;
    }
    public Date getTestTime(){
        return this.testTime;
    }
    public int getscore(){
        return score;
    }
    
    public String toString(){
        return this.code+" "+this.courseName+" "+this.testTime+" "+this.score;
    }




   /******End************/

}
