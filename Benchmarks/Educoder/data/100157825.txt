package step;
public class Sort {
	/**
	 *@param arr 待排序数组
	 */
	public void BubbleSort(int[] arr){
		/********* Begin *********/
	for(int a=0;a<arr.length-1;a++)
    {
    for(int c=0;c<arr.length-a-1;c++)
    {
        if(arr[c]>arr[c+1])
        {
            int b=arr[c];
            arr[c]=arr[c+1];
            arr[c+1]=b;
        }    
    }
        for(int d=0;d<arr.length;d++)
            System.out.print(arr[d]+" ");
        System.out.println();
        
    }



		/********* End *********/
	}
}
