package step4;

import java.util.Arrays;

public class Task {
    
	/********** BEGIN **********/
	 public void sortArray(int[][] arr){
		int i,j;
		int n = arr.length;
		int [] sum = new int[n];
		for(i=0;i<n;i++)
		{
			bubbleSort(arr[i]);
			sum[i] = sum(arr[i]);	
		}
		
		bubbleSortAll(arr,sum);
		
		for(i=0;i<n;i++)
		{
			System.out.print("[");
			for(j=0;j<arr[i].length-1;j++)
			{
				System.out.print(arr[i][j]+", ");
			}
			System.out.println(arr[i][j]+"]");
		}
    }
    public void bubbleSortAll(int[][] arr,int [] sum) {
    	boolean flag = true;
    	int i,j;
    	int n = arr.length;
    	for(i=1;i<n&&flag;i++)
    	{
    		flag = false;
    		for(j=n-1;j>=i;j--)
    		{
    			if(sum[j]>sum[j-1])
    			{
    				int[] temp = arr[j];
    				arr[j] = arr[j-1];
    				arr[j-1] = temp;
    				flag = true;
    			}
    		}
    	}
	}
    public int sum(int[] arr) {
    	int sum = 0;
    	for(int i =0;i<arr.length;i++)
    	{
    		sum+=arr[i];
    		
    	}
		return sum;
	}
	//降序，冒泡
    public void bubbleSort(int [] arr)
    {
    	boolean flag = true;
    	int i,j;
    	int n = arr.length;
    	for(i=1;i<n&&flag;i++)
    	{
    		flag = false;
    		for(j=n-1;j>=i;j--)
    		{
    			if(arr[j]>arr[j-1])
    			{
    				int temp = arr[j];
    				arr[j] = arr[j-1];
    				arr[j-1] = temp;
    				flag = true;
    			}
    		}
    	}
    	
    }
    
    
    /********** END **********/
}
