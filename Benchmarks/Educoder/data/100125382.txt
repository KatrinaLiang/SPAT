package step2;

import javax.swing.JOptionPane;
import java.util.Scanner;

public class InterRectangle {
	private double width;
	private double length;
	/*********请在此编写构造函数及求周长和面积的方法*********/
	public double perimeter(double w,double l)
    {
        return (w+l)*2;
    }
	public double area(double w,double l)
    {
        return w*l;
    }

	/*********end**********/
	public static void main(String[] args) {
 		Scanner scanner = new Scanner(System.in);
		double total=scanner.nextDouble();//获取土地数目
		/*********begin*********/
        //请获取到每块土地的长和宽，并输出土地块数，总面积和总周长
		double per = 0;double are = 0;
        for(double i = total;i > 0;i--)
        {
		double width=scanner.nextDouble();
        double length=scanner.nextDouble();
		InterRectangle l = new InterRectangle();
        per = per + l.perimeter(width,length);
        are = are + l.area(width,length);
        }
		System.out.println("总共有"+total+"块土地");
        System.out.println("土地总面积为:"+are+"平米");
		System.out.println("土地总周长为:"+per+"米");

		/*********end*********/
	}

}
