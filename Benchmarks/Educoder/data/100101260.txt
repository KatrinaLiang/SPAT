package step2;

import java.util.Scanner;

public class SystemUpdate {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String name = scanner.next();
        String birth = scanner.next();
        String position = scanner.next();
        double salary = scanner.nextDouble();
        Salary employee = new Salary(name, birth, position, salary);
        employee.introduction();
    }
}

/********** Begin *********/
class Employee {
    Employee(String name, String birth, String zhiwei) {
        this.name = name;
        this.birth = birth;
        this.zhiwei = zhiwei;
    }
    public void introduction() {
        System.out.print("员工姓名：" + name + " 出生年月：" + birth + " 职位：" + zhiwei);
    }
    private String name;
    private String birth;
    private String zhiwei;
}

class Salary extends Employee {
    Salary(String name, String birth, String zhiwei, double salary) {
        super(name, birth, zhiwei);
        this.salary = salary;
    }
    public void introduction() {
        super.introduction();
        System.out.print(" 薪水：" + salary);
    }
    private double salary;
}
/********** End *********/
