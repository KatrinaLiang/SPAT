package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
        int sum = 0 ;
        int a[] = new int[100];
		for(int i = num1 ; i <= num2 ; i ++ ){
            for(int j = i , x = 0 ;j > 0  ;x++){
                a[x] = j%10 ;
                j = j/10 ;
			}
           for(int x = 0 ; x < 100 ; x++ )
            sum += a[x] ;
        }
        return sum ;
    }
}
