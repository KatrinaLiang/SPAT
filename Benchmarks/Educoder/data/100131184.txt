package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int min = Math.min(num1, num2);
		int max = Math.max(num1, num2);
		int sum = 0;
		for (int i = min; i <= max; i++) {
			int num = i;
			while (num != 0) {
				sum += num % 10;
				num = num / 10;
			}
		}
		return sum;
		/********** END **********/
	}
}
