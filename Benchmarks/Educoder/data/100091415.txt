package step3;
public class NarcissisticNumber {

    //判断一个三位数是否是水仙花数
    public boolean NarcissisticNumber(int n){
        // 请在此添加实现代码
        String x=n+"";
        int mishu=x.length();
        int sum=0;
        for(int i=0;i<mishu;i++)
        {
        	sum+=Math.pow(x.charAt(i)-48, mishu);
        }
        if(sum==n)
        {
        	return true;
        }
        return false;
    }
}
