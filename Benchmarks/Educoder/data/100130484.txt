package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		/********** BEGIN **********/
		double res=0.0;
		int i;
		if(n%2==0) {
			for(i=2;i<=n;i+=2) {
				res=res+3.0/i;
			}
		}
		else{
			for(i=1;i<=n;i+=2) {
				res=res+4.0/i;
			}
		}
		System.out.println(res);
		/********** END **********/
	}
}
