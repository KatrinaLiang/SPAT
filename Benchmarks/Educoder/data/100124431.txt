package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int []num=new int[20];
	    num[0]=1;
	    num[1]=1;
        int s=0;
	    for(int i=2;i<num.length;i++){
	    	num[i]=num[i-1]+num[i-2];
            
        }
        for(int i=0;i<num.length;i++){
            s=s+num[i];
        }
	    System.out.print("前20个元素的和为"+s);
	/********end********/
	}
}
