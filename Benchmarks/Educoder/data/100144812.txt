package step3;

public class Task {
	public void rotate(int[][] matrix){
		/********* Begin *********/
		
		int r=matrix.length;
		int c=matrix[1].length;
		int [][] m=new int[c][r];
		
		for(int p=0;p<r;p++){
			for(int q=0;q<c;q++){
					m[q][c-1-p]=matrix[p][q];	
					}
				}
			
		for(int p=0;p<c;p++){
			for(int q=0;q<r;q++){
				matrix[p][q]=m[p][q];
			}
		}

		
		
		
		/********* End *********/
	}
}
