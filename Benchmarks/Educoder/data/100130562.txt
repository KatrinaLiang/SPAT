package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		/********** BEGIN **********/
		double res=0;
		double i;
		if( n%2 == 0 ){
			i=2;
			while( i<=n ){
				res += 3/i;
				i += 2;
			}
        }else{
        	i=1;
        	while( i<=n ){
				res += 4/i;
				i += 2;
			}
        }
		System.out.println(res);
		
		
		
		
		
		/********** END **********/
	}
}
