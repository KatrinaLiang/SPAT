package step2;

// 利用 Runnable 接口实现
/********** Begin *********/
public class MyThread implements Runnable{
/********** End *********/
	private String name;
    
    //初始化成员变量name的值
    public MyThread(String Name) {
        /********** Begin *********/
           
        name=Name;

        /********** End *********/
    }
    public void run() {
        /********** Begin *********/

		System.out.println("运行:"+name);

        /********** End *********/
    }  
}
