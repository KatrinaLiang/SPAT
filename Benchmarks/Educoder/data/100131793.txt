package step3;

import java.io.File;
import java.io.FilenameFilter;
public class Task {
	int flag =0;
	/********** BEGIN **********/
	public void showDirStructure(File dir)	{
        if(flag == 0){
        	System.out.println("+--"+dir.getName());
            flag +=1;
        }
        String kong = "";
        for(int i=0;i<flag;i++){
        	kong += "  ";
        }
        FilenameFilter filter = new FilenameFilter(){
        	public boolean accept(File dirr,String na){
            	File currentFile = new File(dirr,na);
                if(currentFile.isFile()&&(na.endsWith(".jpg")||na.endsWith(".png")||na.endsWith(".bmp"))){
                	return true;
                }else{return false;}
            }
        };
		String[] names = dir.list();
        for(String name : names){
            String address = dir+"//"+name;
            File ff = new File(address);
            if(ff.isDirectory()){                
            	System.out.println(kong+"+--"+name);
                flag+=1;
                showDirStructure(ff);
                flag-=1;
            }else if(name.endsWith(".jpg")||name.endsWith(".png")||name.endsWith(".bmp")){
            	System.out.println(kong+"--"+name);
            }
        }






	}
	

	/********** END **********/
}
