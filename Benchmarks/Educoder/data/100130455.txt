package step2;

import java.util.Scanner;

public class Task {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		/********** BEGIN **********/
		double m = n%2+3,s=0;
		for(int z=2-n%2;z<=n;z+=2){
			s+=m/z;
		}
		System.out.println(s);	
		/********** END **********/
	}
}
