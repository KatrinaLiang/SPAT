package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int[] arr =new int[20];
    int s=0;
	for(int i=0;i<20;i++){
		if(i==0||i==1)
			arr[i]=1;
		else
			arr[i]=arr[i-1]+arr[i-2];
        s+=arr[i];
		}	
	System.out.print("前20个元素的和为"+s);
	/********end********/
	}
}
