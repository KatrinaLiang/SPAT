package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
        int[] f=new int[20];
        f[0]=1;
        f[1]=1;
        int sum=0;
        for(int i=2;i<f.length;i++){
            f[i]=f[i-2]+f[i-1];
            sum+=f[i];
        }
		System.out.print("前20个元素的和为"+(sum+2));

	/********end********/
	}
}
