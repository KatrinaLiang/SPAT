package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int res = 0;
        
for(int i=num1;i<=num2;i++){
        String[] s1 = (""+i).split("");
        for(String s:s1)
          res+=Integer.parseInt(s);        
}
        return res;
		/********** END **********/
	}
}
