package step4;

import java.util.*;

public class HelloWorld {
	
	
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		//程序会输入三次数据
		/********** Begin **********/
		List<String> list = new ArrayList<String>();
        list.add(sc.nextLine());
		list.add(sc.nextLine());
        list.add(sc.nextLine());
		
        int index = 0;
		for (String str : list) {
        	index++;
            System.out.println("集合的第" + index + "个数据为：" + str);
        }
		
		/********** End **********/
	}
	
}
