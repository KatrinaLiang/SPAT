package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int [] arr=new int [999];
        int sum=0;
        for(int i=num1,j=0;i<=num2;i++,j++)
        {
            arr[j]=i;
        }
        for(int x=0;x<arr.length;x++)
        {
            int a=arr[x]/100;
            int b=(arr[x]-100*a)/10;
            int c=arr[x]%10;
            sum=sum+a+b+c;
        }
		return sum;
		
		/********** END **********/
	}
}
