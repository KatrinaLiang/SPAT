package step3;

import java.io.File;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

public class DeleteFiles {
     public static void start() throws IOException {
	        StringBuilder data = new StringBuilder("");
	        /********** Begin *********/
	        //读取readme.txt文件
	        FileReader reader = new FileReader("src/step3/readme.txt");
	        int ch;
	        while((ch=reader.read())!=-1) {
	        	data.append((char)ch);
	        }
	        reader.close();
	        
	        //写入readmeCopy.txt文件
	        File file = new File("src/step3/readmeCopy.txt");
	    	FileWriter fileWriter = new FileWriter(file, false);
	    	String str;
	    	str = data.toString();
	    	fileWriter.write(str);
	    	fileWriter.close();
	    	
	        //删除readme.txt文件
	    	File file2 = new File("src/step3/readme.txt");   //file对象指向src文件夹下的step1文件夹下的Hello World.txt文件
	    	file2.delete();
	    	File file1 = new File("src/step3");   //file对象指向src文件夹下的step1文件夹
			file1.delete();
	        /********** End *********/
	 }
	
}
