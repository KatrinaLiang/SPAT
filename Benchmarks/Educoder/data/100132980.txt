package step1;

public class Task {
	
	public void sortArray(int[] arr){
        for(int i=1;i<arr.length;i++)
        {
            int flag=i;
            int x=arr[i];
        	for(int p=i-1;p>=0;p--)
            {
            	if(arr[p]<arr[i]){
                flag=p+1;
                break;
                } 
            }
            if(arr[0]>=arr[i]) {
            	for(int p=i;p>0;p--)
                    arr[p]=arr[p-1];
                arr[0]=x;
            }
            else{
            for(int p=i;p>flag;p--)
                arr[p]=arr[p-1];
            arr[flag]=x;
            }
            for(int q=0;q<arr.length;q++)
                System.out.print(arr[q]+" ");
            System.out.println();
        }
		
		
	}
}
