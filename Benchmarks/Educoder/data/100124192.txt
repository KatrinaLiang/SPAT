package step2;
    public class Fibonacci{
        public static void main(String args[]){
        /*********begin********/
	int[] a=new int[20];
			int s=2;
			a[0]=1;
			a[1]=1;
			for (int i = 2; i <= a.length-1; i++) {
				a[i]=a[i-1]+a[i-2];
				s=s+a[i];
			
			}
			System.out.print("前20个元素的和为"+s);

        /********end********/
        }
    }
