package step3;

public class MyThread implements Runnable {   
    private String name;
    private Object prev1;
    private Object prev2;
    private Object self;
    private MyThread(String name, Object prev1,Object prev2, Object self){
        this.name=name;
        this.prev1=prev1;
        this.prev2=prev2;
        this.self=self;
    }
    public void run(){
        int count =5;
        while(count>0){
            synchronized (prev1){
               
                    synchronized (self){
                        System.out.print(name);
                        count--;
                        self.notify();
                    }
                    try{
                        prev1.wait();
                    }catch (InterruptedException e){
                        e.printStackTrace();
                    }
            }
        }
        System.exit(0);
    }
	    
    public static void main(String[] args) throws Exception {   
        Object a =new Object();
        Object b =new Object();
        Object c =new Object();
        MyThread ta =new MyThread("E",c,b,a);
        MyThread tb =new MyThread("D",a,c,b);
        MyThread tc =new MyThread("U",b,a,c);
        new Thread(ta).start();
        Thread.sleep(100);
        new Thread(tb).start();
        Thread.sleep(100);
        new Thread(tc).start();
        Thread.sleep(100);
    }   
}
