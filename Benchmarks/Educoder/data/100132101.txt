package step1;
public class Task {
	public  int[] dailyTemps(Integer[] temps){
		/********** BEGIN **********/
		int[] out=new int[temps.length];
        for(int i=0;i<temps.length;i++){
            out[i]=0;
            for(int j=i+1;j<temps.length;j++){
                if(temps[i]<temps[j]){
                    out[i]=j-i;
                    break;
                }
            }
        }
        return out;
		/********** END **********/
	}
}
