package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int Sum[]=new int [1000] ;
        int j=0;
        int sum=0;
		for(int i=num1;i<=num2;i++)
            
        {
         	int a=i/100;
            int b=(i-a*100)/10;
            int c=i-a*100-b*10;
            int sum1=a+b+c;
            Sum[j]=sum1;
            j++;
        }
		for(int i=0;i<=num2-num1;i++)
        {
            sum+=Sum[i];
        }
		return sum;
		/********** END **********/
	}
}
