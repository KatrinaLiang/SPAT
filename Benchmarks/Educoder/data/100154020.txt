	public void first() {
		/*********Start********/

        int b=1;
        while(true){
        	if(a>100){
            	break;
            }
            System.out.print(a+"\t");
            a++;
            if(b%10==0){
            	System.out.println();
                
            }
            b++;
        }
        	System.out.println();

	}
	
	// 第二题
	public void second() {

            }
        	System.out.print(a+"\t");

            if(b%10==0){

  	      System.out.println();

	}
