package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
	int[] fibonacci = new int[20];
    fibonacci[0] = 1;
    fibonacci[1] = 1;
    int sum=0;
    for(int i=2;i<20;i++){
        fibonacci[i] = fibonacci[i-1]+fibonacci[i-2];
    }
    for(int i=0;i<20;i++){
        sum += fibonacci[i];
    }
    System.out.print("前20个元素的和为"+sum);
	/********end********/
	}
}
