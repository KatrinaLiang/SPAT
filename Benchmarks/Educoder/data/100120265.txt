package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int x,y;
        for(x=1;x<=9;x++)
        {
            for(y=1;y<=x;y++)
            {
                System.out.print(y+"*"+x+"="+x*y+"\t");
            }
            System.out.println();
        }
		
		
		
		
		
		/*****end*****/
	}
}
