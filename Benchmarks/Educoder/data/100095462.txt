package step4;

public class DayCount {
	public void dayCount(int year, int month, int day) {
		/********** Begin *********/
		int count = 0;
		int[] arr = {31,28,31,30,31,30,31,31,30,31,30,31};
		if(year%400==0||(year%4==0&&year%100!=0)) {
			arr[1]=29;
		}
		for(int i=1;i<month;i++) {
			count += arr[i-1];
		}
		count += day;
		System.out.print(count);


		/********** End *********/
	}
}
