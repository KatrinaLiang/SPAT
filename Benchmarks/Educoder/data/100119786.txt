package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		for(int i=1;i<=10;i++){
            for(int n=1;n<=i;n++){
                System.out.print("*");
            }System.out.println();
        }
		
		
		
		
		System.out.println("——————————我是华丽的分界线——————————");
		
		for(int i=1;i<=10;i++){
            for(int n=10;n>=i;n--){
                System.out.print("*");
            }System.out.println();
        }
		
		
		
		
		/*****end*****/
	}
}
