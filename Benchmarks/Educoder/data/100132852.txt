package step3;

public class Task {

	 public int countOfShips(char[][] ferry) {
	  /**************BEGIN************/
        if(ferry==null||ferry.length==0){
            return 0;
        }
	 	int n=ferry.length;
        int m=ferry[0].length;
        int s=0;
        boolean f;
	 	for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                f=false;
                if(ferry[i][j]=='+'){
                    f=true;
                }
                if(i>0&&ferry[i-1][j]=='+'){
                    f=false;
                }
                if(j>0&&ferry[i][j-1]=='+'){
                    f=false;
                }
                if(f){
                    s++;
                }
            }
        }
         return s;
      /***********END ***************/
	 }
}
