package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum = 0;
		for(int i=num1;i<=num2;i++){
        	if(i>99&&i<1000){
                int a = i/100;
                int b = i%100/10;
                int c = i%10;
                sum += a+b+c;
            }else if(i>9&&i<100){
                int a = i/10;
                int b = i%10;
                sum += a+b;
            }else{
                sum+=i;
            }
        }
        return sum;
		
		
		/********** END **********/
	}
}
