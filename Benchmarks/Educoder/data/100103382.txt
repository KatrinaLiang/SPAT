package step3;

import java.io.File;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

public class DeleteFiles {
    public static void start() throws IOException {
        StringBuilder data = new StringBuilder("");
        /********** Begin *********/
        //读取readme.txt文件
       
File file = new File("src/step3/readme.txt");
FileReader fileReader = new FileReader(file);
int read = fileReader.read();
while (read != -1) {   //循环读取，直到read的值为-1，表示文件读到末尾
       data.append((char) read);//输出读取的数据，使用(char)方法对read的值进行类型转换
    read = fileReader.read();   //读取下一个字符
}
         //String data1=data.toString();
        fileReader.close();
        //写入readmeCopy.txt文件
File file1 = new File("src/step3/readmeCopy.txt");
        FileWriter file1Writer = new FileWriter(file1, true);////file对象指向src文件夹下的step1文件夹下的Hello World.txt文件（此时此文件不存在）
if (!file1.exists()) {   //如果文件不存在
    file1.createNewFile();   //创建文件，并输出布尔值
}
file1Writer.write(data.toString());
        file1Writer.close();
        //删除readme.txt文件
File file2 = new File("src/step3/readme.txt");   //file对象指向src文件夹下的step1文件夹下的Hello World.txt文件
if (file2.exists()) {   //如果文件存在
    file2.delete();   //删除该文件，并输出布尔值
}

        /********** End *********/
    }
}
