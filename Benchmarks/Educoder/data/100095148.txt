package step3;

public class Factorial {
	public void factorial(int n) {
		/********** Begin *********/
        long j=1;
        long k=1;
      
        for(int i=1;i<=n;i++){
            j=j*k;
            k++;
            
        }
		/********** End *********/
        System.out.print(j);
    }
}
