package step1;

public class Yard {
    int length;
    int width;
    Yard (int length,int width)
    {
        this.length=length;
        this.width=width;
    }
    double getPerimeter()
    {
        return 2.0*(length+width);
    }
    double getAcreage()
    {
        return 1.0*length*width;
    }
}
