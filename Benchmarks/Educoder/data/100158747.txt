#include <stdio.h>
int n,k;
int min = 99999999;
int arr[100][100];
void dfs(int i,int count,int flag){
    if(i == n){
        if(min>count)min = count;
        return ;
    }
    for(int j=0;j<k;j++)
    {
        if(j!=flag)dfs(i+1,count+arr[i][j],j);
    }
}
int main()
{
    scanf("%d %d",&n,&k);
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<k;j++)
            scanf("%d",&arr[i][j]);
    }
    dfs(0,0,-1);
    printf("%d",min);
}
