package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int[] _num1 = new int[3];
        int sum= 0;
		for(;num1<=num2;num1++)
        {
        	_num1[2] = (num1%100)%10;
			_num1[1] = (num1%100)/10;
			_num1[0] = num1/100;
            sum = sum + _num1[0] + _num1[1] + _num1[2];
   	    }
		return sum;
		/********** END **********/
	}
}
