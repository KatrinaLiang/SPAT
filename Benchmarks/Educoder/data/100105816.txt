package step1;

public class Yard {
    private int length,width;
    public Yard(int length,int width)
    {
        this.length=length;
        this.width=width;
    }
    public double getPerimeter()
    {
        double c=length+width;
        return c*2;
    }
    public double getAcreage()
    {
        double d=length*width;
        return d;
    }
    
        
}
