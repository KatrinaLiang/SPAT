package step1;

public class Person {
    private String name; //姓名
    private String number; //身份证号
    private Date birthday; //出生日期
    
   /************Begin*************/
	public Person(){
    	this.name = "null";
        this.number = "null";
        birthday = new Date();
    }
    public Person(String name,String num,Date birthday){
        this.name = name;
        this.number = num;
        this.birthday = birthday;
    }
	public Person(String name,String num,int year,int month,int day){
        Date birthDate = new Date(year, month, day);
        this.name = name;
        this.number = num;
        this.birthday = birthDate;
    }
    
    public String toString(){
        return "name="+this.name+" number="+this.number+" birthday:"+birthday.toString();
    }

	public void setinfo(String name, String num, Date birthday){
        this.name = name;
        this.number = num;
        this.birthday = birthday;
    }
    
    public String getname(){
        return this.name;
    }
    public String getnum(){
        return this.number;
    }
    public Date getBirthday(){
        return this.birthday;
    }
    
    
    
   /************End*************/
}
