package Pid_108;
import java.util.*;
public class Main{
public static void main(String args[]){
    int number,a,hang;
    Scanner in=new Scanner(System.in);
    number=in.nextInt();
    a=number;
    hang=0;
    for(int i=1;i<=number;i++)
        if(a-i>=0){
            hang++;
            a-=i;
        }   
    else
        break;
        
    System.out.print(hang);		
	}
}
