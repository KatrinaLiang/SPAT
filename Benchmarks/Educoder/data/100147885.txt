public class ThreadHelper{
    static public void calcOnNewThread(int n){
        new Thread(new Runnable(){
            public void run(){
                int result=1;
                for(int i=1;i<=n;i++){
                    result=result*i;
                }
                System.out.println(result);
            }
        }).start();
    }
}
