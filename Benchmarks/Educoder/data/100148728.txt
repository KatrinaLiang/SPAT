package step1;

public class Task {
	
	public void sort(Integer[] arr) {
      /********* Begin *********/

      boolean isSort = true;
      boolean type = true;
      boolean finish = false;
      int s = 10;

      while(!finish && s-- >0) {
         boolean flag = true;
         if(type) {
            for(int i = 0; i < arr.length - 1; i++) {
               if(arr[i] > arr[i + 1]) {
                  int k = arr[i];
                  arr[i] = arr[i + 1];
                  arr[i + 1] = k;
                  print(arr);
                  flag = false;
                  isSort = false;
               }
            }
         }
         else {
            for(int i = arr.length - 1; i > 0 ; i--) {
               if(arr[i] < arr[i - 1]) {
                  int k = arr[i];
                  arr[i] = arr[i - 1];
                  arr[i - 1] = k;
                  print(arr);
                  flag = false;
                  isSort = false;
               }
            }
         }
         finish = flag;
         type = !type;
         if(isSort) {
            print(arr);
         }
      }

      /********* End *********/

   }

   private void print(Integer[] arr) {
      for(int i = 0; i < arr.length; i++) {
         if(i == arr.length - 1) {
            System.out.print(arr[i] + " ");
            System.out.println();
         }
         else {
            System.out.print(arr[i] + " ");
         }
      }
   }
    
}
