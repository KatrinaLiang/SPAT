#include <iostream>

#include <stdio.h>
using namespace std;
int sum=0;
int dfs(int *a,int i,int j,int k,int t,int len)
{
    
	if (i!=j&&i!=k&&j!=k)
    {
    	if (a[i]+a[j]+a[k]<t)
        {
        	sum++;
            if (i+1!=j&&i+1!=k&&j!=k&&i+1<len)
            {
            	dfs(a, i+1, j, k, t, len);
            }
             if (i!=j+1&&i!=k&&j+1!=k&&j+1<len)
            {
            	dfs(a, i, j+1, k, t, len);
            }
             if (i!=j&&i!=k+1&&j!=k+1&&k+1<len)
            {
            	dfs(a, i, j, k+1, t, len);
            }
        }
    }
    else if(i+1<len&&j+1<len&&k+1<len) 
    {
    	if (i+1!=j&&i+1!=k&&j!=k&&i+1<len)
            {
            	dfs(a, i+1, j, k, t, len);
            }
             if (i!=j+1&&i!=k&&j+1!=k&&j+1<len)
            {
            	dfs(a, i, j+1, k, t, len);
            }
             if (i!=j&&i!=k+1&&j!=k+1&&k+1<len)
            {
            	dfs(a, i, j, k+1, t, len);
            }
    }
    return sum;
}
int main()
{
    
    int a[1000];
    int i,j,k,len,t;
    i=0;

    
    while (scanf("%d",&a[i])!=EOF)
    {
        i++;
    }
    len = i-1;
    t=a[len];
    for (i=0;i<len-2;i++)
        for (j=i+1;j<len-1;j++)
            for (k=j+1;k<len;k++)
            {
            if (a[i]+a[j]+a[k]<t)
                sum++;
            }
    
    printf ("%d",sum);
    return 0;
}
