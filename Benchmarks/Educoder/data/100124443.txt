package step2;
public class Fibonacci{
	public static void main(String args[]){
	/*********begin********/
		int[] a=new int[20];
        int sum=0,i;
        a[0]=1;
        a[1]=1;
        for(i=2;i<20;i++){
            a[i]=a[i-1]+a[i-2];
        sum=sum+a[i];
        }
        System.out.print("前20个元素的和为"+(sum+2));

	/********end********/
	}
}
