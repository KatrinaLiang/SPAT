package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
		int m,n;
		//在这里打印出正三角形 
		for(m=1;m<=10;m++){
		for(n=1;n<=m;n++){ 
		System.out.print("*");}
		System.out.println();}
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
		for(m=10;m>0;m--){
		for(n=1;n<=m;n++){ 
		System.out.print("*");}
		System.out.println();}
		
		
		
		
		/*****end*****/
	}
}
