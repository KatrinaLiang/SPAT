package step4;

import java.util.*;

public class HelloWorld {
	
	
	public static void main(String[] args) {
		
        Scanner sc = new Scanner(System.in);
		//程序会输入三次数据
		/********** Begin **********/
        ArrayList<String> list2 = new ArrayList<String>();
        for(int i=0;i<3;i++){
		list2.add(sc.next());
		String str2 = list2.get(i);
		System.out.println("集合的第"+(i+1)+"个数据为："+str2);
        }
		
		
		
		/********** End **********/
	}
	
}
