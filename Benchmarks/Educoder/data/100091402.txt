package step3;
public class NarcissisticNumber {

    //判断一个三位数是否是水仙花数
    public boolean NarcissisticNumber(int n){
        // 请在此添加实现代码
        int save,r,cnt=0, sum=0;
        save = n;
        while(n>0){
          cnt++;
          r = n % 10;
          sum += r*r*r;
          n = n/10;  
        }
        if(cnt==3 && sum==save)
            return true;
        else
            return false;
    }
}
