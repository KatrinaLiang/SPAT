
import java.util.Scanner;
public class Main {

	public static void main(String[] args) {
		// write your code here
        int n;
		Scanner scan=new Scanner(System.in);
		n=scan.nextInt();
		if(cal(n)==true){
			System.out.print("YES");
		}
			else if(cal(n)==false)
			{
				System.out.print("NO");
			}
        
    }
    static boolean cal(int n){
		int a[]=new int[5];
		a[0]=n%10;
		a[1]=(n/10)%10;
		a[2]=(n/100)%10;
		a[3]=(n/1000)%10;
        a[4]=(n/10000)%10;
		int k=a[0]*a[0]+a[1]*a[1]+a[2]*a[2]+a[3]*a[3]+a[4]*a[4];
		if(k==1)
			return true;
        else if(k<10&&k>=0)
            return false;
		else
			return cal(k);
	}
}
