package step4;
		//创建类 添加属性和方法
class WuMingFen{
    String Liang;
    int quantity;
    boolean withSoup;
    public WuMingFen(String ma,int q,boolean soup)
    {
        Liang=ma;
        quantity=q;
        withSoup=soup;
    }
    public WuMingFen(String ma,int q)
    {
        Liang=ma;
        quantity=q;
        withSoup=false;
    }
    public WuMingFen()
    {
        Liang="酸辣";
        quantity=2;
        withSoup=true;
    }
    public void check()
    {
        System.out.println("面码："+Liang+"，粉的份量："+quantity+"两，是否带汤："+withSoup);
    }
}
