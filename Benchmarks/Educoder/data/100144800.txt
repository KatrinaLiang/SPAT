package step3;

public class Task {
	public void rotate(int[][] matrix){
		 if (matrix.length == 1) {
            return;
        }
        if (matrix.length != matrix[0].length) {
            return;
        }

        int n = matrix.length;
        int[][] matrix2 = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                matrix2[j][i] = matrix[i][j];
            }
        }

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                matrix[i][j] = matrix2[i][n - j - 1];
            }
        }
	}
}
