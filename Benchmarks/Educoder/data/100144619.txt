package step1;

public class Task {
	public String inversion(String str){
		/********* Begin *********/	
    char[] chararr = str.toCharArray();
    StringBuilder builder = new StringBuilder();
    for (int i = chararr.length-1; i >= 0; i--) {
        builder.append(chararr[i]);
    }
    return builder.toString();
				/********* End *********/
	}
}
