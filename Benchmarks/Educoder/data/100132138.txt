package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temps){
	 /********** BEGIN **********/

        int[] array = new int[temps.length];

        for (int i = 0; i < temps.length; ++i) {
            int temp = 0;
            for (int j = i + 1; j < temps.length; ++j) {
                if (temps[j] > temps[i]) {
                    temp = j - i;
                    break;
                }
            }
            array[i] = temp;
        }

        return array;
	}
}
