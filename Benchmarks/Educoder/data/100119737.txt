package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
		
		//在这里打印出正三角形 
		int y,p;
        
        
            for(y=1;y<=10;y++)
            {
                for(p=1;p<=y;p++)
                {
                    System.out.print("*");
                }
               System.out.println();
            }
            
        

		
		
		
		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
		int h,j;
        for(h=1;h<=10;h++){
            for(j=1;j<=11-h;j++){
                System.out.print("*");
            }
            System.out.println();
        }
		/*****end*****/
	}
}
