package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
		int i = 1;
		int j;
		while (i <= 9) {
		    j = 1;
		    while (j <= i) { 
                System.out.print(j + "*" + i + "=" +(i * j) + '\t');    
		        j++;
            }  
             System.out.print('\n');
             i++;  
		/*****end*****/
	}
}
}
