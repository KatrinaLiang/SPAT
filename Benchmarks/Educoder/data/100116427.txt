package step3;

public class HelloWorld {
	public static void main(String[] args) {
		int count= 0;	//定义变量存储6的倍数出现的次数
		/*****start*****/
		int i=6;
        int j=1;
        while(j<=100){
            j++;
        
        if(j%i==0)
            count++;
        }
		/*****end*****/
		System.out.println("6的倍数出现的次数为：" + count);
	}
}
