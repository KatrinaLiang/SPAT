package step1;

public class Task {
	public  int[] dailyTemps(Integer[] temperatures){
		/********** BEGIN **********/
		int length = temperatures.length;
        int[] up = new int[length];
        for (int i = 0; i < length; i++){
        	for (int j = i; j < length; j++){
            	if (temperatures[j] > temperatures[i]){
                    up[i] = j - i;
                    break;
            	}
        	}
       }
		return up;
		/********** END **********/
	}
}
