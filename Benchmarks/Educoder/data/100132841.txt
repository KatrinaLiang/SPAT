package step3;

public class Task {

	 public int countOfShips(char[][] ferry) {
	  /**************BEGIN************/
		 int num = 0;
	      for(int i = 0;i<ferry.length;i++){
	      		for(int j = 0;j<ferry[i].length;j++){
	            		if(ferry[i][j]=='+'){
	                    	num++;
	                    	ferry[i][j] = 'o'; 
	                    	if(j<ferry[i].length-1) {
		                    	if(ferry[i][j+1]=='+'){
		                    		for(int x = j+1;x<ferry[i].length;x++) {
		                    			if(ferry[i][x]=='+') {
		                    				ferry[i][x]='o';
		                    			}
		                    			else break;
		                    			
		                    		}
		                        }
	                    	}
	                    	if(i<ferry.length-1) {
	                        if(ferry[i+1][j]=='+') {
	                        	for(int x = i+1;x<ferry.length;x++) {
	                    			if(ferry[x][j]=='+') {
	                    				ferry[x][j]='o';
	                    			}
	                    			else break;
	                    			
	                    		}
	                        }
	                    	}
	                    }
	            }

	      } 
	      return num;
      /***********END ***************/
	 }
}
