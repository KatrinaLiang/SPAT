package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int num=0;
		for (int i = num1; i <= num2; i++) {
			for (int j = 0; j < (i+"").length(); j++) {
				String sub = (i+"").substring(j, j+1);
				num+=Integer.parseInt(sub);
			}
		}
		return num;
		
		
		/********** END **********/
	}
}
