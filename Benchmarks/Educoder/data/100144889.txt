package step3;

public class Task {
	public void rotate(int[][] matrix){
		int len = matrix.length;
		for(int i = 0 ; i < len  ;i++) {
			for(int j = 0 ; j < len -i ; j++) {
				int t = matrix[i][j];
				matrix[i][j] = matrix[len-1-j][len-1-i];
				matrix[len-1-j][len-1-i] =t;
			}
		}
		for(int i = 0 ; i < len/2 ;i++) {
			for(int j = 0 ; j < len ; j++) {
				int t = matrix[i][j];
				matrix[i][j] = matrix[len-1-i][j];
				matrix[len-1-i][j] =t;
			}
		}
		
	
		/********* End *********/
	}
}
