package step3;

import java.util.Arrays;
import java.util.Scanner;

public class RecordRecovery {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int[] record = new int[10];
        int[] recordRecovery = new int[10];
        /********** Begin *********/
       /**用record数组保存从命令行输入的10个int型数据。**/ 
        int num=0;
        for (int i=0;i<10;i++) {
        	record[i]=scanner.nextInt();
        }
        /**recordRecovery数组保存去除重复数据后的元素。示例：数组1 1 2 2 3 3中删去重复的数字，即转变为1 2 3**/
        int []flag=new int[10];
        for (int i=0;i<10;i++) {
        	flag[i]=1;
        }
        for(int i=0;i<record.length;i++) {
        	int a=record[i];
        	for(int m=i+1;m<record.length;m++) {
        		if(a==record[m]) {
        			flag[m]=0;
        		}  		
        	}
        }
        for(int i=0;i<record.length;i++) {
        	if(flag[i]==1) {
        		recordRecovery[num]=record[i];
        		num++;
        	}
        	
        } 
        /**按照数字大小顺序，循环输出recordRecovery数组。**/
        Arrays.sort(recordRecovery,0,num);
        /********** End *********/
        for (int i = 0; i < num; i++) {
            System.out.print(recordRecovery[i]);
            if (i != num - 1) {
                System.out.print(' ');
            }
        }
    }
}
