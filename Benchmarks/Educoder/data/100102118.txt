package step2;

import java.io.File;
import java.io.IOException;

public class FileUtil {
	public static void start() {
		/********** Begin *********/
		File file=new File("src/step2");
		File[] fileFile=file.listFiles();
		for(File s:fileFile)
		{
			if(s.isDirectory()==true)
			{
				System.out.println(s.getName());
			}
		}
		for(File s:fileFile)
		{
			if(s.isDirectory()==false)
			{
				System.out.println(s.getName());
			}
		}
		/********** End *********/
	}
}
