public class Task {
	public String inversion(String str){
		/********* Begin *********/
		 char[] arr = str.toCharArray();
        String reverse = "";
        for(int i=arr.length-1;i>=0;i--){
            reverse +=arr[i];
        }   
        return reverse;
		
		/********* End *********/
	}
}
