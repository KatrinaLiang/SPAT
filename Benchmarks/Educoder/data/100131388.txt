package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
        int sum = 0;
        for(int i=num1;i<=num2;i++){
        	int m = i;
            while(m>0){
                int c = m % 10;
                sum += c;
            	m = m / 10;
            }
        }
		return sum;
		/********** END **********/
	}
}
