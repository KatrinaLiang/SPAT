package step3;

public class ForPractice3 {
	public static void main(String[] args) {
		/*****start*****/
	for(int i=1;i<=9;i++){
		for(int j=1;j<=i;j++){
			int sum=i*j;
			System.out.print(j+"*"+i+"="+sum+"\t");
		    
		}
	System.out.println();
	}	
	
		
		
		
		
		/*****end*****/
	}
}
