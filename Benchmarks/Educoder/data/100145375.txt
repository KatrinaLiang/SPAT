package step2;

public class Task {

	public int getNum(int n){
		/********* Begin *********/
        int ren = 0;
		for(;n != 1;)
		{
			if(n%2 == 1)
			{
				n = (n - 1)/2;
				ren += 1;
			}
			else 
			{
				n = n/2;
			}
		}
		ren += 1;
		return ren;
		
		
		
		
		/********* End *********/
	}
	
}
