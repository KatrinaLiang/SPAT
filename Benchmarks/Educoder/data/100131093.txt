package step1;

public class Task {

	
	public int getSum(int num1, int num2) {
		/********** BEGIN **********/
		int a,b;
		int sum=0;
		if(num2<num1) {
			int tem;
			tem=num1;
			num1=num2;
			num2=tem;
		}
		for(int j=num1;j<=num2;j++) {
			b=j;
			while(j!=0){
				a=j%10;
				j=j/10;
				sum=sum+a;
			}
			j=b;
		}
        return sum;
		/********** END **********/
	}
}
