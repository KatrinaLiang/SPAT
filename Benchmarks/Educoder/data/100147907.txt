//注意文件名，这里应该写MyRunnable类
/***begin your code here***/
 public class MyRunnable implements Runnable{
int num;
     MyRunnable(int num){
         this.num=num;}
     public void  run(){
     try   
       {   
          Thread.sleep(2000);// 延迟2秒   
        }   
        catch (InterruptedException e)   
        {   
            System.out.println(e.getMessage());   
       } 

         int result=1;
         for(int i=1;i<=num;i++)
             result*=i;
      System.out.println(result);
     }
 }
