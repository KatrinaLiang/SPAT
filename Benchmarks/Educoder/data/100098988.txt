package step1;

public class Yard {
    private int length=1;
    private int width=1;
    Yard()
    {
        length=1;
    	width=1;
    }
    Yard(int length,int width)
    {
        this.length=length;
        this.width=width;
    }
    public double getPerimeter()
    {
        return (length+width)*2.0;
    }
	public double getAcreage()
    {
     	return length*width*1.0;   
    }
}
