package step1;

public class ShuShu {

	// 第一题
	public void first() {
		/*********Start********/
					int k = 0;
			while (true) {
				k++;
				System.out.print(k + "\t");
				if (k % 10 == 0) {
					System.out.print("\n");
				}
				if (k >= 100) {
					break;
				}
			}
		/*********End********/
	}
	
	// 第二题
	public void second() {
		/*********Start********/
		      System.out.print("\n");
int k = 0;  
			do {
				k++;
				System.out.print(k + "\t");
				if (k % 10 == 0) {
					System.out.print("\n");
				}
				if (k >= 100) {
					break;
				}
	
			} while (true);
		/*********End********/
	}
	
	// 第三题
	public void third() {
		/*********Start********/
        System.out.print("\n");
					for (int k = 1; true; k++) {
				System.out.print(k + "\t");
				if (k % 10 == 0) {
					System.out.print("\n");
				}
				if (k >= 100) {
					break;
				}
                    }
		/*********End********/
	}
}
