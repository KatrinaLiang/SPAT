package step2;

import java.io.File;
import java.io.IOException;

public class FileUtil {
	public static void start() {
		/********** Begin *********/
		File file = new File("src/step2");
        File[] files = file.listFiles();
		for(File f:files){
            if(f.isDirectory()){
                System.out.println(f.getName());
            }
        }
        for(File f2:files){
            if(f2.isFile()){
                System.out.println(f2.getName());
            }
        }
		/********** End *********/
	}
}
