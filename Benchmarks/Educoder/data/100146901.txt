package step1;

public class GetScores {

	/********* Begin *********/
    public int getLostScores(int[] nums){
       //请在此补全你的代码
    	int[] a = new int[10001];
        int i,j,k=0;
        int result;
        long result2;
        long[] c = new long[3];
        int[] clen = new int[3];
        for(i=1;i<=10000;i++)a[i]=0;
        for(i=0;i<9997;i++) a[nums[i]]=1;
        for(i=1;i<=10000;i++){
            if(a[i]==0){
                c[k]=i;
                k++;
            }
        }
		for(j=0;j<3;j++)
        {
            if(c[j]<10)clen[j]=1;
            else if(c[j]<100)clen[j]=2;
            else if(c[j]<1000)clen[j]=3;
            else clen[j]=4;
        }
        for(i=clen[2];i>0;i--)
        {
            c[1]=c[1]*10;
        }
        for(i=clen[2]+clen[1];i>0;i--)
        {
            c[0]=c[0]*10;
        }
        result2 = (c[0]+c[1]+c[2])%11;
        result = Integer.parseInt(String.valueOf(result2));
		return result;
	}

	/********* End *********/
}
