package step2;
import java.math.*;

public class AdditionCalculator {
	public void addition(int a, int n) {
		/********** Begin *********/
        int i=1,sum=2,b=a;
		while(i<n){
        	b=(int)(b+a*Math.pow(10, i));
            sum=sum+b;
            i++;
        }
		System.out.print(sum);
		/********** End *********/
	}
}
