package task4;

public class Box {

    private double width;
    private double height;
    private double depth;

  /********* Begin *********/        
	void setWidth(double width){
        this.width=width;
    }
    void setHeight(double height){
        this.height=height;
    }
    void setDepth(double depth){
        this.depth=depth;
    }
	double getWidth(){
        return width;
    }
    double getHeight(){
        return height;
    }
	double getDepth(){
        return depth;
    }

  /********* End *********/        

}
